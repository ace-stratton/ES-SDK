179void crypto_init(void)
180{
181 MODID_VersionInfo_t ver_info;
183 crtc.os_task_hnd = NULL;
184 crtc.drbg_hnd = NULL;
185
186 for (uint16_t user_entry = 0; user_entry < CDEF_ELEMENT_COUNT(crtc.user_queue); user_entry++)
187 {
188 (void) memset(crtc.user_queue[user_entry].src_file_name, 0, sizeof(crtc.user_queue[user_entry].src_file_name));
189 (void) memset(crtc.user_queue[user_entry].target_file_name, 0, sizeof(crtc.user_queue[user_entry].target_file_name));
190 crtc.user_queue[user_entry].p_user_notif = NULL;
191 }
192
193 csq_algo_clear(&crypto_file_queue);
194
195 crtc.cmox_init_retval = cmox_initialize(NULL);
196
197 // no need to check this but still keep it during debugging (if it is invalid, one of the other CMOX lib API calls will fail
198 (void) crtc.cmox_init_retval;
199
200 module_id_get_info(&ver_info);
201
202 crtc.drbg_hnd = cmox_ctr_drbg_construct(&crtc.drbg_ctr_hnd, CMOX_CTR_DRBG_AES256);
203
204 CRIT_ASSERT(NULL != crtc.drbg_hnd);
205
206 uint8_t nonce[NONCE_SIZE_BYTES];
207
208 // initialise entropy and nonce buffers with TRNG values
209 rng_fill(crtc.entropy, sizeof(crtc.entropy));
210 rng_fill(nonce, sizeof(nonce));
211
212 // initialise the pseudo-random generator
213 crtc.drbg_retval = cmox_drbg_init(crtc.drbg_hnd,
214 crtc.entropy,
215 sizeof(crtc.entropy),
216 (const uint8_t *) &ver_info,
217 sizeof(ver_info),
218 nonce,
219 sizeof(nonce));
220
221 crypto_cfg_t crypto_cfg;
222
223 Nvm_GetBlockById(NVM_BLOCK_CRYPTO_CFG, &crypto_cfg);
224
225 crtc.is_key_empty = !is_key_valid(crypto_cfg.aes_key, sizeof(crypto_cfg.aes_key));
226
227 // clear out key-related data from stack
228 (void) memset(&crypto_cfg, 0, sizeof(crypto_cfg));
229}

315{

327 p_stream_hnd->cipher_text_bytes_read = 0U;
328 p_stream_hnd->cipher_ctx = NULL;
329

331
332 safe_copy_str(p_stream_hnd->src_file_name, p_src_file_name, sizeof(p_stream_hnd->src_file_name));
333

344
345 set_next_stage(p_stream_hnd, &read_from_file);

348 }
349 }

358crypto_api_res_t crypto_stream_open(crypto_stream_hnd_t * const p_stream_hnd, const char * const p_src_file_name, const cipher_alg_type_t impl_type)
359{
360 CRIT_ASSERT(NULL != p_stream_hnd);
361 CRIT_ASSERT(NULL != p_src_file_name);
362 CRIT_ASSERT(impl_type < OP_CMOX_MAX);
363
364 crypto_api_res_t init_res = CRYPTO_API_RES_ERROR;
365
366 if (crtc.is_key_empty)
367 {
368 return CRYPTO_API_RES_NO_KEY;
369 }
370
371 if ((NULL != p_stream_hnd) && (NULL != p_src_file_name) && (impl_type < OP_CMOX_MAX))
372 {
373 p_stream_hnd->impl_type = impl_type;
374 p_stream_hnd->stage_bytes_read = 0U;
375 p_stream_hnd->cipher_text_bytes_read = 0U;
376 p_stream_hnd->cipher_ctx = NULL;
377
378 set_next_stage(p_stream_hnd, NULL);
379
380 safe_copy_str(p_stream_hnd->src_file_name, p_src_file_name, sizeof(p_stream_hnd->src_file_name));
381
382 FRESULT f_src_res = SdMngr_f_open(&p_stream_hnd->f_src_hnd, p_src_file_name, FA_OPEN_EXISTING | FA_READ);
383
384 if (FR_OK != f_src_res)
385 {
386 CRYPTO_TRACE_DBG("[crypto_stream_open] file system error: %d\n", f_src_res);
387 return CRYPTO_API_RES_FILE_ERROR;
388 }
389
390 if (OP_CMOX_AES_GCM_ENC == impl_type)
391 {
392 init_res = prep_file_for_stream_encr(p_stream_hnd);
393 } else if (OP_CMOX_AES_GCM_DEC == impl_type)
394 {
395 init_res = prep_file_for_stream_decr(p_stream_hnd);
396 }
397 else
398 {
399 CRIT_ASSERT(false);
400 }
401
402 if (CRYPTO_API_RES_OK == init_res)
403 {
404 p_stream_hnd->cipher_ctx = cmox_gcm_construct(&p_stream_hnd->gcm_hnd, *(cipher_alg_map[(uint8_t) impl_type].cmox_gcm_impl));
405
406 // prepare CMOX library context
407 cmox_cipher_retval_t cipher_retval = cmox_cipher_init(p_stream_hnd->cipher_ctx);
408
409 if (cipher_retval != CMOX_CIPHER_SUCCESS)
410 {
411 CRYPTO_TRACE_DBG("cmox_cipher_init() failed with code %d\n", cipher_retval);
412
413 return CRYPTO_API_RES_CRYPTOLIB_ERROR;
414 }
415
416 cipher_retval = cmox_cipher_setTagLen(p_stream_hnd->cipher_ctx, AUTH_TAG_SIZE);
417
418 if (cipher_retval != CMOX_CIPHER_SUCCESS)
419 {
420 CRYPTO_TRACE_DBG("cmox_cipher_setTagLen() failed with code %d\n", cipher_retval);
421
422 return CRYPTO_API_RES_CRYPTOLIB_ERROR;
423 }
424
425 crypto_cfg_t crypto_cfg_blk;
426
427 Nvm_GetBlockById(NVM_BLOCK_CRYPTO_CFG, &crypto_cfg_blk);
428
429 cipher_retval = cmox_cipher_setKey(p_stream_hnd->cipher_ctx, crypto_cfg_blk.aes_key, sizeof(crypto_cfg_blk.aes_key));
430
431 if (cipher_retval != CMOX_CIPHER_SUCCESS)
432 {
433 CRYPTO_TRACE_DBG("cmox_cipher_setKey() failed with code %d\n", cipher_retval);
434
435 return CRYPTO_API_RES_CRYPTOLIB_KEY_ERROR;
436 }
437
438 // clear out key-related data from the stack after writing the keys to the STM32 key register
439 (void) memset(&crypto_cfg_blk, 0, sizeof(crypto_cfg_blk));
440
441 cipher_retval = cmox_cipher_setIV(p_stream_hnd->cipher_ctx, p_stream_hnd->encr_block_hdr.iv, sizeof(p_stream_hnd->encr_block_hdr.iv));
442
443 if (cipher_retval != CMOX_CIPHER_SUCCESS)
444 {
445 CRYPTO_TRACE_DBG("cmox_cipher_setIV() failed with code %d\n", cipher_retval);
446
447 return CRYPTO_API_RES_CRYPTOLIB_IV_ERROR;
448 }
449
450 cipher_retval = cmox_cipher_appendAD(p_stream_hnd->cipher_ctx, (const uint8_t *) p_stream_hnd->encr_block_hdr.aad, sizeof(p_stream_hnd->encr_block_hdr.aad));
451
452 if (cipher_retval != CMOX_CIPHER_SUCCESS)
453 {
454 CRYPTO_TRACE_DBG("cmox_cipher_appendAD() failed with code %d\n", cipher_retval);
455
456 return CRYPTO_API_RES_CRYPTOLIB_AAD_ERROR;
457 }
458 }
459 }
460
461 return init_res;
462}

464crypto_api_res_t crypto_stream_read(crypto_stream_hnd_t * const p_stream_hnd, uint8_t * const p_output_buf, const uint32_t output_buf_size, uint32_t * const p_bytes_read)
465{
466 CRIT_ASSERT(NULL != p_stream_hnd);
467 CRIT_ASSERT(NULL != p_output_buf);
468 CRIT_ASSERT(NULL != p_bytes_read);
469 CRIT_ASSERT(output_buf_size > 0);
470
471 crypto_api_res_t op_res = CRYPTO_API_RES_FILE_ERROR;
472 uint32_t remaining_bytes = output_buf_size;
473 uint32_t bytes_read = 0U;
474 uint32_t actual_bytes_to_read = 0U;
475 uint8_t tmp_read_buf[CIPHER_OP_CHUNK_SIZE_BYTES];
476 p_stream_stage_pipeline_t p_read_func = NULL;
477
478 while (remaining_bytes > 0)
479 {
480 actual_bytes_to_read = (uint32_t) fmin(sizeof(tmp_read_buf), remaining_bytes);
481
482//this is done to ignore the warning "ISO C forbids conversion of object pointer to function pointer type [-Wpedantic]"
483 static_assert(sizeof(p_stream_hnd->p_current_stage_hnd) == sizeof(p_read_func));
484PRAGMA_DIAG_PUSH()
485PRAGMA_DIAG_IGNORED(-Wpedantic)
486 p_read_func = (p_stream_stage_pipeline_t) p_stream_hnd->p_current_stage_hnd;
487PRAGMA_DIAG_POP()
488
489 if (NULL != p_read_func)
490 {
491 op_res = (*p_read_func)(p_stream_hnd, tmp_read_buf, &p_output_buf[output_buf_size - remaining_bytes], actual_bytes_to_read, &bytes_read);
492
493 if (CRYPTO_API_RES_OK == op_res)
494 {
495 remaining_bytes -= bytes_read;
496 *p_bytes_read += bytes_read;
497 }
498 else
499 {
500 *p_bytes_read = 0;
501
502 return op_res;
503 }
504 }
505 else
506 {
507 // all sections covered, streaming ends here
508 return CRYPTO_API_RES_EOS;
509 }
510 }
511
512 // we end up here if the stream is not over and we filled-in the complete user-provided buffer with
513 // data
514
515 return CRYPTO_API_RES_OK;
516}

603STATIC crypto_req_status_t queue_cipher_op(const char * const p_src_file_name, const char * const p_target_file_name, const p_crypto_notif p_user_cbk, cipher_alg_type_t alg_type)
604{
605 CRIT_ASSERT(NULL != p_src_file_name);
606 CRIT_ASSERT(NULL != p_target_file_name);
607 CRIT_ASSERT(alg_type < OP_CMOX_MAX);
608
609 crypto_req_status_t res = CRYPTO_REQ_STATUS_QUEUE_FULL;
610
611 FILINFO fno;
612 FRESULT fres = SdMngr_f_stat(p_src_file_name, &fno);
613 if ((NULL == p_src_file_name) || (FR_NO_FILE == fres) || (FR_NO_PATH == fres))
614 {
615 res = CRYPTO_REQ_STATUS_NO_FILE;
616 }
617 else if (crtc.is_key_empty)
618 {
619 res = CRYPTO_REQ_STATUS_NO_KEY;
620 }
621 else
622 {
623 if ((NULL != p_src_file_name) && (NULL != p_target_file_name))
624 {
625 osMutexAcquire(crypto_lock_id, osWaitForever);
626
627 uint32_t reserved_slot_idx = 0U;
628 bool is_slot_reserved = csq_algo_reserve(&crypto_file_queue, &reserved_slot_idx);
629
630 // if source file name is not assigned, then the slot is assumed to be free
631 if (is_slot_reserved)
632 {
633 safe_copy_str(crtc.user_queue[reserved_slot_idx].src_file_name,
634 p_src_file_name,
635 sizeof(crtc.user_queue[reserved_slot_idx].src_file_name));
636
637 safe_copy_str(crtc.user_queue[reserved_slot_idx].target_file_name,
638 p_target_file_name,
639 sizeof(crtc.user_queue[reserved_slot_idx].target_file_name));
640
641 crtc.user_queue[reserved_slot_idx].p_user_notif = p_user_cbk;
642 crtc.user_queue[reserved_slot_idx].cipher_type = alg_type;
643
644 CRYPTO_TRACE_DBG("new file queued for processing (alg: %d): %s\n", (uint8_t) alg_type, p_src_file_name);
645
646 res = CRYPTO_REQ_STATUS_OK;
647 }
648 else
649 {
650 CRYPTO_TRACE_DBG("could not enqueue file for processing (alg: %d): %s\n", (uint8_t) alg_type, p_src_file_name);
651 }
652
653 osMutexRelease(crypto_lock_id);
654
655 if (CRYPTO_REQ_STATUS_OK == res)
656 {
657 // signal main task to start encryption on existing queue entries
658 osThreadFlagsSet(crtc.os_task_hnd, UNBLOCK_CRYPTO_TASK);
659 }
660 }
661 }
662
663 return res;
664}

670{
671 // currently unused
672 (void) p_task_arg;
673
674 for (;;)
675 {
676 bool is_queue_data_valid = false;
677 uint32_t wait_ret = osThreadFlagsWait(UNBLOCK_CRYPTO_TASK, osFlagsWaitAny, CRYPTO_SERVICE_IDLE_PERIOD_MS);
678
679 if (UNBLOCK_CRYPTO_TASK == wait_ret)
680 {
681 CRYPTO_TRACE_DBG("crypto service queue processing triggered!\n");
682
683 // notify pending users for the abort operation
684 if (CRYPTO_ABORT_OP_ALL == crtc.abort_op)
685 {
686 osMutexAcquire(crypto_lock_id, osWaitForever);
687
688 do
689 {

713 osMutexRelease(crypto_lock_id);

717 timer_handle_t profile_timer;
718
719 timer_start(&profile_timer);

721 crypto_api_res_t cipher_op_res = crypt_decrypt_file(crtc.tmp_data.src_file_name, crtc.tmp_data.target_file_name, crtc.tmp_data.cipher_type);

730 CRYPTO_TRACE_DBG("user notification sent with result %d!\n", (uint16_t) cipher_op_res);

732 }
733 }
734 }
735 }

737 TaskMonitor_IamAlive(TASK_CRYPTO_SRV);
738 vApplicationLowStackCheck(TASK_CRYPTO_SRV);

799STATIC crypto_api_res_t crypt_decrypt_file(const char * const p_src_file_name, const char * const p_target_file_name, const cipher_alg_type_t impl_type)
800{
801 crypto_api_res_t op_res = CRYPTO_API_RES_FILE_ERROR;
802 crypto_stream_hnd_t crypto_stream_hnd;
803 uint8_t tmp_cipher_buf[CIPHER_OP_CHUNK_SIZE_BYTES];
804 FIL target_file_hnd;
805
806 if ((NULL != p_src_file_name) && (NULL != p_target_file_name) && (impl_type < OP_CMOX_MAX))
807 {
808 crypto_api_res_t stream_res = crypto_stream_open(&crypto_stream_hnd, p_src_file_name, impl_type);
809 FRESULT f_target_res = SdMngr_f_open(&target_file_hnd, p_target_file_name, FA_CREATE_ALWAYS | FA_WRITE);
810
811 if (CRYPTO_API_RES_OK != stream_res)
812 {
813 op_res = stream_res;
814 goto stream_cleanup;
815 }
816
817 if (FR_OK == f_target_res)
818 {
819 while (CRYPTO_API_RES_OK == stream_res)
820 {
821 uint32_t bytes_read = 0U;
822 uint32_t bytes_written = 0U;
823
824 // make sure we are not reset by the WD if bigger streams get processed
825 TaskMonitor_IamAlive(TASK_CRYPTO_SRV);
826
827 stream_res = crypto_stream_read(&crypto_stream_hnd, tmp_cipher_buf, sizeof(tmp_cipher_buf), &bytes_read);
828
829 if ((CRYPTO_API_RES_OK == stream_res) || (CRYPTO_API_RES_EOS == stream_res))
830 {
831 // write the encrypted data to storage
832 f_target_res = SdMngr_f_write(&target_file_hnd, tmp_cipher_buf, bytes_read, (UINT *) &bytes_written);
833
834 if ((FR_OK != f_target_res) || (bytes_written != bytes_read))
835 {
836 CRYPTO_TRACE_DBG("file write failed with code %d\n", f_target_res);
837
838 op_res = CRYPTO_API_RES_FILE_ERROR;
839
840 break;
841 }
842
843 if (CRYPTO_API_RES_EOS == stream_res)
844 {
845 CRYPTO_TRACE_DBG("streaming completed!\n", stream_res);
846
847 op_res = stream_res;
848
849 break;
850 }
851 }
852 else
853 {
854 CRYPTO_TRACE_DBG("stream error %d\n", stream_res);
855
856 op_res = stream_res;
857
858 break;
859 }
860 }
861 }
862 }
863
864stream_cleanup: (void) SdMngr_f_close(&target_file_hnd);
865 crypto_stream_close(&crypto_stream_hnd);
866
867 return op_res;
868}

871STATIC crypto_api_res_t prep_file_for_stream_encr(crypto_stream_hnd_t * const p_stream_hnd)
872{
873 CRIT_ASSERT(NULL != p_stream_hnd);
874
875 crypto_api_res_t res = CRYPTO_API_RES_BAD_PARAMS;
876
877 if (NULL != p_stream_hnd)
878 {
879 // prepare the encrypted_data_hdr_t block at the beginning of the file
880 (void) memcpy(p_stream_hnd->encr_block_hdr.data_sig, ENCRYPTED_DATA_SIGNATURE, sizeof(ENCRYPTED_DATA_SIGNATURE) - 1);
881
882 p_stream_hnd->encr_block_hdr.major_ver = DATA_HDR_MAJOR;
883 p_stream_hnd->encr_block_hdr.minor_ver = DATA_HDR_MINOR;
884
885 (void) memset(p_stream_hnd->encr_block_hdr.cipher_algo, '\0', sizeof(p_stream_hnd->encr_block_hdr.cipher_algo));
886 (void) memcpy(p_stream_hnd->encr_block_hdr.cipher_algo, CIPHER_ALGO_NAME, (size_t) fmin(sizeof(CIPHER_ALGO_NAME), sizeof(p_stream_hnd->encr_block_hdr.cipher_algo)));
887
888 // regenerate init vectors for the new encryption
889 generate_iv(p_stream_hnd->encr_block_hdr.iv, sizeof(p_stream_hnd->encr_block_hdr.iv));
890
891 // set the associated data field to the name of the file being encrypted
892 (void) memset(p_stream_hnd->encr_block_hdr.aad, (int) 0, sizeof(p_stream_hnd->encr_block_hdr.aad));
893
894 safe_copy_str((char * const) p_stream_hnd->encr_block_hdr.aad, p_stream_hnd->src_file_name, sizeof(p_stream_hnd->encr_block_hdr.aad));
895
896 // in this version of the file format implementation, the cipher text is placed immediately after the header
897 p_stream_hnd->encr_block_hdr.cipher_text_offset = sizeof(encrypted_data_hdr_t);
898
899 // the size of the cipher text will be the same as the size of the plain text
900 p_stream_hnd->encr_block_hdr.cipher_text_size = SdMngr_f_size(&p_stream_hnd->f_src_hnd);
901
902 set_next_stage(p_stream_hnd, &read_from_header);
903
904 res = CRYPTO_API_RES_OK;
905 }
906
907 return res;
908}

911STATIC crypto_api_res_t prep_file_for_stream_decr(crypto_stream_hnd_t * const p_stream_hnd)
912{
913 CRIT_ASSERT(NULL != p_stream_hnd);
914
915 uint32_t bytes_read = 0U;
916
917 // read encrypted file header
918 FRESULT f_src_res = SdMngr_f_read(&p_stream_hnd->f_src_hnd, &p_stream_hnd->encr_block_hdr, sizeof(encrypted_data_hdr_t), (UINT *) &bytes_read);
919
920 if ((FR_OK != f_src_res) || (bytes_read != sizeof(encrypted_data_hdr_t)))
921 {
922 CRYPTO_TRACE_DBG("[prep_file_for_stream_decr] encrypted file header read error => read code '%d' / bytes_read: %d => expected: %d!\n", f_src_res, bytes_read, sizeof(encrypted_data_hdr_t));
923
924 return CRYPTO_API_RES_PARTIAL_FILE;
925 }
926 else
927 {
928 if ((p_stream_hnd->encr_block_hdr.major_ver != DATA_HDR_MAJOR) ||
929 (p_stream_hnd->encr_block_hdr.minor_ver != DATA_HDR_MINOR) ||
930 (0 != memcmp(p_stream_hnd->encr_block_hdr.cipher_algo, CIPHER_ALGO_NAME, (size_t) fmin(sizeof(CIPHER_ALGO_NAME), sizeof(p_stream_hnd->encr_block_hdr.cipher_algo)))) ||
931 (0 == p_stream_hnd->encr_block_hdr.cipher_text_size) ||
932 (p_stream_hnd->encr_block_hdr.cipher_text_offset < sizeof(encrypted_data_hdr_t)))
933 {
934 CRYPTO_TRACE_DBG("[prep_file_for_stream_decr] invalid file header detected!\n");
935
936 return CRYPTO_API_RES_INVALID_HEADER;
937 }
938 else
939 {
940 // position the file pointer to the beginning of the cipher text
941 f_src_res = SdMngr_f_lseek(&p_stream_hnd->f_src_hnd, p_stream_hnd->encr_block_hdr.cipher_text_offset);
942
943 if (FR_OK != f_src_res)
944 {
945 CRYPTO_TRACE_DBG("[prep_file_for_stream_decr] invalid cipher text offset attempted: %ld!\n", p_stream_hnd->encr_block_hdr.cipher_text_offset);
946
947 return CRYPTO_API_RES_INVALID_CIPHER_OFS;
948 }
949 }
950 }
951
952 set_next_stage(p_stream_hnd, &read_chunk_and_process);
953
954 return CRYPTO_API_RES_OK;
955}

1029STATIC crypto_api_res_t prepare_temp_buf(crypto_stream_hnd_t * const p_stream_hnd, uint8_t * p_read_buf, uint8_t * const p_output_buf, const uint32_t bytes_to_read, uint32_t * const p_bytes_read)
1030{
1031 uint32_t bytes_read = 0U;
1032
1033 (void) p_output_buf;
1034 (void) bytes_to_read;
1035
1036 // if the remaining bytes of the cipher text to read are less than the internal buffer, it must be only partially filled
1037 // (this is important for the decryption step because SdMngr_f_read() will read after the cipher text block, crossing over to the
1038 // footer block... and we don't want to process the footer block data as cipher text);
1039 // in the case of streaming encryption, SdMngr_f_read() will be reading from the original file and it cannot go beyond the end of the file,
1040 // thus no issue there
1041 uint32_t actual_bytes_to_read = (uint32_t) fmin(sizeof(p_stream_hnd->tmp_cipher_buf), (double) (p_stream_hnd->encr_block_hdr.cipher_text_size - p_stream_hnd->cipher_text_bytes_read));
1042
1043 FRESULT f_src_res = SdMngr_f_read(&p_stream_hnd->f_src_hnd, p_read_buf, actual_bytes_to_read, (UINT *) &bytes_read);
1044
1045 if (FR_OK == f_src_res)
1046 {
1047 if (bytes_read > 0)
1048 {
1049 size_t processed_data_size = 0U;
1050
1051 // encrypt what was read from the file
1052 cmox_cipher_retval_t cipher_retval = cmox_cipher_append(p_stream_hnd->cipher_ctx,
1053 p_read_buf,
1054 bytes_read,
1055 p_stream_hnd->tmp_cipher_buf,
1056 &processed_data_size);
1057
1058 if ((cipher_retval != CMOX_CIPHER_SUCCESS) || (processed_data_size != bytes_read))
1059 {
1060 CRYPTO_TRACE_DBG("cmox_cipher_append() failed with code %d\n", cipher_retval);
1061
1062 return CRYPTO_API_RES_CRYPTOLIB_ERROR;
1063 }
1064 else
1065 {
1066 // this step only prepares the temporary buffer, the next one will start depleting it without
1067 // calling the CMOX library
1068 *p_bytes_read = 0;
1069
1070 // set the prepared size for use by the next stage...
1071 p_stream_hnd->tmp_cipher_buf_prep_size = processed_data_size;
1072
1073 set_next_stage(p_stream_hnd, &read_from_temp_buf);
1074
1075 return CRYPTO_API_RES_OK;
1076 }
1077 }
1078 }
1079 else
1080 {
1081 CRYPTO_TRACE_DBG("file read failed with code %d\n", f_src_res);
1082
1083 return CRYPTO_API_RES_FILE_ERROR;
1084 }
1085
1086 return CRYPTO_API_RES_ERROR;
1087}

1090STATIC crypto_api_res_t read_chunk_and_process(crypto_stream_hnd_t * const p_stream_hnd, uint8_t * p_read_buf, uint8_t * const p_output_buf, const uint32_t bytes_to_read, uint32_t * const p_bytes_read)
1091{
1092 CRIT_ASSERT(NULL != p_stream_hnd);
1093 CRIT_ASSERT(NULL != p_read_buf);
1094 CRIT_ASSERT(NULL != p_output_buf);
1095 CRIT_ASSERT(NULL != p_bytes_read);
1096
1097 uint32_t actual_bytes_to_read = 0U;
1098 uint32_t bytes_read = 0U;
1099
1100 *p_bytes_read = 0;
1101
1102 if (p_stream_hnd->cipher_text_bytes_read < p_stream_hnd->encr_block_hdr.cipher_text_size)
1103 {
1104 uint64_t rem_cipher_text_size = p_stream_hnd->encr_block_hdr.cipher_text_size - p_stream_hnd->cipher_text_bytes_read;
1105
1106 actual_bytes_to_read = (uint32_t) fmin(bytes_to_read, rem_cipher_text_size);
1107
1108 // correct the amount of read bytes to a multiple of MIN_CRYPTO_BUF_SIZE due to a specific requirement by the cmox_cipher_append() call
1109 if (actual_bytes_to_read >= MIN_CRYPTO_BUF_SIZE)
1110 {
1111 actual_bytes_to_read -= actual_bytes_to_read % MIN_CRYPTO_BUF_SIZE;
1112 }
1113 else
1114 {
1115 // there is not enough space in the user-provided buffer to accommodate the minimum required CMOX library buffer size =>
1116 // therefore we must perform the cipher_append operation in a temporary buffer of the correct size (usually a multiple of 16)
1117 // and deplete the buffer before reading again in the user buffer
1118 set_next_stage(p_stream_hnd, &prepare_temp_buf);
1119
1120 // we will not return any bytes at this time...
1121 *p_bytes_read = 0;
1122
1123 return CRYPTO_API_RES_OK;
1124 }
1125
1126 FRESULT f_src_res = SdMngr_f_read(&p_stream_hnd->f_src_hnd, p_read_buf, actual_bytes_to_read, (UINT *) &bytes_read);
1127
1128 if (FR_OK == f_src_res)
1129 {
1130 if (bytes_read > 0)
1131 {
1132 size_t processed_data_size = 0U;
1133
1134 // encrypt what was read from the file
1135 cmox_cipher_retval_t cipher_retval = cmox_cipher_append(p_stream_hnd->cipher_ctx,
1136 p_read_buf,
1137 bytes_read,
1138 p_output_buf,
1139 &processed_data_size);
1140
1141 if ((cipher_retval != CMOX_CIPHER_SUCCESS) || (processed_data_size != bytes_read))
1142 {
1143 CRYPTO_TRACE_DBG("cmox_cipher_append() failed with code %d\n", cipher_retval);
1144
1145 return CRYPTO_API_RES_CRYPTOLIB_ERROR;
1146 }
1147 else
1148 {
1149 p_stream_hnd->cipher_text_bytes_read += bytes_read;
1150
1151 *p_bytes_read = bytes_read;
1152
1153 return CRYPTO_API_RES_OK;
1154 }
1155 }
1156 }
1157 else
1158 {
1159 CRYPTO_TRACE_DBG("file read failed with code %d\n", f_src_res);
1160
1161 return CRYPTO_API_RES_FILE_ERROR;
1162 }
1163 }
1164
1165 *p_bytes_read = 0U;
1166
1167 if (OP_CMOX_AES_GCM_ENC == p_stream_hnd->impl_type)
1168 {
1169 set_next_stage(p_stream_hnd, &prepare_footer);
1170 }
1171 else
1172 {
1173 set_next_stage(p_stream_hnd, &verify_footer);
1174 }
1175
1176 // no more data to read...
1177 return CRYPTO_API_RES_OK;
1178}

1181STATIC crypto_api_res_t prepare_footer(crypto_stream_hnd_t * const p_stream_hnd, uint8_t * p_read_buf, uint8_t * const p_output_buf, const uint32_t bytes_to_read, uint32_t * const p_bytes_read)
1182{
1183 CRIT_ASSERT(OP_CMOX_AES_GCM_ENC == p_stream_hnd->impl_type);
1184 CRIT_ASSERT(NULL != p_stream_hnd);
1185 CRIT_ASSERT(NULL != p_read_buf);
1186 CRIT_ASSERT(NULL != p_output_buf);
1187 CRIT_ASSERT(NULL != p_bytes_read);
1188
1189 uint32_t processed_data_size = 0U;
1190
1191 // this is a generation step only, we don't return any bytes... next section will provide the footer block data
1192 *p_bytes_read = 0U;
1193
1194 (void) memcpy(p_stream_hnd->data_footer.footer_sig, FOOTER_SIGNATURE, sizeof(FOOTER_SIGNATURE) - 1);
1195
1196 cmox_cipher_retval_t cipher_retval = cmox_cipher_generateTag(p_stream_hnd->cipher_ctx,
1197 p_stream_hnd->data_footer.auth_tag,
1198 (size_t *) &processed_data_size);
1199
1200 if (cipher_retval != CMOX_CIPHER_SUCCESS)
1201 {
1202 CRYPTO_TRACE_DBG("cmox_cipher_generateTag() failed with code %d\n", cipher_retval);
1203
1204 return CRYPTO_API_RES_AUTH_ERROR;
1205 }
1206
1207 if (processed_data_size != sizeof(p_stream_hnd->data_footer.auth_tag))
1208 {
1209 CRYPTO_TRACE_DBG("expected tag size is %d but actual calculated size is %d\n", sizeof(p_stream_hnd->data_footer.auth_tag), processed_data_size);
1210
1211 return CRYPTO_API_RES_AUTH_ERROR;
1212 }
