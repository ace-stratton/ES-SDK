94payload_ctrl_op_res_t payload_ctrl_init(void)
95{
96 STATIC const osThreadAttr_t task_attr =
97 {
98 .attr_bits = osThreadDetached,
99 .name = "PayloadCtrlMainTask",
100 .priority = osPriorityNormal,
101 .cb_mem = NULL,
102 .cb_size = 0U,
103 .stack_mem = NULL,
104 .stack_size = 3000 // TODO: Measure worst-case stack usage!
105 };
106 payload_ctrl_op_res_t res = PL_CTRL_OP_RES_FAILED;
107
108 EXEH_INIT_MODULE_FILENAME();
109
110 // call configured payload init functions...
111 for (uint8_t pl_id = 0; pl_id < PAYLOAD_COUNT; pl_id++)
112 {
113 const payload_immutable_cfg_t *p_pl_cfg = payload_cfg_get(pl_id);
114
115 CRIT_ASSERT((const payload_immutable_cfg_t *) NULL != p_pl_cfg);
116 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface);
117 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface->init);
118
119 pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state = PL_STATE_NOT_INIT;
120 timer_stop(&pl_ctrl_runtime_context.state_mon_data[pl_id].state_timer);
121 pl_ctrl_runtime_context.state_mon_data[pl_id].pending_flags = PENDING_FLAG_CLEAR_ALL;
122 pl_ctrl_runtime_context.point_target_pl_id = PAYLOAD_COUNT;
123
124 p_pl_cfg->p_pl_interface->init(p_pl_cfg->internal_inst_id,
125 (pl_config_t * const) &p_pl_cfg->init_config,
126 p_pl_cfg->p_pl_notif_cbk);
127 }
128
129 payload_ctrl_trigger_hsm(eHSM_StdEvent_Reset);
130
131 // create thread only if not already existing...
132 if (NULL == pl_ctrl_runtime_context.os_task_hnd)
133 {
134 payload_ctrl_lock_id = osMutexNew(&payload_ctrl_lock_attr);
135
136 CRIT_ASSERT(NULL != payload_ctrl_lock_id);
137
138 pl_ctrl_runtime_context.os_task_hnd = osThreadNew(payload_ctrl_task,
139 (void *) NULL,
140 &task_attr);
141
142 if (NULL == pl_ctrl_runtime_context.os_task_hnd)
143 {
144 EXEH_HANDLE(eEXEHSeverity_Fatal,
145 (int32_t) ERR_CANNOT_SPAWN_TASK);
146 }
147 else
148 {
149 TaskMonitor_TaskInitialized(TASK_MONITOR_PAYLOAD_CTRL);
150 SET_TRUE(pl_ctrl_runtime_context.is_initialized);
151
152 res = PL_CTRL_OP_RES_OK;
153 }
154 }
155
156 return res;
157}

178payload_ctrl_op_res_t payload_ctrl_start(const payload_ctrl_payload_t payload_id,
179 payload_ctrl_start_cfg_t * const p_start_cfg)
180{
181 const payload_immutable_cfg_t *p_pl_cfg = payload_cfg_get(payload_id);
182
183 CRIT_ASSERT((const payload_immutable_cfg_t *) NULL != p_pl_cfg);
184 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface);
185 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface->start);
186
187 pl_state_t pl_state = p_pl_cfg->p_pl_interface->get_state(p_pl_cfg->internal_inst_id);
188
189 osMutexAcquire(payload_ctrl_lock_id, osWaitForever);
190
191 switch (pl_state)
192 {
193 case PL_STATE_STOPPED:
194 case PL_STATE_ERROR:
195 {
196 (void) memcpy((void * volatile) &pl_ctrl_runtime_context.state_mon_data[payload_id].start_cfg,
197 (const void * volatile) p_start_cfg,
198 sizeof(payload_ctrl_start_cfg_t));
199
200 pl_ctrl_runtime_context.state_mon_data[payload_id].pending_flags |= PENDING_FLAG_START;
201
202 uint32_t set_flags = osThreadFlagsSet(pl_ctrl_runtime_context.os_task_hnd, PAYLOAD_TASK_ACTIVATE_FLAGS);
203
204 (void) set_flags;
205
206 break;
207 }
208
209 default:
210 break;
211 }
212
213 osMutexRelease(payload_ctrl_lock_id);
214
215 return PL_CTRL_OP_RES_OK;
216}

218payload_ctrl_op_res_t payload_ctrl_stop(const payload_ctrl_payload_t payload_id,
219 const pl_op_stop_mode_t stop_mode)
220{
221 pl_op_status_t paylod_result = PL_OP_STATUS_FAILURE;
222
223 const payload_immutable_cfg_t *p_pl_cfg = payload_cfg_get(payload_id);
224
225 CRIT_ASSERT((const payload_immutable_cfg_t *) NULL != p_pl_cfg);
226 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface);
227 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface->stop);
228
229 paylod_result = p_pl_cfg->p_pl_interface->stop(p_pl_cfg->internal_inst_id, stop_mode);
230
231 if (payload_id == pl_ctrl_runtime_context.point_target_pl_id)
232 {
233 payload_ctrl_trigger_hsm(ev_payload_ctrl_sm_requesting_payload_off);
234 }
235
236 return (PL_OP_STATUS_OK == paylod_result) ? (PL_CTRL_OP_RES_OK) : (PL_CTRL_OP_RES_FAILED);
237}

266void payload_ctrl_event_notify(const payload_ctrl_payload_t payload_id, pl_state_t to_state)
267{
268 pl_ctrl_runtime_context.state_mon_data[payload_id].last_reported_state = to_state;
269
270 if ((payload_id == pl_ctrl_runtime_context.point_target_pl_id) &&
271 ((PL_STATE_STOPPED == to_state) || (PL_STATE_ERROR == to_state)))
272 {
273 payload_ctrl_trigger_hsm(ev_payload_ctrl_sm_requesting_payload_off);
274 }
275
276 switch (to_state)
277 {
278 case PL_STATE_STARTING:
279 case PL_STATE_STOPPING:
280 {
281 // start tracking payload transient state time...
282 timer_start(&pl_ctrl_runtime_context.state_mon_data[payload_id].state_timer);
283 break;
284 }
285
286 default:
287 {
288 timer_stop(&pl_ctrl_runtime_context.state_mon_data[payload_id].state_timer);
289 break;
290 }
291 }
292
293 if (NULL != pl_ctrl_runtime_context.p_user_event_cbk)
294 {
295 const payload_immutable_cfg_t *p_pl_cfg = payload_cfg_get(payload_id);
296
297 CRIT_ASSERT((const payload_immutable_cfg_t *) NULL != p_pl_cfg);
298 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface);
299 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface->get_state);
300
301 payload_ctrl_event_t pl_event =
302 {
303 .payload_id = payload_id,
304 .op_state = to_state,
305 };
306
307 pl_event.error_code = p_pl_cfg->p_pl_interface->get_last_error(p_pl_cfg->internal_inst_id);
308
309 pl_ctrl_runtime_context.p_user_event_cbk(&pl_event);
310 }
311}

316STATIC void payload_ctrl_task(void * p_task_arg)
317{
318 CRIT_ASSERT(IS_TRUE(pl_ctrl_runtime_context.is_initialized));
319
320 // currently unused
321 (void) p_task_arg;
322
323 for (;;)
324 {
325 volatile uint32_t task_flag_status = osThreadFlagsWait(PAYLOAD_TASK_ACTIVATE_FLAGS,
326 osFlagsWaitAny,
327 PAYLOAD_TASK_TIMEOUT_MS);
328
329 (void) task_flag_status;
330
331 // execute periodic SM activities...
332 payload_ctrl_trigger_hsm(eHSM_StdEvent_Periodic);
333
334 //
335 for (uint8_t pl_id = 0; pl_id < PAYLOAD_COUNT; pl_id++)
336 {
337 if (pl_ctrl_runtime_context.state_mon_data[pl_id].state_timer.is_running)
338 {
339 if (timer_is_elapsed(&pl_ctrl_runtime_context.state_mon_data[pl_id].state_timer, PAYLOAD_TRANSIENT_STATE_TIMEOUT_MS))
340 {
341 timer_stop(&pl_ctrl_runtime_context.state_mon_data[pl_id].state_timer);
342
343 ES_TRACE_WARN("WARN: Payload '%d' is staying in a transient state '%d' for too long.",
344 pl_id,
345 pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state);
346 }
347 }
348 }
349
350 TaskMonitor_IamAlive(TASK_MONITOR_PAYLOAD_CTRL);
351 vApplicationLowStackCheck(TASK_MONITOR_PAYLOAD_CTRL);
352 }
353}
