18eCPDispatchResult_t cp_fault_commands_read_all_faults(sCPDispatchContext_t * const psDispCtx)
19{
20 eCPDispatchResult_t res = ECPDISPATCHRESULT_ERROR;
21
22 if (CPCmdHandler_IsDispContextValid(psDispCtx, CP_CMD_FAULT_READ_ALL))
23 {
24 if (psDispCtx->u32CmdDataSize >= sizeof(cmd_fault_read_all_in_args_t))
25 {
26 cmd_fault_read_all_in_args_t *p_cmd_in_args = (cmd_fault_read_all_in_args_t *) psDispCtx->pu8CmdData;
27
28 fault_persistor_read_all(psDispCtx->pu8ResponseBuf, psDispCtx->u32ResponseBufSize, &psDispCtx->u32ResponseActualSize);
29
30 if (0U == psDispCtx->u32ResponseActualSize)
31 {
32 (void) memset(psDispCtx->pu8ResponseBuf, 0U, psDispCtx->u32ResponseBufSize);
33 }
34
35 if (p_cmd_in_args->clear_storage)
36 {
37 fault_persistor_clear();
38 }
39
40 psDispCtx->u8TLError = EESTL_SFERR_SUCCESS;
41 }
42 else
43 {
44 psDispCtx->u32ResponseActualSize = 0U;
45 psDispCtx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
46 }
47
48 res = ECPDISPATCHRESULT_RESULT_OK;
49 }
50
51 return res;
52}
