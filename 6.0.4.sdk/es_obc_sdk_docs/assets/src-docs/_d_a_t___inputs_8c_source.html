84ES_ReturnType Magnitometers_LIS3MDL_Init(uint8_t Dev_No) {
85 uint8_t val;
86 ES_ReturnType ESRetValue = E_OK;
87
88 // Turn on device and change measurement mode to continuous
89 if ( !(LIS3MDL_MAG_W_SystemOperatingMode (Dev_No, LIS3MDL_MAG_MD_CONTINUOUS)))
90 return E_INV_CFG;
91
92 // Update output value 10 times per second
93 if ( !(LIS3MDL_MAG_W_OutputDataRate (Dev_No, LIS3MDL_MAG_DO_40Hz)))
94 return E_INV_CFG;
95
96 // Set range to 4 Gauss
97 if ( !(LIS3MDL_MAG_W_FullScale(Dev_No, LIS3MDL_MAG_FS_4Ga )))
98 return E_INV_CFG;
99
100 // Set operating mode to ultra-high performance for the X and Y axis ... to get smallest RMS noise figure
101 if ( !(LIS3MDL_MAG_W_OperatingModeXY(Dev_No, LIS3MDL_MAG_OM_ULTRA_HIGH )))
102 return E_INV_CFG;
103
104 // Set operating mode to ultra-high performance for the Z axis ... to get smallest RMS noise figure
105 if ( !(LIS3MDL_MAG_W_OperatingModeZ(Dev_No, LIS3MDL_MAG_OMZ_ULTRA_HIGH )))
106 return E_INV_CFG;
107
108 // Block data update
109 if ( !(LIS3MDL_MAG_W_BlockDataUpdate(Dev_No,LIS3MDL_MAG_BDU_ENABLE)))
110 return E_INV_CFG;
111
112 // Check status
113 if ( !(LIS3MDL_MAG_R_WHO_AM_I_(Dev_No, &val )))
114 return E_INV_CFG;
115
116 // invalid configuration if device magnetometer answered with id different than 0x3D
117 if (val ^ 0x3D)
118 return E_INV_CFG;
119
120 return ESRetValue;
121}

133ES_ReturnType Magnitometers_LIS3MDL_Read_Data (Compass_Axis_t *MagOutData, uint8_t Dev_No, eMagnetoDataFormat_t u8DataFormat) {
134
135 int16_t pDataRaw[3];
136
137 ES_ReturnType ESRetValue = E_OK;
138
139 uint8_t regValue[6] = {0, 0, 0, 0, 0, 0};
140
141 if ( !(LIS3MDL_MAG_Get_Magnetic(Dev_No, (uint8_t*)regValue )))
142 return E_INV_CFG;
143
144 /* Format the data. */
145 pDataRaw[0] = ( ( ( ( int16_t )regValue[1] ) << 8 ) + ( int16_t )regValue[0] );
146 pDataRaw[1] = ( ( ( ( int16_t )regValue[3] ) << 8 ) + ( int16_t )regValue[2] );
147 pDataRaw[2] = ( ( ( ( int16_t )regValue[5] ) << 8 ) + ( int16_t )regValue[4] );
148
149 if ( MAGNETODATAFORMAT_GAUSS == u8DataFormat )
150 {
151 MagOutData->AXIS_X = ( pDataRaw[0] * LIS3MDL_4GAUSS_LSB );
152 MagOutData->AXIS_Y = ( pDataRaw[1] * LIS3MDL_4GAUSS_LSB );
153 MagOutData->AXIS_Z = ( pDataRaw[2] * LIS3MDL_4GAUSS_LSB );
154 }
155 else
156 if ( MAGNETODATAFORMAT_nTESLA == u8DataFormat )
157 {
158 MagOutData->AXIS_X = ( pDataRaw[0] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_nTESLA;
159 MagOutData->AXIS_Y = ( pDataRaw[1] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_nTESLA;
160 MagOutData->AXIS_Z = ( pDataRaw[2] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_nTESLA;
161 }
162 else
163 if ( MAGNETODATAFORMAT_TESLA == u8DataFormat )
164 {
165 MagOutData->AXIS_X = ( pDataRaw[0] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_TESLA;
166 MagOutData->AXIS_Y = ( pDataRaw[1] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_TESLA;
167 MagOutData->AXIS_Z = ( pDataRaw[2] * LIS3MDL_4GAUSS_LSB ) * LIS3MDL_CONV_GAUSS_TO_TESLA;
168 }
169 else
170 {
171 MagOutData->AXIS_X = ( pDataRaw[0] );
172 MagOutData->AXIS_Y = ( pDataRaw[1] );
173 MagOutData->AXIS_Z = ( pDataRaw[2] );
174 }
175
176 return ESRetValue;
177
178}
