142void eps_ii_batt_info_notif(const RespContext_t *const pRespCtx, const EPSII_BP_1GetBatteryInfoResponseData_t *const battery_info)
143{
144
145 uint8_t battery_pack_id = get_id_by_addr(pRespCtx->nAddr);
146
147 if ((NULL != battery_info) && (BATTERY_PACKS_COUNT > battery_pack_id))
148 {
149 DATA_CACHE_Eps2Inst0BatteryInfoType_t bp_info = { 0 };
150
151 // Populate the DATA_CACHE_EPS_7_t struct
152 bp_info.u8Bp_mode = battery_info->eBp_mode;
153 if (BATTERY_INFO_ERROR_VAL == battery_info->i64BattChargeCapacity)
154 {
155 bp_info.u32Capacity = UINT32_ERROR_VAL;
156 }
157 else
158 {
159 bp_info.u32Capacity = (uint32_t)battery_info->i64BattChargeCapacity;
160 }
161
162 if (BATTERY_INFO_ERROR_VAL == battery_info->i64BattCurrent)
163 {
164 bp_info.i16Current = INT16_ERROR_VAL;
165 }
166 else
167 {
168 bp_info.i16Current = (int16_t)battery_info->i64BattCurrent;
169 }
170
171 if (BATTERY_INFO_ERROR_VAL == battery_info->i64BattVoltage)
172 {
173 bp_info.u16Voltage = UINT16_ERROR_VAL;
174 }
175 else
176 {
177 bp_info.u16Voltage = (uint16_t)battery_info->i64BattVoltage;
178 }
179
180 if (BATTERY_INFO_ERROR_VAL == battery_info->i64BattTemperature)
181 {
182 bp_info.i32Temperature = INT32_ERROR_VAL;
183 }
184 else
185 {
186 bp_info.i32Temperature = (int32_t)battery_info->i64BattTemperature;
187 }
188
189 // Update the datacache
190 dc_set_raw_data(DC_DATA_INPUT_INTERNAL, battery_packs[battery_pack_id].datacache_cmd_list[BP_INFO], &bp_info, sizeof(bp_info));
191 }

310// Response callback for the FP command getRawSensors for BP and PDM
311void eps_ii_raw_sensors_notif(const RespContext_t *const pRespCtx, const void *p_raw_sensors)
312{
313#pragma GCC diagnostic push
314#pragma GCC diagnostic ignored "-Waddress-of-packed-member"
315 uint8_t battery_pack_id = get_id_by_addr(pRespCtx->nAddr);
316
317 if ((NULL != p_raw_sensors) && (BATTERY_PACKS_COUNT > battery_pack_id))
318 {
319 // This could also be treated as EPSII_BP_1GetRAWSensorsResponseData_t
320 EPSII_PDM_1GetRAWSensorsResponseData_t const *p_data = p_raw_sensors;
321 DATA_CACHE_EPS_14_t eps_14_data;
322
323 switch (p_data->u32PktID)
324 {
325 case EPS_II_RAW_SENSORS_BP_REQ_INA_BALANCER:
326
327 update_ina_values(&p_data->ai32SensorValues[0],
328 &battery_packs[battery_pack_id]);
329
330 update_balancer_values(&p_data->ai32SensorValues[EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_BALANCER1_BAT_U_ID - EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_INA_BAT_OUT_1_I_ID],
331 &battery_packs[battery_pack_id]);
332
333 break;
334
335 case EPS_II_RAW_SENSORS_BP_REQ_XP_XR_YP:
336
337 update_charger_values(battery_packs[battery_pack_id].datacache_cmd_list[BP_CHARGERS_MAIN_INFO_XP], &p_data->ai32SensorValues[0]);
338
339 update_charger_values(battery_packs[battery_pack_id].datacache_cmd_list[BP_CHARGERS_MAIN_INFO_XR], &p_data->ai32SensorValues[EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_CHRG_XR_STAT_CHARGER_ID - EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_CHRG_XP_STAT_CHARGER_ID]);
340
341 update_charger_values(battery_packs[battery_pack_id].datacache_cmd_list[BP_CHARGERS_MAIN_INFO_YP], &p_data->ai32SensorValues[EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_CHRG_YP_STAT_CHARGER_ID - EPSII_BP_1_EPSIIT_SENSORIDS_EPSIIT_CHRG_XP_STAT_CHARGER_ID]);
342
343 break;
344
345 case EPS_II_RAW_SENSORS_BP_REQ_YR_ZP_ZR:
346
347 update_charger_values(battery_packs[battery_pack_id].datacache_cmd_list[BP_CHARGERS_MAIN_INFO_YR], &p_data->ai32SensorValues[0]);
348

682{

694 0U,
695 sizeof(DATA_CACHE_Eps2Inst0BatteryInfoType_t));
696
697 if ((DC_DATA_STATUS_OK == res) && (EPSII_BP_1_BP_MODE_TYPE_BATTERY == bp_info.u8Bp_mode))
698 {
699 if (UINT32_ERROR_VAL != bp_info.u32Capacity)

701 overall_capacity += bp_info.u32Capacity;

706 normal_state_bps_count++;
707 average_voltage += bp_info.u16Voltage;
708 }
709 }
710 }
711
712 if (0 < normal_state_bps_count)
713 {
714 overall_bp_info.u16Voltage = (uint16_t)(average_voltage / normal_state_bps_count);
715 }
716
717 overall_bp_info.u32Capacity = overall_capacity;
