111static void telemetry_file_sink_init(void)
112{
113 static const osThreadAttr_t task_attr =
114 {
115 .attr_bits = osThreadDetached,
116 .name = "Telemetry_File_Sink_Task",
117 .priority = osPriorityNormal,
118 .cb_mem = NULL,
119 .cb_size = 0U,
120 .stack_mem = NULL,
121 .stack_size = 2500,
122 .reserved = 0U,
123 .tz_module = 0
124 };
125
126 for (uint8_t idx = 0; idx < TLM_FILE_SINK_OP_MAX; idx++)
127 {
128 tlm_filesink_rt_context.main_task_sync_op_flags[idx] = false;
129 }
130
131 CRIT_ASSERT(sizeof(TLM_FILE_SINK_EXT) <= 4U);
132
133 (void) strcat(tlm_filesink_rt_context.wildcard, "*.");
134 (void) strcat(tlm_filesink_rt_context.wildcard, TLM_FILE_SINK_EXT);
135
136 if (NULL == telemetry_file_lock_id)
137 {
138 telemetry_file_lock_id = osMutexNew(&telemetry_file_lock_attr);
139 }
140
141 // create thread only if not already existing...
142 if (NULL == tlm_filesink_rt_context.os_task_hnd)
143 {
144 tlm_filesink_rt_context.os_task_hnd = osThreadNew(telemetry_file_sink_task,
145 NULL,
146 &task_attr);
147
148 if (NULL == tlm_filesink_rt_context.os_task_hnd)
149 {
150 EXEH_HANDLE(eEXEHSeverity_Fatal,
151 (int32_t) EXEH_TLM_FS_CANNOT_SPAWN_TASK);
152 }
153 else
154 {
155 TaskMonitor_TaskInitialized(TASK_TELEMETRY_FILE_SINK);
156 }
157 }
158}

160static void telemetry_file_sink_exec_syncd_tasks(void)
161{
162 for (uint8_t idx = 0; idx < (uint8_t) TLM_FILE_SINK_OP_MAX; idx++)
163 {
164 if (true == tlm_filesink_rt_context.main_task_sync_op_flags[idx])
165 {
166 if (idx == (uint8_t) TLM_FILE_SINK_OP_PAUSE)
167 {
168 (void) telemetry_file_trigger_hsm(ev_telemetry_file_sink_sm_stop);
169 }
170
171 if (idx == (uint8_t) TLM_FILE_SINK_OP_CFG_UPDATE)
172 {
173 (void) telemetry_file_trigger_hsm(ev_telemetry_file_sink_sm_restart);
174 }
175
176 if (idx == (uint8_t) TLM_FILE_SINK_OP_CLEAN)
177 {
178 telemetry_file_sink_clean();
179 }
180
181 tlm_filesink_rt_context.main_task_sync_op_flags[idx] = false;
182 }
183 }

186static void telemetry_file_sink_adapter_cmd(const telemetry_sink_adapter_cmd_t cmd)
187{
188 switch (cmd)
189 {
190 case TELEMETRY_SINK_ADAPTER_CMD_INIT:
191 {
192 telemetry_file_sink_init();
193
194 (void) telemetry_file_trigger_hsm(eHSM_StdEvent_Reset);
195 (void) telemetry_file_trigger_hsm(ev_telemetry_file_sink_sm_init);
196 break;
197 }
198
199 case TELEMETRY_SINK_ADAPTER_CMD_START:
200 {
201 (void) telemetry_file_trigger_hsm(ev_telemetry_file_sink_sm_start);
202
203 break;
204 }
205
206 case TELEMETRY_SINK_ADAPTER_CMD_STOP:
207 {
208 (void) telemetry_file_trigger_hsm(ev_telemetry_file_sink_sm_stop);
209 break;
210 }
211
212 default:
213 {
214 CRIT_ASSERT(false);
215 break;
216 }
217 }
218}

220static void telemetry_file_sink_task(void * p_task_arg)
221{
222 (void) p_task_arg;
223
224 while (true)
225 {
226 telemetry_file_sink_exec_syncd_tasks();
227
228 (void) telemetry_file_trigger_hsm(eHSM_StdEvent_Periodic);
229
230 osDelay(TLM_FILE_SINK_TASK_PERIOD_MS);
231
232 TaskMonitor_IamAlive(TASK_TELEMETRY_FILE_SINK);
233 vApplicationLowStackCheck(TASK_TELEMETRY_FILE_SINK);
234 }
235}

237static bool inline telemetry_file_trigger_hsm(const uint32_t event_id)
238{
239 bool res = false;
240
241 osMutexAcquire(telemetry_file_lock_id, osWaitForever);
242 res = HSM_TriggerEvent(telemetry_file_sink_sm_instance, event_id);
243 osMutexRelease(telemetry_file_lock_id);
244
245 return res;
246}
