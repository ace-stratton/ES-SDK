73#define SLIP_END 0xC0 /* 0300: start and end of every packet */
74#define SLIP_ESC 0xDB /* 0333: escape start (one byte escaped data follows) */

116static err_t
117slipif_output(struct netif *netif, struct pbuf *p)
118{
119 struct slipif_priv *priv;
120 struct pbuf *q;
121 u16_t i;
122 u8_t c;
123
124 LWIP_ASSERT("netif != NULL", (netif != NULL));
125 LWIP_ASSERT("netif->state != NULL", (netif->state != NULL));
126 LWIP_ASSERT("p != NULL", (p != NULL));
127
128 LWIP_DEBUGF(SLIP_DEBUG, ("slipif_output: sending %"U16_F" bytes\n", p->tot_len));
129 priv = (struct slipif_priv *) &((slipif_net_cfg_state_t *)netif->state)->priv_static;
130
131 /* Send pbuf out on the serial I/O device. */
132 /* Start with packet delimiter. */
133 sio_send(SLIP_END, priv->sd);
134
135 for (q = p; q != NULL; q = q->next) {
136 for (i = 0; i < q->len; i++) {
137 c = ((u8_t *)q->payload)[i];
138 switch (c) {
139 case SLIP_END:
140 /* need to escape this byte (0xC0 -> 0xDB, 0xDC) */
141 sio_send(SLIP_ESC, priv->sd);
142 sio_send(SLIP_ESC_END, priv->sd);
143 break;
144 case SLIP_ESC:
145 /* need to escape this byte (0xDB -> 0xDB, 0xDD) */
146 sio_send(SLIP_ESC, priv->sd);
147 sio_send(SLIP_ESC_ESC, priv->sd);
148 break;
149 default:
150 /* normal byte - no need for escaping */
151 sio_send(c, priv->sd);
152 break;
153 }
154 }
155 }
156 /* End with packet delimiter. */
157 sio_send(SLIP_END, priv->sd);
158 return ERR_OK;
159}

207static struct pbuf *
208slipif_rxbyte(struct netif *netif, u8_t c)
209{
210 struct slipif_priv *priv;
211 struct pbuf *t;
212
213 LWIP_ASSERT("netif != NULL", (netif != NULL));
214 LWIP_ASSERT("netif->state != NULL", (netif->state != NULL));
215
216 priv = (struct slipif_priv *) &((slipif_net_cfg_state_t *)netif->state)->priv_static;
217
218 switch (priv->state) {
219 case SLIP_RECV_NORMAL:
220 switch (c) {
221 case SLIP_END:
222 if (priv->recved > 0) {
223 /* Received whole packet. */
224 /* Trim the pbuf to the size of the received packet. */
225 pbuf_realloc(priv->q, priv->recved);
226
227 LINK_STATS_INC(link.recv);
228
229 LWIP_DEBUGF(SLIP_DEBUG, ("slipif: Got packet (%"U16_F" bytes)\n", priv->recved));
230 t = priv->q;
231 priv->p = priv->q = NULL;
232 priv->i = priv->recved = 0;
233 return t;
234 }
235 return NULL;
236 case SLIP_ESC:
237 priv->state = SLIP_RECV_ESCAPE;
238 return NULL;
239 default:
240 break;
241 } /* end switch (c) */
242 break;
243 case SLIP_RECV_ESCAPE:
244 /* un-escape END or ESC bytes, leave other bytes
245 (although that would be a protocol error) */
246 switch (c) {
247 case SLIP_ESC_END:
248 c = SLIP_END;
249 break;
250 case SLIP_ESC_ESC:
251 c = SLIP_ESC;
252 break;
253 default:
254 break;
255 }
256 priv->state = SLIP_RECV_NORMAL;
257 break;
258 default:
259 break;
260 } /* end switch (priv->state) */
261
262 /* byte received, packet not yet completely received */
263 if (priv->p == NULL) {
264 /* allocate a new pbuf */
265 LWIP_DEBUGF(SLIP_DEBUG, ("slipif_input: alloc\n"));
266 priv->p = pbuf_alloc(PBUF_LINK, (PBUF_POOL_BUFSIZE - PBUF_LINK_HLEN - PBUF_LINK_ENCAPSULATION_HLEN), PBUF_POOL);
267
268 if (priv->p == NULL) {
269 LINK_STATS_INC(link.drop);
270 LWIP_DEBUGF(SLIP_DEBUG, ("slipif_input: no new pbuf! (DROP)\n"));
271 /* don't process any further since we got no pbuf to receive to */
272 return NULL;
273 }
274
275 if (priv->q != NULL) {
276 /* 'chain' the pbuf to the existing chain */
277 pbuf_cat(priv->q, priv->p);
278 } else {
279 /* p is the first pbuf in the chain */
280 priv->q = priv->p;
281 }
282 }
283
284 /* this automatically drops bytes if > SLIP_MAX_SIZE */
285 if ((priv->p != NULL) && (priv->recved <= SLIP_MAX_SIZE)) {
286 ((u8_t *)priv->p->payload)[priv->i] = c;
287 priv->recved++;
288 priv->i++;
289 if (priv->i >= priv->p->len) {
290 /* on to the next pbuf */
291 priv->i = 0;
292 if (priv->p->next != NULL && priv->p->next->len > 0) {
293 /* p is a chain, on to the next in the chain */
294 priv->p = priv->p->next;
295 } else {
296 /* p is a single pbuf, set it to NULL so next time a new
297 * pbuf is allocated */
298 priv->p = NULL;
299 }
300 }
301 }
302 return NULL;
303}

360err_t
361slipif_init(struct netif *netif)
362{
363 LWIP_ASSERT("netif is NULL", NULL != netif);
364
365 struct slipif_priv *priv = NULL;
366 slipif_net_cfg_state_t *p_state = (slipif_net_cfg_state_t *) netif->state;
367 u8_t sio_num;
368
369 LWIP_ASSERT("slipif needs an input callback", netif->input != NULL);
370
371 NETIF_SET_CHECKSUM_CTRL(netif, NETIF_CHECKSUM_ENABLE_ALL);
372
373 /* netif->state contains serial port number */
374 sio_num = LWIP_PTR_NUMERIC_CAST(u8_t, p_state->devnum);
375
376 LWIP_DEBUGF(SLIP_DEBUG, ("slipif_init: netif->num=%"U16_F"\n", (u16_t)sio_num));
377
378// /* Allocate private data */
379// priv = (struct slipif_priv *)mem_malloc(sizeof(struct slipif_priv));
380// if (!priv) {
381// return ERR_MEM;
382// }
383
384 priv = (struct slipif_priv *) &(p_state->priv_static);
385
386 netif->name[0] = 's';
387 netif->name[1] = 'l';
388#if LWIP_IPV4
389 netif->output = slipif_output_v4;
390#endif /* LWIP_IPV4 */
391#if LWIP_IPV6
392 netif->output_ip6 = slipif_output_v6;
393#endif /* LWIP_IPV6 */
394 netif->mtu = SLIP_MAX_SIZE;
395
396 /* Try to open the serial port. */
397 priv->sd = sio_open(sio_num);
398 if (!priv->sd) {
399 /* Opening the serial port failed. */
400 mem_free(priv);
401 return ERR_IF;
402 }
403
404 /* Initialize private data */
405 priv->p = NULL;
406 priv->q = NULL;
407 priv->state = SLIP_RECV_NORMAL;
408 priv->i = 0;
409 priv->recved = 0;
410#if SLIP_RX_FROM_ISR
411 priv->rxpackets = NULL;
412#endif
413
414// netif->state = priv;
415
416 /* initialize the snmp variables and counters inside the struct netif */
417 MIB2_INIT_NETIF(netif, snmp_ifType_slip, SLIP_SIO_SPEED(priv->sd));
418
419#if SLIP_USE_RX_THREAD
420 /* Create a thread to poll the serial line. */
421 sys_thread_new(SLIPIF_THREAD_NAME, slipif_loop_thread, netif,
422 SLIPIF_THREAD_STACKSIZE, SLIPIF_THREAD_PRIO);
423#endif /* SLIP_USE_RX_THREAD */
424 return ERR_OK;
425}

456void
457slipif_process_rxqueue(struct netif *netif)
458{
459 struct slipif_priv *priv;
460 SYS_ARCH_DECL_PROTECT(old_level);
461
462 LWIP_ASSERT("netif != NULL", (netif != NULL));
463 LWIP_ASSERT("netif->state != NULL", (netif->state != NULL));
464
465 priv = (struct slipif_priv *) &((slipif_net_cfg_state_t *)netif->state)->priv_static;
466
467 SYS_ARCH_PROTECT(old_level);
468 while (priv->rxpackets != NULL) {
469 struct pbuf *p = priv->rxpackets;
470#if SLIP_RX_QUEUE
471 /* dequeue packet */
472 struct pbuf *q = p;
473 while ((q->len != q->tot_len) && (q->next != NULL)) {
474 q = q->next;
475 }
476 priv->rxpackets = q->next;
477 q->next = NULL;
478#else /* SLIP_RX_QUEUE */
479 priv->rxpackets = NULL;
480#endif /* SLIP_RX_QUEUE */
481 SYS_ARCH_UNPROTECT(old_level);
482 if (netif->input(p, netif) != ERR_OK) {
483 pbuf_free(p);
484 }
485 SYS_ARCH_PROTECT(old_level);
486 }
487 SYS_ARCH_UNPROTECT(old_level);
488}

495static void
496slipif_rxbyte_enqueue(struct netif *netif, u8_t data)
497{
498 struct pbuf *p;
499 struct slipif_priv *priv = (struct slipif_priv *) &((slipif_net_cfg_state_t *)netif->state)->priv_static;
500 SYS_ARCH_DECL_PROTECT(old_level);
501
502 p = slipif_rxbyte(netif, data);
503 if (p != NULL) {
504 SYS_ARCH_PROTECT(old_level);
505 if (priv->rxpackets != NULL) {
506#if SLIP_RX_QUEUE
507 /* queue multiple pbufs */
508 struct pbuf *q = p;
509 while (q->next != NULL) {
510 q = q->next;
511 }
512 q->next = p;
513 } else {
514#else /* SLIP_RX_QUEUE */
515 pbuf_free(priv->rxpackets);
516 }
517 {
518#endif /* SLIP_RX_QUEUE */
519 priv->rxpackets = p;
520 }
521 SYS_ARCH_UNPROTECT(old_level);
522 }
523}
