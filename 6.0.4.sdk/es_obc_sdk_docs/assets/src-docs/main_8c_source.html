336void Error_Handler(void)
337{
338 static atomic_bool in_error_handler = false;
339
340 // this check is needed to avoid recursive calls if the
341 // fault_persistor_store() function fails with a call to Error_Handler()
342 if (!in_error_handler)
343 {
344 in_error_handler = true;
345
346 drv_led_set(DRV_LED_ID_GREEN, DRV_LED_STATE_ON);
347 drv_led_set(DRV_LED_ID_AMBER, DRV_LED_STATE_ON);
348
349 // TODO: Remove when problem with noinit-ram is resolved!
350 fault_persistor_store();
351
352 TaskMonitor_ImmediateReset(TASK_MONITOR_FAULT_RESET);
353 }
354}
