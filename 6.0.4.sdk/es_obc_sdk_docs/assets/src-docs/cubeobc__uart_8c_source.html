229static bool fs_uart_tx(const uint8_t* data, const uint32_t data_sz)
230{
231 CRIT_ASSERT(NULL != data);
232 CRIT_ASSERT(0 < data_sz);
233
234 bool ret = false;
235 HAL_StatusTypeDef status = HAL_ERROR;
236
237
238 if ((NULL != data) &&
239 (0 < data_sz) &&
240 (NULL != fs_lock) &&
241 (osOK == osMutexAcquire(fs_lock, osWaitForever)))
242 {
243 uint32_t timeout = data_sz * 100;
244 status = HAL_UART_Transmit(&fs_huart, (uint8_t*)data, (uint16_t)data_sz, timeout);
245
246 if (HAL_OK == status)
247 {
248 ret = true;
249 }
250 else
251 {
252 ES_TRACE_ERROR("tx fail");
253
254 // signal the thread to re-initialize the whole communication
255 fs_uart_req_t req = FS_UART_REQ_REINIT;
256
257 (void)osMessageQueuePut(fs_iniq, (const void*)&req, 0, 0);
258 }
259
260 (void)osMutexRelease(fs_lock);
261 }
262
263 return ret;
264}

313static bool fs_uart_init(void)
314{
315 RCC_PeriphCLKInitTypeDef periph_clk = { 0 };
316
317 periph_clk.PeriphClockSelection = CUBEOBC_UART_RCC_PERIPHCLK;
318 periph_clk.Usart234578ClockSelection = RCC_USART234578CLKSOURCE_D2PCLK1;
319
320 if ((HAL_OK != HAL_RCCEx_PeriphCLKConfig(&periph_clk)) ||
321 (HAL_OK != HAL_UART_Init(&fs_huart)))
322 {
323 return false;
324 }
325
326 cubeobc_dma_rx_clk_enable();
327
328 if (HAL_OK != HAL_DMA_Init(&fs_hdma_uart_rx))
329 {
330 return false;
331 }
332
333 __HAL_LINKDMA(&fs_huart, hdmarx, fs_hdma_uart_rx);
334
335 if ((HAL_OK != HAL_UART_RegisterCallback(&fs_huart, HAL_UART_ERROR_CB_ID, fs_uart_err_callback)) ||
336 (HAL_OK != HAL_UART_RegisterRxEventCallback(&fs_huart, fs_uart_rx_event_callback)) ||
337 (HAL_OK != HAL_UART_AbortReceive(&fs_huart)) ||
338 (HAL_OK != HAL_UARTEx_ReceiveToIdle_DMA(&fs_huart, (uint8_t*)fs_rx_msg.data, sizeof(fs_rx_msg.data))))
339 {
340 return false;
341 }
342
343 __HAL_DMA_DISABLE_IT(&fs_hdma_uart_rx, DMA_IT_HT);
344
345 HAL_NVIC_SetPriority(CUBEOBC_DMA_RX_IRQN, CUBEOBC_IRQ_PRIO, 0);
346 HAL_NVIC_EnableIRQ(CUBEOBC_DMA_RX_IRQN);
347
348 HAL_NVIC_SetPriority(CUBEOBC_UART_IRQN, CUBEOBC_IRQ_PRIO, 0);
349 HAL_NVIC_EnableIRQ(CUBEOBC_UART_IRQN);
350
351 return true;
352}

456static void fs_uart_rx_event_callback(UART_HandleTypeDef* huart, uint16_t size)
457{
458 if (CUBEOBC_UART_INSTANCE == huart->Instance)
459 {
460 CRIT_ASSERT(NULL != fs_rxq);
461
462 // try to add the message to the RX queue
463
464 fs_rx_msg.size = size;
465
466 if ((NULL == fs_rxq) || (osOK != osMessageQueuePut(fs_rxq, (const void*)&fs_rx_msg, 0, 0)))
467 {
468 // no logging from isr
469 }
470
471 // re-enable RX interrupt
472 if (HAL_OK != HAL_UARTEx_ReceiveToIdle_DMA(&fs_huart, (uint8_t*)fs_rx_msg.data, sizeof(fs_rx_msg.data)))
473 {
474 fs_uart_err_callback(&fs_huart);
475 }
476
477 // disable the half transfer interrupt because we are not interested in it
478 __HAL_DMA_DISABLE_IT(&fs_hdma_uart_rx, DMA_IT_HT);
479 }
480}

486static void fs_uart_thread(void* args)
487{
488 (void)args;
489
490 uint8_t msg_prio;
491 fs_uart_req_t action;
492 bool init_done = false;
493 bool failed = true;
494
495 TaskMonitor_TaskInitialized(TASK_MONITOR_CUBEADCS_GEN2_COMM);
496 task_mon_set_task_policy(TASK_MONITOR_CUBEADCS_GEN2_COMM, TASK_MON_POLICY_MONITOR);
497
498 while (1)
499 {
500 TaskMonitor_IamAlive(TASK_MONITOR_CUBEADCS_GEN2_COMM);
501
502 if ((osOK == osMessageQueueGet(fs_iniq, &action, &msg_prio, 1000)) &&
503 (osOK == osMutexAcquire(fs_lock, osWaitForever)))
504 {
505 if (0 != (FS_UART_REQ_DEINIT & action))
506 {
507 if (false == fs_uart_deinit())
508 {
509 failed = true;
510 ES_TRACE_ERROR("deinit fail");
511 }
512 else
513 {
514 failed = false;
515 init_done = false;
516 }
517 }
518
519 // consecutive init not allowed
520 if ((0 != (FS_UART_REQ_INIT & action)) && (false == init_done))
521 {
522 init_done = fs_uart_init();
523
524 if (false == init_done)
525 {
526 failed = true;
527 ES_TRACE_ERROR("init fail");
528 }
529 else
530 {
531 failed = false;
532 }
533 }
534
535 (void)osMutexRelease(fs_lock);
536 }
537
538 // TX might get blocked if no delay is added here.
539 // Also, since when cube adcs is powered off there is
540 // a LOT of error interrupts in DMA dew to floating RX/TX line levels,
541 // other tasks might be blocked from execution.
542 osDelay(100);
543
544 // deinit or init failed... wait and try again
545 if (true == failed)
546 {
547 osDelay(1000);
548
549 action = FS_UART_REQ_REINIT;
550
551 (void)osMessageQueuePut(fs_iniq, (const void*)&action, 0, 0);
552 }
553
554 vApplicationLowStackCheck(TASK_MONITOR_CUBEADCS_GEN2_COMM);
555 }
556}

607ErrorCode cubeObc_uartIfc_rx(U8* data, U32 size, U32* size_read)
608{
609 ErrorCode err = CUBEOBC_ERROR_READ;
610
611 if ((false == fs_rx_state.read_done) &&
612 (true == fs_uart_rx(&fs_rx_state.msg)) &&
613 (0 != fs_rx_state.msg.size) &&
614 (sizeof(fs_rx_state.msg.data) >= fs_rx_state.msg.size))
615 {
616 fs_rx_state.read_done = true;
617 fs_rx_state.offset = fs_rx_state.msg.size;
618 }
619
620 if ((true == fs_rx_state.read_done) &&
621 (size <= fs_rx_state.offset) &&
622 (0 != fs_rx_state.offset))
623 {
624 memcpy(data, &fs_rx_state.msg.data[fs_rx_state.msg.size - fs_rx_state.offset], size);
625 fs_rx_state.offset -= size;
626 *size_read = size;
627
628 err = CUBEOBC_ERROR_OK;
629 }
630
631 return err;
632}
