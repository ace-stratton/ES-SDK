34ESSATMAC_ErrCodes upy_fp_gw_enq_message(ESSATMAC_Context *pMACContext, uint8_t *pMsgData, uint8_t nMsgDataSize) // NOSONAR
35{
36 ESSATMAC_ErrCodes result = ESSATMAC_EC_ERROR;
37
38 if ((pMACContext != NULL) && (pMsgData != NULL) &&
39 (nMsgDataSize >= sizeof(sMicroPythonGwHdr_t)))
40 {
41 if ((pMACContext->netType == ESSASNETT_MICROPYTHON_GW) &&
42 (pMACContext->nInterfaceNumber == MICROPYTHON_GW_IF_ID) &&
43 (pMACContext->nPayloadProtocol ==
44 ES_SAT_MAC_PROTOCOL_ID_FP_LAYER) &&
45 (uPyRespCtx.pu8DataBuf != NULL) &&
46 (uPyRespCtx.u32MaxDataSize >= nMsgDataSize))
47 {
48 // transfer message data back to CP buffer
49 (void)memcpy(uPyRespCtx.pu8DataBuf, pMsgData, nMsgDataSize);
50
51 // store actual data size for CP command reply dispatch in
52 // cp_cmd_handler_mac_fp_gw_send_cmd(...)
53 uPyRespCtx.u32ResponseSize = nMsgDataSize;
54
55 upy_release_thread_wait();
56
57 result = ESSATMAC_EC_OK;
58 }
59 else
60 {
61 result = ESSATMAC_EC_INVALID_PARAMS;
62 }
63 }
64
65 return result;
66}

68eCPDispatchResult_t upy_fp_gw_send_cmd(uint8_t *pMsgData, uint8_t nMsgDataSize,
69 uint8_t *pRespBuff, uint8_t nRespBuffSize,
70 uint8_t *const u8TLError)
71{
72 eCPDispatchResult_t res = ECPDISPATCHRESULT_ERROR;
73 ESSATMAC_Context MACContext;
74 eThreadBlockEvt_t waitResult = THREADBLOCKEVT_ERROR;
75
76 if (NULL != pMsgData)
77 {
78 (void)memset((void *)&MACContext, (uint8_t)0, sizeof(ESSATMAC_Context));
79
80 // setup response buffer for ESSA Stack
81 uPyRespCtx.pu8DataBuf = pRespBuff;
82 uPyRespCtx.u32MaxDataSize = nRespBuffSize;
83 uPyRespCtx.u32ResponseSize = 0U;
84 uPyRespCtx.onCmdResponseReady = NULL;
85 uPyRespCtx.pUserCtx = NULL;
86
87 MACContext.nInterfaceNumber = MICROPYTHON_GW_IF_ID;
88 MACContext.nPayloadProtocol = ES_SAT_MAC_PROTOCOL_ID_FP_LAYER;
89 MACContext.nSourceAddr = 0x00U;
90 MACContext.nTargetAddr = 0x00U;
91 MACContext.netType = ESSASNETT_MICROPYTHON_GW;
92
93 upy_prepare_thread_wait();
94
95 // inject message to ESSA Stack for FP dispatching...
96 ESSA_Stack_DataDispatch(&MACContext, pMsgData, nMsgDataSize);
97
98 // wait for CPCmdHandler_uPyFpGwEnqMessage() to be called or timeout...
99 waitResult = upy_thread_wait_for_event(UPYFPGW_RESPONSE_TIMEOUT_MS);
100
101 if (waitResult == THREADBlOCKEVT_WAITFINISHED)
102 {
103 *u8TLError = EESTL_SFERR_SUCCESS;
104 }
105 else
106 {
107 if (waitResult == THREADBLOCKEVT_WAITTIMEOUT)
108 {
109 *u8TLError = EESTL_SFERR_RECEIVE_TIMEOUT;
110 }
111 else
112 {
113 *u8TLError = EESTL_SFERR_UNKNOWN;
114 }
115 }
116
117 res = ECPDISPATCHRESULT_RESULT_OK;
118 }
119
120 return res;
