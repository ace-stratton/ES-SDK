85STATIC sds_op_status_t sensor_css_get_data_raw(sds_css_data_t * const p_data_dst)
86{
87 CRIT_ASSERT(NULL != p_data_dst);
88
89 uint16_t css_raw_data[MAX_PAN] = { 0, 0, 0, 0, 0, 0 };
90
91 for (uint8_t i = 0; i < MAX_PAN; i++)
92 {
93 if ( Pan_PD_ADC_Measure(i, &(css_raw_data[i])) != SEN_SUCCESS )
94 {
95 css_err_code = SDS_CSS_ERR_CODE_ERR;
96 }
97 }
98
99 p_data_dst->css_1_adc = css_raw_data[0];
100 p_data_dst->css_2_adc = css_raw_data[1];
101 p_data_dst->css_3_adc = css_raw_data[2];
102 p_data_dst->css_4_adc = css_raw_data[3];
103 p_data_dst->css_5_adc = css_raw_data[4];
104 p_data_dst->css_6_adc = css_raw_data[5];
105
106 SDS_TRACE_DBG("%s CSS RAW DATA---1:%d, 2:%d, 3:%d, 4:%d, 5:%d, 6:%d\r", SDS_DBG_HEAD,
107 p_data_dst->css_1_adc, p_data_dst->css_2_adc, p_data_dst->css_3_adc,
108 p_data_dst->css_4_adc, p_data_dst->css_5_adc, p_data_dst->css_6_adc);
109
110 return SDS_OP_STATUS_OK;
111}
