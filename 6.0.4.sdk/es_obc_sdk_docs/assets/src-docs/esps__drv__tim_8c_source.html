42bool MAC_TIM_bInit(void)
43{
44 uint32_t u32TIMIdx;
45
46 for(u32TIMIdx = 0U; u32TIMIdx < CDEF_ELEMENT_COUNT(TIM_CFG_asTimerEntries); u32TIMIdx++)
47 {
48 NVIC_EnableIRQ(TIM_CFG_asTimerEntries[u32TIMIdx].u32NVICInterrupt);
49
50 HAL_NVIC_SetPriority(TIM_CFG_asTimerEntries[u32TIMIdx].u32NVICInterrupt, TIMER_ISR_PRIO, u32TIMIdx);
51
52 MAC_RCC_vPeripheralClockEnable(TIM_CFG_asTimerEntries[u32TIMIdx].eClockSrc,
53 TIM_CFG_asTimerEntries[u32TIMIdx].u32ClockMask);
54
55 LL_TIM_SetPrescaler(TIM_CFG_asTimerEntries[u32TIMIdx].psTIM,
56 MacCfg_TIM_u16CalculatePrescaler(&TIM_CFG_asTimerEntries[u32TIMIdx]));
57
58 LL_TIM_DisableCounter(TIM_CFG_asTimerEntries[u32TIMIdx].psTIM);
59
60 LL_TIM_DisableIT_UPDATE(TIM_CFG_asTimerEntries[u32TIMIdx].psTIM);
61
62 }
63
64 return true;
65}

85void MAC_TIM_vMicroTimerStart(eTIMTimerID_t fp_eTIM, uint32_t fp_u32Timeout, pfTIMCallback_t fp_pfCallback,
86 void *fp_pvData)
87{
88 sTIMTimerEntry_t *psTIM;
89
90 if((fp_pfCallback == NULL) || (fp_u32Timeout == 0U) || (fp_eTIM >= eTIM_TIM_ID_Count))
91 {
92 EXEH_HANDLE(eEXEHSeverity_Warning, (int32_t) eMACException_InvalidParameter);
93 }
94 else
95 {
96 psTIM = &TIM_CFG_asTimerEntries[fp_eTIM];
97
98 if(psTIM->bRunning)
99 {
100 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_TIMUsed);
101 }
102 else
103 {
104 psTIM->bRunning = true;
105
106 psTIM->pfCallback = fp_pfCallback;
107 psTIM->pvData = fp_pvData;
108 LL_TIM_SetAutoReload(psTIM->psTIM,
109 (fp_u32Timeout <=
110 MAX_TIMER_RANGE_US) ? ((uint16_t) fp_u32Timeout) : ((uint16_t) MAX_TIMER_RANGE_US));
111 LL_TIM_SetCounter(psTIM->psTIM, 0U);
112 LL_TIM_EnableCounter(psTIM->psTIM);
113 LL_TIM_ClearFlag_UPDATE(psTIM->psTIM);
114 LL_TIM_EnableIT_UPDATE(psTIM->psTIM);
115 }
116 }
117}

126void MAC_TIM_vMicroTimerCancel(eTIMTimerID_t fp_eTIM, bool fp_bSuppressCancelCbk)
127{
128 pfTIMCallback_t pfCallback = (pfTIMCallback_t) NULL;
129
130 void *pvData = (void *)NULL;
131
132 if(fp_eTIM >= eTIM_TIM_ID_Count)
133 {
134 EXEH_HANDLE(eEXEHSeverity_Warning, (int32_t) eMACException_InvalidParameter);
135 }
136 else
137 {
138 if(TIM_CFG_asTimerEntries[fp_eTIM].bRunning)
139 {
140 pfCallback = TIM_CFG_asTimerEntries[fp_eTIM].pfCallback;
141 pvData = TIM_CFG_asTimerEntries[fp_eTIM].pvData;
142
143 LL_TIM_DisableCounter(TIM_CFG_asTimerEntries[fp_eTIM].psTIM);
144
145 TIM_CFG_asTimerEntries[fp_eTIM].bRunning = false;
146 TIM_CFG_asTimerEntries[fp_eTIM].pfCallback = NULL;
147 TIM_CFG_asTimerEntries[fp_eTIM].pvData = NULL;
148
149 if((pfCallback) && (!fp_bSuppressCancelCbk))
150 {
151 pfCallback(eTIMTimer_Canceled, pvData);
152 }
153 }
154 }

188static void fs_vProcessExpiration(eTIMTimerID_t fp_eTIMID)
189{
190 pfTIMCallback_t pfCallback = TIM_CFG_asTimerEntries[fp_eTIMID].pfCallback;
191
192 void *pvData = TIM_CFG_asTimerEntries[fp_eTIMID].pvData;
193
194 LL_TIM_DisableCounter(TIM_CFG_asTimerEntries[fp_eTIMID].psTIM);
195 LL_TIM_SetCounter(TIM_CFG_asTimerEntries[fp_eTIMID].psTIM, 0U);
196 LL_TIM_DisableIT_UPDATE(TIM_CFG_asTimerEntries[fp_eTIMID].psTIM);
197
198 if((TIM_CFG_asTimerEntries[fp_eTIMID].bRunning) && (TIM_CFG_asTimerEntries[fp_eTIMID].pfCallback))
199 {
200 TIM_CFG_asTimerEntries[fp_eTIMID].pfCallback = NULL;
201 TIM_CFG_asTimerEntries[fp_eTIMID].pvData = NULL;
202 TIM_CFG_asTimerEntries[fp_eTIMID].bRunning = false;
203
204 pfCallback(eTIMTimer_Expired, pvData);
205 }
