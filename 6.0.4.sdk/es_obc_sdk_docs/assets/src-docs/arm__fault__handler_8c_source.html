60static void __attribute__((used)) capture_registers(uint32_t * const p_fault_stack_frame_start)
61{
62 // store stack pointer when the exception occurred
63 temp_fault_memory_dump.sp = (uint32_t) p_fault_stack_frame_start;
64
65 // for the exact register layout on the stack, refer to ARMv7-M Architecture Reference Manual
66 // (https://developer.arm.com/documentation/ddi0403/ee/)
67 // - Ch. "Stack Alignment on exception entry"
68 // - Ch. "Context state stacking on exception entry with the FP extension"
69 temp_fault_memory_dump.hdr.reserved[0] = 0;
70 temp_fault_memory_dump.hdr.reserved[1] = 0;
71 temp_fault_memory_dump.hdr.reserved[2] = 0;
72 temp_fault_memory_dump.hdr.reserved[3] = 0;
73 temp_fault_memory_dump.stack_frame_len = sizeof(basic_x_frame_t);
74
75 // copy exception frame register values from stack
76 temp_fault_memory_dump.stack_frame.basic_regs.r0 = p_fault_stack_frame_start[0];
77 temp_fault_memory_dump.stack_frame.basic_regs.r1 = p_fault_stack_frame_start[1];
78 temp_fault_memory_dump.stack_frame.basic_regs.r2 = p_fault_stack_frame_start[2];
79 temp_fault_memory_dump.stack_frame.basic_regs.r3 = p_fault_stack_frame_start[3];
80 temp_fault_memory_dump.stack_frame.basic_regs.r12 = p_fault_stack_frame_start[4];
81 temp_fault_memory_dump.stack_frame.basic_regs.lr = p_fault_stack_frame_start[5];
82 temp_fault_memory_dump.stack_frame.basic_regs.pc = p_fault_stack_frame_start[6];
83 temp_fault_memory_dump.stack_frame.basic_regs.psr = p_fault_stack_frame_start[7];
84
85 // is FP extension enabled
86 if (CONTROL_FPCA_Msk == (__get_CONTROL() & CONTROL_FPCA_Msk))
87 {
88 temp_fault_memory_dump.stack_frame_len = sizeof(extended_x_frame_t);
89
90 // copy extended FP register values from stack
91 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 0] = p_fault_stack_frame_start[8];
92 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 1] = p_fault_stack_frame_start[9];
93 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 2] = p_fault_stack_frame_start[10];
94 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 3] = p_fault_stack_frame_start[11];
95 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 4] = p_fault_stack_frame_start[12];
96 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 5] = p_fault_stack_frame_start[13];
97 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 6] = p_fault_stack_frame_start[14];
98 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 7] = p_fault_stack_frame_start[15];
99 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 8] = p_fault_stack_frame_start[16];
100 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 9] = p_fault_stack_frame_start[17];
101 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[10] = p_fault_stack_frame_start[18];
102 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[11] = p_fault_stack_frame_start[19];
103 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[12] = p_fault_stack_frame_start[20];
104 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[13] = p_fault_stack_frame_start[21];
105 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[14] = p_fault_stack_frame_start[22];
106 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[15] = p_fault_stack_frame_start[23];
107
108 temp_fault_memory_dump.stack_frame.fpscr = __get_FPSCR();
109 }
110 else
111 {
112 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 0] = 0;
113 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 1] = 0;
114 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 2] = 0;
115 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 3] = 0;
116 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 4] = 0;
117 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 5] = 0;
118 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 6] = 0;
119 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 7] = 0;
120 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 8] = 0;
121 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[ 9] = 0;
122 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[10] = 0;
123 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[11] = 0;
124 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[12] = 0;
125 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[13] = 0;
126 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[14] = 0;
127 temp_fault_memory_dump.stack_frame.fp_ext_reg_s[15] = 0;
128
129 temp_fault_memory_dump.stack_frame.fpscr = 0;
130 }
131
132 // copy additional CPU registers which may be useful for analyzing the problem (not residing on stack)
133 if (NULL != SCB)
134 {
135 temp_fault_memory_dump.cfsr = SCB->CFSR;
136 temp_fault_memory_dump.abfsr = SCB->ABFSR;
137 temp_fault_memory_dump.hfsr = SCB->HFSR;
138 temp_fault_memory_dump.mmfar = SCB->MMFAR;
139 temp_fault_memory_dump.bfar = SCB->BFAR;
140 temp_fault_memory_dump.vtor = SCB->VTOR;
141 }
142
143 // copy environment data
144 temp_fault_memory_dump.hdr.record_type = ARM_FAULT_REC_TYPE_USER;
145 temp_fault_memory_dump.hdr.exception_rtc_ts = env_data.rtc_time;
146 temp_fault_memory_dump.hdr.uptime = env_data.uptime;
147
148 // calculate CRC but skip the CRC field itself
149 temp_fault_memory_dump.crc = crc32(CRC32_INIT_REMAINDER, (uint8_t *) &temp_fault_memory_dump, sizeof(temp_fault_memory_dump) - sizeof(temp_fault_memory_dump.crc));
150
151 // handle stored frame and reset the system...
152 Error_Handler();
153}

155void arm_fault_handler_init(void)
156{
157 if (NULL != SCB)
158 {
159 SCB->CCR |= SCB_CCR_DIV_0_TRP_Pos; // enable zero devision fault
160 SCB->SHCSR |= (SCB_SHCSR_USGFAULTENA_Msk | SCB_SHCSR_BUSFAULTENA_Msk | SCB_SHCSR_MEMFAULTENA_Msk);
161 }

172bool arm_fault_handler_read_last(persisted_ram_x_frame_t * const p_buf)
173{
174 bool res = false;
175
176 if (NULL != p_buf)
177 {
178 if (arm_fault_handler_is_frame_valid(&temp_fault_memory_dump))
179 {
180 (void) memcpy((void *) p_buf, (void *) &temp_fault_memory_dump, sizeof(persisted_ram_x_frame_t));
181 (void) memset((void *) &temp_fault_memory_dump, 0U, sizeof(persisted_ram_x_frame_t));
182 temp_fault_memory_dump.hdr.record_type = ARM_FAULT_REC_TYPE_EMPTY; // just mark the region as empty (CRC will not be valid)
183
184 res = true;
185 }
186 }
187
188 return res;
189}

191bool arm_fault_handler_is_frame_valid(persisted_ram_x_frame_t * const p_frame)
192{
193 bool res = false;
194
195 if (NULL != p_frame)
196 {
197 uint32_t ram_crc = crc32(CRC32_INIT_REMAINDER, (uint8_t *) p_frame, sizeof(temp_fault_memory_dump) - sizeof(temp_fault_memory_dump.crc));
198
199 res = (ram_crc == p_frame->crc);
200 }
201
202 return res;
203}

213void arm_fault_handler_generic(const uint8_t * const p_user_data, const uint32_t user_data_sz)
214{
215 if ((NULL == p_user_data) || (0U == user_data_sz))
216 {
217 return;
218 }
219
220 static_assert(sizeof(persisted_ram_x_frame_t) == sizeof(persisted_ram_generic_frame_t));
221
222 // the following cast is ok because persisted_ram_generic_frame_t and persisted_ram_x_frame_t are the same size
223 persisted_ram_generic_frame_t *p_generic_frame = (persisted_ram_generic_frame_t *) &temp_fault_memory_dump;
224
225 // copy environment data (add 'G' to record type for generic to indicate this fault doesn't contain register capture)
226 p_generic_frame->hdr.record_type = ARM_FAULT_REC_TYPE_USER | ('G' << 8);
227 p_generic_frame->hdr.exception_rtc_ts = env_data.rtc_time;
228 p_generic_frame->hdr.uptime = env_data.uptime;
229
230 uint32_t sz_to_copy = (user_data_sz > sizeof(p_generic_frame->data)) ? sizeof(p_generic_frame->data) : user_data_sz;
231
232 (void) memset(p_generic_frame->data, 0U, sizeof(p_generic_frame->data));
233 (void) memcpy(p_generic_frame->data, p_user_data, sz_to_copy);
234
235 p_generic_frame->crc = crc32(CRC32_INIT_REMAINDER, (uint8_t *) p_generic_frame, sizeof(persisted_ram_generic_frame_t) - sizeof(p_generic_frame->crc));
236}
