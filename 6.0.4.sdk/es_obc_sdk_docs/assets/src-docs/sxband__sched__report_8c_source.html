63static bool copy_sxband_status_to_report(FIL *report_file_hnd)
64{
65 bool res = true;
66
67 if (report_rt_ctx.file_opened)
68 {
69 uint8_t copy_buf[SXBAND_LOG_COPY_BUF_SIZE];
70 FIL sxband_log_hnd;
71 FRESULT fres = SdMngr_f_open(&sxband_log_hnd, SX_BAND_FOUTPUT_NAME, FA_READ);
72 UINT bytes_read = 0U;
73 bool write_ok = false;
74
75 if (FR_OK == fres)
76 {
77 fres = SdMngr_f_read(&sxband_log_hnd, copy_buf, sizeof(copy_buf), &bytes_read);
78
79 if (FR_OK == fres)
80 {
81 while (bytes_read > 0U)
82 {
83 write_ok = write_and_verify(report_file_hnd, copy_buf, bytes_read);
84
85 if (write_ok)
86 {
87 fres = SdMngr_f_read(&sxband_log_hnd, copy_buf, sizeof(copy_buf), &bytes_read);
88
89 if (FR_OK != fres)
90 {
91 bytes_read = 0;
92 res = false;
93 }
94 }
95 else
96 {
97 bytes_read = 0;
98 res = false;
99
100 EXEH_HANDLE(eEXEHSeverity_Error,
101 (int32_t) EXEH_SXB_SCHED_REPORT_CANT_WRITE_FILE);
102 }
103 }
104 }
105 else
106 {
107 EXEH_HANDLE(eEXEHSeverity_Error,
108 (int32_t) EXEH_SXB_SCHED_REPORT_SXB_LOG_READ_ERR);
109 }
110 }
111 else
112 {
113 EXEH_HANDLE(eEXEHSeverity_Error,
114 (int32_t) EXEH_SXB_SCHED_REPORT_SXB_LOG_OPEN_ERR);
115 }
116
117 (void) SdMngr_f_close(&sxband_log_hnd);
118 }
119 else
120 {
121 CRIT_ASSERT(false);
122 }
123
124 return res;
125}

138static inline bool write_and_verify(FIL *fhnd, const uint8_t *p_buf, const uint32_t buf_size)
139{
140 UINT bytes_written;
141 FRESULT fres = SdMngr_f_write(fhnd, (const void *) p_buf, (UINT) buf_size, &bytes_written);
142
143 return (FR_OK == fres) && (bytes_written == buf_size);
144}

146void sxband_sched_report_open(const char * const p_fname)
147{
148 report_rt_ctx.file_opened = false;
149
150 CRIT_ASSERT(NULL != p_fname);
151
152 // remove temporary report if it exists
153 (void) SdMngr_f_unlink(TEMP_REPORT_FNAME);
154 report_rt_ctx.p_actual_rep_fname = p_fname;
155
156 FRESULT fres = SdMngr_f_open(&report_rt_ctx.rep_file_hnd, TEMP_REPORT_FNAME, FA_CREATE_ALWAYS | FA_WRITE);
157
158 if (FR_OK == fres)
159 {
160 UINT bytes_written = 0U;
161
162 fres = SdMngr_f_write(&report_rt_ctx.rep_file_hnd, CSV_HEADER_COL_NAMES, sizeof(CSV_HEADER_COL_NAMES) - 1U, &bytes_written);
163
164 if ((FR_OK == fres) && (bytes_written == sizeof(CSV_HEADER_COL_NAMES) - 1U))
165 {
166 report_rt_ctx.file_opened = true;
167 }
168 else
169 {
170 (void) SdMngr_f_close(&report_rt_ctx.rep_file_hnd);
171
172 EXEH_HANDLE(eEXEHSeverity_Error,
173 (int32_t) EXEH_SXB_SCHED_REPORT_CANT_WRITE_FILE);
174 }
175 }
176 else
177 {
178 EXEH_HANDLE(eEXEHSeverity_Error,
179 (int32_t) EXEH_SXB_SCHED_REPORT_CANT_OPEN_FILE);
180 }
181}

183void sxband_sched_report_append(const sxband_sched_cmd_t * const p_cmd, const sxband_sched_cmd_status_id cmd_user_status)
184{
185 char format_buf[8U];
186 uint8_t size = 0U;
187 bool write_success = false;
189 CRIT_ASSERT(NULL != p_cmd);
190
191 if (report_rt_ctx.file_opened)
192 {
193 // format command index
194 (void) memset(format_buf, 0U, sizeof(format_buf));
195 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "%ld,", p_cmd->cmd_idx);
196
197 write_success = write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
198
199 // format command id
200 (void) memset(format_buf, 0U, sizeof(format_buf));
201 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "0x%02x,", p_cmd->p_cmd_hdr->cmd);
202
203 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
204
205 // format type 'R'/'W'
206 (void) memset(format_buf, 0U, sizeof(format_buf));
207 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "%c,", p_cmd->p_cmd_hdr->type);
208
209 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
210
211 // format size
212 (void) memset(format_buf, 0U, sizeof(format_buf));
213 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "%d,", p_cmd->p_cmd_hdr->size);
214
215 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
216
217 if (S_X_BAND_TRNSM_GW_CMD_ID_UPLOAD_FILE == p_cmd->p_cmd_hdr->cmd)
218 {
219 // add opening quote only if there are command args
220 if (p_cmd->p_cmd_hdr->size > 0U)
221 {
222 (void) memset(format_buf, 0U, sizeof(format_buf));
223 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "'");
224
225 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
226 }
227
228 // format data bytes
229 (void) memset(format_buf, 0U, sizeof(format_buf));
230
231 for (uint8_t data_idx = 0U; data_idx < p_cmd->p_cmd_hdr->size; data_idx++)
232 {
233 if ('\0' != p_cmd->p_data[data_idx])
234 {
235 (void) memset(format_buf, 0U, sizeof(format_buf));
236 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "%c", p_cmd->p_data[data_idx]);
237
238 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
239 }
240 }
241
242 // add closing quote only if there are command args
243 if (p_cmd->p_cmd_hdr->size > 0U)
244 {
245 (void) memset(format_buf, 0U, sizeof(format_buf));
246 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "'");
247
248 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
249 }
250 }
251 else
252 {
253 // add opening quote only if there are command args
254 if (p_cmd->p_cmd_hdr->size > 0U)
255 {
256 (void) memset(format_buf, 0U, sizeof(format_buf));
257 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "hex('");
258
259 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
260 }
261
262 // format data bytes
263 (void) memset(format_buf, 0U, sizeof(format_buf));
264
265 for (uint8_t data_idx = 0U; data_idx < p_cmd->p_cmd_hdr->size; data_idx++)
266 {
267 (void) memset(format_buf, 0U, sizeof(format_buf));
268 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "%02x", p_cmd->p_data[data_idx]);
269
270 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
271 }
272
273 // add closing quote only if there are command args
274 if (p_cmd->p_cmd_hdr->size > 0U)
275 {
276 (void) memset(format_buf, 0U, sizeof(format_buf));
277 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), "')");
278
279 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
280 }
281 }
282
283 // add comma (', ') after data field
284 (void) memset(format_buf, 0U, sizeof(format_buf));
285 size = (uint8_t) snprintf((char *) format_buf, sizeof(format_buf), ",");
286
287 write_success = write_success && write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t *) format_buf, size);
288
289 // put double quotes to open the 'status' column of the CSV file
290 (void) write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t * const) "\"", 1U);
291
292 switch (cmd_user_status)
293 {
294 case SXBAND_SCHED_CMD_STATUS_TIMEDOUT:
295 {
296 (void) SdMngr_f_write(&report_rt_ctx.rep_file_hnd, TIMEOUT_STR, sizeof(TIMEOUT_STR) - 1U, NULL);
297
298 break;
299 }
300
301 case SXBAND_SCHED_CMD_STATUS_NO_FILES:
302 {
303 (void) SdMngr_f_write(&report_rt_ctx.rep_file_hnd, NO_FILES_STR, sizeof(NO_FILES_STR) - 1U, NULL);
304
305 break;
306 }
307
308 case SXBAND_SCHED_CMD_STATUS_DEFAULT:
309 default:
310 {
311 if (!copy_sxband_status_to_report(&report_rt_ctx.rep_file_hnd))
312 {
313 // if copying of SXBand command output failed for some reason, put a general error message in the report
314 (void) SdMngr_f_write(&report_rt_ctx.rep_file_hnd, ERROR_STR, sizeof(ERROR_STR) - 1U, NULL);
315 }
316
317 break;
318 }
319 }
320
321 // put an ending double quotes a new line to the status column in the CSV file to end the current command record
322 (void) write_and_verify(&report_rt_ctx.rep_file_hnd, (const uint8_t * const) "\"\n\r", 2U);
323
324 // explicitly flush buffered record to storage
325 SdMngr_f_sync(&report_rt_ctx.rep_file_hnd);
326
327 if (!write_success)
328 {
329 EXEH_HANDLE(eEXEHSeverity_Error,
330 (int32_t) EXEH_SXB_SCHED_REPORT_CANT_WRITE_FILE);
331 }
332 }
333 else
334 {
335 CRIT_ASSERT(false);
336 }
337}

350void sxband_sched_report_close(void)
351{
352 if (report_rt_ctx.file_opened)
353 {
354 (void) SdMngr_f_close(&report_rt_ctx.rep_file_hnd);
355
356 // delete any previous report with the same name if it exists
357 (void) SdMngr_f_unlink(report_rt_ctx.p_actual_rep_fname);
358
359 // rename the temporary report to the actual name requested by the user
360 if (FR_OK != SdMngr_f_rename(TEMP_REPORT_FNAME, report_rt_ctx.p_actual_rep_fname))
361 {
362 EXEH_HANDLE(eEXEHSeverity_Warning,
363 (int32_t) EXEH_SXB_SCHED_REPORT_SXB_LOG_RENAME_ERR);
364 }
365 }
