106uint8_t crc8(uint8_t *buf, uint16_t size, uint8_t bEsc, uint8_t u8EsC)
107{
108 if ( size == 0 )
109 return 0xFF;
110
111 uint16_t i;
112 uint8_t crc = 0;
113
114 for ( i = 0; i < size; i++ ) {
115 if ( (bEsc) && (buf[i] == u8EsC) && (buf[i + 1] == u8EsC) )
116 continue;
117
118 crc = crc8_tab[ crc ^ buf[i] ];
119 }
120
121 return crc;
122
123}
