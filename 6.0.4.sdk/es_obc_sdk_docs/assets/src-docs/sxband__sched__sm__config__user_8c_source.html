69static void trigger_cmd_send(void)
70{
71 ES_TRACE_DEBUG("trigger_cmd_send\r\n");
72
73 sxband_sched_rt_context.cmd_resp_ctx.onCmdResponseReady = &sxband_sched_on_cmd_complete;
74 sxband_sched_rt_context.cmd_resp_ctx.pUserCtx = &sxband_sched_rt_context.cmd_resp_ctx;
75 sxband_sched_rt_context.cmd_resp_ctx.is_all_data_in_buf = true;
76 // no need to setup data buffers, we just need the onCmdResponseReady notification and
77 // the full command output will be taken from the produced file by the SXBand driver
78 sxband_sched_rt_context.cmd_resp_ctx.pu8DataBuf = NULL;
79 sxband_sched_rt_context.cmd_resp_ctx.u32MaxDataSize = 0U;
80 sxband_sched_rt_context.cmd_resp_ctx.u32ResponseSize = 0U;
81
82 // remove the last stored file containing SXBand command output status
83 (void) SdMngr_f_unlink(SX_BAND_FOUTPUT_NAME);
84
85 sxband_sched_rt_context.sxband_result_notified = false;
86
87 timer_start(&sxband_sched_rt_context.cmd_timer);
88
89 S_X_BAND_TRNSM_ESTTC_StartCmd(sxband_sched_rt_context.file_iter_ctx.cmd_list_hdr.sxband_dev_id,
90 sxband_sched_rt_context.current_cmd_entry.p_cmd_hdr->cmd,
91 (uint8_t) sxband_sched_rt_context.current_cmd_entry.p_cmd_hdr->type,
92 sxband_sched_rt_context.current_cmd_entry.p_cmd_hdr->size,
93 (sxband_sched_rt_context.current_cmd_entry.p_cmd_hdr->size > 0U) ?
94 (sxband_sched_rt_context.current_cmd_entry.p_data) : (NULL),
95 &sxband_sched_rt_context.cmd_resp_ctx,
96 &sxband_sched_rt_context.cmd_timeout_ms);
97
98 trace_cmd(&sxband_sched_rt_context.current_cmd_entry);
99}

278bool sxband_sched_sm_guard_can_iterate_files_with_pattern(sState_t* const pSMRoot, const sHSM_EventType_t * const pEvent)
279{
280 // USER_CODE_START::guard_can_iterate_files_with_pattern
281 sxband_sched_rt_context.file_iter_ctx.has_more_files = false;
282
283 if (is_file_cmd(&sxband_sched_rt_context.file_iter_ctx.iter_cmd_entry))
284 {
285 uint8_t pattern_copy_sz = 0U;
286
287 if (sizeof(sxband_sched_rt_context.file_iter_ctx.file_pattern) >= sxband_sched_rt_context.file_iter_ctx.iter_cmd_entry.hdr.size)
288 {
289 pattern_copy_sz = sxband_sched_rt_context.file_iter_ctx.iter_cmd_entry.hdr.size;

291 else
292 {
293 pattern_copy_sz = sizeof(sxband_sched_rt_context.file_iter_ctx.file_pattern);
294 }
295
296 // make a temporary copy of the file iterator pattern from the data field because the data buffer will be
297 // used to store the file names during iteration and the FATFS iterator only keeps a pointer to the pattern
298 (void) memcpy(sxband_sched_rt_context.file_iter_ctx.file_pattern,
299 sxband_sched_rt_context.file_iter_ctx.iter_cmd_entry.data,
300 pattern_copy_sz);
301
302 CRIT_ASSERT(pattern_copy_sz >= 1);
303
304 // add terminating zero to pattern string (just in case)
305 sxband_sched_rt_context.file_iter_ctx.file_pattern[pattern_copy_sz - 1U] = '\0';
306
307 sxband_sched_rt_context.file_iter_ctx.find_iter_res = \
308 SdMngr_f_findfirst(&sxband_sched_rt_context.file_iter_ctx.dir_obj,
309 &sxband_sched_rt_context.file_iter_ctx.fno,
310 "",
311 (const TCHAR *) sxband_sched_rt_context.file_iter_ctx.file_pattern);
312
313 // check if the pattern iterator initialization points to a real file
314 sxband_sched_rt_context.file_iter_ctx.has_more_files = \
315 (FR_OK == sxband_sched_rt_context.file_iter_ctx.find_iter_res) &&
316 ('\0' != sxband_sched_rt_context.file_iter_ctx.fno.fname[0]);
317 }
318
319 return sxband_sched_rt_context.file_iter_ctx.has_more_files;
320
321 // USER_CODE_END::guard_can_iterate_files_with_pattern
322}
