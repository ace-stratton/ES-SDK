98void sxband_sched_start(void)
99{
100 STATIC const osThreadAttr_t task_attr =
101 {
102 .attr_bits = osThreadDetached,
103 .name = "SXBand_Sched_Task",
104 .priority = osPriorityNormal,
105 .cb_mem = NULL,
106 .cb_size = 0U,
107 .stack_mem = NULL,
108 .stack_size = 2500, // TODO: Measure worst-case stack usage!
109 .tz_module = 0U,
110 .reserved = 0U
111 };
112
113 (void) sxband_sched_trigger_hsm(eHSM_StdEvent_Reset);
114
115 // create thread only if not already existing...
116 if (NULL == sxband_sched_rt_context.os_task_hnd)
117 {
118 sxband_sched_rt_context.os_task_hnd = osThreadNew(sxband_sched_task,
119 NULL,
120 &task_attr);
121
122 if (NULL == sxband_sched_rt_context.os_task_hnd)
123 {
124 EXEH_HANDLE(eEXEHSeverity_Fatal,
125 (int32_t) EXEH_SXB_SCHED_CANT_SPAWN_TASK);
126 }
127 else
128 {
129 TaskMonitor_TaskInitialized(TASK_SXBAND_SCHED);
130 }
131 }
132
133 (void) sxband_sched_trigger_hsm(ev_sxband_sched_sm_init);
134}

136sxband_sched_status_t sxband_sched_start_file_seq(const char * const p_fname, const uint8_t fname_size, sxband_sched_notif_t p_sched_notif)
137{
138 sxband_sched_status_t res = SXBAND_SCHED_STATUS_FAILED;
139 const sState_t *p_hsm_active_state = HSM_GetActiveState(sxband_sched_sm_instance);
140
141 CRIT_ASSERT(NULL != p_hsm_active_state);
142
143 sxband_sched_rt_context.file_iter_status = SXB_INIT_STATUS_MAX;
144
145 if (STATE_IDLE == p_hsm_active_state->state_id)
146 {
147 sxband_sched_rt_context.file_iter_status = sxband_cmd_iter_file_init_ctx(p_fname, fname_size, &sxband_sched_rt_context.file_iter_ctx);
148
149 if (SXB_INIT_STATUS_OK == sxband_sched_rt_context.file_iter_status)
150 {
151 sxband_sched_rt_context.p_sched_notif = p_sched_notif;
152 sxband_sched_rt_context.final_sched_status = SXBAND_SCHED_STATUS_OK;
153
154 res = SXBAND_SCHED_STATUS_OK;
155
156 (void) sxband_sched_trigger_hsm(ev_sxband_sched_sm_start);
157 }
158 else
159 {
160 if (NULL != p_sched_notif)
161 {
162 FIL def_fhnd;
163 char err_buf[50U];
164
165 uint8_t char_count = (uint8_t) snprintf(err_buf, sizeof(err_buf), "SXBand command iterator init error: %d\r\n", (uint16_t) sxband_sched_rt_context.file_iter_status);
166
167 // attempt to write an error report because the iterator context could not be initialized for some reason
168 // we don't care about the result of this operation as this is only an emergency measure and we just hope the file
169 // can be written but there is no recovery if it failed for some reason
170 (void) SdMngr_f_open(&def_fhnd, DEFAULT_SUMMARY_FILE_NAME, FA_CREATE_ALWAYS | FA_WRITE);
171 (void) SdMngr_f_write(&def_fhnd, err_buf, char_count, NULL);
172 (void) SdMngr_f_close(&def_fhnd);
173
174 // usually the report file name comes from the commands file as a configuration parameter but we use default file name for reporting here
175 // because the file iterator could not parse the commands file for some reason
176 (*p_sched_notif)(SXBAND_SCHED_STATUS_FAILED, DEFAULT_SUMMARY_FILE_NAME);
177 }
178 }
179 }
180 else
181 {
182 // only one sequence can be executed at a time - no buffering of user requests is performed
183 res = SXBAND_SCHED_STATUS_IN_PROGRESS;
184 }
185
186 return res;
187}

211STATIC void sxband_sched_task(void * p_task_arg)
212{
213 // currently unused
214 (void) p_task_arg;
215
216 for (;;)
217 {
218 osDelay(SXBAND_SCHED_TASK_PERIOD_MS);
219
220 (void) sxband_sched_trigger_hsm(ev_sxband_sched_sm_failure_check);
221
222 (void) sxband_sched_trigger_hsm(eHSM_StdEvent_Periodic);
223
224 // the flag is used here to prevent corrupting the HSM context if
225 // the SXBand driver calls the callback function while the HSM engine
226 // hasn't yet finished with the action function execution
227 if (sxband_sched_rt_context.sxband_result_notified)
228 {
229 (void) sxband_sched_trigger_hsm(ev_sxband_sched_sm_cmd_confirmed);
230 sxband_sched_rt_context.sxband_result_notified = false;
231 }
232
233 TaskMonitor_IamAlive(TASK_SXBAND_SCHED);
234 vApplicationLowStackCheck(TASK_SXBAND_SCHED);
235 }
236}

238STATIC bool inline sxband_sched_trigger_hsm(const uint32_t event_id)
239{
240 bool res = false;
241
242 osMutexAcquire(sxband_sched_lock_id, osWaitForever);
243 res = HSM_TriggerEvent(sxband_sched_sm_instance, event_id);
244 osMutexRelease(sxband_sched_lock_id);
245
246 return res;
247}
