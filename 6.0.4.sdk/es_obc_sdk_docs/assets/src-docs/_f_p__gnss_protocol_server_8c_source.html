89static bool fs_HandleData(ESSA_Stack_DataDispatchInfo_t *fp_DataInfo)
90{
91 ESSA_Stack_FP_MsgHdr_t *pHdr = NULL;
92 bool bIsFuncSupported = false;
93 uint8_t i;
94
95 // Basic header validity check
96 if ((fp_DataInfo == NULL) ||
97 (fp_DataInfo->u16DataSize < sizeof(ESSA_Stack_FP_MsgHdr_t)) ||
98 (fp_DataInfo->pu8Data == NULL))
99 {
100 return false;
101 }
102
103 pHdr = (ESSA_Stack_FP_MsgHdr_t *) fp_DataInfo->pu8Data;
104
105 if (IS_REQUEST(*pHdr))
106 {
107 for (i = 0; i < COUNT_OF(fs_aFuncArray); i++)
108 {
109 if (fs_aFuncArray[i].funcId == pHdr->funcId)
110 {
111 if (fs_aFuncArray[i].pfFunc != NULL)
112 {
113 bIsFuncSupported = true;
114 fs_aFuncArray[i].pfFunc(fp_DataInfo);
115 }
116
117 break;
118 }
119 }
120
121 if (!bIsFuncSupported)
122 (void) ProtocolSendErrorResp(fp_DataInfo, (uint8_t) ESSA_FP_ERRCODE_FUNC_NOT_SUPPORTED);
123 }
124
125 return bIsFuncSupported;
126}

202ESSATMAC_ErrCodes gnss_set_powerResp(
203 RespContext_t* ctx,
204 const GNSS_operation_t eOp_status
205)
206{
207#ifndef BIG_ENDIAN_PLATFORM
208 gnssset_powerProtocolResponseData_t responseParams;
209#else // #ifndef BIG_ENDIAN_PLATFORM
210 uint8_t responseParams[sizeof(gnssset_powerProtocolResponseData_t)] = { 0 };
211#endif // #ifndef BIG_ENDIAN_PLATFORM
212
213 ESSATMAC_ErrCodes sendResult = ESSATMAC_EC_NULL;
214
215 STATIC_ASSERT_SIZE_CHECK(gnssset_powerProtocolResponseData_t);
216
217 if (ctx != NULL)
218 {
219#ifndef BIG_ENDIAN_PLATFORM
220 // fill message header
221 // RS485 wire protocol also uses LE byte order -> serialization of params is not needed
222 responseParams.hdr.protoId = ES_SAT_FUNC_PROTOCOL_ID_GNSS;
223 responseParams.hdr.funcId = GNSS_SET_POWER_FUNCRESP_ID;
224 responseParams.hdr.seqId = ctx->seqId;
225 responseParams.hdr.errCode = ESSA_FP_ERRCODE_NOERROR;
226 SET_RESPONSE(responseParams.hdr);
227
228 // fill message data
229 responseParams.data.eOp_status = eOp_status;
230#else // #ifndef BIG_ENDIAN_PLATFORM
231 // TBD: Serialize request fields
232#endif // #ifndef BIG_ENDIAN_PLATFORM
233
234 sendResult = ESSA_Stack_SendFrameEx((ESSASNetInterface) ctx->nInterfaceNumber,
235 ctx->netType,
236 ctx->nAddr,
237 ES_SAT_MAC_PROTOCOL_ID_FP_LAYER,
238#ifndef BIG_ENDIAN_PLATFORM
239 (uint8_t *) &responseParams,
240#else // #ifndef BIG_ENDIAN_PLATFORM
241 (uint8_t *) responseParams,
242#endif // #ifndef BIG_ENDIAN_PLATFORM
243 sizeof(gnssset_powerProtocolResponseData_t),
244 STACK_SENDFRAME_DEFAULT_PRIO,
245 (ESSATMAC_DrvResult_Cbk_t) NULL,
246 (uint32_t *) NULL);
247 }
248
249 return sendResult;
250}
