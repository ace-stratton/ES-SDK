128void TaskMonitor_Init(void)
129{
130 sDelayedResetData.eDelayedResetCountdownActive = ECOUNTDOWN_INACTIVE;
131 sDelayedResetData.u32MsUntilReset = 0U;
132
133 for(uint8_t i = 0; i < TASK_MONITOR_TASKS_NUMBER; i ++ )
134 {
135 TaskMonitor_RunCounter[i] = 0;
136 TaskMonitor_InitComplete[i] = 0;
137 task_mon_policy_list[i] = TASK_MON_POLICY_MONITOR;
138 Task_UnusedStackSize[i] = -1;
139 }
140
141#if (defined(DEBUG_ENABLED) && defined(DEBUG_MODE_WD_ENABLED)) || !defined(DEBUG_ENABLED)
142 TaskMonitor_TaskHandle = osThreadNew(TaskMonitor_Task, NULL, &TaskMonitor_attributes);
143 CRIT_ASSERT( TaskMonitor_TaskHandle );
144#endif
145
146 uint8_t elem_cnt = sizeof(task_data.ai16TaskStackMaxUnusedSize) / sizeof(task_data.ai16TaskStackMaxUnusedSize[0]);
147
148 for (uint8_t j = 0; j < elem_cnt; j++)
149 {
150 task_data.ai16TaskStackMaxUnusedSize[j] = Task_UnusedStackSize[0U];
151 }
152
153 // calculate the minimum number of tasks to report in data cache...
154 min_task_cnt = (elem_cnt >= TASK_MONITOR_TASKS_NUMBER) ? (TASK_MONITOR_TASKS_NUMBER) : (elem_cnt);
155
156 timer_stop(&wd_periodic_check_timer_hnd);
157 timer_stop(&heap_check_timer);
158
159 EXEH_INIT_MODULE_FILENAME();
160}

162void TaskMonitor_Task(void * argument)
163{
164 static uint8_t task_period_cntr = 0U;
165
166 timer_start(&wd_periodic_check_timer_hnd);
167 timer_start(&heap_check_timer);
168
169 while (true)
170 {
171 osDelay(TASK_MONITOR_REFRESH_WDG_PERIOD);
172 fs_vManageDelayedReset(TASK_MONITOR_REFRESH_WDG_PERIOD);
173
174 if (timer_is_elapsed(&wd_periodic_check_timer_hnd, TASK_MONITOR_TASK_CHECK_PERIOD_MS))
175 {
176 volatile uint8_t u8LastTaskChecked = TaskMonitor_CheckTasks();
177
178 // sanity check (if timer is not actually running, timer_is_elapsed() will return true right away;
179 // this situation should not happen but if it does, we will not be checking task
180 // activity because we may end up with a false positive where some tasks didn't have a
181 // chance to report aliveness yet; therefore we just restart the timer hoping that next
182 // time what caused it to stop, will not happen again
183 if (timer_is_running(wd_periodic_check_timer_hnd))
184 {
185 if (u8LastTaskChecked == TASK_MONITOR_TASKS_NUMBER)
186 {
187 drv_iwdg_refresh();
188
189 // restart timer
190 timer_start(&wd_periodic_check_timer_hnd);
191 }
192 else
193 {
194 const char * const resetTypeDescr = fs_pcGetResetTypeDesc(TASK_MONITOR_TASK_TIMEOUT);
195 (void) resetTypeDescr;
196
197 ES_TRACE_DEBUG("%s (tid: %d)\r\n", resetTypeDescr, u8LastTaskChecked);
198 TaskMonitor_ImmediateReset(TASK_MONITOR_TASK_TIMEOUT);
199 }
200 }
201
202 // restart timer
203 timer_start(&wd_periodic_check_timer_hnd);
204 }
205 else
206 {
207 drv_iwdg_refresh();
208 }
209
210 TaskMonitor_IamAlive(TASK_MONITOR_TASK);
211 vApplicationLowStackCheck(TASK_MONITOR_TASK);
212
213 // simple periodic counter is used because the time value between reports is not
214 // critical; we just want to avoid reporting too often
215 if (task_period_cntr++ == DC_TASK_STATS_REPORT_PERIOD_CNTR)
216 {
217 for (uint8_t j = 0; j < min_task_cnt; j++)
218 {
219 task_data.ai16TaskStackMaxUnusedSize[j] = Task_UnusedStackSize[j];
220 }
221
222 dc_set_os_task_data(&task_data);
223
224 task_period_cntr = 0U;
225 }
226#ifdef DEBUG_ENABLED
227 vApplicationLowTotalHeapCheck();
228#endif
229 }
230}

232void TaskMonitor_IamAlive(TASK_MONITOR_TasksEnum task)
233{
234 if( task < TASK_MONITOR_TASKS_NUMBER )
235 {
236 if( TaskMonitor_RunCounter[task] < 0xFE ) /* Prevent from rolling over */
237 {
238 /* Set the task as alive (run one more time) */
239 TaskMonitor_RunCounter[task] ++;
240 }

261uint8_t TaskMonitor_CheckTasksInit(void)
262{
263 uint8_t i;

265
266 for( i = 0; i < TASK_MONITOR_TASKS_NUMBER; i ++ )
267 {

288void vApplicationLowStackCheck(TASK_MONITOR_TasksEnum taskNumber)
289{
290 // uxTaskGetStackHighWaterMark returns words, we want bytes (1 word = 4 bytes)
291 UBaseType_t uxHighWaterMark_bytes = uxTaskGetStackHighWaterMark( NULL ) * 4;
292
293 if(Task_UnusedStackSize[taskNumber] > (uint16_t)uxHighWaterMark_bytes)
294 {
295 Task_UnusedStackSize[taskNumber] = (uint16_t)uxHighWaterMark_bytes;
296 }
297
298 if(Task_UnusedStackSize[taskNumber] < 2 * TASK_MONITOR_LOW_MEMORY_STACK_PER_TASK)
299 EXEH_HANDLE(eEXEHSeverity_Warning, eEXEH_TASK_MONITOR_TASK_STACK_WARN | ((uint32_t) taskNumber << 24U));
300
301 //Minimum stack size over the needed
302 if(Task_UnusedStackSize[taskNumber] < TASK_MONITOR_LOW_MEMORY_STACK_PER_TASK)
303 {
304 Error_Handler();
305 }
306}

308void vApplicationLowTotalHeapCheck(void)
309{
310 extern uint32_t lnk_freertos_heap_size_bytes; // linker symbol

312
313 if (timer_is_elapsed(&heap_check_timer, HEAP_FREE_TRACE_INTERVAL_MS))

315 ES_TRACE_DEBUG("RTOS heap [free: %lu (%0.2f%%), total: %lu] (bytes)", free_heap, (float) free_heap * 100. / (uint32_t) &lnk_freertos_heap_size_bytes, (uint32_t) &lnk_freertos_heap_size_bytes);

320 CRIT_ASSERT(free_heap != 0);

323void TaskMonitor_TriggerDelayedReset(const uint32_t u32WaitMs)
324{
325 // order is important since this information is used in the TaskMonitor_Task() and
326 // the bDelayedResetCountdownActive flag could be checked before setting the u32MsUntilReset
327 // value causing an immediate reset - no protection is needed here since bDelayedResetCountdownActive
328 // is read atomically
329 sDelayedResetData.u32MsUntilReset = u32WaitMs;
330 sDelayedResetData.eDelayedResetCountdownActive = ECOUNTDOWN_ACTIVE;
331}

344static uint8_t TaskMonitor_CheckTasks(void)
345{
346 uint8_t i;
347 uint8_t u8LastTaskChecked = TASK_MONITOR_TASKS_NUMBER; /* assume all task has passed at least once */
348
349 for( i = 0; i < TASK_MONITOR_TASKS_NUMBER; i ++ )
350 {
351 if(task_mon_policy_list[i] == TASK_MON_POLICY_MONITOR)
352 {
353 if( TaskMonitor_RunCounter[i] == 0 )
354 {
355 // report task which was not alive
356 u8LastTaskChecked = i;
357 break;
358 }else{
359 TaskMonitor_RunCounter[i] = 0;
360 }
361 }
362 }
363
364 return u8LastTaskChecked;
365}

372static void fs_vManageDelayedReset(const uint32_t u32MsSinceLastCall)
373{
374 if (sDelayedResetData.eDelayedResetCountdownActive == ECOUNTDOWN_ACTIVE)
375 {
376 if (sDelayedResetData.u32MsUntilReset >= u32MsSinceLastCall)
377 {
378 sDelayedResetData.u32MsUntilReset -= u32MsSinceLastCall;
379 }
380 else
381 TaskMonitor_ImmediateReset(TASK_MONITOR_USER_DELAYED_RESET);
382 }
383}
