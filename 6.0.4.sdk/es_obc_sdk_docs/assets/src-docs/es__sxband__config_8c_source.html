42 PeriphClkInitStruct.PeriphClockSelection = RCC_PERIPHCLK_USART3;
43 PeriphClkInitStruct.Usart234578ClockSelection = RCC_USART234578CLKSOURCE_D2PCLK1;

53 __HAL_RCC_GPIOB_CLK_ENABLE();
54 __HAL_RCC_GPIOD_CLK_ENABLE();

69 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;

71 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_VERY_HIGH;
72 GPIO_InitStruct.Alternate = GPIO_AF7_USART3;
73 HAL_GPIO_Init(SX_BAND_RX_GPIO_Port, &GPIO_InitStruct);
74
75 /* USART3 DMA Init */
76 /* USART3_RX Init */
77 hdma_usart3_rx.Instance = DMA2_Stream0;
78 hdma_usart3_rx.Init.Request = DMA_REQUEST_USART3_RX;
79 hdma_usart3_rx.Init.Direction = DMA_PERIPH_TO_MEMORY;
80 hdma_usart3_rx.Init.PeriphInc = DMA_PINC_DISABLE;
81 hdma_usart3_rx.Init.MemInc = DMA_MINC_ENABLE;
82 hdma_usart3_rx.Init.PeriphDataAlignment = DMA_PDATAALIGN_BYTE;

112 __HAL_LINKDMA(huart,hdmatx,hdma_usart3_tx);

114 /* USART3 interrupt Init */
115 HAL_NVIC_SetPriority(USART3_IRQn, 5, 0);
116 HAL_NVIC_EnableIRQ(USART3_IRQn);
117}

127 HAL_GPIO_DeInit(SX_BAND_TX_GPIO_Port, SX_BAND_TX_Pin);
128 HAL_GPIO_DeInit(SX_BAND_RX_GPIO_Port, SX_BAND_RX_Pin);

130 /* USART3 DMA DeInit */
