38static char full_filename[FF_LFN_BUF];
39static FIL fhnd_pool[MAX_OPENED_HANDLES];

41static FIL *get_free_file_hnd(void)
42{
43 FIL *p_hnd = NULL;
44
45 for (uint8_t i = 0; i < CDEF_ELEMENT_COUNT(fhnd_pool); i++)
46 {
47 if (NULL == fhnd_pool[i].obj.fs)
48 {
49 p_hnd = &fhnd_pool[i];
50 break;
51 }
52 }
53
54 return p_hnd;
55}

57static void *tftp_open_file(const char* fname, u8_t is_write)
58{
59 FIL *p_fhnd = NULL;
60 FRESULT fres;
61
62 p_fhnd = get_free_file_hnd();
63
64 if (NULL != p_fhnd)
65 {
66 (void) memset(full_filename, 0U, sizeof(full_filename));
67 snprintf(full_filename, sizeof(full_filename), "%s%s", LWIP_TFTP_EXAMPLE_BASE_DIR, fname);
68 full_filename[sizeof(full_filename) - 1] = 0;
69
70 if (is_write)
71 {
72 fres = SdMngr_f_open(p_fhnd, full_filename, FA_WRITE | FA_CREATE_ALWAYS);
73 }
74 else
75 {
76 fres = SdMngr_f_open(p_fhnd, full_filename, FA_READ | FA_OPEN_EXISTING);
77 }
78
79 if (FR_OK != fres)
80 {
81 ES_TRACE_ERROR("could not open file with error: %d\r\n", fres);
82 p_fhnd = NULL;
83 }
84 else
85 {
86 ES_TRACE_INFO("'%s' file successfully opened!\r\n", full_filename);
87 }
88 }
89
90 return p_fhnd;
91}

110static int tftp_read(void* handle, void* buf, int bytes)
111{
112 UINT bytes_read = -1;
113 FRESULT fres;
114
115 fres = SdMngr_f_read((FIL *) handle, buf, bytes, &bytes_read);
116
117 if (FR_OK != fres)
118 {
119 bytes_read = -1;
120 }
121
122 return (int) bytes_read;
123}
124
125static int tftp_write(void* handle, struct pbuf* p)
126{
127 UINT bytes_written = -1;
128 FRESULT fres;
129
130 while (p != NULL)
131 {
132 fres = SdMngr_f_write((FIL *) handle, p->payload, p->len, &bytes_written);
133
134 if (FR_OK != fres)
135 {
136 bytes_written = -1;
137 break;
138 }
139
140 p = p->next;
141 }
142
143 return (int) bytes_written;
144}

148{
149 char message[100];

151 LWIP_UNUSED_ARG(handle);

156 printf("TFTP error: %d (%s)", err, message);

167void tftpd_init_server(void)
168{
169 for (uint8_t i = 0; i < CDEF_ELEMENT_COUNT(fhnd_pool); i++)
170 {
171 fhnd_pool[i].obj.fs = NULL;
172 }
173
174 tftp_init_server(&tftp);

177void tftpd_init_client(void)
178{
179 void *f;
180 err_t err;
181 ip_addr_t srv;
182 int ret = ipaddr_aton(LWIP_TFTP_EXAMPLE_CLIENT_REMOTEIP, &srv);
183 LWIP_ASSERT("ipaddr_aton failed", ret == 1);
184
185 err = tftp_init_client(&tftp);
186 LWIP_ASSERT("tftp_init_client failed", err == ERR_OK);
187
188 f = tftp_open_file(LWIP_TFTP_EXAMPLE_CLIENT_FILENAME, 1);
189 LWIP_ASSERT("failed to create file", f != NULL);
190
191 err = tftp_get(f, &srv, TFTP_PORT, LWIP_TFTP_EXAMPLE_CLIENT_FILENAME, TFTP_MODE_OCTET);
192 LWIP_ASSERT("tftp_get failed", err == ERR_OK);
193}
