45// Calculate the prescaler for a given timer to set it as 1ms per tick (1MHz)
46uint16_t MacCfg_TIM_u16CalculatePrescaler(sTIMTimerEntry_t *fp_psTIM)
47{
48 uint16_t prescaler = 0U;
49 uint32_t clkFreqHz = 0U;
50
51 assert(fp_psTIM != NULL);
52
53 switch (fp_psTIM->eClockSrc)
54 {
55 case eRCCPeripheralClock_APB1:
56 // timer clocks are using 2x the frequency of APB1 domain
57 clkFreqHz = HAL_RCC_GetPCLK1Freq() * 2;
58 break;
59
60 case eRCCPeripheralClock_APB2:
61 // timer clocks are using 2x the frequency of APB2 domain
62 clkFreqHz = HAL_RCC_GetPCLK2Freq() * 2;
63 break;
64
65 default:
66 break;
67 }
68
69 prescaler = (uint16_t) ((uint32_t) (PRESCALER_CFG_VALUE_US * (clkFreqHz / 1000000U) - 1));
70
71 return prescaler;
72}
