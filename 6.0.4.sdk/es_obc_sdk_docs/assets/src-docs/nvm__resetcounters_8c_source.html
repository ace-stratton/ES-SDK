22bool NvmResetCounters_ClearCounter(RstCounterId_t cntrId)
23{
24 bool res = true;
25 boot_struct BootData;
26
27 Nvm_GetBlockById(NVM_BLOCK_BOOTDATA, &BootData);
28
29 switch(cntrId)
30 {
31 case RSTCOUNTERID_WWD:
32 BootData.RST_WWD = 0;
33 break;
34
35 case RSTCOUNTERID_IWD:
36 BootData.RST_IWD = 0;
37 break;
38
39 case RSTCOUNTERID_LPR:
40 BootData.RST_LPR = 0;
41 break;
42
43 case RSTCOUNTERID_POR:
44 BootData.RST_POR = 0;
45 break;
46
47 case RSTCOUNTERID_RSTPIN:
48 BootData.RST_RstPin = 0;
49 break;
50
51 case RSTCOUNTERID_BOR:
52 BootData.RST_BOR = 0;
53 break;
54
55 case RSTCOUNTERID_HARDFAULT:
56 BootData.RST_HardFault = 0;
57 break;
58
59 case RSTCOUNTERID_MEMFAULT:
60 BootData.RST_MemFault = 0;
61 break;
62
63 case RSTCOUNTERID_BUSFAULT:
64 BootData.RST_BusFault = 0;
65 break;
66
67 case RSTCOUNTERID_USAGEFAULT:
68 BootData.RST_UsageFault = 0;
69 break;
70
71 case RSTCOUNTERID_ALL:
72 BootData.RST_WWD = 0;
73 BootData.RST_IWD = 0;
74 BootData.RST_LPR = 0;
75 BootData.RST_POR = 0;
76 BootData.RST_RstPin = 0;
77 BootData.RST_BOR = 0;
78
79 BootData.RST_HardFault = 0; /* HardFault_Handler() */
80 BootData.RST_MemFault = 0; /* MemManage_Handler() */
81 BootData.RST_BusFault = 0; /* BusFault_Handler() */
82 BootData.RST_UsageFault = 0; /* UsageFault_Handler() */
83 break;
84
85 default:
86 res = false;
87 break;
88 }
89
90 Nvm_SetBlockById(NVM_BLOCK_BOOTDATA, &BootData, NVMSETBLOCK_STORE_IMMEDIATELY);
91
92 return res;
93}

101void NvmResetCounters_Update(const bool bClearResetReason)
102{
103 boot_struct BootData;
104
105 Nvm_GetBlockById(NVM_BLOCK_BOOTDATA, &BootData);
106
107 //??? Changed because of H7xx
108
109 if(__HAL_RCC_GET_FLAG(RCC_FLAG_WWDG1RST)) /* WINDOW_WATCHDOG_RESET */
110 BootData.RST_WWD++;
111
112 if(__HAL_RCC_GET_FLAG(RCC_FLAG_IWDG1RST)) /* INDEPENDENT_WATCHDOG_RESET */
113 BootData.RST_IWD++;
114
115 if(__HAL_RCC_GET_FLAG(RCC_FLAG_LPWR1RST)) /* LOW_POWER_RESET */
116 BootData.RST_LPR++;
117
118 if(__HAL_RCC_GET_FLAG(RCC_FLAG_PORRST)) /* POWER-ON_RESET (POR) / POWER-DOWN_RESET (PDR) */
119 BootData.RST_POR++;
120
121 if(__HAL_RCC_GET_FLAG(RCC_FLAG_PINRST)) /* EXTERNAL_RESET_PIN_RESET */
122 BootData.RST_RstPin++;
123
124 if(__HAL_RCC_GET_FLAG(RCC_FLAG_BORRST)) /* BROWNOUT_RESET (BOR) */
125 BootData.RST_BOR++;
126
127 Nvm_SetBlockById(NVM_BLOCK_BOOTDATA, &BootData, NVMSETBLOCK_STORE_IMMEDIATELY);
128
129 if (bClearResetReason)
130 __HAL_RCC_CLEAR_RESET_FLAGS();
131}

133void NvmResetCounters_HardFault(void)
134{
135 boot_struct BootData;
136
137 Nvm_GetBlockById(NVM_BLOCK_BOOTDATA, &BootData);
138
139 if ((uint8_t) SCB->CFSR)
140 BootData.RST_MemFault++;
141 else if ((uint8_t) (SCB->CFSR >> 8))
142 BootData.RST_BusFault++;
143 else if ((uint16_t) (SCB->CFSR >> 16))
144 BootData.RST_UsageFault++;
145 else
146 BootData.RST_HardFault++;
147
148 Nvm_SetBlockById(NVM_BLOCK_BOOTDATA, &BootData, NVMSETBLOCK_STORE_IMMEDIATELY);
149}

151uint32_t NvmResetCounters_GetTotalCount(uint16_t * const rst_reason_bits)
152{
153 boot_struct BootData;
154 uint32_t total_reset_count;
155
156 Nvm_GetBlockById(NVM_BLOCK_BOOTDATA, &BootData);
157
158 total_reset_count = BootData.RST_WWD +
159 BootData.RST_IWD +
160 BootData.RST_LPR +
161 BootData.RST_POR +
162 BootData.RST_RstPin +
163 BootData.RST_BOR +
164 BootData.RST_HardFault +
165 BootData.RST_MemFault +
166 BootData.RST_BusFault +
167 BootData.RST_UsageFault;
168
169 if (NULL != rst_reason_bits)
170 {
171 // LS bit 0: WWD reset occurred at least once
172 // LS bit 1: IWD reset occurred at least once
173 // LS bit 2: LPR reset occurred at least once
174 // LS bit 3: POR reset occurred at least once
175 // LS bit 4: RSTPIN reset occurred at least once
176 // LS bit 5: BOR reset occurred at least once
177 // LS bit 6: HARDFAULT reset occurred at least once
178 // LS bit 7: MEMFAULT reset occurred at least once
179 // LS bit 8: BUSFAULT reset occurred at least once
180 // LS bit 9: USAGEFAULT reset occurred at least once
181
182 *rst_reason_bits =
183 ((BootData.RST_WWD > 0U) ? (RST_REASON_WWD) : (0U)) |
184 ((BootData.RST_IWD > 0U) ? (RST_REASON_IWD) : (0U)) |
185 ((BootData.RST_LPR > 0U) ? (RST_REASON_LPR) : (0U)) |
186 ((BootData.RST_POR > 0U) ? (RST_REASON_POR) : (0U)) |
187 ((BootData.RST_RstPin > 0U) ? (RST_REASON_RSTPIN) : (0U)) |
188 ((BootData.RST_BOR > 0U) ? (RST_REASON_BOR) : (0U)) |
189 ((BootData.RST_HardFault > 0U) ? (RST_REASON_HARDFAULT) : (0U)) |
190 ((BootData.RST_MemFault > 0U) ? (RST_REASON_MEMFAULT) : (0U)) |
191 ((BootData.RST_BusFault > 0U) ? (RST_REASON_BUSFAULT) : (0U)) |
192 ((BootData.RST_UsageFault > 0U) ? (RST_REASON_USAGEFAULT) : (0U));
193 }
194
195 return total_reset_count;
196}
