355void FLASH_IRQHandler(void)
356{
357 __HAL_FLASH_DISABLE_IT(FLASH_IT_SNECCERR_BANK2);
358 __HAL_FLASH_DISABLE_IT(FLASH_IT_DBECCERR_BANK2);
359
360 if (__HAL_FLASH_GET_FLAG(FLASH_FLAG_SNECCERR_BANK2))
361 {
362 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_SNECCERR_BANK2);
363 }
364
365 if (__HAL_FLASH_GET_FLAG(FLASH_FLAG_DBECCERR_BANK2))
366 {
367 __HAL_FLASH_CLEAR_FLAG(FLASH_FLAG_DBECCERR_BANK2);
368 }
369
370 HAL_FLASH_IRQHandler();
371
372 __HAL_FLASH_ENABLE_IT(FLASH_IT_SNECCERR_BANK2);
373 __HAL_FLASH_ENABLE_IT(FLASH_IT_DBECCERR_BANK2);
374}
