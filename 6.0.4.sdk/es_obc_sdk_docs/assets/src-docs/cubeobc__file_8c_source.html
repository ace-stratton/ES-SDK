16 cubeadcs_gen2_file_context_t* h = (cubeadcs_gen2_file_context_t*)userData;

18

22 case CUBEADCS_GEN2_BDT_CTX_FTP_AP:
23 {

75ErrorCode cubeObc_bulkDataTransfer_commitFrameBuffer(void* userData, U8* frame, U16 size)
76{
77 ErrorCode result = CUBEOBC_ERROR_COMMIT;
78 cubeadcs_gen2_file_context_t* h = (cubeadcs_gen2_file_context_t*)userData;
79 UINT bw = 0;
80
81 if ((NULL != frame) &&
82 (0 < size) &&
83 (frame == h->frame_buffer) &&
84 (size == h->last_frame_size))
85 {
86 switch (h->context)
87 {
88 case CUBEADCS_GEN2_BDT_CTX_FTP_BL:
89 case CUBEADCS_GEN2_BDT_CTX_FTP_AP:
90 {
91 h->file_seek_index += size;
92 result = CUBEOBC_ERROR_OK;
93 } break;
94
95 case CUBEADCS_GEN2_BDT_CTX_IMG_CC:
96 case CUBEADCS_GEN2_BDT_CTX_EVT_CC:
97 case CUBEADCS_GEN2_BDT_CTX_TLM_CC:
98 {
99 if (CUBEADCS_GEN2_FILE_SZ_LIMIT > h->file_seek_index)
100 {
101 // Opens the file, if it is existing, in writing mode. If not, a new file will be created.
102 if (FR_OK == SdMngr_f_open(&h->file_handle, h->file_name, FA_OPEN_EXISTING | FA_WRITE))
103 {
104 // set the write pointer to the EOF and attempt to write the data
105 if ((FR_OK == SdMngr_f_lseek(&h->file_handle, h->file_seek_index)) &&
106 (FR_OK == SdMngr_f_write(&h->file_handle, h->frame_buffer, size, &bw)))
107 {
108 if (bw != size)
109 {
110 result = CUBEOBC_ERROR_SIZE;
111 }
112 else
113 {
114 h->file_seek_index += size;
115 result = CUBEOBC_ERROR_OK;
116 }
117 }
118
119 (void)SdMngr_f_close(&h->file_handle);
120 }
121 }
122 else
123 {
124 result = CUBEOBC_ERROR_SIZE;
125 }
126 } break;
127
128 default:
129 {
130 result = CUBEOBC_ERROR_USER_DATA;
131 } break;
132 }
133 }
134
135 return result;
136}
