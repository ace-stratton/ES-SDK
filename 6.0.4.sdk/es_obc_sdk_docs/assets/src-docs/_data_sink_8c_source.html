24static void append_formatted_buf_va(sUserResponseCtx_t * const pUserRespCtx, const char * format, va_list args)
25{
26 if (NULL != pUserRespCtx)
27 {
28 if (NULL != pUserRespCtx->pu8DataBuf)
29 {
30 int chCount;
31 chCount = vsnprintf((char *) &pUserRespCtx->pu8DataBuf[pUserRespCtx->u32ResponseSize],
32 (size_t ) pUserRespCtx->u32MaxDataSize - pUserRespCtx->u32ResponseSize,
33 format,
34 args);
35
36 if ((pUserRespCtx->u32ResponseSize + (uint32_t) chCount) <= pUserRespCtx->u32MaxDataSize)
37 {
38 pUserRespCtx->u32ResponseSize += (uint32_t) chCount;
39 }
40 else
41 {
42 pUserRespCtx->is_all_data_in_buf = false;
43 pUserRespCtx->u32ResponseSize = pUserRespCtx->u32MaxDataSize;
44 }
45 }
46 }
47}

73void append_buf(sUserResponseCtx_t * const pUserRespCtx, const uint8_t *buf, const uint32_t bufSize)
74{
75 if (pUserRespCtx != NULL)
76 {
77 if (NULL != pUserRespCtx->pu8DataBuf)
78 {
79 uint32_t remBufSize = pUserRespCtx->u32MaxDataSize - pUserRespCtx->u32ResponseSize;
80 if (remBufSize >= bufSize)
81 {
82 (void) memcpy((char *) &pUserRespCtx->pu8DataBuf[pUserRespCtx->u32ResponseSize],
83 buf,
84 bufSize);
85
86 pUserRespCtx->u32ResponseSize += bufSize;
87 }
88 else
89 {
90 (void) memcpy((char *) &pUserRespCtx->pu8DataBuf[pUserRespCtx->u32ResponseSize],
91 buf,
92 remBufSize);
93
94 pUserRespCtx->is_all_data_in_buf = false;
95 pUserRespCtx->u32ResponseSize = pUserRespCtx->u32MaxDataSize;
96 }
97 }
98 }
99}

101void commit_buf(sUserResponseCtx_t * const pUserRespCtx, const uint8_t *buf, const uint32_t bufSize)
102{
103 if (pUserRespCtx != NULL)
104 {
105 pUserRespCtx->u32ResponseSize = 0U;
106 append_buf(pUserRespCtx, buf, bufSize);
107 notify_buf_complete(pUserRespCtx);
108 }
109}
