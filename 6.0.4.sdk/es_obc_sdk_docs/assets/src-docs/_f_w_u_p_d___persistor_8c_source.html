57static bool fs_bGetPendingTransferInfo(sTransferInfo_t * const pTransferInfo)
58{
59 FIL tempHandle;
60 FRESULT fres;
61 bool bPendingTransferExists = false;
62 UINT uintBytesRead = (UINT) 0;
63
64 CRIT_ASSERT(pTransferInfo != NULL);
65
66 fres = SdMngr_f_open(&tempHandle, (const TCHAR *) aActiveTidFileName, FA_OPEN_EXISTING | FA_READ);
67
68 if (fres == FR_OK)
69 {
70 fres = SdMngr_f_read(&tempHandle,
71 (void *) pTransferInfo,
72 sizeof(sTransferInfo_t),
73 &uintBytesRead);
74
75 (void) fres;
76 (void) SdMngr_f_close(&tempHandle);
77
78 if (uintBytesRead == sizeof(sTransferInfo_t))
79 {
80 bPendingTransferExists = true;
81 }
82 }
83 else
84 {
85 (void) memset(pTransferInfo, 0U, sizeof(sTransferInfo_t));
86 }
87
88 return bPendingTransferExists;
89}

91static bool fs_u32GetStartingPosForTransfer(const uint64_t u64TransferId, uint32_t * pu32StartPos)
92{
93 FRESULT fres;
94 uint32_t u32StartPos = 0U;
95 sTransferInfo_t tinfo;
96 bool bPendingTransferExists = fs_bGetPendingTransferInfo(&tinfo);
97
98 if (bPendingTransferExists)
99 {
100 if (u64TransferId == tinfo.u64TransferId)
101 {
102 // TID file exists and TID matches the new transfer ID -->
103 // try to obtain the size of the downloaded image to
104 // attempt resuming of the active transfer (probably an OBC restart occurred)
105 fres = SdMngr_f_open(&persistCtx.fHnd, (const TCHAR *) aTempImageFileName, FA_OPEN_EXISTING | FA_READ);
106
107 if (fres == FR_OK)
108 {
109 u32StartPos = SdMngr_f_size(&persistCtx.fHnd);
110 (void) SdMngr_f_close(&persistCtx.fHnd);
111 }
112 }
113 }
114
115 // if the starting pos is more than zero, then the transfer size shall also include the size of the update header
116 // which is not written as part of the downloaded file but GS assumes the start of transfer from the beginning of this header;
117 // providing a value 0 here, means that we also want to receive the header again
118 if (pu32StartPos != NULL)
119 {
120 *pu32StartPos = bPendingTransferExists ? u32StartPos + sizeof(SESSoftwareUPD_Bundle) : u32StartPos;
121 }
122
123 return bPendingTransferExists;
124}

148eFwUpdPersistorResult FWUPD_Persistor_StartNewTransfer(sTransferInfo_t * const pTransferInfo, uint32_t * const pu32StartPos)
149{
150 FRESULT fres = FR_INT_ERR;
151 FIL tempHandle;
152 uint32_t u32BytesWritten = 0U;
153 bool bPendingTransferExists = false;
154
155 fs_vInit(&persistCtx);
156
157 if (pTransferInfo != NULL)
158 {
159 bPendingTransferExists = fs_u32GetStartingPosForTransfer(pTransferInfo->u64TransferId, pu32StartPos);
160
161 if (pu32StartPos != NULL)
162 {
163 if (*pu32StartPos >= sizeof(SESSoftwareUPD_Bundle))
164 {
165 *pu32StartPos -= sizeof(SESSoftwareUPD_Bundle);
166 }
167 }
168
169 if (!bPendingTransferExists)
170 {
171 // create the file which contains the ID of the currently active transfer
172 fres = SdMngr_f_open(&tempHandle, (const TCHAR *) aActiveTidFileName, FA_WRITE | FA_CREATE_ALWAYS);
173 if (fres == FR_OK)
174 {
175 fres = SdMngr_f_write(&tempHandle, (const void *) pTransferInfo, sizeof(sTransferInfo_t), (UINT *) &u32BytesWritten);
176 (void) SdMngr_f_close(&tempHandle);
177
178 if ((fres == FR_OK) && (u32BytesWritten == sizeof(sTransferInfo_t)))
179 {
180 // attempt to create a zero-length file
181 fres = SdMngr_f_open(&persistCtx.fHnd, (const TCHAR *) aTempImageFileName, FA_WRITE | FA_CREATE_ALWAYS);
182 (void) fres;
183 fres = SdMngr_f_close(&persistCtx.fHnd);
184 }
185 else
186 {
187 FWUPD_DBG_PRINT("fres = %d\r\n", fres);
188 fres = FR_INT_ERR;
189 }
190 }
191 else
192 FWUPD_DBG_PRINT("SdMngr_f_open(%d)\r\n", fres);
193 }
194 else
195 {
196 fres = FR_OK;
197 }
198 }
199
200 return (fres == FR_OK) ? EFWUPDPERSISTORRESULT_OK : EFWUPDPERSISTORRESULT_ERROR;
201}

203eFwUpdPersistorResult FWUPD_Persistor_CommitBuffer(const uint32_t u32Pos,
204 const uint8_t * const pu8Buf,
205 const uint32_t u32BufSize,
206 uint32_t * const pu32BytesWritten)
207{
208 FRESULT fres;
209 UINT bytesWritten = 0U;
210 eFwUpdPersistorResult retRes = EFWUPDPERSISTORRESULT_ERROR;
211 uint32_t u32WriteTime;
212 uint32_t u32StartTime;
213
214 u32StartTime = osKernelGetTickCount();
215
216 fres = SdMngr_f_open(&persistCtx.fHnd, (const TCHAR *) aTempImageFileName, FA_WRITE | FA_OPEN_APPEND);
217
218 if (fres == FR_OK)
219 {
220 fres = SdMngr_f_lseek(&persistCtx.fHnd, u32Pos);
221
222 if (fres == FR_OK)
223 {
224 fres = SdMngr_f_write(&persistCtx.fHnd, pu8Buf, u32BufSize, &bytesWritten);
225
226 if (fres == FR_OK)
227 {
228 if (pu32BytesWritten != NULL)
229 {
230 *pu32BytesWritten = bytesWritten;
231 }
232
233 if (fres == FR_OK)
234 {
235 retRes = EFWUPDPERSISTORRESULT_OK;
236 }
237 }
238 else
239 FWUPD_DBG_PRINT("SdMngr_f_write(%d)\r\n", fres);
240
241 }
242 else
243 FWUPD_DBG_PRINT("SdMngr_f_lseek(%d)\r\n", fres);
244 }
245 else
246 FWUPD_DBG_PRINT("SdMngr_f_open(%d)\r\n", fres);
247
248 u32WriteTime = osKernelGetTickCount() - u32StartTime;
249
250 if (persistCtx.u32MaxWriteTime < u32WriteTime)
251 {
252 persistCtx.u32MaxWriteTime = u32WriteTime;
253 }
254
255 (void) SdMngr_f_close(&persistCtx.fHnd);
256
257 return retRes;
258}

260eFwUpdPersistorResult FWUPD_Persistor_FinalizeTransfer(const uint32_t num_bytes_to_strip)
261{
262 FRESULT fres = FR_INT_ERR;
263 FIL fHnd;
264
265 fres = SdMngr_f_open(&fHnd, aTempImageFileName, FA_WRITE | FA_OPEN_APPEND);
266
267 if (fres == FR_OK)
268 {
269 // strip the specified number of bytes at the end of the file
270 if (SdMngr_f_size(&fHnd) >= num_bytes_to_strip)
271 {
272 fres = SdMngr_f_lseek(&fHnd, SdMngr_f_size(&fHnd) - num_bytes_to_strip);
273
274 if (fres == FR_OK)
275 {
276 fres = SdMngr_f_truncate(&fHnd);
277 (void) SdMngr_f_close(&fHnd);
278 }
279 }
280 }
281 else
282 FWUPD_DBG_PRINT("SdMngr_f_open(%d)\r\n", fres);
283
284 return (fres == FR_OK) ? EFWUPDPERSISTORRESULT_OK : EFWUPDPERSISTORRESULT_ERROR;
285}

288{

291 if (pRequestedImageName != NULL)
