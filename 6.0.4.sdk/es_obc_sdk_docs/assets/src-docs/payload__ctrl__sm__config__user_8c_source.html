78bool payload_ctrl_sm_guard_is_point_mode_failed(sState_t* const pSMRoot, const sHSM_EventType_t * const pEvent)
79{
80 // USER_CODE_START::guard_is_point_mode_failed
81 volatile adcs_cntrl_result_t adcs_res = aocs_cntrl_get_status();
82 CONOPS_TRACE_DBG("%s: ----- POINT_MODE_FAIL__FEEDBACK(=2?): %d\r",CONOPS_AOCS_MODE_CHANGE, adcs_res);
83 payload_ctrl_payload_t pl_id = pl_ctrl_runtime_context.point_target_pl_id;
84
85 pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state = PL_STATE_NOT_INIT;
86
87 if (ADCS_CNTRL_RESULT_FAILED == adcs_res)
88 {
89 pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state = PL_STATE_ADCS_ERROR;
90 }
91 else
92 {
93 if (timer_is_elapsed(&pl_ctrl_sm_runtime.adcs_timer, pl_ctrl_runtime_context.point_mode_timeout_ms))
94 {
95 pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state = PL_STATE_ADCS_TIMEOUT;
96 }
97 }
98
99 return (PL_STATE_NOT_INIT != pl_ctrl_runtime_context.state_mon_data[pl_id].last_reported_state);
100 // USER_CODE_END::guard_is_point_mode_failed
101}

112
113 aocs_cntrl_reference_params_t aocs_ref =
114 {

119
120 // trigger CubeADCS pointing...
121 TODO(Status of the AOCS state request should be taken into account)

124
125 // convert time to milliseconds because it comes from the payload scheduler in seconds

128 start_cfg.pointing_cfg.adcs_prep_time;
129
130 timer_start(&pl_ctrl_sm_runtime.adcs_timer);
131 CONOPS_TRACE_DBG("%s: ----- PREP_TIME: %5d\r", CONOPS_AOCS_MODE_CHANGE, pl_ctrl_runtime_context.point_mode_timeout_ms);

184void payload_ctrl_sm_action_serve_additional_payloads(sState_t * const pSMRoot, const sHSM_EventType_t * const pEvent)
185{
186 // USER_CODE_START::action_serve_additional_payloads
187
188 for (uint8_t pl_id = 0; pl_id < PAYLOAD_COUNT; pl_id++)
189 {
190 if (pl_ctrl_runtime_context.state_mon_data[pl_id].pending_flags & PENDING_FLAG_START)
191 {
192 // if pointing mode is compatible to the currently active one, activate the payload, otherwise ignore
193 if (pl_ctrl_runtime_context.state_mon_data[pl_id].start_cfg.pointing_cfg.cntrl_mode ==
194 pl_ctrl_runtime_context.state_mon_data[pl_ctrl_runtime_context.point_target_pl_id].start_cfg.pointing_cfg.cntrl_mode)
195 {
196 const payload_immutable_cfg_t *p_pl_cfg = payload_cfg_get(pl_id);
197
198 CRIT_ASSERT((const payload_immutable_cfg_t *) NULL != p_pl_cfg);
199 CRIT_ASSERT(NULL != p_pl_cfg->p_pl_interface->start);
200
201 pl_config_t start_cfg = {
202 .p_config_data = &pl_ctrl_runtime_context.state_mon_data[pl_id].start_cfg.payload_args,
203 .size = sizeof(pl_ctrl_runtime_context.state_mon_data[pl_id].start_cfg.payload_args)
204 };
205
206 pl_ctrl_runtime_context.state_mon_data[pl_id].pending_flags &= (uint8_t) (~PENDING_FLAG_START);
207
208 (void) p_pl_cfg->p_pl_interface->start(p_pl_cfg->internal_inst_id,
209 &start_cfg);
210
211 break;
212 }
213 else
214 {
215 pl_ctrl_runtime_context.state_mon_data[pl_id].pending_flags &= (uint8_t) (~PENDING_FLAG_START);
216 }
217 }
218 }
