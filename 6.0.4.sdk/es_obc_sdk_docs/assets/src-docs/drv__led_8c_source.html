56drv_led_status_t drv_led_init(void)
57{
58 LED_GREEN_CLK_ENABLE();
59 LED_AMBER_CLK_ENABLE();
60
61 GPIO_InitTypeDef init_struct;
62
63 init_struct.Mode = GPIO_MODE_OUTPUT_PP;
64 init_struct.Pull = GPIO_NOPULL;
65 init_struct.Speed = GPIO_SPEED_FREQ_LOW;
66
67 for (uint8_t id = 0; id < DRV_LED_ID_MAX_COUNT; id++)
68 {
69 init_struct.Pin = fs_led[id].pin;
70 HAL_GPIO_Init(fs_led[id].port, &init_struct);
71 drv_led_set(id, DRV_LED_STATE_OFF);
72 }
73
74 return DRV_LED_STATUS_OK;
75}

90drv_led_status_t drv_led_set(drv_led_id_t id, drv_led_state_t state)
91{
92 if (DRV_LED_ID_MAX_COUNT <= (uint8_t)id)
93 {
94 return DRV_LED_STATUS_ERR_PARAM;
95 }
96
97 if (DRV_LED_STATE_MAX_COUNT <= (uint8_t)state)
98 {
99 return DRV_LED_STATUS_ERR_PARAM;
100 }
101
102 if (DRV_LED_STATE_OFF == state)
103 {
104 HAL_GPIO_WritePin(fs_led[id].port, fs_led[id].pin, GPIO_PIN_RESET);
105 }
106 else
107 {
108 HAL_GPIO_WritePin(fs_led[id].port, fs_led[id].pin, GPIO_PIN_SET);
109 }
110
111 return DRV_LED_STATUS_OK;
112}
