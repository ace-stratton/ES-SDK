107mod_op_status_id_t sp_init(sys_init_level_id_t level)
108{
109 mod_op_status_id_t ret = STATUS_NOT_SUPPORTED;
110
111 // Eventually, to split the initialisation into the different levels
112 if (level == INIT_MEM)
113 {
114 // Read the MAC Addresses from the NVM
115 mac_address_book_t addresses = {0};
116
117 // Read the NVM
118 Nvm_GetBlockById(NVM_MAC_ADDRESS_BOOK, &addresses);
119
120 // Update all the mac addresses
121 CRIT_ASSERT(4 == CDEF_ELEMENT_COUNT(ssp_dc_config));
122 ssp_dc_config[0].mac_addr = addresses.a_mac_address[MAC_ADDR_SP1];
123 ssp_dc_config[1].mac_addr = addresses.a_mac_address[MAC_ADDR_SP2];
124 ssp_dc_config[2].mac_addr = addresses.a_mac_address[MAC_ADDR_SP3];
125 ssp_dc_config[3].mac_addr = addresses.a_mac_address[MAC_ADDR_SP4];
126
127 ret = STATUS_OK;
128 }
129 else if (level == INIT_HW)
130 {
131 ret = STATUS_NOT_SUPPORTED;
132 }
133 else if (level == INIT_OS)
134 {
135 osThreadAttr_t attr =
136 {
137 .attr_bits = osThreadDetached,
138 .name = "sp_service_task",
139 .priority = osPriorityNormal,
140 .cb_mem = NULL,
141 .cb_size = 0U,
142 .stack_mem = NULL,
143 .stack_size = 128 * 16,
144 .tz_module = 0U,
145 .reserved = 0U,
146 };
147
148 if (NULL == h_sp_service_task)
149 {
150 h_sp_service_task = osThreadNew(sp_service_task, NULL, &attr);
151 CRIT_ASSERT(h_sp_service_task);
152 }
153
154 // Set the task monitor policy
155 task_mon_set_task_policy(TASK_MONITOR_SP, TASK_MON_POLICY_MONITOR);
156 ret = STATUS_OK;
157 }
158 else
159 {
160 ret = STATUS_NOT_SUPPORTED;
161 }
162
163 return ret;
164}

202static void sp_service_task(void* argument)
203{
204 (void)argument;
205 int32_t wait_ret;
206 ESSATMAC_ErrCodes reqStatus;
207
208 TaskMonitor_TaskInitialized(TASK_MONITOR_SP);
209
210 while(1)
211 {
212 TaskMonitor_IamAlive(TASK_MONITOR_SP);
213
214 for (uint8_t ssp_index = 0; ssp_index < CDEF_ELEMENT_COUNT(ssp_dc_config); ssp_index++)
215 {
216 for (uint8_t retry = 0; retry < SP_MAX_RETRIES; retry++)
217 {
218 ctx.nAddr = ssp_dc_config[ssp_index].mac_addr;
219 ctx.nInterfaceNumber = ESSASNI_SYS_PRI;
220 ctx.seqId = 0;
221 ctx.netType = ESSASNETT_INTERNAL;
222
223 reqStatus = SP_GetTelemetryReq(&ctx);
224
225 if (reqStatus == ESSATMAC_EC_OK)
226 {
227 // wait for response
228 wait_ret = osThreadFlagsWait(TH_FLAG_RESPONSE, osFlagsWaitAny, SP_RESP_TIMEOUT_MS);
229
230 if (TH_FLAG_RESPONSE == wait_ret)
231 {
232 dc_set_raw_data(DC_DATA_INPUT_INTERNAL,
233 ssp_dc_config[ssp_index].dc_id,
234 (void *) &tlm_ssp,
235 sizeof(tlm_ssp));
236
237 break;
238 }
239 }
240 }
241 }
242
243 osDelay(SP_POLING_INTERVAL_MS);
244
245 vApplicationLowStackCheck(TASK_MONITOR_SP);
246 }
247}
