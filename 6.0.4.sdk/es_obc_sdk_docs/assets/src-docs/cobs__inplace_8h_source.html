80/*
81 * @brief Encode in-place the contents of the provided buffer |buf| of length |len|.
82 * Returns True on successful encoding.
83 *
84 * Because encoding adds leading and trailing bytes, your buffer must reserve
85 * bytes 0 and len-1 for the encoding. If the first and last bytes of |buf|
86 * are not set to COBS_INPLACE_SENTINEL_VALUE, the function will fail with
87 * False.
88 *
89 * If a null pointer or invalid length are provided, the function will fail
90 * with False.
91 *
92 * If |len| is less than or equal to COBS_INPLACE_SAFE_BUFFER_SIZE, the
93 * contents of |buf| will never cause encoding to fail. If |len| is larger
94 * than COBS_INPLACE_SAFE_BUFFER_SIZE, encoding can possibly fail with
95 * False if there are more than 254 bytes between zeros.
96 *
97 * If the function returns False, the contents of |buf| are
98 * left indeterminate and must not be relied on to be fully encoded or decoded.
99 */
100bool cobs_inplace_encode(void* buf, cobs_inplace_sz_t len);

103/*
104 * @brief Decode in-place the contents of the provided buffer |buf| of length |len|.
105 * Returns True on successful decoding.
106 *
107 * Because decoding is in-place, the first and last bytes of |buf| will be set
108 * to the value COBS_INPLACE_SENTINEL_VALUE if decoding succeeds. The decoded

111 * If a null pointer or invalid length are provided, the function will fail
112 * with False.
113 *
