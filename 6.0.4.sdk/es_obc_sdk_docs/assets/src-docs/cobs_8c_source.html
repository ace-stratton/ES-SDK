18{

23

51uint32_t cobs_decode(const uint8_t * input, uint32_t length, uint8_t * output)
52{
53 uint32_t read_index = 0;
54 uint32_t write_index = 0;
55 uint8_t code;
56 uint8_t i;
57
58 while(read_index < length)
59 {
60 code = input[read_index];
61
62 if(read_index + code > length && code != 1)
63 {
64 return 0;
65 }
66
67 read_index++;
68
69 for(i = 1; i < code; i++)
70 {
71 output[write_index++] = input[read_index++];
72 }
73 if(code != 0xFF && read_index != length)
74 {
75 output[write_index++] = '\0';
76 }
77 }
78
79 return write_index;
80}
