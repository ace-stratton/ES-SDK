115uint16_t crc16_ccitt_table_ex(uint8_t const *pDataP, uint32_t iLenP, uint16_t seed)
116{
117 uint16_t crc = seed;
118
119 while( iLenP-- ) {
120#ifdef REFLECTED
121 crc = ccrc16_ccitt_table[(crc ^ *pDataP++) & MASK_BYTE] ^ (crc >> BITS_IN_BYTE);
122#else // REFLECTED
123 crc = ccrc16_ccitt_table[((crc >> BITS_IN_BYTE) ^ *pDataP++) & MASK_BYTE] ^ (crc << BITS_IN_BYTE);
124#endif // REFLECTED
125 }
126
127#ifdef INVERTED
128 crc ^= MASK_WORD;
129#endif // INVERTED
130
131 return crc;
132}

134uint16_t crc16_ccitt_calc(uint8_t const *pDataP, unsigned int iLenP)
135{
136 uint16_t crc = INIT_VALUE;
137 unsigned char i;
138
139 while( iLenP-- ) {
140 i = BITS_IN_BYTE + 1u;
141
142#ifdef REFLECTED
143 uint16_t data = (uint16_t)MASK_BYTE & *pDataP++;
144
145 while( --i )
146 {
147 crc = ( (crc & MASK_BYTE_FIRST) ^ (data & MASK_BYTE_FIRST) ) ?
148 (crc >> 1u) ^ POLYNOMICAL_REFLECTED :
149 crc >> 1u;
150
151 data >>= 1u;
152 }
153#else // REFLECTED
154 crc ^= (*pDataP++) << BITS_IN_BYTE;
155
156 while( --i )
157 {
158 crc = (crc & MASK_WORD_LAST) ?
159 (crc << 1) ^ POLYNOMICAL :
160 (crc << 1);
161 }
162#endif // REFLECTED
163 }
164
165#ifdef INVERTED
166 crc ^= MASK_WORD;
167#endif // INVERTED
168
169 return crc;
170}

172uint16_t CRC16_8(uint8_t *fp_pu8Data, int fp_u16Count)
173{
174 uint16_t crc;
175 char i;
176 crc = INIT_VALUE;
177 while (--fp_u16Count >= 0)
178 {
179 crc = crc ^ (int) *fp_pu8Data++ << 8;
180 i = 8;
181 do
182 {
183 if (crc & 0x8000)
184 crc = crc << 1 ^ POLYNOMICAL;
185 else
186 crc = crc << 1;
187 } while(--i);
188 }
189 return crc;
190}
