75STATIC bool arm_dma_rx(sio_descriptor_t * const p_descr)
76{
77 bool res = false;
78
79 CRIT_ASSERT(NULL != p_descr);
80
81 HAL_StatusTypeDef hal_stat = HAL_ERROR;
82
83 hal_stat = HAL_UART_AbortReceive(p_descr->p_slip_cfg->p_uart);
84
85 if (HAL_OK != hal_stat)
86 {
87 ES_TRACE_ERROR("HAL_UART_AbortReceive() failed with code '%d'!", hal_stat);
88 }
89 else
90 {
91 hal_stat = HAL_UARTEx_ReceiveToIdle_DMA(p_descr->p_slip_cfg->p_uart, (uint8_t *) p_descr->dma_rx_buf, sizeof(p_descr->dma_rx_buf));
92
93 if (HAL_OK != hal_stat)
94 {
95 if (HAL_OK != hal_stat)
96 {
97 ES_TRACE_ERROR("HAL_UARTEx_ReceiveToIdle_DMA() failed with code '%d'!", hal_stat);
98
99 hal_stat = HAL_UART_AbortReceive(p_descr->p_slip_cfg->p_uart);
100
101 if (HAL_OK != hal_stat)
102 {
103 ES_TRACE_ERROR("HAL_UART_AbortReceive() failed with code '%d'!", hal_stat);
104 }
105 else
106 {
107 ES_TRACE_WARN("DMA Rx rearmed...\r\n");
108 }
109 }
110 }
111 else
112 {
113 res = true;
114 }
115 }
116
117 return res;
118}

373sio_fd_t sio_open(u8_t devnum)
374{
375 sio_fd_t dev_fd = NULL;
376
377 CRIT_ASSERT(devnum < CDEF_ELEMENT_COUNT(sio_descriptor));
378
379 if (devnum < CDEF_ELEMENT_COUNT(sio_descriptor))
380 {
381 sio_descriptor[devnum].p_slip_cfg = slipif_uart_cfg_get(devnum);
382 sio_descriptor[devnum].p_slipif_net_cfg = slipif_net_cfg_get(devnum);
383
384 uart_init(devnum);
385
386 (void) memset((void *) sio_descriptor[devnum].dma_rx_buf, 0U, sizeof(sio_descriptor[devnum].dma_rx_buf));
387
388 if (NULL == sio_descriptor[devnum].evt_id)
389 {
390 sio_descriptor[devnum].evt_id = osEventFlagsNew(NULL);
391 }
392
393 if (NULL == sio_descriptor[devnum].com_monitor_task_id)
394 {
395 sio_descriptor[devnum].com_monitor_task_id = osThreadNew(&slip_com_recovery_task,
396 (void *) ((uint32_t) devnum),
397 &slip_com_mon_task_attr);
398 }
399
400 dev_fd = &sio_descriptor[devnum];
401
402 sio_descriptor[devnum].init_finished = true;
403
404 (void) arm_dma_rx(&sio_descriptor[devnum]);
405 }
406
407 return dev_fd;
408}

497u32_t sio_write(sio_fd_t fd, u8_t *data, u32_t len)
498{
499 sio_descriptor_t *p_descr = (sio_descriptor_t *) fd;
500 u32_t bytes_sent = 0;
501
502 if (p_descr->init_finished)
503 {
504 for (uint8_t b_idx = 0; b_idx < len; b_idx++)
505 {
506 HAL_StatusTypeDef hal_stat = HAL_UART_Transmit(p_descr->p_slip_cfg->p_uart, &data[b_idx], (uint16_t) 1, 10);
507
508 if (HAL_OK != hal_stat)
509 {
510 ES_TRACE_ERROR("HAL_UART_Transmit() failed with code '%d'!", hal_stat);
511 break;
512 }
513 else
514 {
515 bytes_sent++;
516 }
517 }
518 }
519
520 return bytes_sent;
521}

549STATIC void slip_com_recovery_task(void *arg)
550{
551 uint8_t devnum = (uint8_t) ((uint32_t) arg);
552
553 CRIT_ASSERT(devnum < CDEF_ELEMENT_COUNT(sio_descriptor));
554
555 while (true)
556 {
557 if (sio_descriptor[devnum].init_finished)
558 {
559 uint32_t evt_res = osEventFlagsWait(sio_descriptor[devnum].evt_id, DMA_RX_ARM_FAILURE_EVENT, osFlagsWaitAny, 5000);
560
561 if ((int32_t) DMA_RX_ARM_FAILURE_EVENT == (int32_t) evt_res)
562 {
563 (void) arm_dma_rx(&sio_descriptor[devnum]);
564 }
565 }
566
567 TaskMonitor_IamAlive(TASK_SLIP);
568 vApplicationLowStackCheck(TASK_SLIP);
569
570 osDelay(10);
571 }
