106void trace(const char* const filename, const int32_t line, trace_level_t lvl, const char* fmt, ...)
107{
108 static const char* lvl_str[TRACE_LVL_COUNT] =
109 { "INFO", "WARN", "ERROR", "FATAL", "DEBUG" };
110
111 if ((false == irq_active()) &&
112 (NULL != filename) &&
113 (TRACE_LVL_COUNT > lvl) &&
114 (NULL != fmt) &&
115 (false != fs_init_done))
116 {
117 // local buffer to format user data
118 uint8_t trc_buf[TRACE_BUF_SZ];
119 uint32_t trc_sz = 0;
120
121 // 1st byte set as required by COBS implementation
122 trc_buf[trc_sz++] = COBS_INPLACE_SENTINEL_VALUE;
123
124 // add header in readable format
125 int ret = sys_snprintf(
126 (char*)&trc_buf[trc_sz], TRACE_HDR_FMT_SZ,
127 TRACE_HDR_FMT,
128 TRACE_HDR_FILENAME_SZ, filename,
129 line,
130 TRACE_HDR_LVL_SZ, lvl_str[lvl],
131 TRACE_HDR_TSTAMP_SZ, datetime_to_unix_now());
132
133 ret = (ret > 0) ? ret : 0;
134 trc_sz += ret;
135
136 // add user data in readable format
137 va_list va;
138 va_start(va, fmt);
139
140 ret = sys_vsnprintf((char*)&trc_buf[trc_sz], TRACE_USER_SZ, fmt, va);
141 ret = (ret > 0) ? ret : 0;
142 trc_sz += (ret > TRACE_USER_SZ) ? TRACE_USER_SZ : ret;
143
144 va_end(va);
145
146 // send data to the active channels without encoding
147 fs_send_data_to_channels(
148 &trc_buf[TRACE_COBS_FIRST_IDX + 1], trc_sz - 1, lvl, false);
149
150 // calculate data crc, format and append it
151 uint16_t trc_crc = crc16_ccitt_calc(
152 &trc_buf[TRACE_COBS_FIRST_IDX + 1], trc_sz - 1);
153
154 ret = sys_snprintf(
155 (char*)&trc_buf[trc_sz], TRACE_FTR_FMT_SZ,
156 TRACE_FTR_FMT, TRACE_FTR_CRC_SZ, trc_crc);
157
158 ret = (ret > 0) ? ret : 0;
159 trc_sz += ret;
160
161 // last byte set as required by COBS implementation
162 trc_buf[trc_sz++] = COBS_INPLACE_SENTINEL_VALUE;
163
164 trace_assert(COBS_INPLACE_SAFE_BUFFER_SIZE >= trc_sz);
165
166 // send data to the active channels with encoding
167 if (cobs_inplace_encode(trc_buf, trc_sz))
168 {
169 fs_send_data_to_channels(trc_buf, trc_sz, lvl, true);
170 }
171 else
172 {
173 trace_assert(false);
174 }
175 }

179void trace_serve_chan(const uint32_t channel, uint8_t* buf, const uint32_t buf_sz)
180{
181 trace_assert(fs_init_done);
182 trace_assert(NULL != buf);
183 trace_assert(fs_chan_list_sz > channel);
184
185 trace_channel_t* chan = &fs_chan_list[channel];
186
187 trace_lock_acquire(chan->lock);
188
189 uint32_t read_sz = cbuf_peek(&chan->cbuf, buf, buf_sz);
190
191 trace_lock_release(chan->lock);
192
193 uint32_t sent_sz = 0;
194
195 if ((0 < read_sz) && (buf_sz >= read_sz))
196 {
197 sent_sz = chan->data_send(buf, read_sz);
198 }
199
200 if (0 < sent_sz)
201 {
202 trace_lock_acquire(chan->lock);
203
204 (void)cbuf_free(&chan->cbuf, sent_sz);
205
206 trace_lock_release(chan->lock);
207 }
208}
