117void Panels_ChipSelect2_Init(void)
118{
119 GPIO_InitTypeDef GPIO_InitStruct = { 0 };
120
121 // Enable GPIO and SPI Clocks
122 __HAL_RCC_GPIOI_CLK_ENABLE();
123 __HAL_RCC_GPIOE_CLK_ENABLE();
124 __HAL_RCC_GPIOG_CLK_ENABLE();
125 __HAL_RCC_GPIOF_CLK_ENABLE();
126
127 PanelStat = 0;
128
129 /*Configure GPIO pin Output Level */
130 // Configure all panel temperature sensors CS as inputs with pull-down
131 GPIO_InitStruct.Pin = PAN5_CS2_Pin;
132 GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
133 GPIO_InitStruct.Pull = GPIO_PULLDOWN;
134 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
135 HAL_GPIO_Init(PAN5_CS2_GPIO_Port, &GPIO_InitStruct);
136
137 GPIO_InitStruct.Pin = PAN1_CS2_Pin | PAN3_CS2_Pin | PAN4_CS2_Pin;
138 GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
139 GPIO_InitStruct.Pull = GPIO_PULLDOWN;
140 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
141 HAL_GPIO_Init(GPIOI, &GPIO_InitStruct);
142
143 GPIO_InitStruct.Pin = PAN6_CS2_Pin;
144 GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
145 GPIO_InitStruct.Pull = GPIO_PULLDOWN;
146 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
147 HAL_GPIO_Init(PAN6_CS2_GPIO_Port, &GPIO_InitStruct);
148
149 GPIO_InitStruct.Pin = PAN2_CS2_Pin;
150 GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
151 GPIO_InitStruct.Pull = GPIO_PULLDOWN;
152 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
153 HAL_GPIO_Init(PAN2_CS2_GPIO_Port, &GPIO_InitStruct);
154
155 sysDelay(2);
156
157 if (HAL_GPIO_ReadPin(PAN1_CS2_GPIO_Port, PAN1_CS2_Pin))
158 PanelStat |= (1 << PAN_X_P);
159 if (HAL_GPIO_ReadPin(PAN2_CS2_GPIO_Port, PAN2_CS2_Pin))
160 PanelStat |= (1 << PAN_Y_P);
161 if (HAL_GPIO_ReadPin(PAN3_CS2_GPIO_Port, PAN3_CS2_Pin))
162 PanelStat |= (1 << PAN_Z_P);
163 if (HAL_GPIO_ReadPin(PAN4_CS2_GPIO_Port, PAN4_CS2_Pin))
164 PanelStat |= (1 << PAN_X_M);
165 if (HAL_GPIO_ReadPin(PAN5_CS2_GPIO_Port, PAN5_CS2_Pin))
166 PanelStat |= (1 << PAN_Y_M);
167 if (HAL_GPIO_ReadPin(PAN6_CS2_GPIO_Port, PAN6_CS2_Pin))
168 PanelStat |= (1 << PAN_Z_M);
169
170 // Configure all panel temperature sensors CS as push-pull outputs
171 HAL_GPIO_WritePin(PAN5_CS2_GPIO_Port, PAN5_CS2_Pin, GPIO_PIN_SET); //Chip selects is OFF
172 HAL_GPIO_WritePin(GPIOI, PAN1_CS2_Pin | PAN3_CS2_Pin | PAN4_CS2_Pin,
173 GPIO_PIN_SET);
174 HAL_GPIO_WritePin(PAN6_CS2_GPIO_Port, PAN6_CS2_Pin, GPIO_PIN_SET);
175 HAL_GPIO_WritePin(PAN2_CS2_GPIO_Port, PAN2_CS2_Pin, GPIO_PIN_SET);
176
177 GPIO_InitStruct.Pin = PAN5_CS2_Pin;
178 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
179 GPIO_InitStruct.Pull = GPIO_PULLUP;
180 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
181 HAL_GPIO_Init(PAN5_CS2_GPIO_Port, &GPIO_InitStruct);
182
183 GPIO_InitStruct.Pin = PAN1_CS2_Pin | PAN3_CS2_Pin | PAN4_CS2_Pin;
184 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
185 GPIO_InitStruct.Pull = GPIO_PULLUP;
186 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
187 HAL_GPIO_Init(GPIOI, &GPIO_InitStruct);
188
189 GPIO_InitStruct.Pin = PAN6_CS2_Pin;
190 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
191 GPIO_InitStruct.Pull = GPIO_PULLUP;
192 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;

256void SPI_SlaveSelect(uint8_t Mode)
257{
258 GPIO_PinState GPIO_pstate;
259
260 if (Mode & CS_OFF) {
261 GPIO_pstate = GPIO_PIN_SET;
262 } else {
263 GPIO_pstate = GPIO_PIN_RESET;
264 }
265
266 switch (Mode & (MAX_PAN + 1)) {
267 case 0:
268 if (Mode & TEMP_CS) {
269 HAL_GPIO_WritePin(PAN1_CS2_GPIO_Port, PAN1_CS2_Pin,
270 GPIO_pstate);
271 } else {
272 HAL_GPIO_WritePin(PAN1_CS1_GPIO_Port, PAN1_CS1_Pin,
273 GPIO_pstate);
274 }
275 break;
276 case 1:
277 if (Mode & TEMP_CS) {
278 HAL_GPIO_WritePin(PAN2_CS2_GPIO_Port, PAN2_CS2_Pin,
279 GPIO_pstate);
280 } else {
281 HAL_GPIO_WritePin(PAN2_CS1_GPIO_Port, PAN2_CS1_Pin,
282 GPIO_pstate);
283 }
284 break;
285 case 2:
286 if (Mode & TEMP_CS) {
287 HAL_GPIO_WritePin(PAN3_CS2_GPIO_Port, PAN3_CS2_Pin,
288 GPIO_pstate);
289 } else {
290 HAL_GPIO_WritePin(PAN3_CS1_GPIO_Port, PAN3_CS1_Pin,
291 GPIO_pstate);
292 }
293 break;
294 case 3:
295 if (Mode & TEMP_CS) {
296 HAL_GPIO_WritePin(PAN4_CS2_GPIO_Port, PAN4_CS2_Pin,
297 GPIO_pstate);
298 } else {
299 HAL_GPIO_WritePin(PAN4_CS1_GPIO_Port, PAN4_CS1_Pin,
300 GPIO_pstate);
301 }
302 break;
303 case 4:
304 if (Mode & TEMP_CS) {
305 HAL_GPIO_WritePin(PAN5_CS2_GPIO_Port, PAN5_CS2_Pin,
306 GPIO_pstate);
307 } else {
308 HAL_GPIO_WritePin(PAN5_CS1_GPIO_Port, PAN5_CS1_Pin,
309 GPIO_pstate);
310 }
311 break;
312 case 5:
313 if (Mode & TEMP_CS) {
314 HAL_GPIO_WritePin(PAN6_CS2_GPIO_Port, PAN6_CS2_Pin,
315 GPIO_pstate);
316 } else {
317 HAL_GPIO_WritePin(PAN6_CS1_GPIO_Port, PAN6_CS1_Pin,
318 GPIO_pstate);
319 }
320 break;
321 default:
322 break;
323 }
324}

380status_t SetMagnetorque(uint8_t Panel, uint8_t perc, uint8_t dir)
381{
382 uint8_t u8PwrSettingPercentage = (perc > 100U) ? (100U) : (perc);
383 uint32_t tmp = (uint32_t) u8PwrSettingPercentage;
384
385 TIM_OC_InitTypeDef sConfigOC;
386
387 if ((Panel < PAN_X_M) || (Panel > PAN_Z_M))
388 return SEN_ERROR;
389 else
390 Panel -= (uint8_t) PAN_X_M; /* PAN_X_M is the base panel *///todo VVS: This depends on the enum value ...not good
391
392 if (dir > 1)
393 dir = 1;
394
395 HAL_TIM_PWM_Stop(&htim5, sPWMCntrl[Panel].TIM_CHANNEL);
396
397 sConfigOC.OCMode = TIM_OCMODE_PWM1;
398 sConfigOC.Pulse = (tmp * (MAGTORQ_PWM_PERIOD + 1) + 50) / 100;
399 sConfigOC.OCPolarity = TIM_OCPOLARITY_HIGH;
400 sConfigOC.OCNPolarity = TIM_OCPOLARITY_HIGH;
401 sConfigOC.OCFastMode = TIM_OCFAST_DISABLE;
402 sConfigOC.OCIdleState = TIM_OCIDLESTATE_RESET;
403 sConfigOC.OCNIdleState = TIM_OCNIDLESTATE_RESET;
404 HAL_TIM_PWM_ConfigChannel(&htim5, &sConfigOC,
405 sPWMCntrl[Panel].TIM_CHANNEL);
406
407 HAL_TIM_PWM_Start(&htim5, sPWMCntrl[Panel].TIM_CHANNEL);
408
409 Panel_Magnetorqer_Power[Panel] = u8PwrSettingPercentage;
410 Panel_Magnetorqer_Direction[Panel] = dir;
411
412 if (dir) {
413 /* Current in the coil flows from PWM_A to PWM_B (check magnetorquer schematic) */
414 HAL_GPIO_WritePin(sPWMCntrl[Panel].INA.port,
415 sPWMCntrl[Panel].INA.pin, GPIO_PIN_SET);
416 } else {
417 /* Current in the coil flows from PWM_B to PWM_A (check magnetorquer schematic) */
418 HAL_GPIO_WritePin(sPWMCntrl[Panel].INA.port,
419 sPWMCntrl[Panel].INA.pin, GPIO_PIN_RESET);
420 }
421
422 return SEN_SUCCESS;
423}

438void Magnetorquers_Update(Magnetorquer_Axis_t MT_level)
439{
440 static uint8_t LevX = 0, LevY = 0, LevZ = 0; /* PWM value in percentage */
441 static uint8_t DirX = 1, DirY = 1, DirZ = 1; /* PWM direction */
442
443 /* Set direction */
444 if (MT_level.AXIS_X >= 0)
445 DirX = 1;
446 else
447 DirX = 0;
448 if (MT_level.AXIS_Y >= 0)
449 DirY = 1;
450 else
451 DirY = 0;
452 if (MT_level.AXIS_Z >= 0)
453 DirZ = 1;
454 else
455 DirZ = 0;
456
457 LevX = (uint8_t) (fabs(MT_level.AXIS_X * MTQ_DIP_MOM_TO_PWM_CONV_MULT));
458 if (LevX > 100)
459 LevX = 100;
460 LevY = (uint8_t) (fabs(MT_level.AXIS_Y * MTQ_DIP_MOM_TO_PWM_CONV_MULT));
461 if (LevY > 100)
462 LevY = 100;
463 LevZ = (uint8_t) (fabs(MT_level.AXIS_Z * MTQ_DIP_MOM_TO_PWM_CONV_MULT));
464 if (LevZ > 100)
465 LevZ = 100;
466
467 SetMagnetorque(PAN_X_M, LevZ, DirZ); /* PAN4 */
468 SetMagnetorque(PAN_Y_M, LevX, DirX); /* PAN5 */
469 SetMagnetorque(PAN_Z_M, LevY, DirY); /* PAN6 */
470
471}

573status_t TMP122_GetTemperatureP(Temperature_t * tmp)
574{
575 uint16_t txdata = 0;
576 HAL_StatusTypeDef res = HAL_OK;
577
578 if (tmp == NULL)
579 return SEN_ERROR;
580
581 tmp->Temp_X = tmp->Temp_Y = tmp->Temp_Z = (int16_t) 0xFFFF;
582
583 SPI_SlaveSelect(TEMP_CS | CS_ON | PAN_X_P);
584 res =
585 HAL_SPI_TransmitReceive(&PANEL_SPI_HANDLER, (uint8_t *) & txdata,
586 (uint8_t *) (&tmp->Temp_X), 1, 10);
587 SPI_SlaveSelect(TEMP_CS | CS_OFF | PAN_X_P);
588 if (res != HAL_OK)
589 return SEN_ERROR;
