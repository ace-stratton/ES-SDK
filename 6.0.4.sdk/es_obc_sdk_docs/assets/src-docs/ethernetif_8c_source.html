70/*
71 @Note: This interface is implemented to operate in zero-copy mode only:
72 - Rx Buffers will be allocated from LwIP stack memory heap,
73 then passed to ETH HAL driver.
74 - Tx Buffers will be allocated from LwIP stack memory heap,
75 then passed to ETH HAL driver.
76
77 @Notes:
78 1.a. ETH DMA Rx descriptors must be contiguous, the default count is 4,
79 to customize it please redefine ETH_RX_DESC_CNT in ETH GUI (Rx Descriptor Length)
80 so that updated value will be generated in stm32xxxx_hal_conf.h
81 1.b. ETH DMA Tx descriptors must be contiguous, the default count is 4,
82 to customize it please redefine ETH_TX_DESC_CNT in ETH GUI (Tx Descriptor Length)
83 so that updated value will be generated in stm32xxxx_hal_conf.h
84
85 2.a. Rx Buffers number: ETH_RX_BUFFER_CNT must be greater than ETH_RX_DESC_CNT.
86 2.b. Rx Buffers must have the same size: ETH_RX_BUFFER_SIZE, this value must
87 passed to ETH DMA in the init field (heth.Init.RxBuffLen)
88 */
89typedef enum
90{
91 RX_ALLOC_OK = 0x00,
92 RX_ALLOC_ERROR = 0x01
93} RxAllocStatusTypeDef;

157static void low_level_init(struct netif *netif)
158{
159 osThreadAttr_t ethernetif_task_attr =
160 {
161 .attr_bits = osThreadDetached,
162 .name = "ethernetif_task",
163 .priority = osPriorityAboveNormal,
164 .cb_mem = NULL,
165 .cb_size = 0U,
166 .stack_mem = NULL,
167 .stack_size = INTERFACE_THREAD_STACK_SIZE
168 };
169
170 int32_t PHYLinkState = 0U;
171 uint8_t macaddress[6] = { ETH_MAC_ADDR0, ETH_MAC_ADDR1, ETH_MAC_ADDR2, ETH_MAC_ADDR3, ETH_MAC_ADDR4, ETH_MAC_ADDR5 };
172
173#ifdef USE_CACHE
174 /* Configure the MPU attributes as Device memory for ETH DMA descriptors */
175 mpu_config();
176
177 /* Enable the CPU Cache */
178 cpu_cache_enable();
179#endif
180
181 EthHandle.Instance = ETH;
182 EthHandle.Init.MACAddr = macaddress;
183 EthHandle.Init.MediaInterface = HAL_ETH_MII_MODE;
184 EthHandle.Init.RxDesc = DMARxDscrTab;
185 EthHandle.Init.TxDesc = DMATxDscrTab;
186 EthHandle.Init.RxBuffLen = ETH_RX_BUFFER_SIZE;
187
188 /* configure ethernet peripheral (GPIOs, clocks, MAC, DMA) */
189 HAL_ETH_Init(&EthHandle);
190
191 /* set MAC hardware address length */
192 netif->hwaddr_len = ETH_HWADDR_LEN;
193
194 /* set MAC hardware address */
195 netif->hwaddr[0] = EthHandle.Init.MACAddr[0];
196 netif->hwaddr[1] = EthHandle.Init.MACAddr[1];
197 netif->hwaddr[2] = EthHandle.Init.MACAddr[2];
198 netif->hwaddr[3] = EthHandle.Init.MACAddr[3];
199 netif->hwaddr[4] = EthHandle.Init.MACAddr[4];
200 netif->hwaddr[5] = EthHandle.Init.MACAddr[5];
201
202 /* maximum transfer unit */
203 netif->mtu = ETH_MAX_PAYLOAD;
204
205 /* device capabilities */
206 /* don't set NETIF_FLAG_ETHARP if this device is not an ethernet one */
207 netif->flags |= NETIF_FLAG_BROADCAST | NETIF_FLAG_ETHARP;
208
209 /* Initialize the RX POOL */
210 LWIP_MEMPOOL_INIT(RX_POOL);
211
212 /* Set Tx packet config common parameters */
213 memset(&TxConfig, 0, sizeof(ETH_TxPacketConfig));
214 TxConfig.Attributes = ETH_TX_PACKETS_FEATURES_CSUM | ETH_TX_PACKETS_FEATURES_CRCPAD;
215 TxConfig.ChecksumCtrl = ETH_CHECKSUM_IPHDR_PAYLOAD_INSERT_PHDR_CALC;
216 TxConfig.CRCPadCtrl = ETH_CRC_PAD_INSERT;
217
218 /* create a binary semaphore used for informing ethernetif of frame reception */
219 RxPktSemaphore = osSemaphoreNew(1, 1, NULL);
220
221 /* create a binary semaphore used for informing ethernetif of frame transmission */
222 TxPktSemaphore = osSemaphoreNew(1, 1, NULL);
223
224 /* create the task that handles the ETH_MAC */
225 if (NULL == os_task_hnd)
226 {
227 os_task_hnd = osThreadNew(ethernetif_input,
228 (void *) netif,
229 &ethernetif_task_attr);
230 }
231
232 PHYLinkState = TJA1100_GetLinkState(&PHY_OBJ);
233
234 /* Get link state */
235 if (PHYLinkState <= PHY_STATUS_LINK_DOWN)
236 {
237 netif_set_link_down(netif);
238 netif_set_down(netif);
239 }
240 else
241 {
242 HAL_ETH_Start_IT(&EthHandle);
243 netif_set_up(netif);
244 netif_set_link_up(netif);
245 }
246}

260static err_t low_level_output(struct netif *netif, struct pbuf *p)
261{
262 uint32_t i = 0U;
263 err_t errval = ERR_OK;
264 ETH_BufferTypeDef Txbuffer[ETH_TX_DESC_CNT] = {0};
265
266 (void) netif;
267
268 memset(Txbuffer, 0, ETH_TX_DESC_CNT * sizeof(ETH_BufferTypeDef));
269
270 for (struct pbuf *q = p; q != NULL; q = q->next)
271 {
272 if (i >= ETH_TX_DESC_CNT)
273 {
274 return ERR_IF;
275 }
276
277 Txbuffer[i].buffer = q->payload;
278 Txbuffer[i].len = q->len;
279
280 if (i > 0)
281 {
282 Txbuffer[i - 1].next = &Txbuffer[i];
283 }
284
285 if (q->next == NULL)
286 {
287 Txbuffer[i].next = NULL;
288 }
289
290 i++;
291 }
292
293 TxConfig.Length = (NULL != p) ? p->tot_len : 0;
294 TxConfig.TxBuffer = Txbuffer;
295 TxConfig.pData = p;
296
297 pbuf_ref(p);
298
299 HAL_ETH_Transmit_IT(&EthHandle, &TxConfig);
300
301 osStatus_t status = osSemaphoreAcquire(TxPktSemaphore, TIME_WAITING_FOR_TX);
302
303 switch (status)
304 {
305 case osOK:
306 {
307 errval = ERR_OK;
308 break;
309 }
310
311 case osErrorTimeout:
312 {
313 errval = ERR_TIMEOUT;
314 break;
315 }
316
317 case osErrorNoMemory:
318 {
319 errval = ERR_MEM;
320 break;
321 }
322
323 default:
324 {
325 errval = ERR_IF;
326 break;
327 }
328 }
329
330 HAL_ETH_ReleaseTxPacket(&EthHandle);
331
332 return errval;
333}

343static struct pbuf* low_level_input(struct netif *netif)
344{
345 struct pbuf *p = NULL;
346
347 (void) netif;
348
349 if (RxAllocStatus == RX_ALLOC_OK)
350 {
351 HAL_StatusTypeDef read_status = HAL_ETH_ReadData(&EthHandle, (void**) &p);
352
353 if (HAL_OK != read_status)
354 {
355 p = NULL;
356 }
357 }
358
359 return p;
360}

368static __attribute__((noreturn)) void ethernetif_input(void *argument)
369{
370 struct pbuf *p = NULL;
371 struct netif *netif = (struct netif*) argument;
372
373 for (;;)
374 {
375 osStatus_t semStatus = osSemaphoreAcquire(RxPktSemaphore, TIME_WAITING_FOR_INPUT);
376
377 if (osOK == semStatus)
378 {
379 do
380 {
381 p = low_level_input(netif);
382 if (p != NULL)
383 {
384 if (netif->input(p, netif) != ERR_OK)
385 {
386 pbuf_free(p);
387 }
388 }
389
390 } while (p != NULL);
391 }
392
393 TaskMonitor_IamAlive(TASK_ETHERNET);
394 vApplicationLowStackCheck(TASK_ETHERNET);
395 }
396}

410err_t ethernetif_init(struct netif *netif)
411{
412 LWIP_ASSERT("netif != NULL", (netif != NULL));
413
414#if LWIP_NETIF_HOSTNAME == 1
415 /* Initialize interface hostname */
416 netif->hostname = "lwip";
417#endif /* LWIP_NETIF_HOSTNAME */
418
419 /*
420 * Initialize the snmp variables and counters inside the struct netif.
421 * The last argument should be replaced with your link speed, in units
422 * of bits per second.
423 */
424 MIB2_INIT_NETIF(netif, snmp_ifType_ethernet_csmacd, LINK_SPEED_OF_YOUR_NETIF_IN_BPS);
425
426 netif->name[0] = IFNAME0;
427 netif->name[1] = IFNAME1;
428
429 /* We directly use etharp_output() here to save a function call.
430 * You can instead declare your own function an call etharp_output()
431 * from it if you have to do some checks before sending (e.g. if link
432 * is available...) */
433 netif->output = etharp_output;
434 netif->linkoutput = low_level_output;
435
436 /* initialize the hardware */
437 low_level_init(netif);
438
439 return ERR_OK;
440}

516void HAL_ETH_MspInit(ETH_HandleTypeDef *ethHandle)
517{
518 GPIO_InitTypeDef GPIO_InitStruct = { 0 };
519
520 if (ethHandle->Instance == ETH)
521 {
522 __HAL_RCC_ETH1MAC_CLK_ENABLE();
523 __HAL_RCC_ETH1TX_CLK_ENABLE();
524 __HAL_RCC_ETH1RX_CLK_ENABLE();
525 __HAL_RCC_GPIOE_CLK_ENABLE();
526 __HAL_RCC_GPIOC_CLK_ENABLE();
527 __HAL_RCC_GPIOA_CLK_ENABLE();
528 __HAL_RCC_GPIOH_CLK_ENABLE();
529 __HAL_RCC_GPIOB_CLK_ENABLE();
530
552// KEEP tja1100 in reset state while configuring pins
553// once configuration is done
554// ETH_PHY_keep_in_reset();
555// keep phy in disabled state while preparing the pins. Schematic keeps the PHY in disabled state but we ensure this as well
556 HAL_ETH_Disable_PHY();
557
558 // Initialize Ethernet Control pins RST and EN
559 GPIO_InitStruct.Pin = ETH_RST_PIN | ETH_EN_PIN;
560 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
561 GPIO_InitStruct.Pull = GPIO_NOPULL;
562 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
563 HAL_GPIO_Init(GPIOH, &GPIO_InitStruct);
564
565 GPIO_InitStruct.Pin = GPIO_PIN_2;
566 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
567 GPIO_InitStruct.Pull = GPIO_NOPULL;
568 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
569 GPIO_InitStruct.Alternate = GPIO_AF11_ETH;
570 HAL_GPIO_Init(GPIOE, &GPIO_InitStruct);
571
572 GPIO_InitStruct.Pin = GPIO_PIN_1 | GPIO_PIN_2 | GPIO_PIN_3 | GPIO_PIN_4
573 | GPIO_PIN_5;
574 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
575 GPIO_InitStruct.Pull = GPIO_NOPULL;
576 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
577 GPIO_InitStruct.Alternate = GPIO_AF11_ETH;
578 HAL_GPIO_Init(GPIOC, &GPIO_InitStruct);
579
580 GPIO_InitStruct.Pin = GPIO_PIN_1 | GPIO_PIN_2 | GPIO_PIN_7;
581 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
582 GPIO_InitStruct.Pull = GPIO_NOPULL;
583 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
584 GPIO_InitStruct.Alternate = GPIO_AF11_ETH;
585 HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);
586
587 GPIO_InitStruct.Pin = GPIO_PIN_6;
588 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
589 GPIO_InitStruct.Pull = GPIO_NOPULL;
590 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
591 GPIO_InitStruct.Alternate = GPIO_AF11_ETH;
592 HAL_GPIO_Init(GPIOH, &GPIO_InitStruct);
593
594 GPIO_InitStruct.Pin = GPIO_PIN_1 | GPIO_PIN_11 | GPIO_PIN_12
595 | GPIO_PIN_13;
596 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
597 GPIO_InitStruct.Pull = GPIO_NOPULL;
598 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
599 GPIO_InitStruct.Alternate = GPIO_AF11_ETH;
600 HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);
601
602 /* Enable the Ethernet global Interrupt */
603 HAL_NVIC_SetPriority(ETH_IRQn, 0x7, 0);
604 HAL_NVIC_EnableIRQ(ETH_IRQn);
605
606 // reset phy and enable it
607 HAL_ETH_Reset_PHY();
608 HAL_ETH_Enable_PHY();
609
610 /* Set PHY IO functions */
611 TJA1100_RegisterBusIO(&PHY_OBJ, &PHY_IOCtx);
612
613 /* Initialize the ETH PHY */
614
615 uint32_t tickstart = HAL_GetTick();
616
617 while(PHY_STATUS_OK != TJA1100_Init(&PHY_OBJ))
618 {
619 if ((HAL_GetTick() - tickstart) > 2000U)
620 {
621 Error_Handler();
622 //should not come here
623 break;
624 }
625 }
626 }

743__attribute__((noreturn)) void ethernet_link_thread(void *arg)
744{
745 int32_t PHYLinkState = 0U;
746 struct netif *netif = (struct netif*) arg;
747
748 for (;;)
749 {
750 PHYLinkState = TJA1100_GetLinkState(&PHY_OBJ);
751
752 if (netif_is_link_up(netif) > 0)
753 {
754 if (PHYLinkState <= PHY_STATUS_LINK_DOWN)
755 {
756 HAL_ETH_Stop_IT(&EthHandle);
757 netif_set_down(netif);
758 netif_set_link_down(netif);
759 }
760 }
761 else
762 {
763 if (PHYLinkState > PHY_STATUS_LINK_DOWN)
764 {
765 HAL_ETH_Start_IT(&EthHandle);
766 netif_set_up(netif);
767 netif_set_link_up(netif);
768 }
769 }
770
771 osDelay(100);
772 }
773}

775void HAL_ETH_RxAllocateCallback(uint8_t **buff)
776{
777 struct pbuf_custom *p = LWIP_MEMPOOL_ALLOC(RX_POOL);
778
779 if (NULL != p)
780 {
781 /* Get the buff from the struct pbuf address. */
782 *buff = (uint8_t*) p + offsetof(RxBuff_t, buff);
783 p->custom_free_function = pbuf_free_custom;
784 /* Initialize the struct pbuf.
785 * This must be performed whenever a buffer's allocated because it may be
786 * changed by lwIP or the app, e.g., pbuf_free decrements ref. */
787 pbuf_alloced_custom(PBUF_RAW, 0, PBUF_REF, p, *buff, ETH_RX_BUFFER_SIZE);
788 }
789 else
790 {
791 RxAllocStatus = RX_ALLOC_ERROR;
792 *buff = NULL;
793 }
794}

796void HAL_ETH_RxLinkCallback(void **pStart, void **pEnd, uint8_t *buff, uint16_t Length)
797{
798 struct pbuf **ppStart = (struct pbuf**) pStart;
799 struct pbuf **ppEnd = (struct pbuf**) pEnd;
800 struct pbuf *p = NULL;
801
802 /* Get the struct pbuf from the buff address. */
803 p = (struct pbuf*) (buff - offsetof(RxBuff_t, buff));
804 p->next = NULL;
805 p->tot_len = 0;
806 p->len = Length;
807
808 /* Chain the buffer. */
809 if (!*ppStart)
810 {
811 /* The first buffer of the packet. */
812 *ppStart = p;
813 }
814 else
815 {
816 /* Chain the buffer to the end of the packet. */
817 (*ppEnd)->next = p;
818 }
819 *ppEnd = p;
820
821 /* Update the total length of all the buffers of the chain. Each pbuf in the chain should have its tot_len
822 * set to its own length, plus the length of all the following pbufs in the chain. */
823 for (p = *ppStart; p != NULL; p = p->next)
824 {
825 p->tot_len += Length;
826 }
827
828#ifdef USE_CACHE
829 /* Invalidate data cache because Rx DMA's writing to physical memory makes it stale. */
830 SCB_InvalidateDCache_by_Addr((uint32_t*) buff, Length);
831#endif
832
833}

854static void mpu_config(void)
855{
856 MPU_Region_InitTypeDef MPU_InitStruct;
857
858 /* Disable the MPU */
859 HAL_MPU_Disable();
860
861 /* Configure the MPU as Strongly ordered for not defined regions */
862 MPU_InitStruct.Enable = MPU_REGION_ENABLE;
863 MPU_InitStruct.BaseAddress = 0x00;
864 MPU_InitStruct.Size = MPU_REGION_SIZE_4GB;
865 MPU_InitStruct.AccessPermission = MPU_REGION_NO_ACCESS;
866 MPU_InitStruct.IsBufferable = MPU_ACCESS_NOT_BUFFERABLE;
867 MPU_InitStruct.IsCacheable = MPU_ACCESS_NOT_CACHEABLE;
868 MPU_InitStruct.IsShareable = MPU_ACCESS_SHAREABLE;
869 MPU_InitStruct.Number = MPU_REGION_NUMBER0;
870 MPU_InitStruct.TypeExtField = MPU_TEX_LEVEL0;
871 MPU_InitStruct.SubRegionDisable = 0x87;
872 MPU_InitStruct.DisableExec = MPU_INSTRUCTION_ACCESS_DISABLE;
873
874 HAL_MPU_ConfigRegion(&MPU_InitStruct);
