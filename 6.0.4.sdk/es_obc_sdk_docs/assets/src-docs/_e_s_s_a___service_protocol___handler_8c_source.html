71// Set initial device configuration
72// This is one time (after JTAG) very basic module configuration
73// Make sure this is saved on several places in memory with recovery mechanism and not overridden by any software update other than JTAG
74// If device loose this configuration it become useless!
75// This configuration is applied with broadcast, before any MAC address can be set.
76// So connect only your device and dongle when set this.
77// Most of all other commands require this to be set to work correctly!
78ESSASP_NetError ESSASP_CB_SetInitConfig(ESSASP_NetSetInitConfig *pInitConfig)
79{
80 ESSASP_NetError res = ESSASPNE_BAD_PARAMS;
81#ifdef COMM_RS485_MAC_LAYER_ENABLED
82 if (pInitConfig != NULL)
83 {
84 sModuleIdPersistentSettings_t modIdSettings;
85 Nvm_GetBlockById(NVM_BLOCK_MODULEID, &modIdSettings);
86
87 modIdSettings.nModuleHWConfig = pInitConfig->nModuleHWConfig;
88 modIdSettings.nModuleHWRevision = pInitConfig->nModuleHWRevision;
89 modIdSettings.nModuleType = pInitConfig->nModuleType;
90 (void) memcpy(modIdSettings.szModuleSerial,
91 pInitConfig->szModuleSerial,
92 ESSASP_DEV_SERIAL_MAX_SIZE);
93
94 Nvm_SetBlockById(NVM_BLOCK_MODULEID, &modIdSettings, NVMSETBLOCK_STORE_IMMEDIATELY);
95
96 res = ESSASPNE_SUCCESS;
97 }
98#endif
99 return res;
100}

102// Restart device in specific mode.
103// If nInMode = ESSASP_OM_UNKNOWN, no mode change, just restart
104// If mode is not supported, restart device in current one
105void ESSASP_CB_RestartDev(ESSASP_OperationalMode nInMode)
106{
107 AppMode_t reqMode;
108
109 switch (nInMode)
110 {
111 case ESSASP_OM_NORMAL:
112 reqMode = APPMODE_APPLICATION;
113 break;
114
115 case ESSASP_OM_FAILSAFE:
116 reqMode = APPMODE_BOOTLOADER;
117 break;
118
119 // only reset without switching modes
120 case ESSASP_OM_UNKNOWN:
121 default:
122#if defined(BOOTLOADER)
123 reqMode = APPMODE_BOOTLOADER;
124#else // #if defined(BOOTLOADER)
125 reqMode = APPMODE_APPLICATION;
126#endif
127 break;
128 }
129
130 WriteHandler_ResetInBootOrAppMode(reqMode);
131}

149// Expected result is send async (no timeout in host) via ESSASP_PrepareShipModeComplete with same params
150// If go async copy/preserve pRC with memcpy till ESSASP_PrepareShipModeComplete call
151void ESSASP_CB_PrepareShipMode(ESSASP_ReplyContext *pRC)
152{
153 ESSASP_NetError psmRes;
154
155 bool rstCntrsRes = NvmResetCounters_ClearCounter(RSTCOUNTERID_ALL);
156
157 // format all SD cards
158 FRESULT fr = SdMngr_RequestPartitionFormat(SD_MNGR_FATFS_SDCARD_PARTITION, true);
159
160 psmRes = ((fr == FR_OK) && (rstCntrsRes)) ? (ESSASPNE_SUCCESS) : (ESSASPNE_GENERAL_ERROR);
161
162 // This can be async operation and it has no timeout on it.
163 // So long prepare operations can be performed
164 ESSASP_PrepareShipModeComplete(pRC, psmRes);
165}

175// Discovery - server
176// Give ability for this node to be discovered dynamically on network
177// Called in response to network request to get our info or ESSASP_SendAnnounce() is called by app layer
178void ESSASP_CB_GetDeviceInfo(ESSASP_NetDiscoveryNodeInfo *pNodeInfo)
179{
180 MODID_VersionInfo_t verInfo;
181
182 if (pNodeInfo != NULL)
183 {
184 module_id_get_info(&verInfo);
185
186 memset(pNodeInfo, 0, sizeof(*pNodeInfo)); // It is a good practice to clear it, in case new fields are added
187
188 strncpy((char *) pNodeInfo->szModuleSerial, (const char *) verInfo.au8HwSerialNumber, (size_t) sizeof(pNodeInfo->szModuleSerial) - 1);
189 pNodeInfo->szModuleSerial[sizeof(pNodeInfo->szModuleSerial) - 1] = 0x0; // just in case
190
191 pNodeInfo->nModuleType = verInfo.u16ModuleType; // or DevName. From ESPLATFORM_COMMON\ESSAT_ModuleTypes.h
192 pNodeInfo->nModuleHWConfig = verInfo.sHwVersion.u16Major; // Device specific, use 0 if not used
193 pNodeInfo->nModuleHWRevision = verInfo.sHwVersion.u16Minor; // Device specific, use 0 if not used
194 pNodeInfo->nOperationalMode = (verInfo.eActiveImageType == MODID_SWIMGTYPE_BOOTLOADER) ? (ESSASP_OM_FAILSAFE) : (ESSASP_OM_NORMAL); // ESSASP_OperationalMode
195 pNodeInfo->nMACAddr = ESSATMAC_GetModuleAddr(); // Current one, may be 0 if not set
196 pNodeInfo->nModuleFWVariant = 0; // or Type. Firmware type if normal mode of operation. Device specific, use 0 if not used
197 pNodeInfo->nFWVersionMaj = verInfo.sFwVersion.u16Major;
198 pNodeInfo->nFWVersionMin = verInfo.sFwVersion.u16Minor;
199 // TODO: Update after Primary/Secondary rules are clarified...
200 pNodeInfo->nFlags = ESSASP_DPF_NONE; // ESSASP_DPF_...
201 }
202}

253// Called in response to network request to get currently executed GS command status. Fill pCmdInfo or return error code
254ESSASP_NetError ESSASP_CB_OnReqGSCmdStatus(ESSASP_NetGS_CP_CmdInfo *pCmdInfo)
255{
256 EESCP_State eState = CESCPS_FREE;
257 uint64_t nCP_CmdID = 0;
258 ESCP_ModDrv_GetCmdStatus(&nCP_CmdID, &eState);
259 pCmdInfo->nCP_CmdID = nCP_CmdID;
260
261 pCmdInfo->nMACAddr = ESSATMAC_GetModuleAddr();
262
263 switch(eState)
264 {
265 case CESCPS_FREE:
266 {
267 if(pCmdInfo->nCP_CmdID)
268 pCmdInfo->nCP_CmdStatus = ESSASP_CCPCS_UPLOADING;
269 else
270 pCmdInfo->nCP_CmdStatus = ESSASP_CCPCS_NONE;
271 break;
272 }
273 case CESCPS_PENDING:
274 {
275 pCmdInfo->nCP_CmdStatus = ESSASP_CCPCS_WAIT_EXEC;
276 break;
277 }
278 case CESCPS_WAIT_SEND_RESULT:
279 {
280 pCmdInfo->nCP_CmdStatus = ESSASP_CCPCS_DOWNLOADING;
281 break;
282 }
283 default:
284 return ESSASPNE_GENERAL_ERROR;
285 }
286 return ESSASPNE_SUCCESS;
287}
