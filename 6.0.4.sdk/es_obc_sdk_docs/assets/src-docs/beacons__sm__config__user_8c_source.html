24static inline uint32_t safe_decrease(const uint32_t num, const uint32_t decr_value)
25{
26 return (num >= decr_value) ? (num - decr_value) : (0U);
27}

242// GEN_OBJ_START::beacons_sm_action_send_beacon
243void beacons_sm_action_send_beacon(sState_t * const pSMRoot, const sHSM_EventType_t * const pEvent)
244{
245 // USER_CODE_START::action_send_beacon
246 uint8_t beacon_slot_idx; // this is used to enumerate the beacon slot indices in the active beacon preset slot
247 uint8_t *p_frame_pos = (uint8_t *) &beacons_rt_context.active_beacon_frame[0U];
248 uint8_t *p_flags_pos = (uint8_t *) &(((beacon_frame_hdr_t *) p_frame_pos)->flags);
249 uint16_t rem_bytes_in_frame = BEACON_FRAME_SIZE_BYTES;
250
251 // initialize beacon buffer...
252 (void) memset(p_frame_pos, INVALID_BEACON_SLOT_ASSIGNMENT, rem_bytes_in_frame);
253
254 // prepare beacon frame header...
255 ((beacon_frame_hdr_t *) p_frame_pos)->rolling_cntr = beacons_rt_context.beacon_frame_rolling_cntr++;
256 ((beacon_frame_hdr_t *) p_frame_pos)->obc_opmode = beacons_get_op_mode();
257 ((beacon_frame_hdr_t *) p_frame_pos)->flags = FRAME_HDR_FLAGS_INIT;
258
259 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, sizeof(beacon_frame_hdr_t));
260 p_frame_pos += sizeof(beacon_frame_hdr_t);
261
262 // start filling-in individual beacon messages from the currently active preset...
263 for (beacon_slot_idx = beacons_rt_context.next_beacon_slot_idx_to_send;
264 beacon_slot_idx < MAX_BEACON_MSG_SLOTS;
265 beacon_slot_idx++)
266 {
267 // obtain the next beacon message type from the presets configuration...
268 uint16_t beacon_type = INVALID_BEACON_SLOT_ASSIGNMENT;
269 dc_did_id dc_data_idx = DC_DID_MAX;
270
271 if ((beacon_slot_idx < MAX_BEACON_MSG_SLOTS) &&
272 (beacons_rt_context.active_preset_id < BEACON_PRESETS_COUNT))
273 {
274 beacon_type = beacons_rt_context.preset_cfg.beacon_slots_setting[beacons_rt_context.active_preset_id][beacon_slot_idx];
275 dc_data_idx = dc_get_index_by_data_id(beacon_type);
276 }
277
278 if ((INVALID_BEACON_SLOT_ASSIGNMENT != beacon_type) &&
279 (DC_DID_MAX != dc_data_idx))
280 {
281 uint32_t bcn_msg_size = dc_get_data_size(dc_data_idx);
282
283 CRIT_ASSERT(bcn_msg_size <= BEACON_MSG_MAX_SIZE_BYTES);
284
285 bool can_package_msg_in_frame = IS_TRUE(beacons_rt_context.gen_cfg.chunk_based_transfer_en) ||
286 (IS_FALSE(beacons_rt_context.gen_cfg.chunk_based_transfer_en) &&
287 (bcn_msg_size <=
288 (sizeof(beacons_rt_context.active_beacon_frame) -
289 sizeof(beacon_frame_hdr_t) -
290 sizeof(beacon_msg_hdr_t))));
291
292 // if message splitting is not configured, then any beacon message shall fit inside the
293 // provided buffer otherwise it will never be able to get sent...
294 CRIT_ASSERT(can_package_msg_in_frame);
295
296 // this check is here to serve the case when CRIT_ASSERT() is empty and will not be
297 // evaluated and the configuration of presets does not allow to package the current frame due
298 // to its bigger size
299 if (!can_package_msg_in_frame)
300 continue;
301
302 // save data status position for later
303 uint8_t * p_beacon_data_status_pos = &(((beacon_msg_hdr_t *) p_frame_pos)->beacon_data_status);
304
305 ((beacon_msg_hdr_t *) p_frame_pos)->beacon_type = beacon_type;
306 ((beacon_msg_hdr_t *) p_frame_pos)->beacon_data_status = 0U;
307
308 beacons_rt_context.next_beacon_slot_idx_to_send = beacon_slot_idx;
309
310 // the full beacon message fits in the current frame?
311 if ((rem_bytes_in_frame >= (sizeof(beacon_msg_hdr_t) + bcn_msg_size)) &&
312 (IS_FALSE(beacons_rt_context.gen_cfg.chunk_based_transfer_en) ||
313 IS_FALSE(beacons_rt_context.partial_msg_info.is_continuation_frame)))
314 {
315 ((beacon_msg_hdr_t *) p_frame_pos)->data_len = bcn_msg_size;
316
317 // jump over individual beacon message header
318 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, sizeof(beacon_msg_hdr_t));
319 p_frame_pos += sizeof(beacon_msg_hdr_t);
320
321 // copy data to beacon buffer...
322 *((uint8_t *) p_beacon_data_status_pos) =
323 (uint8_t) dc_get_raw_data(dc_data_idx, p_frame_pos, rem_bytes_in_frame, 0U, bcn_msg_size);
324
325 // jump over data field to next free position...
326 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, bcn_msg_size);
327 p_frame_pos += bcn_msg_size;
328 }
329 else // the beacon message doesn't fit in the remaining buffer space or this is a continuation from previous frame
330 {
331 // the algorithm will try to split the frame only if the feature is configured and
332 // in case there is not enough space in the remaining buffer to fully accommodate at least
333 // the beacon message header + 1 more byte (headers do not get split - sorry!)
334 if (IS_TRUE(beacons_rt_context.gen_cfg.chunk_based_transfer_en) &&
335 (rem_bytes_in_frame > sizeof(beacon_msg_hdr_t)))
336 {
337 // mark frame as continuation frame
338 *p_flags_pos = (*p_flags_pos) | FRAME_HDR_FLAGS_CONTINUATION;
339
340 if (IS_FALSE(beacons_rt_context.partial_msg_info.is_continuation_frame))
341 {
342 // first part of a continuation frame...
343 SET_TRUE(beacons_rt_context.partial_msg_info.is_continuation_frame);
344
345 // this field indicates the actual remaining number of bytes to be sent across all frames until end of
346 // the message
347 ((beacon_msg_hdr_t *) p_frame_pos)->data_len = bcn_msg_size;
348 /*
349 * | beacon_msg_hdr_t | bytes_until_end_of_frame |<---- rem_msg_bytes_to_send ---->|
350 * |<------------------- rem_bytes_in_frame --------------------->|
351 * |<--------------------------------- bcn_msg_size ---------------------------->|
352 *
353 * The actual number of bytes that will be sent with this frame is:
354 * bytes_until_end_of_frame = rem_bytes_in_frame - sizeof(beacon_msg_hdr_t)
355 *
356 * ...then the actual remaining number of bytes from the message are:
357 * rem_msg_bytes_to_send = bcn_msg_size - bytes_until_end_of_frame
358 */
359
360 uint32_t bytes_until_end_of_frame = rem_bytes_in_frame - sizeof(beacon_msg_hdr_t);
361 beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send = bcn_msg_size - bytes_until_end_of_frame;
362
363 // jump over individual beacon message header
364 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, sizeof(beacon_msg_hdr_t));
365 p_frame_pos += sizeof(beacon_msg_hdr_t);
366
367 // copy data to beacon buffer...
368 *((uint8_t *) p_beacon_data_status_pos) =
369 (uint8_t) dc_get_raw_data(dc_data_idx,
370 p_frame_pos,
371 bytes_until_end_of_frame,
372 0U,
373 bytes_until_end_of_frame);
374
375 break;
376 }
377 else
378 {
379 // subsequent continuation frames...
380 uint32_t bytes_until_end_of_frame = rem_bytes_in_frame - sizeof(beacon_msg_hdr_t);
381
382 ((beacon_msg_hdr_t *) p_frame_pos)->data_len = beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send;
383
384 // jump over individual beacon message header
385 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, sizeof(beacon_msg_hdr_t));
386 p_frame_pos += sizeof(beacon_msg_hdr_t);
387
388 // would message fit this time?
389 if (beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send <= bytes_until_end_of_frame)
390 {
391 // copy data to beacon buffer...
392 *((uint8_t *) p_beacon_data_status_pos) =
393 (uint8_t) dc_get_raw_data(dc_data_idx,
394 p_frame_pos,
395 bytes_until_end_of_frame,
396 bcn_msg_size - beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send,
397 beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send);
398
399 // jump over data field to next free position...
400 rem_bytes_in_frame = safe_decrease(rem_bytes_in_frame, beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send);
401 p_frame_pos += beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send;
402
403 SET_FALSE(beacons_rt_context.partial_msg_info.is_continuation_frame);
404 beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send = 0U;
405 }
406 else // message still doesn't fit, hence package as many bytes as possible and send frame
407 {
408 // copy data to beacon buffer...
409 *((uint8_t *) p_beacon_data_status_pos) =
410 (uint8_t) dc_get_raw_data(dc_data_idx,
411 p_frame_pos,
412 bytes_until_end_of_frame,
413 bcn_msg_size - beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send,
414 rem_bytes_in_frame);
415
416 beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send =
417 safe_decrease(beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send, rem_bytes_in_frame);
418
419 break;
420 }
421 }
422 }
423 else // no message splitting, just leave this message for next frame...
424 {
425 SET_FALSE(beacons_rt_context.partial_msg_info.is_continuation_frame);
426 beacons_rt_context.partial_msg_info.rem_msg_bytes_to_send = 0U;
427
428 // initialize rest of the frame data to invalid values
429 (void) memset(p_frame_pos, INVALID_BEACON_SLOT_ASSIGNMENT, rem_bytes_in_frame);
430
431 break;
432 }
433 }
434 }
435 else
436 {
437 SET_FALSE(beacons_rt_context.more_messages_pending);
438
439 break;
440 }
441 }
442
443 if (MAX_BEACON_MSG_SLOTS == beacon_slot_idx)
444 {
445 SET_FALSE(beacons_rt_context.more_messages_pending);
446 }
447
448 // start sink confirmation timeout tracking...
449 timer_start(&beacons_rt_context.inter_frame_timer);
450
451 SET_FALSE(beacons_rt_context.last_sink_frame_confirmed);
452
453 bool bSendConfirmed = beacons_sink.send(&beacons_rt_context.active_beacon_frame[0U],
454 sizeof(beacons_rt_context.active_beacon_frame),
455 &beacons_rt_context.last_sink_frame_id);
456
457 if (bSendConfirmed)
458 {
459 SET_TRUE(beacons_rt_context.last_sink_frame_confirmed);
460 }
461 else
462 {
463 // start sink confirmation timeout tracking...
464 timer_start(&beacons_rt_context.sink_confirm_timer);
465 }
