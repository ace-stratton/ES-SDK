110void CP_Handler_Task()
111{
112 sCPDispatchContext_t sDispCtx;
113 uint64_t u64CmdId = 0U;
114 pfCPCmdHandler pfCmdHandler = (pfCPCmdHandler) NULL;
115 uint32_t u32ResBufSize = 0U;
116 uint32_t u32CmdType = 0U;
117 uint32_t u32CmdDataBufSize = 0;
118 eCPDispatchResult_t eCmdDispatchRes;
119 uint8_t *pu8CmdDataBuf = (uint8_t *) NULL;
120 uint8_t *pu8ResBuf = (uint8_t *) NULL;
121
122 switch (eCmdState)
123 {
124 case ESMTS_WAIT_CMD_PROCESS :
125 {
126 ESCP_ModDrv_GetPendingCommand(&u64CmdId, &u32CmdType, &pu8CmdDataBuf, &u32CmdDataBufSize);
127
128 // nCmdType actually represents the command ID while nCmdID is a sequence ID which is incremented with
129 // each new command received
130 pfCmdHandler = GetCommandHandler(u32CmdType);
131
132 // try to obtain a response buffer for the command data
133 pu8ResBuf = ESCP_ModDrv_GetResultBuffer(&u32ResBufSize);
134
135 if (pfCmdHandler != NULL)
136 {
137 memset(&sDispCtx, 0, sizeof(sCPDispatchContext_t));
138 sDispCtx.u32CmdType = u32CmdType;
139 sDispCtx.pu8CmdData = pu8CmdDataBuf;
140 sDispCtx.u32CmdDataSize = u32CmdDataBufSize;
141 sDispCtx.pu8ResponseBuf = pu8ResBuf;
142 sDispCtx.u32ResponseBufSize = u32ResBufSize;
143 sDispCtx.u32ResponseActualSize = 0U;
144 sDispCtx.u8TLError = EESTL_SFERR_OUT_OF_MEM;
145 sDispCtx.pChunkRequestCbk = (OnESCP_RequestCommandChunk_t) NULL;
146
147 eCmdDispatchRes = (*pfCmdHandler)(&sDispCtx);
148
149 switch (eCmdDispatchRes)
150 {
151 case ECPDISPATCHRESULT_SIMPLERESULT_OK:
152 ESCP_ModDrv_PrepareSimpleResult(sDispCtx.u8TLError);
153 break;
154
155 case ECPDISPATCHRESULT_RESULT_OK:
156 ESCP_ModDrv_PrepareResult(sDispCtx.u32ResponseActualSize, sDispCtx.u8TLError);
157 break;
158
159 case ECPDISPATCHRESULT_RESULT_BY_CHUNKS:
160 ESCP_ModDrv_PrepareResultOnChunks(sDispCtx.u32ResponseActualSize, sDispCtx.u8TLError, sDispCtx.pChunkRequestCbk);
161 break;
162
163 // It is responsibility of the handler to complete the transfer when the data is present
164 case ECPDISPATCHRESULT_RESULT_POSTPONED:
165 case ECPDISPATCHRESULT_ERROR:
166 default:
167 ESCP_ModDrv_PrepareSimpleResult(EESTL_SFERR_BAD_CMD_PARAMS);
168 break;
169 }
170
171 eCmdState = ESMTS_NONE;
172 }
173 else
174 EXEH_HANDLE(eEXEHSeverity_Warning, EXC_UNSUPPORTED_COMMAND_ID_RCVD);
175
176 break;
177 }
178
179 case ESMTS_NONE:
180 default:
181 break;
182 }
183}

185bool CPCmdHandler_IsDispContextValid(sCPDispatchContext_t * const psDispCtx, const uint32_t u32CmdId)
186{
187 bool res = false;
188
189 if (psDispCtx != NULL)
190 {
191 res = ((psDispCtx->u32CmdType == u32CmdId) &&
192 (psDispCtx->pu8CmdData != NULL) &&
193 (psDispCtx->pu8ResponseBuf != NULL) &&
194 (psDispCtx->u32ResponseBufSize > 0U));
195 }
196
197 return res;
198}
