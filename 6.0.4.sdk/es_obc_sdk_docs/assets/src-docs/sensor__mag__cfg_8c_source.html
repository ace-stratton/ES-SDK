114STATIC sds_op_status_t sensor_mag_prim_get_data_in_uT(sds_mag_data_t * const p_data_dst)
115{
116 CRIT_ASSERT(p_data_dst != NULL);
117
118 uint8_t regValue[6] = {0, 0, 0, 0, 0, 0};
119
120 if ( SEN_SUCCESS != LIS3MDL_MAG_Get_Magnetic(LIS3MDL_MAG_I2C_ADDRESS_LOW, (uint8_t*)regValue ) )
121 {
122 prim_mag_sens_status = SDS_SENSOR_STATUS_ERR;
123
124 prim_mag_err_code = SDS_MTM_ERR_CODE_ERR;
125
126 return SDS_OP_STATUS_FAIL;
127 }
128
129 return sensor_mag_conv_raw_data_to_uT(&regValue[0], p_data_dst);
130}

157STATIC sds_op_status_t sensor_mag_conv_raw_data_to_uT(uint8_t const * const p_raw_data, sds_mag_data_t * const p_data_dst)
158{
159 CRIT_ASSERT(NULL != p_raw_data);
160 CRIT_ASSERT(NULL != p_data_dst);
161
162 int16_t DataRaw[3];
163 uint16_t DataRawTmp[3]; // Necessary to fix MISRA C 2004, 10.5
164 sds_mag_data_t mag_data_uT;
165
166 /* Format the data */
167 DataRawTmp[0] = (uint16_t)p_raw_data[1]; // Necessary to fix MISRA C 2004, 10.5
168 DataRawTmp[1] = (uint16_t)p_raw_data[3];
169 DataRawTmp[2] = (uint16_t)p_raw_data[5];
170
171 DataRaw[0] = (int16_t)( ( (uint16_t)(DataRawTmp[0] << 8) ) + p_raw_data[0] );
172 DataRaw[1] = (int16_t)( ( (uint16_t)(DataRawTmp[1] << 8) ) + p_raw_data[2] );
173 DataRaw[2] = (int16_t)( ( (uint16_t)(DataRawTmp[2] << 8) ) + p_raw_data[4] );
174
175 SDS_TRACE_DBG("%s MTM RAW DATA---X:%d, Y:%d, Z:%d\r", SDS_DBG_HEAD, DataRaw[0], DataRaw[1], DataRaw[2]);
176
177 /* Convert to uT */
178 mag_data_uT.mag_x_uT = (int32_t)( ( DataRaw[0] * LIS3MDL_4GAUSS_LSB ) * 100U );
179 mag_data_uT.mag_y_uT = (int32_t)( ( DataRaw[1] * LIS3MDL_4GAUSS_LSB ) * 100U );
180 mag_data_uT.mag_z_uT = (int32_t)( ( DataRaw[2] * LIS3MDL_4GAUSS_LSB ) * 100U );
181
182 SDS_TRACE_DBG("%s MTM CONV DATA---X:%d, Y:%d, Z:%d\r", SDS_DBG_HEAD, mag_data_uT.mag_x_uT, mag_data_uT.mag_y_uT, mag_data_uT.mag_z_uT);
183
184 /* Copy to destination buffer */
185 *p_data_dst = mag_data_uT;
186
187 return SDS_OP_STATUS_OK;
188}

272STATIC sds_op_status_t sensor_mag_get_register(uint8_t sens_id, sds_axis_enum_t sens_axis, uint32_t reg_address, uint32_t const *reg_value)
273{
274 CRIT_ASSERT(sens_id < SDS_MAG_NUM);
275 CRIT_ASSERT(NULL != reg_value);
276
277 (void)sens_axis;
278
279 sds_op_status_t ret_val = SDS_OP_STATUS_OK;
280
281 uint8_t i2c_addr = (sens_id == SDS_MAG_ENUM_PRIM ? LIS3MDL_MAG_I2C_ADDRESS_LOW : LIS3MDL_MAG_I2C_ADDRESS_HIGH);
282
283 if ( SEN_SUCCESS != LIS3MDL_MAG_ReadReg(i2c_addr, (uint8_t)reg_address, (uint8_t *) reg_value) )
284 {
285 ret_val = SDS_OP_STATUS_FAIL;
286 }
287
288 return ret_val;
289}
