44static void fs_vESSA_SP_Task(void *argument)
45{
46 #define TASK_PERIOD_MS (50)
47 #define WD_KICK_PERIOD (2000)
48 uint8_t u8AliveCntr = 0U;
49
50 while (true)
51 {
52 ESSA_SP_Task();
53
54 osDelay(TASK_PERIOD_MS);
55
56 u8AliveCntr++;
57
58 if ((u8AliveCntr * TASK_PERIOD_MS) >= WD_KICK_PERIOD)
59 {
60 u8AliveCntr = 0U;
61 TaskMonitor_IamAlive(TASK_ESSA_SP_HANDLER);
62 vApplicationLowStackCheck(TASK_ESSA_SP_HANDLER);
63 }
64 }
65}

67void ESSA_Stack_Facade_Init()
68{
69 // Start MAC driver
70 MAC_bStartAllIfs(MACIF_INITTYPE_FULL);
71
72 // Start Dispatcher threads
73 for (uint8_t i = 0; i < eMAC_Interface_Count; i++)
74 DISP_vThreadInit(MacCfg_u8GetStackIfId(i));
75
76 ESSA_SP_Init(true);
77
78 ESSA_StackConfig_Init();
79
80 // Function generated by CMake
81 ESSA_Stack_Services_Init();
82
83 // initialise TP layer
84 ESSA_Stack_TP_Layer_Init();
85
86 for (uint8_t i = 0; i < eMAC_Interface_Count; i++)
87 ESSASP_SendAnnounce(MacCfg_u8GetStackIfId(i));
88
89 is_module_initialized = SAFE_TRUE;
90
91 ESSA_SP_TaskHandle = osThreadNew(fs_vESSA_SP_Task, NULL, &ESSA_SP_Task_attributes);
92 ESSASP_ASSERT(ESSA_SP_TaskHandle);
93
94 TaskMonitor_TaskInitialized(TASK_ESSA_SP_HANDLER);
95}
