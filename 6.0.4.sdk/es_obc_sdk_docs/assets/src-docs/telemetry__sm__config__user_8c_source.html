42static uint32_t prepare_tlm_frame(const uint16_t data_id, uint8_t * const p_data_buf, const uint32_t data_buf_size)
43{
44 uint32_t frame_len = 0U;
45
46 CRIT_ASSERT(NULL != p_data_buf);
47 CRIT_ASSERT(data_buf_size > 0U);
48
49 if (INVALID_TLM_SLOT_ASSIGNMENT != data_id)
50 {
51 dc_did_id dc_item_idx = dc_get_index_by_data_id(data_id);
52
53 if (DC_DID_MAX != dc_item_idx)
54 {
55 uint32_t dc_data_size = dc_get_data_size(dc_item_idx);
56
57 CRIT_ASSERT(dc_data_size <= TLM_BUF_SIZE);
58
59 if (dc_data_size <= TLM_BUF_SIZE)
60 {
61 tlm_frame_hdr_t *p_frame_hdr = (tlm_frame_hdr_t *) p_data_buf;
62 uint8_t *p_frame_data = &p_data_buf[sizeof(tlm_frame_hdr_t)];
63
64 frame_len = sizeof(tlm_frame_hdr_t) + dc_data_size + sizeof(tlm_msg_footer_t);
65
66 CRIT_ASSERT(sizeof(tlm_rt_context.tlm_msg_buffer) >= frame_len);
67
68 p_frame_hdr->timestamp = datetime_to_unix_now();
69 p_frame_hdr->rolling_cntr++;
70 conops_get_hsm_active_state(&p_frame_hdr->obc_opmode);
71
72 dc_data_status_t dc_status =
73 dc_get_raw_data(dc_item_idx,
74 p_frame_data,
75 sizeof(tlm_rt_context.tlm_msg_buffer),
76 0U,
77 dc_data_size);
78
79 p_frame_hdr->msg_hdr.msg_type = data_id;
80 p_frame_hdr->msg_hdr.tlm_data_status = dc_status;
81 CRIT_ASSERT(dc_data_size <= UINT_MAX);
82 p_frame_hdr->msg_hdr.data_len = (uint16_t) dc_data_size;
83
84 p_frame_data += dc_data_size;
85
86 ((tlm_msg_footer_t *) p_frame_data)->crc =
87 crc16_ccitt_table((const uint8_t *) p_frame_hdr, frame_len - sizeof(tlm_msg_footer_t));
88 }
89 }
90 }
91
92 return frame_len;
93}

221// GEN_OBJ_START::telemetry_sm_action_collect_telemetry
222void telemetry_sm_action_collect_telemetry(sState_t * const pSMRoot, const sHSM_EventType_t * const pEvent)
223{
224 // USER_CODE_START::action_collect_telemetry
225
226 telemetry_load_config();
227
228 for (uint8_t tlm_slot_idx = 0U; tlm_slot_idx < TELEMETRY_MAX_CONFIGURABLE_ENTRIES; tlm_slot_idx++)
229 {
230 tlm_preset_cfg_t tlm_entry = tlm_rt_context.preset_cfg.tlm_slots_setting[tlm_rt_context.active_preset_id][tlm_slot_idx];
231
232 tlm_rt_context.tlm_message_acq_counters[tlm_slot_idx]++;
233
234 // the following acquisition period check assumes the telemetry periodic collection happens on 1 sec;
235 // if this is not the case, then the configuration settings will not lead to correct timings...
236 CRIT_ASSERT(1000U == TLM_COLLECTION_PERIOD_MS);
237
238 // has acquisition period elapsed?
239 if ((tlm_entry.active) && (tlm_entry.acq_period > 0U))
240 {
241 if ((tlm_rt_context.tlm_message_acq_counters[tlm_slot_idx] % tlm_entry.acq_period) == 0U)
242 {
243 uint32_t frame_len = prepare_tlm_frame(tlm_entry.data_id, tlm_rt_context.tlm_msg_buffer, sizeof(tlm_rt_context.tlm_msg_buffer));
244
245 if (frame_len > 0U)
246 {
247 uint32_t cobs_frame_len = cobs_encode(tlm_rt_context.tlm_msg_buffer,
248 frame_len,
249 tlm_rt_context.tlm_msg_encode_buffer);
250
251 // the following check ensures there is room for the frame delimiter byte
252 CRIT_ASSERT((cobs_frame_len + 1U) <= sizeof(tlm_rt_context.tlm_msg_encode_buffer));
253
254 tlm_rt_context.tlm_msg_encode_buffer[cobs_frame_len++] = TLM_FRM_DELIMITER;
255
256 bool res = telemetry_queue_push(tlm_rt_context.tlm_msg_encode_buffer, cobs_frame_len);
257
258 (void) res;
259
260 // Note: Even if res is false, which means there was no space in the queue, we will
261 // still continue with other slots because there may be enough space in the queue for them.
262 }
263 else
264 {
265 // Invalid slot found -> this indicates the end of the configured telemetry message sequence.
266 break;
267 }
268
269 // reset acquisition time
270 tlm_rt_context.tlm_message_acq_counters[tlm_slot_idx] = 0U;
271 }
272 }
273 }
