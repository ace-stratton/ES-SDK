63eCPDispatchResult_t CPCmdHandler_SXBand_GwSendCmd(sCPDispatchContext_t * const psDispCtx)
64{
65 const sSXBandGwCmdHdr_t * pCmdHdr = NULL;
66 eThreadBlockEvt_t waitResult = THREADBLOCKEVT_ERROR;
67 static const char data_too_big_hdr[] = " >>";
68 static char data_too_big_msg[sizeof(data_too_big_hdr) + sizeof(SX_BAND_FOUTPUT_NAME) - 1U];
69 static bool is_msg_initialized = false;
70 uint32_t cmd_timeout_ms = SXBAND_RESPONSE_TIMEOUT_MS;
71
72 if (!is_msg_initialized)
73 {
74 (void) snprintf(data_too_big_msg, sizeof(data_too_big_msg), "%s%s", data_too_big_hdr, SX_BAND_FOUTPUT_NAME);
75
76 is_msg_initialized = true;
77 }
78
79 if (sys_conf_is_module_active(SYS_MODULE_ID_SX_BAND))
80 {
81 if (CPCmdHandler_IsDispContextValid(psDispCtx, CP_CMD_SXBAND_GWSEND))
82 {
83 pCmdHdr = (const sSXBandGwCmdHdr_t *) psDispCtx->pu8CmdData;
84 if (psDispCtx->u32CmdDataSize >= sizeof(sSXBandGwCmdHdr_t) + pCmdHdr->u8Size)
85 {
86 userRespCtx.pu8DataBuf = psDispCtx->pu8ResponseBuf;
87 userRespCtx.u32MaxDataSize = psDispCtx->u32ResponseBufSize;
88 userRespCtx.u32ResponseSize = 0U;
89 userRespCtx.onCmdResponseReady = OnCmdResponseReadyCbk;
90 userRespCtx.pUserCtx = (void *) &userRespCtx;
91 // assume that all SXBand response data can fit in the response buffer;
92 // the flag will be set to false in case the response is bigger than the CP output buffer
93 userRespCtx.is_all_data_in_buf = true;
94
95 ESSA_Stack_TP_Layer_PrepareThreadWait();
96
97 S_X_BAND_TRNSM_ESTTC_StartCmd(pCmdHdr->Identifier,
98 pCmdHdr->Cmd,
99 pCmdHdr->Type,
100 pCmdHdr->u8Size,
101 (pCmdHdr->u8Size > 0U) ?
102 &psDispCtx->pu8CmdData[sizeof(sSXBandGwCmdHdr_t)] : NULL,
103 &userRespCtx,
104 &cmd_timeout_ms);
105
106 // block here to wait answer from SXBand... (not the full timeout reported by the SXBand driver is waited here because
107 // this may keep the GS communication thread blocked for too long and other commands would not be executable, hence
108 // there is a limit (the full timeout is accounted for only by the SXBand command scheduler)
109 waitResult = ESSA_Stack_TP_Layer_ThreadWaitForEvent((cmd_timeout_ms > SXBAND_RESPONSE_TIMEOUT_MS) ? SXBAND_RESPONSE_TIMEOUT_MS : cmd_timeout_ms);
110
111 if (waitResult == THREADBlOCKEVT_WAITFINISHED)
112 {
113 if (0U == userRespCtx.u32ResponseSize)
114 {
115 psDispCtx->u32ResponseActualSize = 0U;
116 psDispCtx->u8TLError = EESTL_SFERR_SEND_TIMEOUT;
117 }
118 else
119 {
120 if (userRespCtx.u32ResponseSize > psDispCtx->u32ResponseBufSize)
121 {
122 userRespCtx.u32ResponseSize = psDispCtx->u32ResponseBufSize;
123 }
124
125 psDispCtx->u32ResponseActualSize = userRespCtx.u32ResponseSize;
126 psDispCtx->u8TLError = EESTL_SFERR_SUCCESS;
127
128 // user-data is already written in the response buffer which was passed before the call to
129 // S_X_BAND_TRNSM_ESTTC_StartCmd_Generic;
130 // in case the SXBand response was bigger than what we can accommodate, we provide a hint to the
131 // user at the end of the response buffer that the full output is available in the SXBAND.LOG file
132 // on the OBC SD card
133 if (!userRespCtx.is_all_data_in_buf)
134 {
135 if (userRespCtx.u32ResponseSize >= sizeof(data_too_big_msg))
136 {
137 (void) memcpy((void *) &psDispCtx->pu8ResponseBuf[userRespCtx.u32ResponseSize - sizeof(data_too_big_msg)],
138 data_too_big_msg,
139 sizeof(data_too_big_msg));
140 }
141 else
142 {
143 // directly fail because there is no room in the buffer to even accommodate the message
144 // hinting the user to check the SXBAND.LOG file for the full command output
145 psDispCtx->u32ResponseActualSize = 0U;
146 psDispCtx->u8TLError = EESTL_SFERR_OUT_OF_MEM;
147 }
148 }
149 }
150 }
151 else
152 if (waitResult == THREADBLOCKEVT_WAITTIMEOUT)
153 {
154 psDispCtx->u32ResponseActualSize = 0U;
155 psDispCtx->u8TLError = EESTL_SFERR_RECEIVE_TIMEOUT;
156 }
157 else
158 {
159 (void) memcpy(psDispCtx->pu8ResponseBuf, &waitResult, sizeof(waitResult));
160 psDispCtx->u32ResponseActualSize = sizeof(waitResult);
161 psDispCtx->u8TLError = EESTL_SFERR_UNKNOWN;
162 }
163 }
164 else
165 {
166 psDispCtx->u32ResponseActualSize = 0U;
167 psDispCtx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
168 }
169 }
170 }
171 else
172 {
173 psDispCtx->u32ResponseActualSize = 0U;
174 psDispCtx->u8TLError = EESTL_SFERR_MODULE_NOT_OPERATIONAL;
175 }
176
177 return (0U == psDispCtx->u32ResponseActualSize) ? ECPDISPATCHRESULT_SIMPLERESULT_OK : ECPDISPATCHRESULT_RESULT_OK;
178}

232eCPDispatchResult_t CPCmdHandler_SXBand_Sched_Run(sCPDispatchContext_t * const psDispCtx)
233{
234 const sxband_sched_cp_in_args_t * pCmdHdr = NULL;
235 sxband_sched_cp_out_args_t out_args =
236 {
237 .bytes_written = 0U,
238 .file_op_res = (uint8_t) FR_OK,
239 .sched_status = (uint8_t) SXBAND_SCHED_STATUS_FAILED
240 };
241
242 if (sys_conf_is_module_active(SYS_MODULE_ID_SX_BAND))
243 {
244 if (CPCmdHandler_IsDispContextValid(psDispCtx, CP_CMD_SXBAND_SCHED_RUN))
245 {
246 if (NULL == sxband_cp_lock_id)
247 {
248 sxband_cp_lock_id = osMutexNew(&sxband_cp_lock_attr);
249 }
250
251 pCmdHdr = (const sxband_sched_cp_in_args_t *) psDispCtx->pu8CmdData;
252
253 if (psDispCtx->u32CmdDataSize >= sizeof(sxband_sched_cp_in_args_t))
254 {
255 if (!pCmdHdr->run_from_existing_file)
256 {
257 FIL fhnd;
258 UINT bw;
259
260 out_args.file_op_res = (uint8_t) SdMngr_f_open(&fhnd, (const TCHAR *) pCmdHdr->file_name, FA_WRITE | FA_CREATE_ALWAYS);
261 out_args.bytes_written = 0U;
262
263 if (FR_OK == out_args.file_op_res)
264 {
265 out_args.file_op_res = (uint8_t) SdMngr_f_write(&fhnd, &psDispCtx->pu8CmdData[sizeof(sxband_sched_cp_in_args_t)], psDispCtx->u32CmdDataSize - sizeof(sxband_sched_cp_in_args_t), &bw);
266 out_args.bytes_written = (uint32_t) bw;
267
268 out_args.file_op_res = (uint8_t) SdMngr_f_close(&fhnd);
269 }
270 }
271
272 if (true == pCmdHdr->reset_pwr_before_sched_start)
273 {
274 // perform OBC SXBand driver logical state reset, UART peripheral reset and
275 // power reset of the SXBand module
276 sxband_pl_ctrl_if.stop(0, PL_OP_STOP_MODE_EMERGENCY);
277 sx_band_deinit(INIT_HW);
278 sx_band_init(INIT_MEM);
279 sx_band_init(INIT_HW);
280 sxband_pl_ctrl_if.start(0, NULL);
281 }
282
283 osMutexAcquire(sxband_cp_lock_id, osWaitForever);
284 // initialize poll status before triggering the asynchronous file command sequence
285 sxband_sched_poll_status.sched_status = SXBAND_SCHED_STATUS_IN_PROGRESS;
286 (void) memset(sxband_sched_poll_status.summary_f_name, 0, sizeof(sxband_sched_poll_status.summary_f_name));
287 osMutexRelease(sxband_cp_lock_id);
288
289 out_args.sched_status = (uint8_t) sxband_sched_start_file_seq(pCmdHdr->file_name, (uint8_t) strlen((const char *) pCmdHdr->file_name) + 1, &on_sxband_sched_notif);
290
291 psDispCtx->u32ResponseActualSize = sizeof(out_args);
292 (void) memcpy(psDispCtx->pu8ResponseBuf, &out_args, sizeof(out_args));
293
294 psDispCtx->u8TLError = EESTL_SFERR_SUCCESS;
295 }
296 else
297 {
298 psDispCtx->u32ResponseActualSize = 0U;
299 psDispCtx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
300 }
301 }
302 }
303 else
304 {
305 psDispCtx->u32ResponseActualSize = 0U;
306 psDispCtx->u8TLError = EESTL_SFERR_MODULE_NOT_OPERATIONAL;
307 }
308
309 return (0U == psDispCtx->u32ResponseActualSize) ? ECPDISPATCHRESULT_SIMPLERESULT_OK : ECPDISPATCHRESULT_RESULT_OK;
310}
