82// Initialize the dispatcher.
83void DISP_vThreadInit(eMACInterface_t fp_eInterface)
84{
85 STATIC const osThreadAttr_t sDispAttr =
86 {
87 //TODO: Make joinable, implement MAC interface stop
88 // join the thread at stop.
89 .attr_bits = osThreadDetached,
90 .name = "MAC_DISP",
91 .priority = osPriorityNormal,
92 .cb_mem = NULL,
93 .cb_size = 0U,
94 .stack_mem = NULL,
95 .stack_size = 3000
96 };
97
98 fs_sDispatcherThread_main.psInterfaceState = MacCfg_sGetMacInterfaceById(fp_eInterface);
99
100 if (fs_sDispatcherThread_main.psInterfaceState != NULL)
101 {
102 // create thread only if not already existing...
103 if (fs_sDispatcherThread_main.pvThread == NULL)
104 {
105 fs_sDispatcherThread_main.pvThread = osThreadNew(fs_vDispatchThread, (void*) &fs_sDispatcherThread_main, &sDispAttr);
106
107 if (fs_sDispatcherThread_main.pvThread == NULL)
108 {
109 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_SystemCallFailed);
110 }
111 else
112 TaskMonitor_TaskInitialized(TASK_MACDRV_DISPATCHER);
113 }
114 }
115}

120// Dispathcer thread
121// It should call MAC_DISP_vDispatchMessages() from the esps driver
122STATIC void fs_vDispatchThread (void * fp_pvArg)
123{
124 bool bRunning = true;
125 uint32_t u32Flags;
126 sMACInterfaceState_t* psInterface = ((sDispatcherThread_t*) fp_pvArg)->psInterfaceState;
127
128
129 if (psInterface == NULL)
130 {
131 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_DispatcherBroken);
132 }
133 else
134 {
135 MACDRV_DBG_PRINT("MAC dispatcher thread started (instance: %ld @ 0x%08x)\r\n", (uint32_t) psInterface->macIfId, (uint32_t) fp_pvArg);
136
137 while(bRunning)
138 {
139 // Wait for activation
140 u32Flags = osThreadFlagsWait(MAC_DISPATCH_FLAGS, osFlagsWaitAny, 1000);
141
142 if ((u32Flags & MAC_DISPATCH_ERROR) && (u32Flags != osErrorTimeout))
143 {
144 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_SystemCallFailed);
145 }
146 else
147 {
148 if (((u32Flags != osErrorTimeout) && (u32Flags & MAC_DISPATCH_ACTIVATE_FLAG)) ||
149 (u32Flags == osErrorTimeout))
150 {
151 MACDRV_DBG_PRINT("Trg RxQ %ld [flg: %ld])\r\n", (uint32_t) psInterface->macIfId, (uint32_t) u32Flags);
152
153 MAC_DISP_vDispatchMessages(psInterface);
154 }
155
156 if ((u32Flags != osErrorTimeout) && (u32Flags & MAC_DISPATCH_STOP_FLAG))
157 {
158 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_SystemCallFailed);
159 }
160 }
161 TaskMonitor_IamAlive(TASK_MACDRV_DISPATCHER);
162 vApplicationLowStackCheck(TASK_MACDRV_DISPATCHER);
163 }
164 }
165
166 osThreadExit();
167}
