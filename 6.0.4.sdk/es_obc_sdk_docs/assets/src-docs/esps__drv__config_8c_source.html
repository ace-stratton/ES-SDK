156void MacCfg_vSetBusTerminationState(const eMACInterface_t eMacIf, const bool bNewState, const bool bPersistSetting)
157{
158 GPIO_InitTypeDef GPIO_InitStruct = {0};
159 bool bSettingIsValid = true;
160
161 switch (eMacIf)
162 {
163 case eMAC_Interface_Primary:
164 {
165 // - RS4xx_HS1_TE
166 __HAL_RCC_GPIOD_CLK_ENABLE();
167
168 /* Configure GPIO pin : RS485_TE_Pin */
169 GPIO_InitStruct.Pin = RS485_TE_Pin;
170 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
171 GPIO_InitStruct.Pull = GPIO_NOPULL;
172 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
173 HAL_GPIO_Init(RS485_TE_GPIO_Port, &GPIO_InitStruct);
174
175 HAL_GPIO_WritePin(RS485_TE_GPIO_Port,
176 RS485_TE_Pin,
177 (bNewState) ? (GPIO_PIN_SET) : (GPIO_PIN_RESET));
178
179 break;
180 }
181 default:
182 bSettingIsValid = false;
183 CRIT_ASSERT(false);
184 break;
185 }
186#if defined(COMM_RS485_MAC_LAYER_ENABLED)
187 if ((bSettingIsValid) && (bPersistSetting))
188 {
189 sMacConfig_t macNvmCfg;
190
191 Nvm_GetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg);
192
193 if (bNewState)
194 macNvmCfg.macDrvConfig.u8MacBusTerminationBitField |= (uint8_t) ((0x01U) << (uint8_t) eMacIf);
195 else
196 macNvmCfg.macDrvConfig.u8MacBusTerminationBitField &= (uint8_t) ~((0x01U) << (uint8_t) eMacIf);
197
198 Nvm_SetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg, NVMSETBLOCK_STORE_IMMEDIATELY);
199 }
200#endif
201}

203bool MacCfg_bGetBusTerminationState(const eMACInterface_t eMacIf, const bool bReadPersistentSetting)
204{
205 GPIO_PinState pinState;
206 bool bBusTerminationState = false;
207
208#if defined(COMM_RS485_MAC_LAYER_ENABLED)
209 sMacConfig_t macNvmCfg;
210
211 Nvm_GetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg);
212
213 switch (eMacIf)
214 {
215 case eMAC_Interface_Primary:
216 {
217 if (bReadPersistentSetting)
218 {
219 bBusTerminationState = (bool) \
220 (macNvmCfg.macDrvConfig.u8MacBusTerminationBitField >> ((uint8_t) eMAC_Interface_Primary)) & 0x01U;
221 }
222 else
223 {
224 pinState = HAL_GPIO_ReadPin(RS485_TE_GPIO_Port, RS485_TE_Pin);
225 bBusTerminationState = (pinState == GPIO_PIN_SET) ? (true) : (false);
226 }
227
228 break;
229 }
230 default:
231 CRIT_ASSERT(false);
232 break;
233 }
234#endif
235
236 return bBusTerminationState;
237}

247bool MacCfg_bSetMacAddress(const eMACInterface_t eMacIf, const uint8_t u8MacAddr)
248{
249 bool bResult = false;
250#if defined(COMM_RS485_MAC_LAYER_ENABLED)
251 if ((eMacIf < eMAC_Interface_Count) &&
252 (u8MacAddr != 0x00U) &&
253 (u8MacAddr != ESSATMAC_ADDR_GATEWAY_DEV) &&
254 (u8MacAddr != ESSATMAC_ADDR_BROADCAST))
255 {
256 if (u8MacAddr != MacCfg_u8GetMacAddress(eMacIf))
257 {
258 sMacConfig_t macNvmCfg;
259
260 Nvm_GetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg);
261 macNvmCfg.macDrvConfig.u8MacAddress = u8MacAddr;
262 Nvm_SetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg, NVMSETBLOCK_STORE_IMMEDIATELY);
263 }
264
265 // the new setting will be taken into account after OBC reset
266 bResult = true;
267 }
268#endif
269 return bResult;
270}

272uint8_t MacCfg_u8GetMacAddress(const eMACInterface_t eMacIf)
273{
274#if defined(COMM_RS485_MAC_LAYER_ENABLED)
275 sMacConfig_t macNvmCfg;
276
277 Nvm_GetBlockById(NVM_BLOCK_MACCONFIG, &macNvmCfg);
278
279 // if for some reason, the persisted address is missing or set to one of the
280 // broadcast groups, then use the default MAC SYS1 address
281 if ((macNvmCfg.macDrvConfig.u8MacAddress == 0xFFU) ||
282 (macNvmCfg.macDrvConfig.u8MacAddress == 0x00U) ||
283 (macNvmCfg.macDrvConfig.u8MacAddress == ESSATMAC_ADDR_GATEWAY_DEV) ||
284 (macNvmCfg.macDrvConfig.u8MacAddress == ESSATMAC_ADDR_BROADCAST))
285 return MAC_SYS1_ADDRESS_PRI;
286 else
287 return macNvmCfg.macDrvConfig.u8MacAddress;
288#endif
289}

291static void fs_Uart1Init (struct sMACConfig_t *cfg)
292{
293 assert(cfg != NULL);
294
295 GPIO_InitTypeDef GPIO_InitStruct = {0};
296
297 /* Peripheral clock enable */
298 __HAL_RCC_USART1_CLK_ENABLE();
299 __HAL_RCC_GPIOA_CLK_ENABLE();
300
306 huart1.Instance = USART1;
307 huart1.Init.BaudRate = ((sMACConfig_t *) cfg)->u32Baudrate;
308 huart1.Init.WordLength = UART_WORDLENGTH_9B;
309 huart1.Init.StopBits = UART_STOPBITS_1;
310 huart1.Init.Parity = UART_PARITY_NONE;
311 huart1.Init.Mode = UART_MODE_TX_RX;
312 huart1.Init.HwFlowCtl = UART_HWCONTROL_NONE;
313 huart1.Init.OverSampling = UART_OVERSAMPLING_16;
314 huart1.Init.OneBitSampling = UART_ONE_BIT_SAMPLE_DISABLE;
315 huart1.AdvancedInit.AdvFeatureInit = UART_ADVFEATURE_NO_INIT;
316
317 if (HAL_UART_Init (&huart1) != HAL_OK)
318 {
319 Error_Handler ();
320 }
321
322 GPIO_InitStruct.Pin = RS485_TX_Pin | RS485_RX_Pin;
323 GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
324 GPIO_InitStruct.Pull = GPIO_NOPULL;
325 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_VERY_HIGH;
326 GPIO_InitStruct.Alternate = GPIO_AF7_USART1;
327 HAL_GPIO_Init (GPIOA, &GPIO_InitStruct);
328
329 /*Configure GPIO pin Output Level */
330 HAL_GPIO_WritePin(RS485_DE_GPIO_Port, RS485_DE_Pin, GPIO_PIN_RESET);
331 HAL_GPIO_WritePin(RS485_TE_GPIO_Port, RS485_TE_Pin, GPIO_PIN_RESET);
332 HAL_GPIO_WritePin(RS485_nRE_GPIO_Port, RS485_nRE_Pin, GPIO_PIN_RESET);
333
334 GPIO_InitStruct.Pin = RS485_DE_Pin;
335 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
336 GPIO_InitStruct.Pull = GPIO_NOPULL;
337 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
338 HAL_GPIO_Init (RS485_DE_GPIO_Port, &GPIO_InitStruct);
339
340 /*Configure GPIO pin : RS485_TE_Pin */
341 GPIO_InitStruct.Pin = RS485_TE_Pin;
342 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
343 GPIO_InitStruct.Pull = GPIO_NOPULL;
344 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
345 HAL_GPIO_Init(RS485_TE_GPIO_Port, &GPIO_InitStruct);
346
347 /*Configure GPIO pin : RS485_nRE_Pin */
348 GPIO_InitStruct.Pin = RS485_nRE_Pin;
349 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
350 GPIO_InitStruct.Pull = GPIO_NOPULL;
351 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
352 HAL_GPIO_Init(RS485_nRE_GPIO_Port, &GPIO_InitStruct);
353
354 /* USART1 interrupt Init */
355 HAL_NVIC_SetPriority(USART1_IRQn, MAC_UART_ISR_PRIO, 0);
356 HAL_NVIC_EnableIRQ(USART1_IRQn);
357
358 // Receive Enable ACTIVE
359 fs_vRS485ReceptionEnable();
360
361 // Termination Enable ACTIVE
362 MacCfg_vSetBusTerminationState(eMAC_Interface_Primary, MacCfg_bGetBusTerminationState(eMAC_Interface_Primary, true), true);
363
364 // Start with a disabled communication driver...
365 fs_vRS485DriverDisable();
366}

407// Called from MAC_bProtocolProcessMessage to be notified on a received frame
408void MacCfg_vHandleFrameNotifications (eMACInterface_t fp_eMACId, ESSATMAC_Message * fp_psMsg)
409{
410 // USER CODE
411 switch (fp_psMsg->u8Protocol)
412 {
413 case ES_SAT_MAC_PROTOCOL_ID_FP_LAYER:
414 {
415#ifndef BIG_ENDIAN_PLATFORM
416 ESSA_Stack_FP_MsgHdr_t *pFPHeader = (ESSA_Stack_FP_MsgHdr_t *) fp_psMsg->au8Data;
417#else
418#error Big-endian encoding of the header is not supported currently!
419#endif
420
421 // we are interested only in request FP frames
422 if (pFPHeader != NULL)
423 if (IS_REQUEST(*pFPHeader))
424 MAC__Notif_OnEvent_Cfg(fp_eMACId,
425 EMACNOTIFICATION_ON_FP_REQ_RCVD,
426 (eventData_t) fp_psMsg);
427 break;
428 }
429
430 case ES_SAT_MAC_PROTOCOL_ID_TRANSPORT_LAYER:
431 {
432 MAC__Notif_OnEvent_Cfg(fp_eMACId,
433 EMACNOTIFICATION_ON_TL_FRAME_RCVD,
434 (eventData_t) fp_psMsg);
435 break;
436 }
437
438 default:
439 MAC__Notif_OnEvent_Cfg(fp_eMACId,
440 EMACNOTIFICATION_ON_GEN_FRAME_RCVD,
441 (eventData_t) fp_psMsg);
442 break;
443 }
444 // END OF USER CODE
445 return;
446}

448void MAC__Stats_getEventsBuffer(eMACInterface_t fp_eMACId,
449 uint8_t * const pu8UserBuf,
450 uint32_t u32UserBufSize,
451 uint32_t * const pu32ResponseSize,
452 uint32_t * const pu32EventsCount)
453{
454 if ((pu8UserBuf != NULL) &&
455 (u32UserBufSize > 0U) &&
456 (pu32EventsCount != NULL) &&
457 (pu32ResponseSize != NULL) &&
458 (fp_eMACId < eMAC_Interface_Count))
459 {
460 if (u32UserBufSize >= sizeof(macStats_t))
461 u32UserBufSize = sizeof(macStats_t);
462
463 *pu32EventsCount = eMACStatId_MAX;
464
465 taskENTER_CRITICAL();
466
467 (void) memcpy(pu8UserBuf, (const void *) fs_asMACStates[fp_eMACId].macStats, u32UserBufSize);
468
469 taskEXIT_CRITICAL();
470
471 // size contains also the event count which is part of the returned buffer
472 *pu32ResponseSize = sizeof(uint32_t) + sizeof(fs_asMACStates[fp_eMACId].macStats);
473 }
