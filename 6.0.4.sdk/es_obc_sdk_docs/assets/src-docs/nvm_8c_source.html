59void
60Nvm_Init(void)
61{
62 uint32_t u32CurrentAddress = START_ADDRESS;
63
64 if (NVM_Cfg_Init())
65 {
66 for (uint16_t u16Idx = 0U; u16Idx < NVM_BLOCK_MAX_COUNT; u16Idx++)
67 {
68 CRIT_ASSERT(aBlockConfig[u16Idx].u32Size <=
69 NVM_Cfg_GetMaxPhysBlockSize());
70
71 if ((aBlockConfig[u16Idx].u16ConfigBitfield &
72 BLKCFG_FIXED_ADDRESS) == BLKCFG_FIXED_ADDRESS)
73 {
74 u32CurrentAddress = aBlockConfig[u16Idx].u32FixedAddress;
75 }
76
77 switch (aBlockConfig[u16Idx].eBlockType)
78 {
79 case EBLOCKTYPE_REDUNDANT:
80 for (uint8_t u8Idx = 0U;
81 u8Idx < CDEF_ELEMENT_COUNT(
82 aBlockRTData[u16Idx].u32BlockAddress);
83 u8Idx++)
84 {
85 aBlockRTData[u16Idx].u32BlockAddress[u8Idx] =
86 u32CurrentAddress;
87 u32CurrentAddress += aBlockConfig[u16Idx].u32Size;
88 }
89 break;
90
91 default:
92 aBlockRTData[u16Idx].u32BlockAddress[0U] =
93 u32CurrentAddress;
94 u32CurrentAddress += aBlockConfig[u16Idx].u32Size;
95 break;
96 }
97
98 aBlockRTData[u16Idx].bDefaultsRestored = false;
99 aBlockRTData[u16Idx].bRamMirrorDirty = false;
100 aBlockRTData[u16Idx].bOverrideForceDefaults = false;
101 }
102 }
103 else
104 ES_TRACE_DEBUG("%s init error\r\n", NVM_TRACE_SIGNATURE);
105}

107void
108Nvm_Load(void)
109{
110 eDpOpsResult_t res;
111 uint8_t u8Idx;
112 bool bCrcProtected;
113
114 for (uint16_t u16Idx = 0U; u16Idx < NVM_BLOCK_MAX_COUNT; u16Idx++)
115 {
116 ES_TRACE_DEBUG("%s loading block %d...\r\n", NVM_TRACE_SIGNATURE, u16Idx);
117
118 if (NULL == aBlockConfig[u16Idx].pBufDefaults)
119 {
120 ES_TRACE_DEBUG("FLASH ROM defaults not provided for block id %d!\r\n",
121 u16Idx);
122 }
123
124 bCrcProtected = (aBlockConfig[u16Idx].u16ConfigBitfield &
125 BLKCFG_CRC_PROTECTED) == BLKCFG_CRC_PROTECTED;
126 switch (aBlockConfig[u16Idx].eBlockType)
127 {
128 case EBLOCKTYPE_REDUNDANT:
129 u8Idx = 0U;
130
131 res = fs_eNvmReadBlock(
132 aBlockRTData[u16Idx].u32BlockAddress[u8Idx],
133 aBlockConfig[u16Idx].pRamMirrorBuf,
134 aBlockConfig[u16Idx].u32Size,
135 aBlockConfig[u16Idx].pBufDefaults, bCrcProtected);
136
137 aBlockRTData[u16Idx].bDefaultsRestored =
138 (res == EDPOPSRESULT_DEFAULTS_RESTORED);
139 ES_TRACE_DEBUG("%s NvmReadBlock result -> [%d]...\r\n",
140 NVM_TRACE_SIGNATURE, res);
141
142 while ((++u8Idx < CDEF_ELEMENT_COUNT(
143 aBlockRTData[u16Idx].u32BlockAddress)) &&
144 (res != EDPOPSRESULT_OK))
145 {
146 res = fs_eNvmReadBlock(
147 aBlockRTData[u16Idx].u32BlockAddress[u8Idx],
148 aBlockConfig[u16Idx].pRamMirrorBuf,
149 aBlockConfig[u16Idx].u32Size,
150 aBlockConfig[u16Idx].pBufDefaults, bCrcProtected);
151
152 aBlockRTData[u16Idx].bDefaultsRestored =
153 (res == EDPOPSRESULT_DEFAULTS_RESTORED);
154 ES_TRACE_DEBUG(
155 "%s opted for redundant copy due to integrity failure "
156 "[%d] -> [%d]...\r\n",
157 NVM_TRACE_SIGNATURE, u8Idx, res);
158 }
159 break;
160
161 case EBLOCKTYPE_SINGLE:
162 default:
163 res = fs_eNvmReadBlock(aBlockRTData[u16Idx].u32BlockAddress[0U],
164 aBlockConfig[u16Idx].pRamMirrorBuf,
165 aBlockConfig[u16Idx].u32Size,
166 aBlockConfig[u16Idx].pBufDefaults,
167 bCrcProtected);
168
169 aBlockRTData[u16Idx].bDefaultsRestored =
170 (res == EDPOPSRESULT_DEFAULTS_RESTORED);
171 ES_TRACE_DEBUG("%s NvmReadBlock result -> [%d]...\r\n",
172 NVM_TRACE_SIGNATURE, res);
173 break;
174 }
175 }
176
177 // check layout constraints which are project-specific and force defaults if
178 // something doesn't quite match
179 if (!NVM_Cfg_VerifyLayout())
180 {
181 ES_TRACE_DEBUG("%s Layout verification failed! Forcing defaults...\r\n",
182 NVM_TRACE_SIGNATURE);
183 NVM_Cfg_PrepareLayoutUpgrade();
184 Nvm_Store(NVMSTORE_FORCEDEFAULTS);
185 }
186}

188void
189Nvm_Store(const bool bForceDefaults)
190{
191 NVM_Cfg_EnterCriticalSection();
192
193 for (uint16_t u16Idx = 0U; u16Idx < NVM_BLOCK_MAX_COUNT; u16Idx++)
194 {
195 if (( false != aBlockRTData[u16Idx].bRamMirrorDirty) || bForceDefaults)
196 {
197 if ((bForceDefaults) && (aBlockConfig[u16Idx].pBufDefaults != NULL))
198 {
199 // apply defaults only if block defaults override switch not set
200 if (!aBlockRTData[u16Idx].bOverrideForceDefaults)
201 {
202 (void)memcpy(
203 (void *volatile)aBlockConfig[u16Idx].pRamMirrorBuf,
204 (const void *volatile)aBlockConfig[u16Idx].pBufDefaults,
205 aBlockConfig[u16Idx].u32Size);
206 }
207 }
208
209 ES_TRACE_DEBUG("%s Storing block -> %d...\r\n", NVM_TRACE_SIGNATURE,
210 u16Idx);
211 (void)NvmStoreBlock(
212 u16Idx, (uint8_t *const)aBlockConfig[u16Idx].pRamMirrorBuf);
213 }
214 }
215
216 NVM_Cfg_LeaveCriticalSection();
217}

254void
255Nvm_SetBlockById(uint16_t u16BlockId, void *const pBlockBuffer,
256 const bool bStoreImmediately)
257{
258 if (u16BlockId < NVM_BLOCK_MAX_COUNT)
259 {
260 NVM_Cfg_EnterCriticalSection();
261 (void)memcpy(aBlockConfig[u16BlockId].pRamMirrorBuf, pBlockBuffer,
262 aBlockConfig[u16BlockId].u32Size);
263 aBlockRTData[u16BlockId].bRamMirrorDirty = true;
264
265 if (bStoreImmediately)
266 {
267 ES_TRACE_DEBUG(
268 "%s NvmSetBlockById() block id %d immediate store "
269 "triggered\r\n",
270 NVM_TRACE_SIGNATURE, u16BlockId);
271 NvmStoreBlock(u16BlockId, pBlockBuffer);
272 }
273 else
274 ES_TRACE_DEBUG(
275 "%s NvmSetBlockById() block id %d scheduled for storage "
276 "later\r\n",
277 NVM_TRACE_SIGNATURE, u16BlockId);
278
279 NVM_Cfg_LeaveCriticalSection();
280 }
281 else
282 ES_TRACE_DEBUG("%s NvmSetBlockById() invalid block ID specified -> %d\r\n",
283 NVM_TRACE_SIGNATURE, u16BlockId);
284}

286static eDpOpsResult_t
287fs_eNvmReadBlock(const uint32_t u32Address, uint8_t *const pBuf,
288 const uint32_t u32BufSize, const void *const pBufDefaults,
289 const bool bCrcProtected)
290{
291 bool bReadSuccessful;
292 eDpOpsResult_t eRes = EDPOPSRESULT_ERROR;
293
294 CRIT_ASSERT(pBuf != NULL);
295 CRIT_ASSERT(u32BufSize > 0U);
296
297 bReadSuccessful =
298 NVM_Cfg_ReadBlock(u32Address, (uint8_t *)pBuf, u32BufSize);
299
300 if (bReadSuccessful)
301 {
302 if (bCrcProtected)
303 {
304 CRIT_ASSERT(u32BufSize > CRC32_SIZE);
305
306 uint32_t u32CalcCrc = MACCRC32_Calc32(pBuf, u32BufSize - CRC32_SIZE,
307 CRC32_INIT_VALUE);
308
309 if (u32CalcCrc != *((uint32_t *)&pBuf[u32BufSize - CRC32_SIZE]))
310 {
311 eRes = EDPOPSRESULT_CRC_MISMATCH;
312
313 // load FLASH defaults if available...
314 if (pBufDefaults != NULL)
315 {
316 (void)memcpy(pBuf, pBufDefaults, u32BufSize);
317 eRes = EDPOPSRESULT_DEFAULTS_RESTORED;
318 }
319
320 ES_TRACE_DEBUG("%s NvmReadBlock(%08x) CRC failed\r\n",
321 NVM_TRACE_SIGNATURE, u32Address);
322 }
323 else
324 {
325 eRes = EDPOPSRESULT_OK;
326 ES_TRACE_DEBUG("%s NvmReadBlock(%08x) CRC OK!\r\n",
327 NVM_TRACE_SIGNATURE, u32Address);
328 }
329 }
330 else
331 ES_TRACE_DEBUG("%s NvmReadBlock(%08x) CRC check not configured\r\n",
332 NVM_TRACE_SIGNATURE, u32Address);
333 }
334 else
335 {
336 // load FLASH defaults if available...
337 if (pBufDefaults != NULL)
338 {
339 (void)memcpy(pBuf, pBufDefaults, u32BufSize);
340 eRes = EDPOPSRESULT_DEFAULTS_RESTORED;
341 }
342 }
343
344 return eRes;
345}

347static bool
348fs_bNvmWriteBlock(const uint32_t u32Address, uint8_t *const pBuf,
349 const uint32_t u32BufSize, const bool bCrcProtected)
350{
351 CRIT_ASSERT(pBuf != NULL);
352 CRIT_ASSERT(u32BufSize > 0U);
353
354 if (bCrcProtected)
355 {
356 CRIT_ASSERT(u32BufSize > CRC32_SIZE);
357
358 uint32_t u32CalcCrc =
359 MACCRC32_Calc32(pBuf, u32BufSize - CRC32_SIZE, CRC32_INIT_VALUE);
360
361 *((uint32_t *)&pBuf[u32BufSize - CRC32_SIZE]) = u32CalcCrc;
362 }
363
364 ES_TRACE_DEBUG("%s NvmWriteBlock(%08x) triggered\r\n", NVM_TRACE_SIGNATURE,
365 u32Address);
366 return NVM_Cfg_WriteBlock(u32Address, (uint8_t *)pBuf, u32BufSize);
367}

369static bool
370NvmStoreBlock(const uint16_t u16BlockId, uint8_t *const pBuf)
371{
372 bool bWriteSuccessful = false;
373 bool bCrcProtected;
374 CRIT_ASSERT(u16BlockId < NVM_BLOCK_MAX_COUNT);
375 CRIT_ASSERT(pBuf != NULL);
376
377 bCrcProtected = (aBlockConfig[u16BlockId].u16ConfigBitfield &
378 BLKCFG_CRC_PROTECTED) == BLKCFG_CRC_PROTECTED;
379
380 switch (aBlockConfig[u16BlockId].eBlockType)
381 {
382 case EBLOCKTYPE_REDUNDANT:
383 for (uint8_t u8Idx = 0U;
384 u8Idx <
385 CDEF_ELEMENT_COUNT(aBlockRTData[u16BlockId].u32BlockAddress);
386 u8Idx++)
387 {
388 bWriteSuccessful = fs_bNvmWriteBlock(
389 aBlockRTData[u16BlockId].u32BlockAddress[u8Idx],
390 (uint8_t *const)pBuf, aBlockConfig[u16BlockId].u32Size,
391 bCrcProtected);
392
393 aBlockRTData[u16BlockId].bRamMirrorDirty = !bWriteSuccessful;
394 }
395 break;
396
397 case EBLOCKTYPE_SINGLE:
398 default:
399 bWriteSuccessful = fs_bNvmWriteBlock(
400 aBlockRTData[u16BlockId].u32BlockAddress[0U],
401 (uint8_t *const)pBuf, aBlockConfig[u16BlockId].u32Size,
402 bCrcProtected);
403
404 aBlockRTData[u16BlockId].bRamMirrorDirty = !bWriteSuccessful;
405 break;
406 }
407
408 return bWriteSuccessful;
409}
