90bool MAC_bStartAllIfs(const eMAC_Interface_InitType_t fp_eInitType)
91{
92 bool res = true;
93
94 uint8_t i;
95
96 MAC_bCoreInit();
97
98 for(i = 0; i < eMAC_Interface_Count; i++)
99 {
100 MacCfg_vApplyMacAddrConfig((eMACInterface_t) i, MacCfg_u8GetMacAddress(i));

121bool MAC_bStart(eMACInterface_t fp_eInterface, const eMAC_Interface_InitType_t fp_eInitType)
122{
123 bool bStartOK = false;
124
125 sMACInterfaceState_t *pMacIntf = NULL;
126
127 if(fp_eInterface >= eMAC_Interface_Count)
128 {
129 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_InvalidParameter);
130 }
131 else
132 {
133 pMacIntf = MacCfg_sGetMacInterfaceById(fp_eInterface);
134
135 assert(pMacIntf != NULL);
136
137 fs_vInitInterface(pMacIntf, fp_eInterface, fp_eInitType);
138
139 if(!pMacIntf->bInitDone)
140 {
141 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_InitFailed);
142 }
143 else
144 {
145 // Enable the reception
146 fs_vEnableReception(pMacIntf);
147 }
148
149 bStartOK = true;
150 }
151
152 return bStartOK;
153}

179bool MAC_bProtocolProcessMessage(eMACInterface_t fp_eMACId, ESSATMAC_Message * fp_psMsg)
180{
181 ESSATMAC_Context MACContext;
182
183 bool res = false;
184
185 if((fp_eMACId < eMAC_Interface_Count) && (fp_psMsg != (ESSATMAC_Message *) NULL))
186 {
187 (void)memset((void *)&MACContext, (uint8_t) 0, sizeof(ESSATMAC_Context));
188
189 // This is to be defined by the user in the configuration
190 MacCfg_vHandleFrameNotifications(fp_eMACId, fp_psMsg);
191
192 MACContext.nInterfaceNumber = MacCfg_u8GetStackIfId(fp_eMACId);
193 MACContext.nPayloadProtocol = fp_psMsg->u8Protocol;
194 MACContext.nSourceAddr = fp_psMsg->u8SourceAddr;
195 MACContext.nTargetAddr = fp_psMsg->u8TargetAddr;
196 MACContext.netType = ESSASNETT_INTERNAL;
197
198#if defined(USE_CUSTOM_ESPS_MSG_DISPATCH_HANDLER)
199 custom_esps_data_dispatch(&MACContext, fp_psMsg->au8Data, fp_psMsg->u8DataSize);
200#else
201 // This must be defined in the application. See ESSA_Stack.h
202 ESSA_Stack_DataDispatch(&MACContext, fp_psMsg->au8Data, fp_psMsg->u8DataSize);
203#endif
204
205 res = true;
206 }
207
208 return res;
209}

211ESSATMAC_ErrCodes MAC_DrvEnqMessage(ESSATMAC_Context * fp_pMACContext, uint8_t * fp_pMsgData, uint8_t fp_nMsgDataSize)
212{
213 sMACInterfaceState_t *pMacIntf = NULL;
214
215 sMACTXPoolEntry_t *psEntry;
216
217 bool txQueueAddResult;
218
219 ESSATMAC_ErrCodes result = ESSATMAC_EC_ERROR;
220
221 eMACInterface_t drvIf;
222
223 if((fp_pMACContext != NULL) && (fp_pMsgData != NULL) && (fp_nMsgDataSize > 0))
224 {
225 drvIf = MacCfg_eGetMacDrvIf(fp_pMACContext->nInterfaceNumber);
226
227 if(drvIf >= eMAC_Interface_Count)
228 {
229 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_InvalidParameter);
230 }
231 else
232 {
233 // Get the interface (which RS485 connection)
234 pMacIntf = MacCfg_sGetMacInterfaceById(drvIf);
235 assert(pMacIntf != NULL);
236
237 psEntry = MAC_POOL_psPoolTXNew(pMacIntf);
238
239 if(psEntry != NULL)
240 {
241 psEntry->nMsgID = fp_pMACContext->nMsgID;
242
243 // prepare message header
244 psEntry->sMessage.u8TargetAddr = fp_pMACContext->nTargetAddr;
245 psEntry->sMessage.u8SourceAddr = (fp_pMACContext->netType != ESSASNETT_UART_BRIDGE) ?
246 ((uint8_t) pMacIntf->u16Address) :
247 ((uint8_t) fp_pMACContext->nSourceAddr);
248 psEntry->sMessage.u8Protocol = fp_pMACContext->nPayloadProtocol;
249 psEntry->sMessage.u8DataSize = fp_nMsgDataSize;
250 psEntry->sMessage.uHdrCRC16 = crc16_ccitt_table((const uint8_t *)&psEntry->sMessage,
251 ESSATMAC_HEADER_SIZE -
252 sizeof(psEntry->sMessage.uHdrCRC16));
253
254 (void)memcpy(psEntry->sMessage.au8Data, fp_pMsgData, fp_nMsgDataSize);
255
256 psEntry->sPrio.u8BasePriority = pMacIntf->u8BasePrio;
257 psEntry->sPrio.u8Priority = fp_pMACContext->nSendBasePriority;
258 psEntry->pSendCallback = fp_pMACContext->pSendCallback;
259
260 // Calculate the CRC prior to attempting to send.
261 fs_vUpdateCRC(psEntry);
262
263 // Queue the message for sending.
264 txQueueAddResult = MAC_QUEUE_vTXQueueAdd(pMacIntf, psEntry);
265
266 if(txQueueAddResult)
267 {
268 // Indicate that a new message has been added to the queue
269 MAC_IRQ_vTXRequestIndication(pMacIntf);
270
271 result = ESSATMAC_EC_OK;
272 }
273 else
274 result = ESSATMAC_EC_Q_FULL;
275 }
276 else
277 {
278 MAC_Stats_triggerEvent(pMacIntf->macStats, eMACStatId_TxMsgBufUnavailCntr);
279 result = ESSATMAC_EC_NO_MEM;
280 }
281 }
282 }
283 else
284 result = ESSATMAC_EC_NULL;
285
286 return result;
287}

290{
291
292 sMACInterfaceState_t *psInterfaceState;

295

298

302 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_InvalidState);

311 }

315void MAC_vDispatchUARTEvent(sMACInterfaceState_t * fp_psState)
316{
317 if(LL_USART_IsEnabledIT_RXNE(fp_psState->psCfg->sHwConfig.psUART) &&
318 LL_USART_IsActiveFlag_RXNE(fp_psState->psCfg->sHwConfig.psUART))
319 {
320 MAC_IRQ_vRXIndication(fp_psState);
321
322 LL_USART_RequestRxDataFlush(fp_psState->psCfg->sHwConfig.psUART);
323 }
324
325 if(LL_USART_IsEnabledIT_TXE(fp_psState->psCfg->sHwConfig.psUART) &&
326 LL_USART_IsActiveFlag_TXE(fp_psState->psCfg->sHwConfig.psUART))
327 {
328 MAC_IRQ_vTXIndication(fp_psState);
329 }
330
331 if(LL_USART_IsEnabledIT_TC(fp_psState->psCfg->sHwConfig.psUART) &&
332 LL_USART_IsActiveFlag_TC(fp_psState->psCfg->sHwConfig.psUART))
333 {
334 LL_USART_ClearFlag_TC(fp_psState->psCfg->sHwConfig.psUART);
335 MAC_IRQ_vTXFinishedIndication(fp_psState);
336 }
337
338 if(LL_USART_IsEnabledIT_PE(fp_psState->psCfg->sHwConfig.psUART) &&
339 LL_USART_IsActiveFlag_PE(fp_psState->psCfg->sHwConfig.psUART))
340 {
341 LL_USART_ClearFlag_PE(fp_psState->psCfg->sHwConfig.psUART);
342 MAC_IRQ_vRXProcessEvent(fp_psState, eRXEvent_RxError);
343 }
344
345 if(LL_USART_IsEnabledIT_ERROR(fp_psState->psCfg->sHwConfig.psUART))
346 {
347 if(LL_USART_IsActiveFlag_NE(fp_psState->psCfg->sHwConfig.psUART))
348 {
349 LL_USART_ClearFlag_NE(fp_psState->psCfg->sHwConfig.psUART);
350 MAC_IRQ_vTXProcessEvent(fp_psState, eTXEvent_CollisionDetected);
351 MAC_IRQ_vRXProcessEvent(fp_psState, eRXEvent_RxError);
352 }
353
354 if(LL_USART_IsActiveFlag_FE(fp_psState->psCfg->sHwConfig.psUART))
355 {
356 LL_USART_ClearFlag_FE(fp_psState->psCfg->sHwConfig.psUART);
357 MAC_IRQ_vTXProcessEvent(fp_psState, eTXEvent_CollisionDetected);
358 MAC_IRQ_vRXProcessEvent(fp_psState, eRXEvent_RxError);
359 }
360
361 if(LL_USART_IsActiveFlag_ORE(fp_psState->psCfg->sHwConfig.psUART))
362 {
363 LL_USART_ClearFlag_ORE(fp_psState->psCfg->sHwConfig.psUART);
364 MAC_IRQ_vTXProcessEvent(fp_psState, eTXEvent_CollisionDetected);
365 MAC_IRQ_vRXProcessEvent(fp_psState, eRXEvent_RxError);
366 }
367 }
368
369 if(LL_USART_IsActiveFlag_IDLE(fp_psState->psCfg->sHwConfig.psUART))
370 LL_USART_ClearFlag_IDLE(fp_psState->psCfg->sHwConfig.psUART);
371}

373void MAC_vEnterCriticalSection(sMACInterfaceState_t * const fp_psIntf)
374{
375 if(fp_psIntf != NULL)
376 {
377 if(fp_psIntf->criticalSectionNestLevel == 0U)
378 {
379 // This must be compatible with the hardware configuration of the MAC driver
380 NVIC_DisableIRQ(fp_psIntf->psCfg->sHwConfig.u32NVICMask);
381 MAC_TIM_vSetTimerIsrStateById(fp_psIntf->psCfg->sHwConfig.eTXTIMID, false);
382 MAC_TIM_vSetTimerIsrStateById(fp_psIntf->psCfg->sHwConfig.eRXTIMID, false);
383 }
384
385 if(fp_psIntf->criticalSectionNestLevel < 255)
386 fp_psIntf->criticalSectionNestLevel++;
387 }
388}

390void MAC_vLeaveCriticalSection(sMACInterfaceState_t * const fp_psIntf)
391{
392 if(fp_psIntf != NULL)
393 {
394 if(fp_psIntf->criticalSectionNestLevel > 0U)
395 fp_psIntf->criticalSectionNestLevel--;

397 if(fp_psIntf->criticalSectionNestLevel == 0U)
398 {
399 // This must be compatible with the hardware configuration of the MAC driver
400 NVIC_EnableIRQ(fp_psIntf->psCfg->sHwConfig.u32NVICMask);
401 MAC_TIM_vSetTimerIsrStateById(fp_psIntf->psCfg->sHwConfig.eTXTIMID, true);
402 MAC_TIM_vSetTimerIsrStateById(fp_psIntf->psCfg->sHwConfig.eRXTIMID, true);
403 }

410STATIC void fs_vInitInterface(sMACInterfaceState_t * fp_psState,
411 eMACInterface_t fp_eInterface, const eMAC_Interface_InitType_t fp_eInitType)
412{
413 if((!fp_psState->bInitDone) || (fp_eInitType == MACIF_INITTYPE_FULL))
414 {
415 // Set the queue head
416 fp_psState->sTXCtx.psQueue = NULL;
417 fp_psState->sTXCtx.u32QueueElementsCount = 0U;
418
419 // Initialize the uart. This calls fs_Uart1Init, previously set in esps_drv_config.c
420 if(fp_psState->psCfg->pfUartInit != NULL)
421 (*fp_psState->psCfg->pfUartInit) ((struct sMACConfig_t *)(&fp_psState->psCfg->sHwConfig));
422
423 // Enable the clock to the peripheral.
424 MAC_RCC_vPeripheralClockEnable(fp_psState->psCfg->sHwConfig.eClockSource,
425 fp_psState->psCfg->sHwConfig.u32RCCMask);
426
427 // initialize the memory pool
428 fp_psState->bInitDone = MAC_POOL_bPoolInit(fp_psState);
429
430 if((fp_psState->psCfg->sHwConfig.u32Baudrate != 0) && (fp_psState->bInitDone))
431 {
432 // Set the initial state of the MAC interface
433 fp_psState->sTXCtx.eState = eMACTransmit_Idle;
434 fp_psState->sTXCtx.ePrevState = eMACTransmit_Off;
435 fp_psState->sTXCtx.u8NestLevel = 0U;
436 fp_psState->sRXCtx.eState = eMACReceive_Idle;
437 fp_psState->sRXCtx.ePrevState = eMACReceive_Off;
438 fp_psState->sRXCtx.u8NestLevel = 0U;
439
440 // Reset to 0 the values of the TX/RX stats
441 MAC_Stats_resetEventCounters(fp_psState->macStats);
442
443 // initialize the dispatcher pool to 0
444 MAC_DISP_vDispatcherInit(fp_psState, fp_eInterface);
445
446 fp_psState->criticalSectionNestLevel = 0U;
447
448 // Enable UART global interrupt
449 NVIC_EnableIRQ(fp_psState->psCfg->sHwConfig.u32NVICMask);
450
451 // Enable the UART
452 LL_USART_Enable(fp_psState->psCfg->sHwConfig.psUART);
453 }
454 }
455 else
456 {
457 // This branch gets executed upon re-initialization after sleep
458 if(fp_psState->psCfg->pfUartInit != NULL)
459 (*fp_psState->psCfg->pfUartInit) ((struct sMACConfig_t *)(&fp_psState->psCfg->sHwConfig));
460
461 // Enable the clock to the peripheral.
462 MAC_RCC_vPeripheralClockEnable(fp_psState->psCfg->sHwConfig.eClockSource,
463 fp_psState->psCfg->sHwConfig.u32RCCMask);
464
465 // Set the initial state of the MAC interface
466 fp_psState->sTXCtx.eState = eMACTransmit_Idle;
467 fp_psState->sTXCtx.ePrevState = eMACTransmit_Off;
468 fp_psState->sTXCtx.u8NestLevel = 0U;
469 fp_psState->sRXCtx.eState = eMACReceive_Idle;
470 fp_psState->sRXCtx.ePrevState = eMACReceive_Off;
471 fp_psState->sRXCtx.u8NestLevel = 0U;
472
473 fp_psState->criticalSectionNestLevel = 0U;
474 // Enable UART global interrupt.
475 NVIC_EnableIRQ(fp_psState->psCfg->sHwConfig.u32NVICMask);
476
477 // Enable the UART
478 LL_USART_Enable(fp_psState->psCfg->sHwConfig.psUART);
479 }
480}

503STATIC void fs_vEnableReception(sMACInterfaceState_t * fp_psState)
504{
505 if(!fp_psState->bInitDone)
506 {
507 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_InvalidState);
508 }
509 else
510 {
511 // ipetrov: Currently, for some unknown reason, Tx and Rx timers
512 // require one single run before they start
513 // to elapse correctly. Otherwise, first timer shot elapses
514 // prematurely and this could lead to first
515 // RX message discarded and first Tx message sent with an
516 // incorrect priority wait time.
517 MAC_TIM_vMicroTimerStart(fp_psState->psCfg->sHwConfig.eRXTIMID, (uint32_t) 5, fs_vTIMCallback, fp_psState);
518 MAC_TIM_vMicroTimerStart(fp_psState->psCfg->sHwConfig.eTXTIMID, (uint32_t) 5, fs_vTIMCallback, NULL);
519 }
520}
