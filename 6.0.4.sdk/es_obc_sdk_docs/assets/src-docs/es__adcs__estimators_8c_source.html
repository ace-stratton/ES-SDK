72es_adcs_result_t es_adcs_estimate(es_adcs_estimator_config_t estimConfig, es_adcs_used_magnetometer_t activeMagnetometer)
73{
74 /* 0. catch if both estimators are the same? also, if both are NONE, then we can proceed
75 * 1. extract number of algorithms to run from estimator config
76 * 2. extract which algorithms to run from estimator config
77 * 3. run algorithms
78 * 4. report if there are errors in the return type
79 */
80
81 /* Execution result variable */
82 es_adcs_result_t result = ES_ADCS_RESULT_FAIL;
83
84 /* Set up the initial conditions for this loop */
85 estimatorStatus.lastWarningGeneral = ES_ADCS_ESTIMATOR_WARN_NONE;
86 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_GENERAL;
87
88 /* If the desired estimators are not NONE and they are both the same */
89 if ( (ES_ADCS_ESTIMATOR_ALGO_NONE != estimConfig.eActiveEstimators[0]) &&
90 (estimConfig.eActiveEstimators[0] == estimConfig.eActiveEstimators[1]) )
91 {
92 /* Two identical estimator algorithms selected - raise an error */
93 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_ALGO_IDENTICAL;
94
95 /* Report through es_exeh */
96 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_ESTIMATOR_ALGO_IDENTICAL);
97 }
98 else
99 {
100 /* Execute all active estimators */
101 for (uint8_t i = 0; i < ES_ADCS_ACTIVE_ESTIMATORS_MAX; i++)
102 {
103 switch( estimConfig.eActiveEstimators[i] )
104 {
105 case ES_ADCS_ESTIMATOR_ALGO_NONE:
106 {
107 /* No estimation performed */
108 break;
109 }
110
111 case ES_ADCS_ESTIMATOR_ALGO_BDOT_MAG:
112 {
113 /* Perform Bdot estimation */
114 estimatorStatus.algoStatus[i] = fs_es_adcs_estimator_detumbling_bdot(activeMagnetometer);
115 break;
116 }
117
118 default:
119 {
120 /* Invalid estimator algorithm chosen - raise an error */
121 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_INVALID_ALGO;
122
123 /* Report through es_exeh */
124 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_ESTIMATOR_INVALID_TYPE);
125 break;
126 }
127 }
128 }
129 }
130
131 /* If both estimators are active */
132 if ( (ES_ADCS_ESTIMATOR_ALGO_NONE != estimConfig.eActiveEstimators[0]) &&
133 (ES_ADCS_ESTIMATOR_ALGO_NONE != estimConfig.eActiveEstimators[1]) )
134 {
135 /* Check that all estimator algorithms have completed successfully */
136 /* If the code reached this place and no errors have been reported, everything is ok */
137 if ( (ES_ADCS_ESTIMATOR_ERROR_GENERAL == estimatorStatus.lastErrorGeneral) &&
138 (ES_ADCS_ESTIMATOR_ERROR_NONE == estimatorStatus.algoStatus[0].lastErrorAlgo) &&
139 (ES_ADCS_ESTIMATOR_ERROR_NONE == estimatorStatus.algoStatus[1].lastErrorAlgo) )
140 {
141 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_NONE;
142 result = ES_ADCS_RESULT_OK;
143 }
144 }
145 /* If only the first estimator is activated */
146 else if ( (ES_ADCS_ESTIMATOR_ALGO_NONE != estimConfig.eActiveEstimators[0]) &&
147 (ES_ADCS_ESTIMATOR_ALGO_NONE == estimConfig.eActiveEstimators[1]) )
148 {
149 /* Check that only the first estimator algorithm has completed successfully */
150 /* If the code reached this place and no errors have been reported, everything is ok */
151 if ( (ES_ADCS_ESTIMATOR_ERROR_GENERAL == estimatorStatus.lastErrorGeneral) &&
152 (ES_ADCS_ESTIMATOR_ERROR_NONE == estimatorStatus.algoStatus[0].lastErrorAlgo) )
153 {
154 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_NONE;
155 result = ES_ADCS_RESULT_OK;
156 }
157 }
158 /* If only the second estimator is activated */
159 else if ( (ES_ADCS_ESTIMATOR_ALGO_NONE == estimConfig.eActiveEstimators[0]) &&
160 (ES_ADCS_ESTIMATOR_ALGO_NONE != estimConfig.eActiveEstimators[1]) )
161 {
162 /* Check that only the second estimator algorithm has completed successfully */
163 /* If the code reached this place and no errors have been reported, everything is ok */
164 if ( (ES_ADCS_ESTIMATOR_ERROR_GENERAL == estimatorStatus.lastErrorGeneral) &&
165 (ES_ADCS_ESTIMATOR_ERROR_NONE == estimatorStatus.algoStatus[1].lastErrorAlgo) )
166 {
167 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_NONE;
168 result = ES_ADCS_RESULT_OK;
169 }
170 }
171 /* If no estimator algorithms are active */
172 else
173 {
174 /* If the code reached this place and no errors have been reported, everything is ok */
175 if (ES_ADCS_ESTIMATOR_ERROR_GENERAL == estimatorStatus.lastErrorGeneral)
176 {
177 estimatorStatus.lastErrorGeneral = ES_ADCS_ESTIMATOR_ERROR_NONE;
178 result = ES_ADCS_RESULT_OK;
179 }
180 }
181
182 return result;
183}

213static es_adcs_estimator_algo_status_t fs_es_adcs_estimator_detumbling_bdot(es_adcs_used_magnetometer_t usedMagnetometer)
214{
215 /* 0. magnetorquer stop time shall be handled outside the estimator (in the task, preferably)
216 * outline:
217 * 1. retrieve sensor data from DataCache (config flag determines primary or secondary)
218 * 2. calculate derivative of magnetic field in each axis
219 * 3. return calculated data to DataCache
220 */
221
222 es_adcs_estimator_algo_status_t algoStatus;
223 algoStatus.lastWarningAlgo = ES_ADCS_ESTIMATOR_WARN_NONE;
224 algoStatus.lastErrorAlgo = ES_ADCS_ESTIMATOR_ERROR_GENERAL;
225
226 REAL x_current;
227 REAL x_previous;
228 REAL y_current;
229 REAL y_previous;
230 REAL z_current;
231 REAL z_previous;
232 REAL vectorMagnitude;
233
234 /* Dodge MISRA-C:2004 10.4 */
235 REAL temp_deriv_x_uT;
236 REAL temp_deriv_y_uT;
237 REAL temp_deriv_z_uT;
238
239 /* Read sensor data service configuration from nvm */
240 sds_sens_read_freq_t sds_chan_freq_cfgs;
241 Nvm_GetBlockById(NVM_BLOCK_SDS_SENS_READ_FREQ_CFG, &sds_chan_freq_cfgs);
242
243 /* Extract magnetometer sampling rate in Hz */
244 uint16_t sampling_rate_mag = sds_chan_freq_cfgs.sens_cfg[0].freq_cfg;
245
246 /* Transform Hz to seconds */
247 REAL sampling_time = 1 / (REAL)sampling_rate_mag;
248
249 /* Local vectors to enable use of vector math */
250 es_adcs_vec_3d_t magMeasureCurrent;
251 es_adcs_vec_3d_t magMeasurePrevious;
252 es_adcs_vec_3d_t magMeasureAverage;
253 es_adcs_vec_3d_t magMeasureDerivative;
254
255 /* Check if primary magnetometer is being used */
256 if (ES_ADCS_USED_MAGNETOMETER_PRIMARY == usedMagnetometer)
257 {
258 /* Retrieve latest measurements from primary magnetometer */
259 data_cache_error_estimator = dc_get_es_adcs_sensor_mag_primary_data(&mag_measurements_primary);
260
261 /* Catch any DataCache read failure */
262 if (DC_DATA_STATUS_OK != data_cache_error_estimator)
263 {
264 /* DataCache readout failed - raise an error */
265 algoStatus.lastErrorAlgo = ES_ADCS_ESTIMATOR_ERROR_DATA_CACHE_GET;
266
267 /* Report through es_exeh */
268 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_ESTIMATOR_BDOT_DC_READ_FAIL);
269 return algoStatus;
270 }
271
272 /* Measurements come in microT, convert to T */
273 x_current = (REAL)mag_measurements_primary.i32MAG_X_CURRENT * ES_ADCS_uT_TO_T;
274 x_previous = (REAL)mag_measurements_primary.i32MAG_X_PREVIOUS * ES_ADCS_uT_TO_T;
275
276 y_current = (REAL)mag_measurements_primary.i32MAG_Y_CURRENT * ES_ADCS_uT_TO_T;
277 y_previous = (REAL)mag_measurements_primary.i32MAG_Y_PREVIOUS * ES_ADCS_uT_TO_T;
278
279 z_current = (REAL)mag_measurements_primary.i32MAG_Z_CURRENT * ES_ADCS_uT_TO_T;
280 z_previous = (REAL)mag_measurements_primary.i32MAG_Z_PREVIOUS * ES_ADCS_uT_TO_T;
281 }
282 /* Check if secondary magnetometer is being used */
283 else if (ES_ADCS_USED_MAGNETOMETER_SECONDARY == usedMagnetometer)
284 {
285 /* Retrieve latest measurements from secondary magnetometer */
286 data_cache_error_estimator = dc_get_es_adcs_sensor_mag_secondary_data(&mag_measurements_secondary);
287
288 /* Catch any DataCache read failure */
289 if (DC_DATA_STATUS_OK != data_cache_error_estimator)
290 {
291 /* DataCache readout failed - raise an error */
292 algoStatus.lastErrorAlgo = ES_ADCS_ESTIMATOR_ERROR_DATA_CACHE_GET;
293
294 /* Report through es_exeh */
295 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_ESTIMATOR_BDOT_DC_READ_FAIL);
296 return algoStatus;
297 }
298
299 /* Measurements come in microT, convert to T */
300 x_current = (REAL)mag_measurements_secondary.i32MAG_X_CURRENT * ES_ADCS_uT_TO_T;
301 x_previous = (REAL)mag_measurements_secondary.i32MAG_X_PREVIOUS * ES_ADCS_uT_TO_T;
302
303 y_current = (REAL)mag_measurements_secondary.i32MAG_Y_CURRENT * ES_ADCS_uT_TO_T;
304 y_previous = (REAL)mag_measurements_secondary.i32MAG_Y_PREVIOUS * ES_ADCS_uT_TO_T;
305
306 z_current = (REAL)mag_measurements_secondary.i32MAG_Z_CURRENT * ES_ADCS_uT_TO_T;
307 z_previous = (REAL)mag_measurements_secondary.i32MAG_Z_PREVIOUS * ES_ADCS_uT_TO_T;
308 }
309 else
310 {
311 /* Magnetometer config is flawed - raise an error */
312 algoStatus.lastErrorAlgo = ES_ADCS_ESTIMATOR_ERROR_MAGNETOMETER_INVALID;
313
314 /* Report through es_exeh */
315 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_ESTIMATOR_INVALID_MAG_CONFIG);
316 return algoStatus;
317 }
318
319 ES_TRACE_DEBUG("\r");
320 ES_TRACE_DEBUG("[es_adcs_estimator] mag measurement current : %f, %f, %f [Tesla]\r", x_current, y_current, z_current);
321 ES_TRACE_DEBUG("[es_adcs_estimator] mag measurement previous: %f, %f, %f [Tesla]\r", x_previous, y_previous, z_previous);
322
323 /* Set measurements as vector components */
324 magMeasureCurrent = es_adcs_math_VecSet(x_current , y_current , z_current);
325 magMeasurePrevious = es_adcs_math_VecSet(x_previous, y_previous, z_previous);
326
327 /* Find the average of the magnetic field vector */
328 /* Scaling by 0.5 is equivalent to dividing by 2 */
329 magMeasureAverage = es_adcs_math_VecScale( es_adcs_math_VecAdd(magMeasureCurrent, magMeasurePrevious), 0.5f);
330
331 /* Find the magnitude of the average magnetic field vector */
332 vectorMagnitude = es_adcs_math_VecNorm(magMeasureAverage);
333
334 /* Find the derivative of the magnetic field vector */
335 magMeasureDerivative = es_adcs_math_VecScale(
336 es_adcs_math_VecSub(magMeasureCurrent, magMeasurePrevious),
337 es_adcs_math_InverseOf(vectorMagnitude * sampling_time)
338 );
339 ES_TRACE_DEBUG("[es_adcs_estimator] mag deriv: %.10f, %.10f, %.10f [rad/s]\r", magMeasureDerivative.X, magMeasureDerivative.Y, magMeasureDerivative.Z);
340
341 /* The result is in rad/s */
342 /* Convert to desired microrad/s unit */
343 temp_deriv_x_uT = magMeasureDerivative.X * ES_ADCS_RADS_TO_uRADS;
344 temp_deriv_y_uT = magMeasureDerivative.Y * ES_ADCS_RADS_TO_uRADS;
345 temp_deriv_z_uT = magMeasureDerivative.Z * ES_ADCS_RADS_TO_uRADS;
346
347 bdot_estimates.i32MAG_FIELD_DERIV_X = (int32_t)temp_deriv_x_uT;
348 bdot_estimates.i32MAG_FIELD_DERIV_Y = (int32_t)temp_deriv_y_uT;
349 bdot_estimates.i32MAG_FIELD_DERIV_Z = (int32_t)temp_deriv_z_uT;
350
351 ES_TRACE_DEBUG("[es_adcs_estimator] mag deriv to DataCache: %d, %d, %d [microrad/s]\r", bdot_estimates.i32MAG_FIELD_DERIV_X, bdot_estimates.i32MAG_FIELD_DERIV_Y, bdot_estimates.i32MAG_FIELD_DERIV_Z);
352
353 /* Send estimates to data cache */
354 dc_set_es_adcs_estimates_bdot_data(&bdot_estimates);
355
356
357 /* If the code reached this place and no errors have been reported, everything is ok */
358 if (ES_ADCS_ESTIMATOR_ERROR_GENERAL == algoStatus.lastErrorAlgo)
359 {
360 algoStatus.lastErrorAlgo = ES_ADCS_ESTIMATOR_ERROR_NONE;
361 }
362
363 return algoStatus;
364}
