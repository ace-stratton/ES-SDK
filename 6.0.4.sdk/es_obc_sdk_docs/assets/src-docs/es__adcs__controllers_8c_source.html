74es_adcs_result_t es_adcs_control(es_adcs_controller_config_t controlConfig)
75{
76 /* 1. extract which algorithm to run from es_adcs_controller_config_t
77 * 2. run algorithm
78 * 3. report if there are errors in the return type
79 */
80
81 /* Execution result variable */
82 es_adcs_result_t result = ES_ADCS_RESULT_FAIL;
83
84 /* Set up the initial conditions for this loop */
85 controllerStatus.lastWarningGeneral = ES_ADCS_CONTROLLER_WARN_NONE;
86 controllerStatus.lastErrorGeneral = ES_ADCS_CONTROLLER_ERROR_GENERAL;
87
88 /* Execute controllers */
89 switch( controlConfig.eActiveController )
90 {
91 case ES_ADCS_CONTROLLER_ALGO_NONE:
92 {
93 /* No control performed */
94 break;
95 }
96
97 case ES_ADCS_CONTROLLER_ALGO_BDOT_MAG:
98 {
99 /* Perform Bdot control */
100 controllerStatus.algoStatus = fs_es_adcs_controller_detumbling_bdot(controlConfig.sBdotConfig);
101 break;
102 }
103
104 default:
105 {
106 /* Invalid controller algorithm chosen - raise an error */
107 controllerStatus.lastErrorGeneral = ES_ADCS_CONTROLLER_ERROR_INVALID_ALGO;
108
109 /* Report through es_exeh */
110 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_CONTROLLER_INVALID_TYPE);
111 break;
112 }
113 }
114
115 /* If the code reached this place and no errors have been reported, everything is ok */
116 if ( (ES_ADCS_CONTROLLER_ERROR_GENERAL == controllerStatus.lastErrorGeneral) &&
117 (ES_ADCS_CONTROLLER_ERROR_NONE == controllerStatus.algoStatus.lastErrorAlgo) )
118 {
119 controllerStatus.lastErrorGeneral = ES_ADCS_CONTROLLER_ERROR_NONE;
120 result = ES_ADCS_RESULT_OK;
121 }
122
123 return result;
124}

154static es_adcs_controller_algo_status_t fs_es_adcs_controller_detumbling_bdot(es_adcs_controller_bdot_config_t activeBdotConfig)
155{
156 /*
157 * 0. magnetorquer starting shall be handled outside the controller (separate OBC task, preferably)
158 * outline:
159 * 1. retrieve estimated data from DataCache
160 * 2. calculate magnetorquer drive strength (from -100 to +100) in each axis
161 * 3. return calculated data to DataCache
162 */
163
164 es_adcs_controller_algo_status_t algoStatus;
165 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_NONE;
166 algoStatus.lastErrorAlgo = ES_ADCS_CONTROLLER_ERROR_GENERAL;
167
168 REAL estimated_magfield_deriv_x;
169 REAL estimated_magfield_deriv_y;
170 REAL estimated_magfield_deriv_z;
171
172 REAL control_value_x;
173 REAL control_value_y;
174 REAL control_value_z;
175
176 /* Get data from the estimator */
177 data_cache_error_controller = dc_get_es_adcs_estimates_bdot_data(&bdot_estimates_controller);
178
179 /* Catch any DataCache read failure */
180 if (DC_DATA_STATUS_OK != data_cache_error_controller)
181 {
182 /* DataCache readout failed - raise an error */
183 algoStatus.lastErrorAlgo = ES_ADCS_CONTROLLER_ERROR_DATA_CACHE_GET;
184
185 /* Report through es_exeh */
186 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ES_ADCS_CONTROLLER_BDOT_DC_READ_FAIL);
187 }
188 else
189 {
190 /* Measurements come in microT/s */
191 estimated_magfield_deriv_x = (REAL)bdot_estimates_controller.i32MAG_FIELD_DERIV_X * ES_ADCS_uT_TO_T;
192 estimated_magfield_deriv_y = (REAL)bdot_estimates_controller.i32MAG_FIELD_DERIV_Y * ES_ADCS_uT_TO_T;
193 estimated_magfield_deriv_z = (REAL)bdot_estimates_controller.i32MAG_FIELD_DERIV_Z * ES_ADCS_uT_TO_T;
194
195 /* Apply B-dot control law => out = - Gain * est_mag_field_deriv */
196 control_value_x = - activeBdotConfig.bdot_coefficient_X * estimated_magfield_deriv_x;
197 control_value_y = - activeBdotConfig.bdot_coefficient_Y * estimated_magfield_deriv_y;
198 control_value_z = - activeBdotConfig.bdot_coefficient_Z * estimated_magfield_deriv_z;
199
200 ES_TRACE_DEBUG("[es_adcs_controller] MTQ values pre-saturation: %.2f, %.2f, %.2f [%%]\r", control_value_x, control_value_y, control_value_z);
201
202 /* Saturate the X signal */
203 if (control_value_x > BDOT_SATURATION_UPPER)
204 {
205 control_value_x = BDOT_SATURATION_UPPER;
206 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_UPPER;
207 }
208 if (control_value_x < BDOT_SATURATION_LOWER)
209 {
210 control_value_x = BDOT_SATURATION_LOWER;
211 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_LOWER;
212 }
213
214 /* Saturate the Y signal */
215 if (control_value_y > BDOT_SATURATION_UPPER)
216 {
217 control_value_y = BDOT_SATURATION_UPPER;
218 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_UPPER;
219 }
220 if (control_value_y < BDOT_SATURATION_LOWER)
221 {
222 control_value_y = BDOT_SATURATION_LOWER;
223 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_LOWER;
224 }
225
226 /* Saturate the Z signal */
227 if (control_value_z > BDOT_SATURATION_UPPER)
228 {
229 control_value_z = BDOT_SATURATION_UPPER;
230 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_UPPER;
231 }
232 if (control_value_z < BDOT_SATURATION_LOWER)
233 {
234 control_value_z = BDOT_SATURATION_LOWER;
235 algoStatus.lastWarningAlgo = ES_ADCS_CONTROLLER_WARN_SATURATION_LOWER;
236 }
237
238 /* Convert to desired unit for DataCache */
239 bdot_control_values.i8MAGTORQUE_VALUE_X = (int8_t)control_value_x;
240 bdot_control_values.i8MAGTORQUE_VALUE_Y = (int8_t)control_value_y;
241 bdot_control_values.i8MAGTORQUE_VALUE_Z = (int8_t)control_value_z;
242
243 ES_TRACE_DEBUG("[es_adcs_controller] MTQ values to DataCache: %d, %d, %d [%%]\r", bdot_control_values.i8MAGTORQUE_VALUE_X, bdot_control_values.i8MAGTORQUE_VALUE_Y, bdot_control_values.i8MAGTORQUE_VALUE_Z);
244
245 /* Send calculated control values to data cache */
246 dc_set_es_adcs_control_values_mtq_data(&bdot_control_values);
247 }
248
249 /* If the code reached this place and no errors have been reported, everything is ok */
250 if (ES_ADCS_CONTROLLER_ERROR_GENERAL == algoStatus.lastErrorAlgo)
251 {
252 algoStatus.lastErrorAlgo = ES_ADCS_CONTROLLER_ERROR_NONE;
253 }
254
255 return algoStatus;
256}
