116bool eps_ii_set_channel_output(eps_ii_ctrl_ch_t channel, bool state){
117 bool res = false;
118
119 // Set a GPIO
120 if (EPS_II_CTRL_GPIO_MAX > channel){
121 // call set gpio
122 res = eps_ii_set_gpio(eps_ii_channel_to_gpio(channel), state);
123 }
124 // Set a Power Output
125 else if (EPS_II_CTRL_MAX > channel){
126 // call set power output
127 res = eps_ii_set_sgpo(eps_ii_channel_to_sgpo(channel), state);
128 }
129 //
130 else{
131 CRIT_ASSERT(false); // Catch this error while in debug
132 res = false;
133 }
134
135 return res;
136}
137bool eps_ii_get_channel_output(eps_ii_ctrl_ch_t channel){
138 (void) channel; // The EPS II request does not need the channel
139 bool res = false;
140
141 // Set a GPIO

143 // call get gpio
144 res = eps_ii_get_gpio();
145 }
146 // Set a Power Output

148 // call get sgpo
149 res = eps_ii_get_sgpo();
150 }
151 else{
152 CRIT_ASSERT(false); // Catch this error while in debug
153 res = false;
154 }
155
156 return res;
157}

234// EPS II channel control
235// Request EPS II module to switch on or off a channel
239static bool eps_ii_set_gpio(EPSII_PDM_1_SGGPIO_Bitmask_t channel_mask, bool state){
240
241 // Channel mask should be a valid value from the enum type
242 if (EPSII_PDM_1_SGGPIO_BITMASK_MAX_CNT <= channel_mask)
243 {
244 CRIT_ASSERT(false); // Catch this error while in debug
245 return false;
246 }
247
248 // Assign the value mask the same value as the channel mask or 0
249 EPSII_PDM_1_SGGPIO_OnOff_mask_t values_mask = state ? channel_mask : (uint8_t) 0U;
250
251 // Send ESPS request
252 ESSATMAC_ErrCodes resp = EPSII_PDM_1_SetPC104GPIOReq(&fp_pdm_context, channel_mask, values_mask);
253 return (bool)(ESSATMAC_EC_OK == resp);
254
255}
