263bool cubeadcs_gen2_trigger_sm(const uint32_t event_id, const sm_event_ctx_t * const event_ctx)
264{
265 bool op_res;
266
267 sm_event_t event;
268
269 if ( osOK != osMutexAcquire(cubeadcs_gen2_sm_lock, osWaitForever) )
270 {
271 EXEH_HANDLE_EX(eEXEHSeverity_Warning, eEXEHModuleID_CUBEADCS_GEN2,
272 EXEH_CUBEADCS_GEN2_OS_OBJ_ERR)
273 }
274
275 event.base_event.id = event_id;
276
277 if ( NULL != event_ctx)
278 {
279 memcpy(&event.event_ctx, event_ctx, sizeof(sm_event_ctx_t));
280 }
281 else
282 {
283 memset(&event.event_ctx, 0, sizeof(sm_event_ctx_t));
284 }
285
286 op_res = HSM_TriggerEvent_raw(cubeadcs_gen2_sm_instance, &event.base_event);
287
288 if ( osOK != osMutexRelease(cubeadcs_gen2_sm_lock) )
289 {
290 EXEH_HANDLE_EX(eEXEHSeverity_Warning, eEXEHModuleID_CUBEADCS_GEN2,
291 EXEH_CUBEADCS_GEN2_OS_OBJ_ERR)
292 }
293
294 return op_res;
295}

530STATIC void fs_cubeadcs_gen2_task(void *arg)
531{
532 (void)cubeadcs_gen2_trigger_sm(eHSM_StdEvent_Reset, NULL);
533
534 static uint32_t wait_mark;
535
536 for(;;)
537 {
538 wait_mark = osKernelGetTickCount();
539
540 CUBEADCS_GEN2_TRACE_INFO("%s: %s\r", CUBEADCS_GEN2_DBG_HEAD, cubeadcs_gen2_get_sm_state()->pName);
541
542 if ( STATE_HANDLER_BUSY == fs_cubeadcs_gen2_state_handler(cubeadcs_gen2_get_sm_state()->state_id) )
543 {
544 osDelayUntil( (wait_mark + CUBEADCS_GEN2_SM_TASK_FAST_PERIOD_MS) );
545 }
546 else
547 {
548 (void)cubeadcs_gen2_trigger_sm(eHSM_StdEvent_Periodic, NULL);
549
550 osDelayUntil( (wait_mark + CUBEADCS_GEN2_SM_TASK_BASE_PERIOD_MS) );
551 }
552
553 TaskMonitor_IamAlive(TASK_MONITOR_CUBEADCS_GEN2);
554 vApplicationLowStackCheck(TASK_MONITOR_CUBEADCS_GEN2);
555 }
556}

565STATIC eStateHandlerStatus_t fs_cubeadcs_gen2_state_handler(const uint8_t sm_state)
566{
567 eStateHandlerStatus_t op_res = STATE_HANDLER_IDLE;
568
569 switch (sm_state)
570 {
571 case STATE_EVENTMARKERACQUISITION:
572 case STATE_CHECK:
573 case STATE_STATECHANGE:
574 case STATE_EVENTDOWNLOAD:
575 case STATE_FDIR:
576 {
577 state_handler[sm_state]();
578
579 op_res = STATE_HANDLER_BUSY;
580 break;
581 }
582 default:
583 {
584 break;
585 }
586 }
587
588 return op_res;
589}

603STATIC void fs_cubeadcs_gen2_state_handler_check(void)
604{
605 CUBEADCS_GEN2_TRACE_INFO("%s: State Handler Check\r", CUBEADCS_GEN2_DBG_HEAD);
606
607 op_status_t op_status_cache = {0};
608
609 op_status_cache.common_tlm.is_valid = true;
610 op_status_cache.app_tlm.is_valid = false;
611
612 if ((CUBEOBC_ERROR_OK == CubeADCS_Gen2_CommonFramework1_getBootStatusRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &op_status_cache.common_tlm.boot_status, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)) &&
613 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerCommon3_getPowerStateRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &op_status_cache.common_tlm.power_state, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)) &&
614 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerCommon3_getComponentErrorCodesRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &op_status_cache.common_tlm.comp_err_codes, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)))
615 {
616 dc_set_raw_data(DC_DATA_INPUT_INTERNAL, DC_DID_CUBEADCS_GEN2_OP_STATUS_COMMON_DATA, &op_status_cache.common_tlm, sizeof(DATA_CACHE_CUBEADCS_GEN2_OP_STATUS_COMMON_t));
617
618 op_status_cache.app_tlm.is_valid = true;
619
620 if ((TCTLM_COMMON_FRAMEWORK_1__APPLICATION_RUNNING == op_status_cache.common_tlm.boot_status.state) &&
621 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerControlProgram8_getAdcsRunModeRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &op_status_cache.app_tlm.run_mode, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)) &&
622 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerControlProgram8_getEstModeRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &op_status_cache.app_tlm.est_mode, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)) &&
623 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerControlProgram8_getConModeRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER,&op_status_cache.app_tlm.control_mode, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)) &&
