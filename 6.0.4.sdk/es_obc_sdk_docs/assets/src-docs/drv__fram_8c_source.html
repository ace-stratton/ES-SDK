54 GPIO_InitStruct.Pin = SPI4_FRAM_CHIP_SELECT_Pin;

60 HAL_GPIO_WritePin(SPI4_FRAM_CHIP_SELECT_GPIO_Port, SPI4_FRAM_CHIP_SELECT_Pin, GPIO_PIN_SET);

62 hspi4.Instance = SPI4;
63 hspi4.Init.Mode = SPI_MODE_MASTER;
64 hspi4.Init.Direction = SPI_DIRECTION_2LINES;
65 hspi4.Init.DataSize = SPI_DATASIZE_8BIT;
66 hspi4.Init.CLKPolarity = SPI_POLARITY_LOW;
67 hspi4.Init.CLKPhase = SPI_PHASE_1EDGE;
68 hspi4.Init.NSS = SPI_NSS_SOFT;
69 hspi4.Init.BaudRatePrescaler = SPI_BAUDRATEPRESCALER_4;
70 hspi4.Init.FirstBit = SPI_FIRSTBIT_MSB;
71 hspi4.Init.TIMode = SPI_TIMODE_DISABLE;
72 hspi4.Init.CRCCalculation = SPI_CRCCALCULATION_DISABLE;
73 hspi4.Init.CRCPolynomial = 7;
74 hspi4.Init.CRCLength = SPI_CRC_LENGTH_DATASIZE;
75 hspi4.Init.NSSPMode = SPI_NSS_PULSE_ENABLE;

77 if (HAL_OK != HAL_SPI_RegisterCallback(&hspi4, HAL_SPI_MSPINIT_CB_ID, FRAM_MspInit))

169bool FRAM_WriteData(uint32_t address, uint16_t size, uint8_t * dataBuff)
170{
171 uint8_t TxData[4] = {0x02, (uint8_t)(address >> 16), (uint8_t)(address >> 8), (uint8_t)(address)}; //Opcode READ (Read memory data)
172 bool bWriteSuccessful = false;
173
174 memcpy(ExtFram_TxBuffer,TxData, 4);
175 memcpy(&ExtFram_TxBuffer[4],dataBuff, size);
176
177 FRAM_WriteEnable();
178
179 for (uint8_t i = 0; i < EXT_FRAM_MAX_RETRY; i++)
180 {
181 (void)ExtFram_Transfer(ExtFram_TxBuffer, 4+size, dataBuff, 0);
182
183 bWriteSuccessful = FRAM_ReadData(address,size, ExtFram_RxBuffer[1]);
184
185 if (memcmp(&ExtFram_TxBuffer[4], ExtFram_RxBuffer[1], size) == 0)
186 {
187 break;
188 }
189 }
190
191 FRAM_WriteDisable();
192
193 return bWriteSuccessful;
194}

239static bool ExtFram_Transfer(uint8_t* txBuffer, uint16_t txSize, uint8_t* rxBuffer, uint16_t rxSize)
240{
241 HAL_StatusTypeDef statusTx = HAL_OK;
242 HAL_StatusTypeDef statusRx = HAL_OK;
243
244 HAL_GPIO_WritePin(SPI4_FRAM_CHIP_SELECT_GPIO_Port, SPI4_FRAM_CHIP_SELECT_Pin, GPIO_PIN_RESET);
245
246 if (txSize > 0U)
247 {
248 statusTx = HAL_SPI_Transmit(&hspi4, txBuffer, txSize, EXT_FRAM_TIMEOUT_MS);
249 }
250
251 if (rxSize > 0U)
252 {
253 statusRx = HAL_SPI_Receive(&hspi4, rxBuffer, rxSize, EXT_FRAM_TIMEOUT_MS);
254 }
255
256 HAL_GPIO_WritePin(SPI4_FRAM_CHIP_SELECT_GPIO_Port, SPI4_FRAM_CHIP_SELECT_Pin, GPIO_PIN_SET);
257
258 return ((statusTx == HAL_OK) && (statusRx == HAL_OK)) ? (true) : (false);
259}

276static uint8_t ExtFram_SafeRead(uint8_t* txBuffer, uint16_t txSize, uint8_t* rxBuffer, uint16_t rxSize)
277{
278 int readStatus;
279 uint8_t retStat = 0;
280
281 if (rxSize <= FRAM_RX_BUFF_SIZE)
282 {
283 for (uint8_t i = 0; i < EXT_FRAM_MAX_RETRY; i++)
284 {
285 ExtFram_Transfer(txBuffer, txSize, ExtFram_RxBuffer[0], rxSize);
286
287 if (rxSize <= 0)
288 {
289 // if no bytes are going to be read, there is nothing to compare
290 break;
291 }
292
293 ExtFram_Transfer(txBuffer, txSize, ExtFram_RxBuffer[1], rxSize);
294
295 readStatus = memcmp(ExtFram_RxBuffer[0], ExtFram_RxBuffer[1], rxSize);
296
297 if (readStatus == 0)
298 {
299 memcpy(rxBuffer, ExtFram_RxBuffer[0], rxSize);
300 retStat = 1;
301 break;
302 }
303 }
304 }
305
306 return retStat;
307}
