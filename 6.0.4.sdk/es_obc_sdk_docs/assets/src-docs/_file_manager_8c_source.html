158static inline bool fs_isHandleFree(const FileManagerFHnd fhnd)
159{
160 CRIT_ASSERT(fhnd < FILEMANAGER_MAX_FILE_HANDLE_COUNT);
161
162 filemng_lock_acquire(aFileMngMutex);
163 bool ret = (bool)(aFileHandles[fhnd].obj.fs == (FATFS *)NULL);
164 filemng_lock_release(aFileMngMutex);
165
166 return ret;
167}
168
169static bool fs_getFreeHandle(FileManagerFHnd *const pHnd)
170{
171 bool ret = false;
172
173 CRIT_ASSERT(pHnd != (FileManagerFHnd *)NULL);
174
175 filemng_lock_acquire(aFileMngMutex);
176
177 *pHnd = FILEMANAGER_INVALID_FILE_HANDLE;
178
179 for (uint8_t i = 0; i < CDEF_ELEMENT_COUNT(aFileHandles); i++)
180 {
181 if (fs_isHandleFree(i))
182 {
183 *pHnd = i;
184 ret = true;
185 break;
186 }
187 }
188
189 filemng_lock_release(aFileMngMutex);
190
191 return ret;
192}

201static uint32_t fs_calcBufChecksum(const uint8_t *const pBuf, const uint32_t u32BufSize, const uint32_t u32StartCs)
202{
203 uint32_t u32Res = u32StartCs;
204
205 CRIT_ASSERT((pBuf != NULL) && (u32BufSize > (uint32_t)0));
206
207 for (uint32_t i = 0; i < u32BufSize; i++)
208 {
209 u32Res += pBuf[i];
210 }
211
212 return u32Res;
213}

215static void fs_release_all_handles(void)
216{
217 filemng_lock_acquire(aFileMngMutex);
218
219 for (uint32_t i = 0; i < CDEF_ELEMENT_COUNT(aFileHandles); i++)

277static FRESULT scan_files(const TCHAR *const file_path, const TCHAR *const pattern, const scan_file_ctx_t *const p_scan_file_ctx)
278{
279 static const TCHAR stack_fname[] = "___fstack.tmp";
280 TCHAR current_folder_path[FF_LFN_BUF] = { '\0' };
281 TCHAR tmp_stack_full_path[FF_LFN_BUF] = { '\0' };
282 file_stack_ctx_t fstack_hnd;
283 FF_DIR dinfo;
284 FILINFO finfo;
285 FRESULT res = FR_INVALID_PARAMETER;
286
287 CRIT_ASSERT(NULL != file_path);
288 CRIT_ASSERT(NULL != p_scan_file_ctx);
289 CRIT_ASSERT(NULL != p_scan_file_ctx->p_file_action_handler);
290
291 res = file_stack_open(&fstack_hnd, stack_fname, FF_LFN_BUF * sizeof(TCHAR));
292
293 if (FR_OK != res)
294 {
295 return res;
296 }
297
298 res = file_stack_push(&fstack_hnd, file_path);
299
300 while ((file_stack_get_elem_cnt(&fstack_hnd) > 0U) && (FR_OK == res))
301 {
302 res = file_stack_pop(&fstack_hnd, current_folder_path);
303
304 if (FR_OK != res)
305 {
306 break;
307 }
308
309 current_folder_path[FF_LFN_BUF - 1] = '\0';
310
311 // find all dirs with any name first and add them to the stack
312 res = SdMngr_f_findfirst(&dinfo, &finfo, current_folder_path, "*");
313
314 if ((FR_OK == res) && ('\0' != finfo.fname[0]))
315 {
316 do
317 {
318 bool is_path_len_valid;
319
320 (void)strncpy(tmp_stack_full_path, current_folder_path, FF_LFN_BUF);
321 tmp_stack_full_path[FF_LFN_BUF - 1] = '\0';
322
323 size_t fname_sz = strlen(finfo.fname);
324 is_path_len_valid = (strlen(tmp_stack_full_path) + strlen(PATH_DELIMITER) + fname_sz) < FF_LFN_BUF;
325
326 if (true == is_path_len_valid)
327 {
328 if (finfo.fattrib == (finfo.fattrib & AM_DIR))
329 {
330 (void)strcat(tmp_stack_full_path, PATH_DELIMITER);
331 (void)strncat(tmp_stack_full_path, finfo.fname, fname_sz);
332 tmp_stack_full_path[FF_LFN_BUF - 1] = '\0';
333
334 res = file_stack_push(&fstack_hnd, tmp_stack_full_path);
335
336 if (FR_OK != res)
337 {
338 break;
339 }
340 }
341 }
342 else
343 {
344 ES_TRACE_WARN("file '%s' in folder '%s' hit the name limit size of %d bytes\n", finfo.fname, current_folder_path, FF_LFN_BUF);
345 }
346
347 res = SdMngr_f_findnext(&dinfo, &finfo);
348 } while ((FR_OK == res) && ('\0' != finfo.fname[0]));
349 }
350
351 SdMngr_f_closedir(&dinfo);
352
353 // list all files/dirs matching the pattern and add them to the dump list
354 res = SdMngr_f_findfirst(&dinfo, &finfo, current_folder_path, pattern);
355
356 if ((FR_OK == res) && ('\0' != finfo.fname[0]))
357 {
358 do
359 {
360 bool is_path_len_valid;
361
362 (void)strncpy(tmp_stack_full_path, current_folder_path, FF_LFN_BUF);
363 tmp_stack_full_path[FF_LFN_BUF - 1] = '\0';
364
365 size_t fname_sz = strlen(finfo.fname);
366 is_path_len_valid = (strlen(tmp_stack_full_path) + strlen(PATH_DELIMITER) + fname_sz) < FF_LFN_BUF;
367
368 // do not call the user handler for the temporary stack file because
369 // it will be deleted once it is used and the file will be reported as
370 // part of the FATFS iteration but will be inexistent after the function finishes
371 if ((true == is_path_len_valid)
372 && (0U != strcmp(stack_fname, finfo.fname)))
373 {
374 // call user action
375 (p_scan_file_ctx->p_file_action_handler)
376 ( &dinfo
377 , &finfo
378 , current_folder_path
379 , p_scan_file_ctx->p_custom_info);
380 }
381
382 res = SdMngr_f_findnext(&dinfo, &finfo);
383 } while ((FR_OK == res) && ('\0' != finfo.fname[0]));
384 }
385
386 res = SdMngr_f_closedir(&dinfo);
387 }
388
389 res = file_stack_close(&fstack_hnd, true);
390
391 return res;
392}

469eFileManagerOpResult_t fileManager_FGetCheckSum(const TCHAR *const filePath, uint32_t *const pu32Checksum)
470{
471 eFileManagerOpResult_t res = FR_INVALID_PARAMETER;
472 FileManagerFHnd fHnd = FILEMANAGER_INVALID_FILE_HANDLE;
473 FIL *physHnd = (FIL *)NULL;
474 uint8_t au8ReadBuf[FILEMANAGER_CS_BUF_SIZE_BYTES] = { 0 };
475 UINT bytesRead = (UINT)0;
476 UINT remainingBytes = (UINT)0;
477 DEF_LOCAL_FILE_PATH_BUFFER;
478
479 // Validate the parameters
480 if ((NULL == filePath) || (NULL == pu32Checksum))
481 {
482 return FR_INVALID_PARAMETER;
483 }
484
485 if (false == fs_formatFullFilePath(filePath, GET_LOCAL_FILE_PATH_BUFFER,
486 sizeof(GET_LOCAL_FILE_PATH_BUFFER)))
487 {
488 return FR_INVALID_PARAMETER;
489 }
490
491 if (!fs_getFreeHandle(&fHnd))
492 {
493 return FR_TOO_MANY_OPEN_FILES;
494 }
495
496 // Open the file
497 physHnd = fs_getPhysHandle(fHnd);
498 res = SdMngr_f_open(physHnd,
499 (const TCHAR *)GET_LOCAL_FILE_PATH_BUFFER,
500 FA_READ | FA_OPEN_EXISTING);
501 if (res != FR_OK)
502 {
503 return res;
504 }
505
506 // Get the Checksum
507 *pu32Checksum = (uint32_t)0;
508 remainingBytes = SdMngr_f_size(physHnd);
509 // a non-zero value is needed to enable entering the loop
510 // condition below...
511 bytesRead = remainingBytes;
512
513 while ((remainingBytes > 0) && (FR_OK == res) &&
514 (bytesRead > 0))
515 {
516 // try to read data in chunks...
517 res = SdMngr_f_read(physHnd, au8ReadBuf,
518 sizeof(au8ReadBuf), &bytesRead);
519
520 if (FR_OK == res)
521 {
522 *pu32Checksum = fs_calcBufChecksum(
523 au8ReadBuf, bytesRead, *pu32Checksum);
524
525 if (remainingBytes >= bytesRead)
526 {
527 remainingBytes -= bytesRead;
528 }
529 else
530 {
531 // this should not happen!
532 res = FR_INT_ERR;
533 break;
534 }
535 }
536 }
537
538 SdMngr_f_close(physHnd);
539
540 return res;
541}

543eFileManagerOpResult_t fileManager_FOpen(const TCHAR *const filePath,
544 const uint8_t mode,
545 FileManagerFHnd *const pFileHandle)
546{
547 FRESULT res = FR_INVALID_PARAMETER;
548 DEF_LOCAL_FILE_PATH_BUFFER;
549
550 if ((NULL == filePath)
551 || (NULL == pFileHandle)
552 || (0 != (mode & ((uint8_t)(~eFileManagerFileModeId_MASK))))
553 || (false == fs_formatFullFilePath(filePath, GET_LOCAL_FILE_PATH_BUFFER, sizeof(GET_LOCAL_FILE_PATH_BUFFER))))
554 {
555 return FR_INVALID_PARAMETER;
556 }
557
558 if (true == fs_getFreeHandle(pFileHandle))
559 {
560 res = SdMngr_f_open(fs_getPhysHandle(*pFileHandle), GET_LOCAL_FILE_PATH_BUFFER, mode);
561 }
562 else
563 {
564 res = FR_TOO_MANY_OPEN_FILES;
565 }
566
567 return res;
568}

570eFileManagerOpResult_t fileManager_FRead(const FileManagerFHnd fileHandle,
571 const uint32_t u32Pos,
572 uint8_t *const pu8Buf,
573 const uint32_t u32BufSize,
574 uint32_t *const pu32BytesRead)
575{
576 FRESULT res = FR_INVALID_PARAMETER;
577 FIL *physHnd = (FIL *)NULL;
578
579 // Validate the parameters
580 if ((NULL == pu8Buf) ||
581 (NULL == pu32BytesRead) ||
582 (false == fs_isHandleValid(fileHandle)) ||
583 (u32BufSize == (uint32_t)0))
584 {
585 return FR_INVALID_PARAMETER;
586 }
587
588 // handle needs to have been opened before using this function
589 if (true == fs_isHandleFree(fileHandle))
590 {
591 return FR_INVALID_OBJECT;
592 }
593
594 // Read the file
595 *pu32BytesRead = 0U;
596 physHnd = fs_getPhysHandle(fileHandle);
597 if (u32Pos <= SdMngr_f_size(physHnd))
598 {
599 res = SdMngr_f_lseek(physHnd, u32Pos);
600
601 if (FR_OK == res)
602 {
603 res = SdMngr_f_read(physHnd, (void *)pu8Buf, (UINT)u32BufSize, (UINT *)pu32BytesRead);
604 }
605 }
606
607 return res;
608}

724eFileManagerOpResult_t fileManager_FDumpDirListToFile(const TCHAR *const filePath,
725 const TCHAR *const pattern)
726{
727
728 FRESULT res = FR_INVALID_PARAMETER;
729 FIL physHnd;
730 DEF_LOCAL_FILE_PATH_BUFFER;
731
732 // Validate parameters
733 if ((NULL == filePath)
734 || (NULL == pattern)
735 || (false == fs_formatFullFilePath(filePath, GET_LOCAL_FILE_PATH_BUFFER, sizeof(GET_LOCAL_FILE_PATH_BUFFER))))
736 {
737 return FR_INVALID_PARAMETER;
738 }
739
740 // Open the dump file
741 res = SdMngr_f_open(&physHnd, GET_LOCAL_FILE_PATH_BUFFER, FA_WRITE | FA_CREATE_ALWAYS);
742
743 // Start dump directory list
744 if (FR_OK == res)
745 {
746 f_printf(&physHnd, "file_name,size(in bytes),attributes,timestamp\r\n");
747
748 for (uint8_t p_id = 0; p_id < (uint8_t)SD_MNGR_PARTITIONS_CNT; p_id++)
749 {
750 char const *p_mnt_name = SdMngr_get_mnt_point_name(p_id);
751
752 dirlist_ctx_t dirlist_ctx = { .h_dirlist_file = &physHnd };
753
754 scan_file_ctx_t scan_file_ctx = { .p_custom_info = (void *)&dirlist_ctx,
755 .p_file_action_handler = &dirlist_file_action_handler };
756
757 if (NULL != p_mnt_name)
758 {
759 res = scan_files(p_mnt_name, pattern, &scan_file_ctx);
760 }
761 }
762
763 (void)SdMngr_f_close(&physHnd);
764 }
765
766 return res;
767}

784eFileManagerOpResult_t fileManager_FGetSize(const TCHAR *const filePath, uint32_t *const pu32Size)

786 FRESULT res = FR_INVALID_PARAMETER;
