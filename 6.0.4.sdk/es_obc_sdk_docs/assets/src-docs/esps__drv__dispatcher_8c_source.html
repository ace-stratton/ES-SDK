48// Add a message to the dispatch queue.
49void MAC_DISP_bDispatchQueueAdd(sMACDispatcher_t * fp_psDisp, sMACRXPoolEntry_t * fp_psEntry)
50{
51 // This function should be called from ISR only.
52 if((fp_psDisp != NULL) && (fp_psEntry != NULL))
53 {
54 fp_psEntry->psNext = NULL;
55
56 if(fp_psDisp->sQueue.psTail == NULL)
57 {
58 fp_psDisp->sQueue.psTail = fp_psEntry;
59 }
60 else
61 {
62 fp_psDisp->sQueue.psTail->psNext = fp_psEntry;
63 fp_psDisp->sQueue.psTail = fp_psEntry;
64 }
65
66 if(fp_psDisp->sQueue.psHead == NULL)
67 {
68 fp_psDisp->sQueue.psHead = fp_psEntry;
69 }
70
71 fp_psDisp->u32RxMsgCount++;
72
73 // Notify dispatcher thread about new message
74 MAC_DISP_vDispatcherNotif(fp_psDisp->eMACId);
75
76 }
77}

79// Return the first message from the dispatch queue.
80sMACRXPoolEntry_t *MAC_DISP_bDispatchQueueTop(sMACDispatcher_t * fp_psDisp)
81{
82 sMACRXPoolEntry_t *psEntry = NULL;
83
84 if(fp_psDisp != NULL)

86 psEntry = fp_psDisp->sQueue.psHead;

88
89 return psEntry;

92// Remove the first message from the queue
93void MAC_DISP_bDispatchQueuePop(sMACDispatcher_t * fp_psDisp)
94{
95 if(fp_psDisp != NULL)
96 {
97 if(fp_psDisp->sQueue.psHead != NULL)
98 {
99 if(fp_psDisp->sQueue.psHead == fp_psDisp->sQueue.psTail)
100 {
101 fp_psDisp->sQueue.psTail = NULL;
102 fp_psDisp->sQueue.psHead = NULL;
103 }
104 else
105 {
106 fp_psDisp->sQueue.psHead = fp_psDisp->sQueue.psHead->psNext;
107 }
108
109 if(fp_psDisp->u32RxMsgCount > 0U)
110 --fp_psDisp->u32RxMsgCount;
111
112 MACDRV_DBG_PRINT(" --> RxQ %ld rem cnt: %ld\r\n", (uint32_t) fp_psDisp->eMACId,
113 (uint32_t) fp_psDisp->u32RxMsgCount);
114 }
115 }
116}

157// Process all queued messages.
158STATIC void fs_vProcessQueuedMessages(sMACInterfaceState_t * fp_psIntf)
159{
160 sMACRXPoolEntry_t *psMsg = NULL;
161
162 uint32_t u32CRC;
163
164 bool bProcessed;
165
166 if(fp_psIntf != NULL)
167 {
168 do
169 {
170 // New message
171 psMsg = MAC_DISP_bDispatchQueueTop(&fp_psIntf->sDisp);
172
173 if(psMsg)
174 {
175 // Check CRC
176 u32CRC =
177 MACCRC32_Calc32((uint8_t *) & psMsg->sMessage, ESSATMAC_HEADER_SIZE + psMsg->sMessage.u8DataSize,
178 CRC32_INIT_VALUE);
179
180 // LSB network byte order is used, hence we can use the in-memory representation of the CRC to directly compare to
181 // the received value on the bus
182 if(u32CRC != psMsg->sMessage.u32CRC)
183 {
184 MAC_Stats_triggerEvent(fp_psIntf->macStats, eMACStatId_IncorrectCrcMsgCntr);
185
186 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_CRCMismatch);
187 }
188 else
189 {
190 bProcessed = MAC_bProtocolProcessMessage(fp_psIntf->sDisp.eMACId, &psMsg->sMessage);
191 if(!bProcessed)
192 {
193 EXEH_HANDLE(eEXEHSeverity_Warning, (int32_t) eMACException_NotProcessed);
194 }
195 else
196 MAC_Stats_triggerEvent(fp_psIntf->macStats, eMACStatId_RxValidFramesCntr);
197 }
198
199 // Remove the messages from the queue.
200 MAC_DISP_bDispatchQueuePop(&fp_psIntf->sDisp);
201
202 // Release the message.
203 MAC_POOL_vPoolRXFree(fp_psIntf, psMsg);
204 }
205 }
206 while(psMsg != NULL);
207 }
