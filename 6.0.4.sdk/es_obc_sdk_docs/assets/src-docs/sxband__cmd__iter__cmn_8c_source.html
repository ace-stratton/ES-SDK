34{
35 CRIT_ASSERT(NULL != p_cmd_hdr);

37
38 uint16_t calc_cmd_crc = crc16_ccitt_table((const uint8_t *) p_cmd_hdr, sizeof(cmd_list_command_hdr_t));
39 calc_cmd_crc = crc16_ccitt_table_ex(p_data, p_cmd_hdr->size, calc_cmd_crc);
