65void cubeadcs_persist_get_nvm_stored_status(CubeADCS_Status_t *psDst)
66{
67 CRIT_ASSERT(NULL != psDst);
68
69 cubeadcs_status_t sTmp_status;

71 Nvm_GetBlockById(NVM_BLOCK_CUBEADCS_STATUS, &sTmp_status);

146bool cubeadcs_persist_set_nvm_sys_state_cfg(cubeADCS_sys_state_cfg_t *psSrc)
147{
148 bool res = true;
149 CRIT_ASSERT(NULL != psSrc);
150
151 cubeADCS_sys_state_cfg_t sTmp_state_cfg;
152
153 memcpy(&sTmp_state_cfg, psSrc, sizeof(cubeADCS_sys_state_cfg_t));
154
155 Nvm_SetBlockById(NVM_BLOCK_CUBEADCS_SYS_STATE_CFG, &sTmp_state_cfg, NVMSETBLOCK_STORE_IMMEDIATELY);
156
157 memset(&sTmp_state_cfg, 0x00, sizeof(cubeADCS_sys_state_cfg_t));
158
159 Nvm_GetBlockById(NVM_BLOCK_CUBEADCS_SYS_STATE_CFG, &sTmp_state_cfg);
160
161 if(memcmp(&sTmp_state_cfg, psSrc, sizeof(cubeADCS_sys_state_cfg_t)))
162 {
163 res = true;
164 }
165 else
166 {
167 res = false;
168 }
169
170 return res;
171}
