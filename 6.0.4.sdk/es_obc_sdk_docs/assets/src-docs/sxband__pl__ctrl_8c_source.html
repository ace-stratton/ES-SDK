84static pl_op_status_t sxband_pl_ctrl_init(const pl_instance_id_t instance_id, pl_config_t * const p_init_cfg, const p_pl_event_notify_t p_event_notify_cb)
85{
86 pl_op_status_t res = PL_OP_STATUS_REJECT;
87 (void) p_init_cfg;
88
89 if (is_valid_sxband_instance(instance_id))
90 {
91 p_sxband_state_notify = p_event_notify_cb;
92 update_state_and_notify(instance_id, PL_STATE_STOPPED);
93 res = PL_OP_STATUS_OK;
94 }
95
96 return res;
97}

115static pl_op_status_t sxband_pl_ctrl_start(const pl_instance_id_t instance_id, pl_config_t * const p_start_cfg)
116{
117 pl_op_status_t res = PL_OP_STATUS_REJECT;
118
119#if defined(SXBAND_SCHEDULER_ENABLED)
120 if ((is_valid_sxband_instance(instance_id)) && (false == sxband_sched_is_file_seq_active()))
121#else // #if defined(SXBAND_SCHEDULER_ENABLED)
122 if (is_valid_sxband_instance(instance_id))
123#endif // #if defined(SXBAND_SCHEDULER_ENABLED)
124 {
125 (void) eps_ctrl_set_channel_output(EPSCTRL_POWER_OUT_X_BAND_EN, true);
126
127 update_state_and_notify(instance_id, PL_STATE_STARTED);
128
129#if defined(SXBAND_SCHEDULER_ENABLED)
130 const char *p_sched_file_name = NULL;
131
132 // Note:
133 // The scheduling args should contain a zero-terminated file name with SXBand commands to execute.
134 // The file has to have been uploaded on the OBC SD card before starting the payload by the main scheduler.
135 if (NULL != p_start_cfg)
136 {
137 p_sched_file_name = (char *) ((pl_args_t *)(p_start_cfg->p_config_data))->pl_args;
138 }
139
140 if ((NULL != p_sched_file_name) && ('\0' != p_sched_file_name[0U]))
141 {
142 sxband_sched_status_t seq_status = sxband_sched_start_file_seq(p_sched_file_name, strlen(p_sched_file_name) + 1, &on_sxband_notif);
143
144 if (SXBAND_SCHED_STATUS_OK == seq_status)
145 {
146 res = PL_OP_STATUS_OK;
147 }
148 }
149 else
150 {
151 // this is just a payload start without a schedule file
152 res = PL_OP_STATUS_OK;
153 }
154#else // #if defined(SXBAND_SCHEDULER_ENABLED)
155 res = PL_OP_STATUS_OK;
156#endif // #if defined(SXBAND_SCHEDULER_ENABLED)
157 }
158
159 return res;
160}

162static pl_op_status_t sxband_pl_ctrl_stop(const pl_instance_id_t instance_id, const pl_op_stop_mode_t stop_mode)
163{
164 pl_op_status_t res = PL_OP_STATUS_REJECT;
165
166 if (is_valid_sxband_instance(instance_id))
167 {
168 // stop the power regardless of the mode
169 (void) stop_mode;
170
171#if defined(SXBAND_SCHEDULER_ENABLED)
172 sxband_sched_stop_file_seq();
173#endif // #if defined(SXBAND_SCHEDULER_ENABLED)
174
175 sx_band_stop();
176 (void) eps_ctrl_set_channel_output(EPSCTRL_POWER_OUT_X_BAND_EN, false);
177
178 update_state_and_notify(instance_id, PL_STATE_STOPPED);
179
180 res = PL_OP_STATUS_OK;
181 }
182
183 return res;
184}
