96static CSP_DEFINE_TASK(fs_csp_service_thread)
97{
98 csp_socket_t* socket = csp_socket(CSP_SO_NONE);
99 csp_bind(socket, CSP_CMP);
100 csp_bind(socket, CSP_PING);
101 csp_bind(socket, CSP_UPTIME);
102 csp_listen(socket, FS_CSP_LISTEN_BACKLOG_SZ);
103 csp_conn_t* conn;
104
105 while (1)
106 {
107 conn = csp_accept(socket, FS_CSP_ACCEPT_TIMEOUT);
108
109 if (NULL != conn)
110 {
111 csp_packet_t* packet;
112
113 while (NULL != (packet = csp_read(conn, FS_CSP_READ_TIMEOUT)))
114 {
115 csp_service_handler(conn, packet);
116 }
117
118 csp_close(conn);
119 }
120 }
121
122 return CSP_TASK_RETURN;
123}

132static CSP_DEFINE_TASK(fs_csp_can_dispatch_thread)
133{
134 can_msg_t msg;
135
136 while (1)
137 {
138 uint32_t flags = osThreadFlagsWait(
139 FS_CSP_DISPATCH_NOTIFY_FLAG, osFlagsWaitAny, FS_CSP_DISPATCH_WAIT_TIMEOUT);
140
141 if (FS_CSP_DISPATCH_NOTIFY_FLAG == flags)
142 {
143 while (true == can_rx(&msg, CAN_ID_EXTENDED))
144 {
145 csp_can_rx(&fs_iface_can, msg.id, msg.data, msg.data_sz, NULL);
146 }
147 }
148 }
149
150 return CSP_TASK_RETURN;
151}

162static bool fs_can_rx_callback(const can_msg_t* const msg)
163{
164 if ((NULL != fs_can_dispatch_h) && (NULL != msg))
165 {
166 // check if this is the last message from a packet.
167 // check csp_if_can.c for more info. if we don't use this we don't know when
168 // a given message has been fully transmitted. This leads to too many interrupts and
169 // the communication gets slower.
170 if (0 == CFP_REMAIN(msg->id))
171 {
172 osThreadFlagsSet(fs_can_dispatch_h, FS_CSP_DISPATCH_NOTIFY_FLAG);
173 }
174 }
175
176 return false;
177}

191static int fs_can_transmit(void* driver_data, uint32_t id, const uint8_t* data, uint8_t data_sz)
192{
193 (void)driver_data;
194
195 int ret = CSP_ERR_TX;
196 can_msg_t msg;
197
198 if ((NULL != data) && (sizeof(msg.data) >= data_sz))
199 {
200 msg.id_type = CAN_ID_EXTENDED;
201 msg.id = id;
202 msg.data_sz = data_sz;
203
204 memcpy(msg.data, data, data_sz);
205
206 if (true == can_tx(&msg))
207 {
208 ret = CSP_ERR_NONE;
209 }
210 else
211 {
212 ret = CSP_ERR_DRIVER;
213 }
214 }
215
216 return ret;
217}

220bool csp_service_init(void)
221{
222 csp_conf_t config;
223
224 csp_conf_get_defaults(&config);
225
226 config.address = CSP_SERVICE_CFG_OWN_ADDRESS;
227 config.buffers = CSP_SERVICE_CFG_NUM_BUFFERS;
228 config.buffer_data_size = CSP_SERVICE_CFG_BUFFER_SZ;
229
230 if (CSP_ERR_NONE != csp_init(&config))
231 {
232 return false;
233 }
234
235 can_reg_rx(CAN_ID_EXTENDED, fs_can_rx_callback);
236
237 if (CSP_ERR_NONE != csp_can_add_interface(&fs_iface_can))
238 {
239 return false;
240 }
241
242 if (CSP_ERR_NONE != csp_thread_create(
243 fs_csp_can_dispatch_thread,
244 "csp_can_disp",
245 FS_CSP_CAN_DISPATCH_TH_STACK_SZ,
246 NULL,
247 osPriorityNormal,
248 &fs_can_dispatch_h))
249 {
250 return false;
251 }
252
253 if (CSP_ERR_NONE != csp_rtable_set(
254 CSP_SERVICE_CFG_OWN_ADDRESS, CSP_ID_HOST_SIZE, &csp_if_lo, CSP_NO_VIA_ADDRESS))
255 {
256 return false;
257 }
258
259 if (CSP_ERR_NONE != csp_rtable_set(
260 CSP_DEFAULT_ROUTE, CSP_ID_HOST_SIZE, &fs_iface_can, CSP_NO_VIA_ADDRESS))
261 {
262 return false;
263 }
264
265 if (CSP_ERR_NONE != csp_route_start_task(
266 FS_CSP_ROUTER_TH_STACK_SZ, osPriorityNormal))
267 {
268 return false;
269 }
270
271 if (CSP_ERR_NONE != csp_thread_create(
272 fs_csp_service_thread,
273 "csp_service",
274 FS_CSP_SERVICE_TH_STACK_SZ,
275 NULL,
276 osPriorityNormal,
277 &fs_csp_service_h))
278 {
279 return false;
280 }
281
282 return true;
283}
