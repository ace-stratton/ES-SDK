60/* DESCRIPTION : queue_flush */
61/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
62/* RETURN VALUE : void (None) */
63/* DESIGN INFORMATION : Flush queue contents */
64/* ========================================================================== */
65void queue_flush(tQueue *q) {
66 int index;
67
68 for (index = 0; index < q->size; index++) {
69 q->data[index] = 0;
70 }
71 q->tail = q->head = 0;
72}
73

75/* DESCRIPTION : queue_quick_flush */
76/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
77/* RETURN VALUE : void (None) */
78/* DESIGN INFORMATION : Flush queue without contents wiping */
79/* ========================================================================== */
80void queue_quick_flush(tQueue *q) {
81 q->tail = q->head = 0;
82}

84/* ========================================================================== */
85/* DESCRIPTION : queue_init */
86/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
87/* tQueuePData pBuf - buffer pointer */
88/* tQueueSize qSize - buffer size */
89/* RETURN VALUE : void (None) */
90/* DESIGN INFORMATION : Initialise the queue contents */
91/* ========================================================================== */
92void queue_init(tQueue *q, tQueuePData pBuf, tQueueSize qSize) {
93 if ((qSize < 1) || (pBuf == NULL)) {
94 q->size = 0;
95 q->data = NULL;
96 } else {
97 int cnt = (sizeof(tQueueSize) * 8) - 1;
98 int mask = (1 << ((sizeof(tQueueSize) * 8) - 1));
99 int size = qSize;
100
101 /* Align buffer size to nearest power of two */
102 while (0 == (size & mask)) {
103 size <<= 1;
104 cnt--;
105 }
106 q->size = 1 << cnt;;
107 q->data = pBuf;
108 queue_flush(q);
109 }
110}

112/* ========================================================================== */
113/* DESCRIPTION : queue_data_len */
114/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
115/* RETURN VALUE : Size of the data in the queue */
116/* DESIGN INFORMATION : Get data size in the queue */
117/* ========================================================================== */
118tQueueSize queue_data_len(tQueue *q) {
119 tQueueSize qLen = 0;
120
121 CRIT_ASSERT(q->data != NULL);
122 CRIT_ASSERT(q->size != 0);
123 if (q->head != q->tail) {
124 if (q->tail > q->head) {
125 qLen = q->tail - q->head;
126 } else {
127 qLen = q->tail + q->size - q->head;
128 }
129 } /* else No data in the queue */
130
131 return (qLen);
132}

134/* ========================================================================== */
135/* DESCRIPTION : queue_data_free */
136/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
137/* RETURN VALUE : Free space in the queue */
138/* DESIGN INFORMATION : Get free space in the queue */
139/* ========================================================================== */
140tQueueSize queue_data_free(tQueue *q) {
141 return (q->size - 1 - queue_data_len(q));
142}

144/* ========================================================================== */
145/* DESCRIPTION : queue_put */
146/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
147/* tQueueData qData - data to store */
148/* RETURN VALUE : void (None) */
149/* DESIGN INFORMATION : Puts data in the queue */
150/* ========================================================================== */
151void queue_put(tQueue *q, const tQueueData qData) {
152 CRIT_ASSERT(q->data != NULL);
153 CRIT_ASSERT(q->size != 0);
154 q->data[q->tail++] = qData;
155 q->tail &= (q->size - 1);
156}

159/* DESCRIPTION : queue_is_empty */
160/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
161/* RETURN VALUE : BOOL - true if there is no data */
162/* DESIGN INFORMATION : Checks if the queue is empty */
163/* ========================================================================== */
164BOOL queue_is_empty(tQueue *q) {
165 return (queue_data_len(q) == 0 ? cTrue : cFalse);

168/* ========================================================================== */
169/* DESCRIPTION : queue_is_full */
170/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
171/* RETURN VALUE : BOOL - true if queue is full */
172/* DESIGN INFORMATION : Checks if the queue is full */
173/* ========================================================================== */
174BOOL queue_is_full(tQueue *q) {
175 return (queue_data_free(q) == 0);
176}

178/* ========================================================================== */
179/* DESCRIPTION : queue_get */
180/* PARAMETERS (Type,Name,Min,Max) : tQueue *q - queue context */
181/* RETURN VALUE : tQueueData - Data from the queue, */
182/* 0xFF if empty */
183/* DESIGN INFORMATION : Returns data byte from the queue */
184/* ========================================================================== */
185int queue_get(tQueue *q) {
186 tQueueData qData = 0xFF;
187
188 if (!queue_is_empty(q)) {
189 qData = q->data[q->head++];
190 q->head &= (q->size - 1);
191 } /* else No data in the queue */
192 else
193 return -1;
194
195 return ((int)qData);
196}
