81void actuator_control_service_init(void)
82{
83 EXEH_INIT_MODULE_FILENAME();
84
85 /* Make sure the task logic doesn't run until told to by the ADCS */
86 activeState = ACTUATOR_CONTROL_SERVICE_STATE_INACTIVE;
87
88 /* Spawn the task */
89 actuator_control_service_Task_TaskHandle = osThreadNew(fs_actuator_control_service_Task, NULL, &actuator_control_service_Task_attributes);
90 CRIT_ASSERT( actuator_control_service_Task_TaskHandle );
91
92 /* Notify task monitor that the actuator control service task is initialized */
93 TaskMonitor_TaskInitialized (TASK_ACTUATOR_CONTROL_SERVICE);
94 task_mon_set_task_policy(TASK_ACTUATOR_CONTROL_SERVICE, TASK_MON_POLICY_MONITOR);
95}

97void actuator_control_service_start(void)
98{
99 /*
100 * (this is to be called when ADCS switches from IDLE to any mode)
101 */
102
103 if (ACTUATOR_CONTROL_SERVICE_STATE_ACTIVE != activeState)
104 {
105 /* Reactivate the task logic */
106 activeState = ACTUATOR_CONTROL_SERVICE_STATE_ACTIVE;
107
108 /* Initialize physical magnetorquer hardware */
109 MCU_MagnetorqersInit();
110 }
111}

113void actuator_control_service_stop(void)
114{
115 /*
116 * (this is to be called when ADCS switches from any mode to IDLE)
117 * stop function should also force all actuators to 0 (or turn off)
118 */
119
120 if (ACTUATOR_CONTROL_SERVICE_STATE_INACTIVE != activeState)
121 {
122 /* Make the task dormant */
123 activeState = ACTUATOR_CONTROL_SERVICE_STATE_INACTIVE;
124
125 /* Clear magnetorquer outputs */
126 fs_magnetorquers_stop();
127
128 /* Deinitialize physical magnetorquer hardware */
129 MCU_MagnetorqersDeInit();
130 }
131}

150static void fs_actuator_control_service_Task(void *arg)
151{
152 /*
153 * this should only call each different type of actuator handler
154 * the rest is done inside the handler itself, keep it clean
155 */
156
157 for (;;)
158 {
159 TaskMonitor_IamAlive(TASK_ACTUATOR_CONTROL_SERVICE);
160
161 switch (activeState)
162 {
163 case ACTUATOR_CONTROL_SERVICE_STATE_ACTIVE:
164 {
165 /* Actuate magnetorquers */
166 fs_magnetorquers_actuate();
167
168 /* Add any future actuators here */
169
170 break;
171 }
172 case ACTUATOR_CONTROL_SERVICE_STATE_INACTIVE:
173 {
174 /* Remain dormant until changed by ADCS mode switch or satellite operator (through FP) */
175 break;
176 }
177 default:
178 {
179 /* State out of bounds - report through es_exeh */
180 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ACTUATOR_CONTROL_SERVICE_INVALID_STATE);
181 break;
182 }
183 }
184
185 /* Update service-wide status variable */
186 fs_actuator_control_service_update_status();
187
188 /* The task sleeps for the rest of the sampling period */
189 vApplicationLowStackCheck(TASK_ACTUATOR_CONTROL_SERVICE);
190 osDelay(ACTUATOR_CONTROL_SERVICE_SAMPLING_TIME_MS);
191 }
192}

202static void fs_magnetorquers_actuate(void)
203{
204 /*
205 * 1. read datacache
206 * 2. use es_coordsys transform
207 * 3. determine directions
208 * 4. scale (if necessary) to appropriate control units
209 * 5. send to actuator
210 */
211
212 lastError = ACTUATOR_CONTROL_SERVICE_ERROR_NONE;
213
214 /* Retrieve control values from DataCache */
215 data_cache_error_actuators = dc_get_es_adcs_control_values_mtq_data(&mtq_control_values_sat_frame_DC);
216
217 /* Catch any DataCache read failure */
218 if (DC_DATA_STATUS_OK != data_cache_error_actuators)
219 {
220 /* DataCache readout failed - raise an error */
221 lastError = ACTUATOR_CONTROL_SERVICE_ERROR_DATA_CACHE_GET;
222
223 /* Report through es_exeh */
224 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ACTUATOR_CONTROL_SERVICE_MTQ_DC_READ_FAIL);
225 return;
226 }
227
228 /* Use es_coordsys to transform from satellite frame to actuator frame */
229 memcpy(mtq_control_values_sat_frame, &mtq_control_values_sat_frame_DC, sizeof(DATA_CACHE_ES_ADCS_CONTROL_VALUES_MTQ_t));
230
231 if (!coordSys_transformMtq(mtq_control_values_sat_frame, mtq_control_values_mtq_frame))
232 {
233 /* Coordinate frame transform failed - report through es_exeh */
234 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_ACTUATOR_CONTROL_SERVICE_MTQ_COORD_SYS_FAIL);
235 lastError = ACTUATOR_CONTROL_SERVICE_ERROR_COORD_SYS_TRANSFORM;
236 return;
237 }
238
239 /* Determine direction of magnetorquer action */
240 fs_get_magnetorquers_direction(mtq_control_values_mtq_frame, mtq_control_values_directions);
241
242 /* No scaling needed - driver functions work with 0-100 range and direction 0-1 */
243
244 /* Call driver function for magnetorquer */
245 /* PAN1 PAN2 PAN3 PAN4 PAN5 PAN6 */
246 /* Satellite frame: Y- X- Z- Z+ Y+ X+ */
247 SetMagnetorque(PAN_X_M, (uint8_t)(abs(mtq_control_values_mtq_frame[0])), mtq_control_values_directions[0]); /* PAN4 */
248 SetMagnetorque(PAN_Y_M, (uint8_t)(abs(mtq_control_values_mtq_frame[1])), mtq_control_values_directions[1]); /* PAN5 */
249 SetMagnetorque(PAN_Z_M, (uint8_t)(abs(mtq_control_values_mtq_frame[2])), mtq_control_values_directions[2]); /* PAN6 */
250 /* Accessing elements of the array like this is very ugly, but probably faster than a for */
251
252 /* AA TODO: the enum (1st parameter) of SetMagnetorque is very, very confusing */
253 /* It should really be redesigned when given the chance */
254}

257static void fs_magnetorquers_stop(void)
258{
259 /*
260 * 1. zero out all control values (or whatever represents a turned off actuator)
261 * 2. send to actuator
262 */
263 /* AA TODO: Check if we really need to zero out the values or just pass temporary zeroes to the driver function */
264 for (uint8_t i = 0; i < MAGNETORQUER_QUANTITY; i++)
265 {
266 mtq_control_values_mtq_frame[i] = 0;
267 }
268
269 /* Accessing elements of the array like this is very ugly, but probably faster than a for */
270 SetMagnetorque(PAN_X_M, (uint8_t)(abs(mtq_control_values_mtq_frame[0])), mtq_control_values_directions[0]); /* PAN4 */
271 SetMagnetorque(PAN_Y_M, (uint8_t)(abs(mtq_control_values_mtq_frame[1])), mtq_control_values_directions[1]); /* PAN5 */

278static void fs_get_magnetorquers_direction(const int8_t *inputData, uint8_t *outputDirections)
279{
280 /* 3 magnetorquers for now */
281 for (uint8_t i = 0; i < MAGNETORQUER_QUANTITY; i++)
282 {
283 if (inputData[i] >= 0)
284 {
285 outputDirections[i] = 1;
286 }
287 else
288 {
289 outputDirections[i] = 0;
290 }
291 }
