50// GEN_OBJ_START::eps_ctrl_sm_on_entry_active_request
51void eps_ctrl_sm_on_entry_active_request(sState_t * const pSMRoot, const sHSM_EventType_t * const pEvent)
52{
53 // USER_CODE_START::onEntry_active_request
54 (void) pEvent;
55 (void) pSMRoot;
56 eps_ctrl_result_t res = EPS_CTRL_FAIL;
57
58 // Call eps_ctrl to send the request
59 eps_ctrl_request_t head_copy;
60 eps_ctrl_request_t const * p_request = eps_ctrl_queue_head(&head_copy);
61
62 if (NULL != p_request)
63 {
64 switch (head_copy.type)
65 {
66 case EPS_CTRL_REQ_SET_CH:
67 res = eps_ctrl_cfg_set_channel_output(&head_copy);
68 break;
69
70 case EPS_CTRL_REQ_GET_CH:
71 res = eps_ctrl_cfg_get_channel_output(&head_copy);
72 break;
73
74 default:
75 // Wrong parameter
76 break;
77 }
78 }
79
80 // If the response was not send correctly, the timeout periodic event will trigger the retry
81 eps_sm_request_sent_ok = (EPS_CTRL_DONE == res);
82
83 // Start the timer
84 timer_start(&response_timer);
85
86 // USER_CODE_END::onEntry_active_request
87
88}

119
