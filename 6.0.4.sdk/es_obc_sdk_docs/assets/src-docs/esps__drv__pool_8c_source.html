29sMACTXPoolEntry_t *MAC_POOL_psPoolTXNew(sMACInterfaceState_t * fp_psMACIntf)
30{
31 sMACTXPoolEntry_t *psEntry = NULL;
32
33 uint8_t idx;
34
35 bool freePoolFound = false;
36
37 if((fp_psMACIntf == NULL)
38 || ((uint8_t *) fp_psMACIntf->sTXCtx.pvMemPool != (uint8_t *) fp_psMACIntf->sTXCtx.aTxPool))
39 {
40 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_InvalidParameter);
41 }
42 else
43 {
44 MAC_vEnterCriticalSection(fp_psMACIntf);
45
46 for(idx = 0; idx < MAC_TXMEMORYPOOL_COUNT; idx++)
47 if(fp_psMACIntf->sTXCtx.aTxPoolFreeMap[idx])
48 {
49 freePoolFound = true;
50
51 // Mark entry as occupied
52 fp_psMACIntf->sTXCtx.aTxPoolFreeMap[idx] = false;
53
54 ++fp_psMACIntf->sTXCtx.txPoolsOccupied;
55 MAC_Stats_storeEventMax(fp_psMACIntf->macStats,
56 eMACStatId_MaxTxSlotsOccupied, fp_psMACIntf->sTXCtx.txPoolsOccupied);
57
58 break;
59 }
60
61 if(!freePoolFound)
62 {
63 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_SystemCallFailed);
64 }
65 else
66 psEntry = (sMACTXPoolEntry_t *) & fp_psMACIntf->sTXCtx.aTxPool[idx];
67
68 MAC_vLeaveCriticalSection(fp_psMACIntf);
69 }
70
71 return psEntry;
72}

124void MAC_POOL_vPoolTXFree(sMACInterfaceState_t * fp_psMACIntf, sMACTXPoolEntry_t * fp_psEntry)
125{
126 uint8_t idx;
127
128 bool poolFound = false;
129
130 if((fp_psMACIntf == NULL) || (fp_psEntry == NULL)
131 || ((uint8_t *) fp_psMACIntf->sTXCtx.pvMemPool != (uint8_t *) fp_psMACIntf->sTXCtx.aTxPool))
132 {
133 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_InvalidParameter);
134 }
135 else
136 {
137 MAC_vEnterCriticalSection(fp_psMACIntf);
138
139 for(idx = 0; idx < MAC_TXMEMORYPOOL_COUNT; idx++)
140 if((fp_psMACIntf->sTXCtx.aTxPoolFreeMap[idx] == false) &&
141 (fp_psEntry == ((sMACTXPoolEntry_t *) & fp_psMACIntf->sTXCtx.aTxPool[idx])))
142 {
143 fp_psMACIntf->sTXCtx.aTxPoolFreeMap[idx] = true;
144 poolFound = true;
145
146 if(fp_psMACIntf->sTXCtx.txPoolsOccupied > 0)
147 --fp_psMACIntf->sTXCtx.txPoolsOccupied;
148
149 break;
150 }
151
152 if(!poolFound)
153 {
154 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t) eMACException_SystemCallFailed);
155 }
156
157 MAC_vLeaveCriticalSection(fp_psMACIntf);
158 }
159}

200bool MAC_POOL_bPoolInit(sMACInterfaceState_t * fp_psMACIntf)
201{
202 bool bInitOK = false;
203
204 if(fp_psMACIntf == NULL)
205 {
206 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_InvalidParameter);
207 }
208 else
209 {
210 // Set the values for the Transmit pools as all free
211 fp_psMACIntf->sTXCtx.pvMemPool = (sMACTXPoolEntry_t *) fp_psMACIntf->sTXCtx.aTxPool;
212 memset(fp_psMACIntf->sTXCtx.aTxPoolFreeMap, true, MAC_TXMEMORYPOOL_COUNT);
213 fp_psMACIntf->sTXCtx.txPoolsOccupied = (uint8_t) 0;
214
215 MAC_Stats_storeEventMax(fp_psMACIntf->macStats, eMACStatId_MaxTxSlotsOccupied,
216 fp_psMACIntf->sTXCtx.txPoolsOccupied);
217
218 // Set the values for the Receive pools as all free
219 fp_psMACIntf->sRXCtx.pvMemPool = (sMACRXPoolEntry_t *) fp_psMACIntf->sRXCtx.aRxPool;
220 memset(fp_psMACIntf->sRXCtx.aRxPoolFreeMap, true, MAC_RXMEMORYPOOL_COUNT);
221 fp_psMACIntf->sRXCtx.rxPoolsOccupied = (uint8_t) 0;
222
223 MAC_Stats_storeEventMax(fp_psMACIntf->macStats, eMACStatId_MaxRxSlotsOccupied,
224 fp_psMACIntf->sRXCtx.rxPoolsOccupied);
225
226 bInitOK = true;
227 }
228
229 return bInitOK;
