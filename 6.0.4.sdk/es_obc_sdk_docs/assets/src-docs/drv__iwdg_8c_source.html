33bool drv_iwdg_init(uint32_t timeout)
34{
35#ifdef DEBUG_ENABLED
36 __HAL_DBGMCU_FREEZE_IWDG1();
37#endif
38
39 if (0 == timeout)
40 {
41 return false;
42 }
43
44 if (FS_CALC_TIMEOUT(FS_RELOAD_MAX, LSI_VALUE, IWDG_PRESCALER_256) < timeout)
45 {
46 timeout = FS_CALC_TIMEOUT(FS_RELOAD_MAX, LSI_VALUE, IWDG_PRESCALER_256);
47 }
48
49 fs_hiwdg.Instance = IWDG1;
50 fs_hiwdg.Init.Prescaler = IWDG_PRESCALER_256;
51 fs_hiwdg.Init.Window = FS_WINDOW_MAX;
52 fs_hiwdg.Init.Reload = FS_CALC_RELOAD(timeout, LSI_VALUE, IWDG_PRESCALER_256);
53
54 if (FS_RELOAD_MAX < fs_hiwdg.Init.Reload)
55 {
56 fs_hiwdg.Init.Reload = FS_RELOAD_MAX;
57 fs_current_timeout_ms = FS_CALC_TIMEOUT(FS_RELOAD_MAX, LSI_VALUE, IWDG_PRESCALER_256);
58 }
59 else
60 {
61 fs_current_timeout_ms = timeout;
62 }
63
64 if (HAL_OK != HAL_IWDG_Init(&fs_hiwdg))
65 {
66 return false;
67 }
68
69 return true;
70}
