28REAL es_adcs_math_VecNorm(es_adcs_vec_3d_t inVec)
29{
30 return sqrt(inVec.X * inVec.X + inVec.Y * inVec.Y + inVec.Z * inVec.Z);
31}

33es_adcs_vec_3d_t es_adcs_math_VecNormalize(es_adcs_vec_3d_t inVec)
34{
35 es_adcs_vec_3d_t tempVec;
36
37 REAL norm = es_adcs_math_VecNorm(inVec);
38 if (norm >= DOUBLE_ZERO_LIMIT)
39 {
40 tempVec.X = inVec.X / norm;
41 tempVec.Y = inVec.Y / norm;
42 tempVec.Z = inVec.Z / norm;
43 }
44 else
45 {
46 tempVec = es_adcs_math_VecSet(1.0, 1.0, 1.0);
47 }
48
49 return tempVec;
50}

57es_adcs_vec_3d_t es_adcs_math_VecCrossProduct(es_adcs_vec_3d_t inVec1, es_adcs_vec_3d_t inVec2)
58{
59 es_adcs_vec_3d_t tempVec;
60
61 tempVec.X = inVec1.Y * inVec2.Z - inVec1.Z * inVec2.Y;
62 tempVec.Y = inVec1.Z * inVec2.X - inVec1.X * inVec2.Z;
63 tempVec.Z = inVec1.X * inVec2.Y - inVec1.Y * inVec2.X;
64
65 return tempVec;
66}

68void es_adcs_math_MatMatMultiply(const es_adcs_vec_3d_t *inMat1, const es_adcs_vec_3d_t *inMat2, es_adcs_vec_3d_t *outMat)
69{
70 memset (outMat, 0x00, sizeof (es_adcs_vec_3d_t) * 3);
71 for (uint8_t i = 0; i < 3; i++)
72 {
73 for (uint8_t j = 0; j < 3; j++)
74 {
75 for (uint8_t k = 0; k < 3; k++)
76 {
77 outMat[i].Vec[j] += inMat1[i].Vec[k] * inMat2[k].Vec[j];
78 }
79 }
80 }
81}

83void es_adcs_math_MatVecMultiply(const es_adcs_vec_3d_t *inMat, es_adcs_vec_3d_t inVec, es_adcs_vec_3d_t *outVec)
84{
85 /* Help vector used to prevent overlap of variables when inVec is the same as outVec */
86 es_adcs_vec_3d_t tempVec = es_adcs_math_VecSet(0.0, 0.0, 0.0);
87
88 memset (outVec, 0x00, sizeof (es_adcs_vec_3d_t));
89 for (uint8_t i = 0; i < 3; i++)
90 {
91 for (uint8_t j = 0; j < 3; j++)
92 {
93 tempVec.Vec[i] += inMat[i].Vec[j] * inVec.Vec[j];
94 }
95 }
96
97 outVec->X = tempVec.X;
98 outVec->Y = tempVec.Y;
99 outVec->Z = tempVec.Z;
100}

168void es_adcs_math_GetDCM(es_adcs_vec_3d_t inVec, es_adcs_vec_3d_t *outMat)
169{
170 es_adcs_vec_3d_t tempVec = inVec;
171
172 /* Standard 3-2-1 DCM, COLUMN 1 */
173 outMat[0].X = cos(tempVec.Y) * cos(tempVec.Z);
174 outMat[0].Y = cos(tempVec.Y) * sin(tempVec.Z);
175 outMat[0].Z = -sin(tempVec.Y);
176
177 /* Standard 3-2-1 DCM, COLUMN 2 */
178 outMat[1].X = sin(tempVec.X) * sin(tempVec.Y) * cos(tempVec.Z) - cos(tempVec.X) * sin(tempVec.Z);
179 outMat[1].Y = sin(tempVec.X) * sin(tempVec.Y) * sin(tempVec.Z) + cos(tempVec.X) * cos(tempVec.Z);
180 outMat[1].Z = sin(tempVec.X) * cos(tempVec.Y);
181
182 /* Standard 3-2-1 DCM, COLUMN 3 */
183 outMat[2].X = cos(tempVec.X) * sin(tempVec.Y) * cos(tempVec.Z) + sin(tempVec.X) * sin(tempVec.Z);
184 outMat[2].Y = cos(tempVec.X) * sin(tempVec.Y) * sin(tempVec.Z) - sin(tempVec.X) * cos(tempVec.Z);
185 outMat[2].Z = cos(tempVec.X) * cos(tempVec.Y);
186}
