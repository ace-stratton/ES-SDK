174sds_op_status_t sds_get_sensor_data(sensor_data_sensors_t sens_type, sds_all_sens_data_t * const p_sens_data,
175 sds_all_sens_data_sat_step_t * const p_sens_data_sat, const void **p_sens_data_mem,
176 const void **p_sens_data_sat_mem)
177{
178 CRIT_ASSERT(sens_type < SENSOR_COUNT);
179 CRIT_ASSERT(p_sens_data != NULL);
180 CRIT_ASSERT(p_sens_data_sat != NULL);
181
182 sds_op_status_t op_res;
183
184 switch (sens_type)
185 {
186 case SENSOR_MAG_PRIMARY:
187 if ( SDS_OP_STATUS_OK == (op_res = sensor_prim_mag_data_cfg.sens_mag_data_if->get_mag_data(&p_sens_data->mag_data_prim)) )
188 {
189 memcpy(&p_sens_data_sat->mag_data_prim.mag_data_prev, &p_sens_data_sat->mag_data_prim.mag_data_curr, sizeof(sds_mag_data_sat_t));
190
191 sds_conv_mag_data_to_sat_frame(&p_sens_data->mag_data_prim, SDS_MAG_ENUM_PRIM, &p_sens_data_sat->mag_data_prim.mag_data_curr);
192 }
193
194 *p_sens_data_mem = &p_sens_data->mag_data_prim;
195 *p_sens_data_sat_mem = &p_sens_data_sat->mag_data_prim;
196
197 break;
198 case SENSOR_MAG_SECONDARY:
199 if ( SDS_OP_STATUS_OK == (op_res = sensor_sec_mag_data_cfg.sens_mag_data_if->get_mag_data(&p_sens_data->mag_data_sec)) )
200 {
201 memcpy(&p_sens_data_sat->mag_data_sec.mag_data_prev, &p_sens_data_sat->mag_data_sec.mag_data_curr, sizeof(sds_mag_data_sat_t));
202
203 sds_conv_mag_data_to_sat_frame(&p_sens_data->mag_data_sec, SDS_MAG_ENUM_SEC, &p_sens_data_sat->mag_data_sec.mag_data_curr);
204 }
205
206 *p_sens_data_mem = &p_sens_data->mag_data_sec;
207 *p_sens_data_sat_mem = &p_sens_data_sat->mag_data_sec;
208
209 break;
210 case SENSOR_GYRO:
211 if ( SDS_OP_STATUS_OK == (op_res = sensor_gyro_data_cfg.sens_gyro_data_if->get_gyro_data(&p_sens_data->gyro_data)) )
212 {
213 memcpy(&p_sens_data_sat->gyro_data.gyro_data_prev, &p_sens_data_sat->gyro_data.gyro_data_curr, sizeof(sds_gyro_data_sat_t));
214
215 sds_conv_gyro_data_to_sat_frame(&p_sens_data->gyro_data, &p_sens_data_sat->gyro_data.gyro_data_curr);
216 }
217
218 *p_sens_data_mem = &p_sens_data->gyro_data;
219 *p_sens_data_sat_mem = &p_sens_data_sat->gyro_data;
220
221 break;
222 case SENSOR_CSS:
223 if ( SDS_OP_STATUS_OK == (op_res = sensor_css_data_cfg.sens_css_data_if->get_css_data(&p_sens_data->css_data)) )
224 {
225 memcpy(&p_sens_data_sat->css_data.css_data_prev, &p_sens_data_sat->css_data.css_data_curr, sizeof(sds_css_data_sat_t));
226
227 sds_conv_css_data_to_sat_frame(&p_sens_data->css_data, &p_sens_data_sat->css_data.css_data_curr);
228 }
229
230 *p_sens_data_mem = &p_sens_data->css_data;
231 *p_sens_data_sat_mem = &p_sens_data_sat->css_data;
232
233 break;
234
235 case SENSOR_ACC_PRIMARY:
236 if ( SDS_OP_STATUS_OK == (op_res = sensor_prim_acc_data_cfg.sens_acc_data_if->get_acc_data(&p_sens_data->acc_data_prim)) )
237 {
238 memcpy(&p_sens_data_sat->acc_data_prim.acc_data_prev, &p_sens_data_sat->acc_data_prim.acc_data_curr, sizeof(sds_acc_data_sat_t));
239
240 sds_conv_acc_data_to_sat_frame(&p_sens_data->acc_data_prim, SDS_ACC_ENUM_PRIM, &p_sens_data_sat->acc_data_prim.acc_data_curr);
241 }
242
243 *p_sens_data_mem = &p_sens_data->acc_data_prim;
244 *p_sens_data_sat_mem = &p_sens_data_sat->acc_data_prim;
245
246 break;
247
248 case SENSOR_ACC_SECONDARY:
249 if ( SDS_OP_STATUS_OK == (op_res = sensor_sec_acc_data_cfg.sens_acc_data_if->get_acc_data(&p_sens_data->acc_data_sec)) )
250 {
251 memcpy(&p_sens_data_sat->acc_data_sec.acc_data_prev, &p_sens_data_sat->acc_data_sec.acc_data_curr, sizeof(sds_acc_data_sat_t));
252
253 sds_conv_acc_data_to_sat_frame(&p_sens_data->acc_data_sec, SDS_ACC_ENUM_SEC, &p_sens_data_sat->acc_data_sec.acc_data_curr);
254 }
255
256 *p_sens_data_mem = &p_sens_data->acc_data_sec;
257 *p_sens_data_sat_mem = &p_sens_data_sat->acc_data_sec;
258
259 break;
260
261 case SENSOR_PANEL_TEMP:
262 if ( SDS_OP_STATUS_OK == (op_res = sensor_panel_temp_data_cfg.sens_panel_temp_data_if->get_panel_temp_data(&p_sens_data->panel_temp_data)) )
263 {
264 sds_conv_panel_temp_data_to_sat_frame(&p_sens_data->panel_temp_data, &p_sens_data_sat->panel_temp_data);
265 }
266
267 *p_sens_data_mem = &p_sens_data->panel_temp_data;
268 *p_sens_data_sat_mem = &p_sens_data_sat->panel_temp_data;
269
270 break;
271
272 case SENSOR_GYRO_TEMP:
273 if ( SDS_OP_STATUS_OK == (op_res = sensor_gyro_temp_data_cfg.sens_gyro_temp_data_if->get_gyro_temp_data(&p_sens_data->gyro_temp_data)) )
274 {
275 sds_conv_gyro_temp_data_to_sat_frame(&p_sens_data->gyro_temp_data, &p_sens_data_sat->gyro_temp_data);
276 }
277
278 *p_sens_data_mem = &p_sens_data->gyro_temp_data;
279 *p_sens_data_sat_mem = &p_sens_data_sat->gyro_temp_data;
280
281 break;
282
283 default:
284
285 op_res = SDS_OP_STATUS_FAIL;
286
287 break;
288 }
289
290 return op_res;
291}
