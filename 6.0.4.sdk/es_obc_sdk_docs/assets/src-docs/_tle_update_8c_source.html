25tle_update_err_t TleUpdateFromFile(const char *file_name)
26{
27 FIL fileHandle;
28 sgp4_two_line_element_t tle;
29 sgp4_parsed_two_line_element_t parsed_tle;
30 CubeADCS_SGP4_Orbit_Params_t orbit_params;
31
32 if (file_name == NULL)
33 return TLE_UPDATE_ERR_FILE;
34
35 if (SdMngr_f_open(&fileHandle, (const TCHAR *)file_name, FA_OPEN_EXISTING | FA_READ) != FR_OK)
36 {
37 return TLE_UPDATE_ERR_FILE;
38 }
39
40 memset(tle.line, 0, sizeof(tle.line));
41 if (SdMngr_f_gets((TCHAR *)&(tle.line[0]), SGP4_TLE_CHARS_PER_LINE+4, &fileHandle) == NULL)
42 {
43 SdMngr_f_close(&fileHandle);
44 return TLE_UPDATE_ERR_FILE;
45 }
46 tle.line[0][SGP4_TLE_CHARS_PER_LINE] = 0;
47
48 if (SdMngr_f_gets((TCHAR *)&(tle.line[1]), SGP4_TLE_CHARS_PER_LINE+4, &fileHandle) == NULL)
49 {
50 SdMngr_f_close(&fileHandle);
51 return TLE_UPDATE_ERR_FILE;
52 }
53 tle.line[1][SGP4_TLE_CHARS_PER_LINE] = 0;
54
55 SdMngr_f_close(&fileHandle);
56
57 if (sgp4_tle_verify_format(&tle) != SGP4_OK)
58 return TLE_UPDATE_ERR_SGP4;
59
60 if (sgp4_tle_verify_checksum(&tle) != SGP4_OK)
61 return TLE_UPDATE_ERR_SGP4;
62
63 if (sgp4_tle_parse(&tle, &parsed_tle, false) != SGP4_OK)
64 return TLE_UPDATE_ERR_SGP4;
65
66 orbit_params.dInc = parsed_tle.incl; // Inclination
67 orbit_params.dEcc = parsed_tle.ecc; // Eccentricity
68 orbit_params.dRAAN = parsed_tle.raan; // Right-ascension of the Ascending Node
69 orbit_params.dAOP = parsed_tle.aop; // Argument of perigee
70 orbit_params.dBStar = parsed_tle.bstar; // B-Star drag term
71 orbit_params.dMeanM = parsed_tle.mmotion; // Mean Motion
72 orbit_params.dMeanA = parsed_tle.manomaly; // Mean Anomaly
73 orbit_params.dEpoch = parsed_tle.epoch; // Epoch
74
75 if (sys_conf_is_module_active(SYS_MODULE_ID_CUBE_ADCS))
76 {
77 if (CubeADCS_SetSGP4OrbitParams(&orbit_params) != eCUBEADCS_STDRPLY_OK)
78 return TLE_UPDATE_ERR_ADCS;
79 }
80 else
81 {
82 (void)orbit_params;
83 }
84
85 return TLE_UPDATE_OK;
86}
