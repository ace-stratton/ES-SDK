89void ServicesTask_Init(void)
90{
91 ServicesTask_RTC_StartTime = datetime_to_unix_now();
92
93 ServicesTask_TaskHandle = osThreadNew(ServicesTask, NULL, &ServicesTask_attributes);
94 CRIT_ASSERT(ServicesTask_TaskHandle);
95}

107static void ServicesTask(void *argument)
108{
109 static safe_bool_t esps_started = SAFE_FALSE;
110
111 TaskMonitor_TaskInitialized(TASK_MONITOR_SERVICES); /* The task is initialized and is ready */
112
113 sds_init();
114
115 /* Initialize the service for controlling actuators */
116 actuator_control_service_init();
117
118 InitCompleted();
119
120 while (1)
121 {
122 TaskMonitor_IamAlive(TASK_MONITOR_SERVICES); /* Prevent from WatchDog reset */
123
124 // NOTE:
125 // ESPS-related initialization is performed in a task context to ensure the OS is
126 // already started when ESPS driver is activated because heavy communication at OBC start
127 // could introduce too many interrupts which will prevent the OS to initialize properly
128 if (IS_FALSE(esps_started))
129 {
130 ESSA_Stack_Facade_Init();
131
132#if defined(BEACONS_ENABLED)
133 beacons_init();
134
135 beacons_start();
136#endif // #if defined(BEACONS_ENABLED)
137
138 esps_started = SAFE_TRUE;
139 }
140
141 osDelay(SERVICE_TASK_PERIOD);
142
143 vApplicationLowStackCheck(TASK_MONITOR_SERVICES);
144 }
145}

168static void InitCompleted(void)
169{
170 static uint8_t complete = 0;
171
172 if ((0 == complete) && (0 != TaskMonitor_CheckTasksInit()))
173 {
174 ES_TRACE_INFO("OK+OBC initialisation completed\r\n"); /* OBC GUI application needs that string to end or start a operation */
175 complete = 1;
176 }
177}
178
179uint32_t ServicesTask_UptimeSeconds(void)
180{
181 return (datetime_to_unix_now() - ServicesTask_RTC_StartTime);
182}
