81void telemetry_init(void)
82{
83 EXEH_INIT_MODULE_FILENAME();
84
85 tlm_rt_context.os_task_hnd = NULL;
86
87 timer_stop(&tlm_rt_context.tlm_timer);

91 telemetry_lock_id = osMutexNew(&telemetry_lock_attr);

93
94 telemetry_trigger_hsm(eHSM_StdEvent_Reset);
95 telemetry_trigger_hsm(ev_telemetry_sm_init);

97 telemetry_file_sink_if.cmd(TELEMETRY_SINK_ADAPTER_CMD_INIT);

100void telemetry_start(void)
101{
102 static const osThreadAttr_t task_attr =
103 {
104 .attr_bits = osThreadDetached,
105 .name = "Telemetry_Service_Task",
106 .priority = osPriorityNormal,
107 .cb_mem = NULL,
108 .cb_size = 0U,
109 .stack_mem = NULL,
110 .stack_size = 2500 // TODO: Measure worst-case stack usage!
111 };
112
113 // create thread only if not already existing...
114 if (NULL == tlm_rt_context.os_task_hnd)
115 {
116 tlm_rt_context.os_task_hnd = osThreadNew(telemetry_task,
117 (void *) NULL,
118 &task_attr);
119
120 if (NULL == tlm_rt_context.os_task_hnd)
121 {
122 EXEH_HANDLE(eEXEHSeverity_Fatal,
123 (int32_t) EXEH_TLM_CANNOT_SPAWN_TASK);
124 }
125 else
126 {
127 TaskMonitor_TaskInitialized(TASK_TELEMETRY);
128
129 (void) telemetry_trigger_hsm(ev_telemetry_sm_start);
130
131 telemetry_file_sink_if.cmd(TELEMETRY_SINK_ADAPTER_CMD_START);
132 }
133 }
134}

142void telemetry_load_config(void)
143{
144 static uint8_t prev_preset_id = 0U;
145
146 Nvm_GetBlockById(NVM_BLOCK_TELEMETRY_GENERAL, &tlm_rt_context.gen_cfg);
147 Nvm_GetBlockById(NVM_BLOCK_TELEMETRY_PRESETS, &tlm_rt_context.preset_cfg);
148 tlm_rt_context.active_preset_id = telemetry_cfg_get_active_preset();
149
150 if (prev_preset_id != tlm_rt_context.active_preset_id)
151 {
152 // reset acquisition counters...
153 (void) memset(tlm_rt_context.tlm_message_acq_counters,
154 0U,
155 sizeof(tlm_rt_context.tlm_message_acq_counters));
156
157 prev_preset_id = tlm_rt_context.active_preset_id;
158 }
159
160 CRIT_ASSERT(tlm_rt_context.active_preset_id < TELEMETRY_PRESETS_COUNT);
161}

166static void telemetry_task(void * p_task_arg)
167{
168 static safe_bool_t last_tlm_en_status = SAFE_FALSE;
169 bool hsm_result = false;
170
171 // currently unused
172 (void) p_task_arg;
173
174 telemetry_load_config();
175
176 for (;;)
177 {
178 if (last_tlm_en_status != tlm_rt_context.gen_cfg.telemetry_en)
179 {
180 switch (tlm_rt_context.gen_cfg.telemetry_en)
181 {
182 case SAFE_TRUE:
183 {
184 hsm_result = telemetry_trigger_hsm(ev_telemetry_sm_start);
185
186 break;
187 }
188
189 case SAFE_FALSE:
190 {
191 hsm_result = telemetry_trigger_hsm(ev_telemetry_sm_stop);
192
193 break;
194 }
195
196 default:
197 {
198 CRIT_ASSERT(false);
199 break;
200 }
201 }
202
203 (void) hsm_result;
204 last_tlm_en_status = tlm_rt_context.gen_cfg.telemetry_en;
205 }
206
207 // execute periodic SM activities...
208 (void) telemetry_trigger_hsm(eHSM_StdEvent_Periodic);
209
210 osDelay(TLM_TASK_PERIOD_MS);
211
212 TaskMonitor_IamAlive(TASK_TELEMETRY);
213 vApplicationLowStackCheck(TASK_TELEMETRY);
214 }
215}

217static bool inline telemetry_trigger_hsm(const uint32_t event_id)
218{
219 bool res = false;
220
221 osMutexAcquire(telemetry_lock_id, osWaitForever);
222 res = HSM_TriggerEvent(telemetry_sm_instance, event_id);
223 osMutexRelease(telemetry_lock_id);
224
225 return res;
226}
