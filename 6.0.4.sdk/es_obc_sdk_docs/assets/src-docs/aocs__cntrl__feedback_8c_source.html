73adcs_cntrl_result_t aocs_cntrl_get_status(void)
74{
75
76 adcs_cntrl_result_t eStateChangeOperationResult;
77
78 /* Retrieve the current ADCS system state configuration */
79 DATA_CACHE_AOCS_CNTRL_SYS_STATE_t sADCSData;
80 aocs_cntrl_sys_state_req_state_t eCurrentSysStateCfgStatus = AOCS_CNTRL_SYS_STATE_STATE_MAX;
81
82 if ( (DC_NOT_INITIALIZED != dc_get_aocs_cntrl_sys_state_data(&sADCSData)) &&
83 (sADCSData.u8AdcsSysStateStatus < AOCS_CNTRL_SYS_STATE_STATE_MAX ) )
84 {
85 eCurrentSysStateCfgStatus = (aocs_cntrl_sys_state_req_state_t)sADCSData.u8AdcsSysStateStatus;
86 }
87
88 /* Return state change operation status based on system state configuration status variable */
89 if (AOCS_CNTRL_SYS_STATE_STATE_REQ_COMPLETED == eCurrentSysStateCfgStatus)
90 {
91 eStateChangeOperationResult = ADCS_CNTRL_RESULT_SUCCESS;
92 }
93 else if (AOCS_CNTRL_SYS_STATE_STATE_REQ_FAILED == eCurrentSysStateCfgStatus)
94 {
95 eStateChangeOperationResult = ADCS_CNTRL_RESULT_FAILED;
96 }
97 else
98 {
99 /* If the system state configuration status is not OK or ERROR, then */
100 /* it is either attempting to change or custom logic is being executed. */
101 /* Either of those means that the system state change operation is in progress */
102 eStateChangeOperationResult = ADCS_CNTRL_RESULT_IN_PROGRESS;
103 }
104
105 return eStateChangeOperationResult;
106}
