20static inline uint32_t fs_GetFreeBytes(CircularBufCtx_t* const ptBufContext)
21{
22 CRIT_ASSERT(ptBufContext != NULL);
23
24 if (ptBufContext->bBufferFull)
25 return 0U;
26
27 if (ptBufContext->u32TailPos > ptBufContext->u32HeadPos)
28 return ptBufContext->u32BufSize - (ptBufContext->u32TailPos - ptBufContext->u32HeadPos);
29 else
30 if (ptBufContext->u32TailPos < ptBufContext->u32HeadPos)
31 return ptBufContext->u32HeadPos - ptBufContext->u32TailPos;
32 else // buffer is empty
33 return ptBufContext->u32BufSize;
34}

36static inline uint32_t fs_GetUsedBytes(CircularBufCtx_t* const ptBufContext)
37{
38 CRIT_ASSERT(ptBufContext != NULL);
39
40 return (ptBufContext->u32BufSize - fs_GetFreeBytes(ptBufContext));
41}

43eCircBufOpResult CircularBuf_Init(CircularBufCtx_t* const ptBufContext,
44 uint8_t* const pu8UserBuf,
45 const uint32_t u32UserBufSize,
46 const uint32_t u32CommitBufSize,
47 uint8_t* const pu8AuxBuf,
48 const uint32_t u32AuxBufSize)
49{
50 eCircBufOpResult res = ECIRCBUFOPRESULT_INVALID_ARGS;
51
52 if ((ptBufContext != (CircularBufCtx_t*)NULL) &&
53 (pu8UserBuf != NULL) &&
54 (u32CommitBufSize <= u32UserBufSize))
55 {
56 ptBufContext->pu8Buf = pu8UserBuf;
57 ptBufContext->u32BufSize = u32UserBufSize;
58 ptBufContext->u32CommitBufSize = u32CommitBufSize;
59
60 ptBufContext->u32HeadPos = 0U;
61 ptBufContext->u32TailPos = 0U;
62 ptBufContext->u32LastBufPtrSize = 0U;
63 ptBufContext->bBufferFull = false;
64 ptBufContext->u32MaxByteCountReached = 0U;
65
66 if (pu8AuxBuf != NULL)
67 {
68 ptBufContext->pu8AuxBuf = pu8AuxBuf;
69 ptBufContext->u32AuxBufSize = u32AuxBufSize;
70 }
71 else
72 {
73 ptBufContext->pu8AuxBuf = NULL;
74 ptBufContext->u32AuxBufSize = 0U;
75 }
76
77 ptBufContext->bAuxBufferInUse = false;
78
79 res = ECIRCBUFOPRESULT_OK;
80 }
81
82 return res;
83}

85eCircBufOpResult CircularBuf_Append(CircularBufCtx_t* const ptBufContext,
86 uint8_t* const pu8Data,
87 const uint32_t u32DataSize)
88{
89 eCircBufOpResult res = ECIRCBUFOPRESULT_INVALID_ARGS;
90 uint32_t u32RemainingBytesUntilEnd;
91 uint32_t u32UsedBytes;
92
93 if ((ptBufContext != NULL) &&
94 (pu8Data != NULL) &&
95 (u32DataSize > 0U))
96 {
97 if (u32DataSize <= fs_GetFreeBytes(ptBufContext))
98 {
99 if ((ptBufContext->u32TailPos + u32DataSize) <= ptBufContext->u32BufSize)
100 {
101 (void)memcpy((void*)&ptBufContext->pu8Buf[ptBufContext->u32TailPos],
102 (void*)pu8Data,
103 (size_t)u32DataSize);
104 }
105 // buffer will overflow, so we need to copy in two chunks:
106 // one until end of buffer and one starting from beginning
107 else
108 {
109 u32RemainingBytesUntilEnd = ptBufContext->u32BufSize - ptBufContext->u32TailPos;
110
111 (void)memcpy((void*)&ptBufContext->pu8Buf[ptBufContext->u32TailPos],
112 (void*)pu8Data,
113 (size_t)u32RemainingBytesUntilEnd);
114
115 if ((u32DataSize - u32RemainingBytesUntilEnd) > 0U)
116 (void)memcpy((void*)&ptBufContext->pu8Buf[0],
117 (void*) &pu8Data[u32RemainingBytesUntilEnd],
118 (size_t)(u32DataSize - u32RemainingBytesUntilEnd));
119 }
120
121 ptBufContext->u32TailPos = (ptBufContext->u32TailPos + u32DataSize) % ptBufContext->u32BufSize;
122
123 // if tail reaches the head position after append operation, then the buffer became completely full
124 if (ptBufContext->u32TailPos == ptBufContext->u32HeadPos)
125 ptBufContext->bBufferFull = true;
126
127 // keep maximum used bytes statistics
128 u32UsedBytes = fs_GetUsedBytes(ptBufContext);
129 if (ptBufContext->u32MaxByteCountReached < u32UsedBytes)
130 ptBufContext->u32MaxByteCountReached = u32UsedBytes;
131
132 res = ECIRCBUFOPRESULT_OK;
133 }
134 else
135 res = ECIRCBUFOPRESULT_BUFFER_FULL;
136 }
137
138 return res;
139}

141uint8_t* CircularBuf_GetNextBufferPtr(CircularBufCtx_t* const ptBufContext, const uint32_t u32BytesRequested, uint32_t* const pu32BytesAvailable)
142{
143 uint8_t* pRetBuf = NULL;
144 uint32_t u32RemainingBytesUntilEnd = 0U;
145 uint32_t u32DataBytesAvailable = 0U;
146
147 // Aux buffer or any previously reserved buffer has to be free before requesting another buffer, otherwise the bytes could be processed in a wrong sequence,
148 // therefore further reading of the circular buffer is not allowed until last reserved buffer is released via CircularBuf_ReleaseLastBufferPtr()
149 if ((ptBufContext != NULL) &&
150 (!ptBufContext->bAuxBufferInUse) &&
151 (ptBufContext->u32LastBufPtrSize == 0U))
152 {
153 // calculate actual data bytes in the buffer which are pending to be processed
154 u32DataBytesAvailable = fs_GetUsedBytes(ptBufContext);
155
156 // check if we already accumulated enough bytes to fulfill the buffer request
157 if (u32DataBytesAvailable >= u32BytesRequested)
158 {
159 if (ptBufContext->u32TailPos == ptBufContext->u32HeadPos)
160 {
161 if (!ptBufContext->bBufferFull)
162 u32RemainingBytesUntilEnd = ptBufContext->u32BufSize;
163 }
164 else
165 if (ptBufContext->u32TailPos > ptBufContext->u32HeadPos)
166 u32RemainingBytesUntilEnd = ptBufContext->u32TailPos - ptBufContext->u32HeadPos;
167 else
168 // tail position is before head position, hence the buffer overflows to beginning
169 u32RemainingBytesUntilEnd = ptBufContext->u32BufSize - ptBufContext->u32HeadPos;
170
171 // can we return a contiguous buffer or the data rolls over to the beginning of the buffer?
172 if (u32RemainingBytesUntilEnd >= u32BytesRequested)
173 {
174 ptBufContext->u32LastBufPtrSize = u32BytesRequested;
175 pRetBuf = &ptBufContext->pu8Buf[ptBufContext->u32HeadPos];
176 }
177 else
178 {
179 // data rolls over to the beginning of the buffer, check if we can use the Aux buffer to provide a
180 // contiguous chunk of memory
181 if (ptBufContext->pu8AuxBuf != NULL)
182 {
183 ptBufContext->bAuxBufferInUse = true;
184
185 // Aux buffer can take the data, hence we need to copy and return a pointer to the Aux buffer
186 if (ptBufContext->u32AuxBufSize >= u32BytesRequested)
187 {
188 // copy first part of the buffer
189 (void)memcpy((void*)ptBufContext->pu8AuxBuf,
190 (void*)&ptBufContext->pu8Buf[ptBufContext->u32HeadPos],
191 (size_t)u32RemainingBytesUntilEnd);
192
193 // ...and remaining bytes
194 (void)memcpy((void*)&ptBufContext->pu8AuxBuf[u32RemainingBytesUntilEnd],
195 (void*)&ptBufContext->pu8Buf[0],
196 (size_t)(u32BytesRequested - u32RemainingBytesUntilEnd));
197
198 // now data is in the Aux buffer, so we can move the head in the circular buffer forward to
199 // make room for more data
200 ptBufContext->u32HeadPos = (ptBufContext->u32HeadPos + u32BytesRequested) % ptBufContext->u32BufSize;
201
202 // finally, return a pointer to the Aux buffer
203 pRetBuf = ptBufContext->pu8AuxBuf;
204 }
205 }
206 }
207
208 u32DataBytesAvailable = u32BytesRequested;
209 }
210
211 if (pu32BytesAvailable != NULL)
212 *pu32BytesAvailable = u32DataBytesAvailable;
213 }
214
215 return pRetBuf;
216}

218void CircularBuf_ReleaseLastBufferPtr(CircularBufCtx_t* const ptBufContext, const bool bDiscardLastBufferData)
219{
220 if (ptBufContext != NULL)
221 {
222 // if Aux buffer was last used, there is nothing to discard in the circular buffer as that was already done in
223 // CircularBuf_GetNextBufferPtr() operation
224 if (ptBufContext->bAuxBufferInUse)
225 ptBufContext->bAuxBufferInUse = false;
226 else
227 {
228 // discard data in the circular buffer because it is not anymore needed by the user
229 if (ptBufContext->u32LastBufPtrSize > 0U)
230 {
231 // if bDiscardLastBufferData flag is set, then the user has consumed the data, otherwise something may have
232 // happened and the user would like to make another try with the same data
233 if (bDiscardLastBufferData)
234 {
235 ptBufContext->u32HeadPos = (ptBufContext->u32HeadPos + ptBufContext->u32LastBufPtrSize) % ptBufContext->u32BufSize;
236 ptBufContext->bBufferFull = false;
237 }
238
239 ptBufContext->u32LastBufPtrSize = 0;
240 }
241 }
242 }
