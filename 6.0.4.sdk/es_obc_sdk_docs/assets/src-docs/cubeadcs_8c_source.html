487void cubeadcs_set_node_pwr_state(eCUBE_ADCS_TC_11_NODES_t node, eCUBE_ADCS_TC_11_PWR_STATES_t state)
488{
489 uint8_t u8Data[CUBEADCS_TC_11_BYTE_LEN] = {0x00, 0x00, 0x00};
490
491 uint16_t u16Resp;
492 uint16_t u16RespSize;
493
494 CubeADCS_CmdCfg_t sCmdStruct;
495
496 CubeADCS_PREPCmd(&sCmdStruct, eCUBEADCS_CMD_ID_11, &u8Data[0], CUBEADCS_TC_11_BYTE_LEN, (void *)&u16Resp, CUBEADCS_TC_LEN, &u16RespSize);
497
498 /* Here we clear the two bits responsible for the node and then set them to the requested value (going from right to left)
499 * - we clear the two bits corresponding to the node in the 3 bytes responsible for the power control
500 * - we logically OR this value with the 0x00AAAAAA pattern, this pattern corresponds to the power control set to no change
501 * - we logically OR with the result with the node state bits
502 * - ...there may be a better way ...
503 * */
504 uint32_t u32PwrData = ((uint32_t)state << ((uint32_t)node * 2)) | ( 0x00AAAAAA & ~((uint32_t)0x03 << ((uint32_t)node * 2)) );
505
506 for ( uint8_t u8Indx = 0; u8Indx < CUBEADCS_TC_11_BYTE_LEN; ++u8Indx )
507 {
508 *(sCmdStruct.pu8CmdData + u8Indx) = *(((uint8_t *)&u32PwrData) + u8Indx); /* Each node occupies 2 bits for the state - see TC 11 */
509 }
510
511 /* Update the node control status on success */
512 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmd(&sCmdStruct, eCUBEADCS_COMMREQTYPE_FETCH))
513 {
514 sNodeCntrlStatus.u32RawData = sResult_TLM197.u32RawData | (((uint32_t)state << ((uint32_t)node * 2)) << 8); /* The last shift is needed because the cmdID is part of the structure */
515 }
516}

518eCubeADCS_StdRply CubeADCS_GetAngularRate(CubeADCS_AngRateFormatted_t *sAngRate, eCubeADCS_AngRateType_t eMeasType)
519{
520 CRIT_ASSERT(sAngRate != NULL);
521 CRIT_ASSERT(eMeasType < eCUBEADCS_ANGRATETYPE_NUM);
522
523 eCubeADCS_AngVelValid_t u8CanRead;
524 eCUBE_ADCS_TC_56_MTM_MODE_t eUsedMtmSMode;
525
526 float fWA0;
527 float fWB0;
528
529 eCubeADCS_StdRply eRetVal;
530
531 uint16_t u16CmdRspLen;
532 CubeADCS_CmdCfg_t sCmdStruct_147;
533 CubeADCS_CmdCfg_t sCmdStruct_155;
534
535 CubeADCS_TLM147_t sCmdRsp147;
536 CubeADCS_TLM155_t sCmdRsp155;
537
538 fWA0 = 0.0F;
539 fWB0 = 0.0F;
540 u8CanRead = eANGVEL_VALID_NONE;
541 eRetVal = eCUBEADCS_STDRPLY_ERR;
542
543 CubeADCS_PREPCmd(&sCmdStruct_147, eCUBEADCS_CMD_ID_147, NULL, 0, (void *)&sCmdRsp147, CUBEADCS_TLM147_LEN, &u16CmdRspLen);
544 CubeADCS_PREPCmd(&sCmdStruct_155, eCUBEADCS_CMD_ID_155, NULL, 0, (void *)&sCmdRsp155, CUBEADCS_TLM155_LEN, &u16CmdRspLen);
545
546 eUsedMtmSMode = CubeADCS_GetMTMSamplingMode();
547
548 /* Check if we can actually read data from the sensors/estimators */
549 if ( eCUBEADCS_TC56_MAIN_MTM_MOTOR_MCU == eUsedMtmSMode )
550 {
551 if ( eCUBE_ADCS_TC11_STATE_ON == CubeADCS_GetADCSNodePwrState(eCUBE_ADCS_TC11_NODES_CUBECONTROL_MOTOR) )
552 {
553 u8CanRead |= eANGVEL_VALID_RATE;
554
555 if ( eCUBE_ADCS_TC_14_ESTIMATIONMODE_NO_EST != CubeADCS_GetEstMode() )
556 u8CanRead |= eANGVEL_VALID_EST;
557 }
558 }
559 else
560 {
561 if ( eCUBE_ADCS_TC11_STATE_ON == CubeADCS_GetADCSNodePwrState(eCUBE_ADCS_TC11_NODES_CUBECONTROL_SIGNAL) )
562 if ( eCUBE_ADCS_TC_14_ESTIMATIONMODE_NO_EST != CubeADCS_GetEstMode() )
563 u8CanRead |= eANGVEL_VALID_EST;
564
565 if ( eCUBE_ADCS_TC11_STATE_ON == CubeADCS_GetADCSNodePwrState(eCUBE_ADCS_TC11_NODES_CUBECONTROL_MOTOR) )
566 u8CanRead |= eANGVEL_VALID_RATE;
567 }
568
569 /* Read data based on the request and sensors / estimation availability */
570 switch (eMeasType)
571 {
572 case eCUBEADCS_ANGRATETYPE_EST:
573 {
574 if ( eANGVEL_VALID_EST == (u8CanRead & eANGVEL_VALID_EST) ) {
575 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmd(&sCmdStruct_147, eCUBEADCS_COMMREQTYPE_FETCH) ) {
576 sAngRate->fX = ( (float)sCmdRsp147.AngRate.i16X * 0.01F ) * CUBEADCS_TO_RAD_CONV;
577 sAngRate->fY = ( (float)sCmdRsp147.AngRate.i16Y * 0.01F ) * CUBEADCS_TO_RAD_CONV;
578 sAngRate->fZ = ( (float)sCmdRsp147.AngRate.i16Z * 0.01F ) * CUBEADCS_TO_RAD_CONV;
579
580 eRetVal = eCUBEADCS_STDRPLY_OK;
581 }
582 }
583
584 break;
585 }
586 case eCUBEADCS_ANGRATETYPE_RATE:
587 {
588 if ( eANGVEL_VALID_RATE == (u8CanRead & eANGVEL_VALID_RATE) ) {
589 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmd(&sCmdStruct_155, eCUBEADCS_COMMREQTYPE_FETCH) ) {
590 sAngRate->fX = ( (float)sCmdRsp155.AngRate.i16X * 0.01F ) * CUBEADCS_TO_RAD_CONV;
591 sAngRate->fY = ( (float)sCmdRsp155.AngRate.i16Y * 0.01F ) * CUBEADCS_TO_RAD_CONV;
592 sAngRate->fZ = ( (float)sCmdRsp155.AngRate.i16Z * 0.01F ) * CUBEADCS_TO_RAD_CONV;
593
594 eRetVal = eCUBEADCS_STDRPLY_OK;
595 }
596 }
597
598 break;
599 }
600 case eCUBEADCS_ANGRATETYPE_WEIGHTED:
601 {
602 if ( eANGVEL_VALID_EST == (u8CanRead & eANGVEL_VALID_EST) ) {
603 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmd(&sCmdStruct_147, eCUBEADCS_COMMREQTYPE_FETCH) )
604 {
605 fWA0 = 1.0F;
606
607 eRetVal = eCUBEADCS_STDRPLY_OK;
608 }

1069bool cubeadcs_set_sys_state_and_ref(cubeadcs_ref_t *p_ref, eCubeADCS_CfgType new_sys_cfg)
1070{
1071 CRIT_ASSERT(p_ref != NULL);
1072 CRIT_ASSERT(new_sys_cfg < eCUBEADCS_CFGTYPE_NUMBER);
1073
1074 bool bRes = true;
1075
1076 switch (new_sys_cfg)
1077 {
1078 case eCUBEADCS_CFGTYPE_Y_MOMENTUM:
1079 case eCUBEADCS_CFGTYPE_Y_MOMENTUM_FULL_STATE_EKF:
1080 case eCUBEADCS_CFGTYPE_3AXIS:
1081 {
1082 CubeADCS_Command_Angles_t sAngles = {
1083 .i16RollAngle = 0,
1084 .i16PitchAngle = 0,
1085 .i16YawAngle = 0
1086 };
1087
1088 /* Clear offset angles */
1089 fs_CubeADCS_SetCommandedAngles(sAngles);
1090
1091 /* Update the system state configuration */
1092 cubeadcs_set_state_cfg(new_sys_cfg, CUBEADCS_EXEC_STATE_REQ);
1093
1094 break;
1095 }
1096 /* If the new system state configuration is for target tracking, the target should be set before the */
1097 /* state is changed to target tracking (CubeADCS User Manual, 4.4.3.3) */
1098 case eCUBEADCS_CFGTYPE_TARGET_TRACKING:
1099 case eCUBEADCS_CFGTYPE_GEO_TARGET_TRACKING:
1100 {
1101 uint16_t u16CmdRsp;
1102 uint16_t u16CmdRspLen;
1103
1104 CubeADCS_CmdCfg_t sCmdStruct;
1105
1106 CubeADCS_PREPCmd(&sCmdStruct, eCUBEADCS_CMD_ID_55, (uint8_t *)p_ref, CUBEADCS_TC_55_BYTE_LEN, (void *)&u16CmdRsp, CUBEADCS_TC_LEN, &u16CmdRspLen);
1107
1108 if ( eCUBEADCS_COMM_OK != CubeADCS_SendCmd(&sCmdStruct, eCUBEADCS_COMMREQTYPE_FETCH) )
1109 {
1110 bRes = false;
1111
1112 /* Update the system state configuration, don't execute it ... */
1113 cubeadcs_set_state_cfg(new_sys_cfg, CUBEADCS_DONT_EXEC_STATE_REQ);
1114 /* ... and then immediately invalidate the request since we don't have the target coordinates set */
1115 fs_cubeadcs_update_adcs_sys_state_cfg_status(eSYSSTATECFGSTATUS_ERR);
1116 }
1117 else
1118 {
1119 /* Update the system state configuration */
1120 cubeadcs_set_state_cfg(new_sys_cfg, CUBEADCS_EXEC_STATE_REQ);
1121 }
1122
1123 break;
1124 }
1125 case eCUBEADCS_CFGTYPE_ORBITAL_CHANGE:
1126 {
1127 /* Save supplied 3D vector as roll/pitch/yaw angles instead of lon/lat/alt set points */
1128 /* Format them as CubeADCS requires */
1129
1130 if ( (fabs(p_ref->ref_param_1) > 180.0F) ||
1131 (fabs(p_ref->ref_param_2) > 180.0F) ||
1132 (fabs(p_ref->ref_param_3) > 180.0F) )
1133 {
1134 bRes = false;
1135
1136 orbital_change_roll = 0;
1137 orbital_change_pitch = 0;

1510static void fs_cubeadcs_task(void *arg)
1511{
1512 for(;;)
1513 {
1514 eCubeADCS_State_t eADCS_NewState;
1515 eCubeADCS_State_t eADCS_CacheState;
1516
1517 eADCS_CacheState = CubeADCS_GetState();
1518 eADCS_NewState = eADCS_CacheState;
1519
1520 switch ( eADCS_CacheState )
1521 {
1522 case eCUBEADCS_STATE_INIT_WAIT:
1523 {
1524 CUBEADCS_TRACE_DBG("%s INIT_WAIT_RUNS\r", CUBEADCS_HEAD);
1525
1526 /* Wait until CubeADCS starts in Application mode */
1527 if ( get_uptime_seconds() >= (u32BaseTime + CUBEADCS_PWR_UP_TO_APP_TIME_S) )
1528 {
1529 eADCS_NewState = eCUBEADCS_STATE_INIT;
1530 }
1531 else
1532 {
1533 CubeADCS_DelayTask(CUBEADCS_TASK_BASE_DELAY_MS);
1534 }
1535
1536 break;
1537 }
1538
1539 case eCUBEADCS_STATE_INIT:
1540 {
1541 CUBEADCS_TRACE_DBG("%s INIT_RUNS\r", CUBEADCS_HEAD);
1542
1543 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmdWithChecks(&aInitState[ eCUBEADCS_INIT_STATE_CMD_ID_128 ], CUBEADCS_DIAG_UPDATE_NOK, CUBEADCS_VALIDATE_DATA_TRUE) )
1544 {
1545 fs_cubeadcs_set_node_type((eCubeADCS_Mode_t)sResult_TLM128.NodeType);
1546
1547 eADCS_NewState = eCUBEADCS_STATE_HEALTH_CHECK;
1548
1549 for (uint8_t u8Indx = 1; u8Indx < eCUBEADCS_INIT_STATE_CMD_ID_NUM; ++u8Indx )
1550 {
1551 (void)CubeADCS_SendCmdWithChecks(&aInitState[u8Indx], CUBEADCS_DIAG_UPDATE_OK, CUBEADCS_VALIDATE_DATA_TRUE);
1552
1553 if ( eCUBEADCS_COMM_OK != fs_cubeadcs_get_last_comm_status() )
1554 {
1555 break;
1556 }
1557 }
1558 }
1559
1560 CubeADCS_DelayTask(CUBEADCS_TASK_BASE_DELAY_MS);
1561
1562 break;
1563 }
1564
1565 case eCUBEADCS_STATE_IDLE:
1566 {
1567 CUBEADCS_TRACE_DBG("%s IDLE_STARTS\r", CUBEADCS_HEAD);
1568
1569 /* Retrieve the current system state configuration */
1570 CubeADCS_SysSCfg_t sCurrentSysStateCfg;
1571 cubeadcs_get_adcs_state_cfg(&sCurrentSysStateCfg);
1572
1573 CUBEADCS_TRACE_DBG("%s CUBE_ADCS_eReqCfgStatus: %d\r", CUBEADCS_HEAD, sCurrentSysStateCfg.eReqCfgStatus);
1574 /* If there is additional logic required by this state configuration */
1575 /* and it -has not- been completed */
1576 if ( (0 != asCubeADCS_SysStates[sCurrentSysStateCfg.eReqCfg].uAdditionalLogic) &&
1577 (eSYSSTATECFGSTATUS_IN_PROGRESS == sCurrentSysStateCfg.eReqCfgStatus) )
1578 {
1579 uint8_t u8Res = 0;
1580 /* Call the additional logic handler */
1581 u8Res = fs_cubeadcs_execute_custom_logic(sCurrentSysStateCfg.eReqCfg);
1582
1583 /* Catch an error if there is one */
1584 if (255 == u8Res)
1585 {
1586 CubeADCS_ErrHandl(); // error handler maybe?
1587 }
1588
1589 /* Default task delay if there is logic being executed */
1590 CubeADCS_DelayTask(CUBEADCS_TASK_BASE_DELAY_MS);
1591 CUBEADCS_TRACE_DBG("%s IDLE_RUNS_CUSTOM_LOGIC\r", CUBEADCS_HEAD);
1592 }
1593 else
1594 {
1595 /* IDLE delay if there is no logic being executed */
1596 CubeADCS_DelayTask(CUBEADCS_TASK_IDLE_DELAY_MS);
1597 CUBEADCS_TRACE_DBG("%s IDLE_RUNS\r", CUBEADCS_HEAD);
1598 }
1599
1600 if ( (eCUBEADCS_STATE_IDLE == CubeADCS_GetState()) && ( eCUBEADCS_FTRANSF_STATE_IDLE == CubeADCS_FTransfer_GetTaskState()))
1601 {
1602 eADCS_NewState = eCUBEADCS_STATE_HEALTH_CHECK;
1603 }
1604
1605 break;
1606 }
1607
1608 case eCUBEADCS_STATE_HEALTH_CHECK:
1609 {
1610 CUBEADCS_TRACE_DBG("%s HEALTH_CHECK_RUNS\r", CUBEADCS_HEAD);
1611
1612 if ( eCUBEADCS_COMM_OK == CubeADCS_SendCmdWithChecks(&aHealthCheckState[ eCUBEADCS_IDLE_STATE_CMD_ID_128 ], CUBEADCS_DIAG_UPDATE_NOK, CUBEADCS_VALIDATE_DATA_TRUE) )
1613 {
1614 fs_cubeadcs_set_node_type((eCubeADCS_Mode_t)sResult_TLM128.NodeType);
1615
1616 for ( uint8_t u8Indx = 1; u8Indx < eCUBEADCS_IDLE_STATE_CMD_ID_NUM; ++u8Indx )
1617 {
1618 (void)CubeADCS_SendCmdWithChecks(&aHealthCheckState[u8Indx], CUBEADCS_DIAG_UPDATE_OK, CUBEADCS_VALIDATE_DATA_TRUE);
1619
1620 if ( eCUBEADCS_COMM_OK != fs_cubeadcs_get_last_comm_status() )
1621 {
1622 break;
1623 }
1624 }
1625 }
1626
1627 if ( eCUBEADCS_MODE_BOOTLOADER == CubeADCS_GetNodeType() )
1628 {
1629 if ( true == fs_cubeadcs_is_boot_commanded() )
1630 {
1631 eADCS_NewState = eCUBEADCS_STATE_IDLE;
1632 }
1633 }
1634 else {
1635 CUBEADCS_TRACE_DBG("%s HEALTH_CHECK_ERR_HANDL_RUNS\r", CUBEADCS_HEAD);
1636
1637 CubeADCS_ErrHandl();
1638
1639 eADCS_NewState = eCUBEADCS_STATE_PREP_DETUMB;
1640 }
1641 break;
1642 }
1643
1644 case eCUBEADCS_STATE_SYSSTATE_CHANGE:
1645 {
1646 CUBEADCS_TRACE_DBG("%s SYS_STATE_CHANGE_ENTRY\r", CUBEADCS_HEAD);
1647
1648 eCubeADCS_StdRply eExecRetVal;
1649 CubeADCS_SysSCfg_t sSysStateCfg;
1650 CubeADCS_SysSCfg_t sUpdatedSysStateCfg;
1651
1652 /* Retrieve the desired system state configuration */
1653 cubeadcs_get_adcs_state_cfg(&sSysStateCfg);
1654
1655 CUBEADCS_TRACE_DBG("%s State Change Attempt: %02d\r", CUBEADCS_HEAD, sSysStateCfg.eReqCfgStatus);
1656
1657 switch( sSysStateCfg.eReqCfgStatus )
1658 {
1659 case eSYSSTATECFGSTATUS_ERR:
1660 {
1661 /* Too many attempts to change state were made, throw an error and fall through to IDLE */
1662 cubeadcs_errhandl_activate(CUBEADCS_DIAG_ERR_HANDL_CNTS_NUM);
1663 CUBEADCS_TRACE_DBG("%s SYS_STATE_CHANGE_ERROR\r", CUBEADCS_HEAD);
1664 }
1665
1666 case eSYSSTATECFGSTATUS_OK:
1667 {
1668 /* Return to IDLE state */
1669 eADCS_NewState = eCUBEADCS_STATE_IDLE;
1670
1671 /* If it's an actual "OK", not just a fall through from ERR above */
1672 if (eSYSSTATECFGSTATUS_OK == sSysStateCfg.eReqCfgStatus)
1673 {
1674 CUBEADCS_TRACE_DBG("%s SYS_STATE_CHANGE_OK\r", CUBEADCS_HEAD);
1675 }
1676
1677 break;
1678 }
1679
1680 case eSYSSTATECFGSTATUS_IN_PROGRESS:
1681 {
1682 /* Return to IDLE, but keep the state change operation IN_PROGRESS */
1683 eADCS_NewState = eCUBEADCS_STATE_IDLE;
1684 CUBEADCS_TRACE_DBG("%s SYS_STATE_CHANGE_PROGRESS\r", CUBEADCS_HEAD);
1685 break;
1686 }
1687
1688 default:
1689 {
1690 /* Attempt to change the CubeADCS state */
1691 eExecRetVal = fs_CubeADCS_ApplyNewSysStateCfg(sSysStateCfg.eReqCfg);
1692
1693 /* Retrieve the now updated system state configuration */
1694 cubeadcs_get_adcs_state_cfg(&sUpdatedSysStateCfg);
1695
1696 /* Before requesting a new state change load the configuration from the NVM */
1697 if(eSYSSTATECFGSTATUS_NEW_REQ == sUpdatedSysStateCfg.eReqCfgStatus)
1698 {
1699 cubeadcs_persist_load_sys_state_cfg(asCubeADCS_SysStates);
1700 }
1701
1702 /* If the system state configuration successfully changed to the desired one */
1703 if ( (sSysStateCfg.eReqCfgStatus == sUpdatedSysStateCfg.eReqCfgStatus) &&
1704 ( sSysStateCfg.eReqCfg == sUpdatedSysStateCfg.eReqCfg) &&
1705 ( eCUBEADCS_STDRPLY_OK == eExecRetVal ))
1706 {
1707 /* If the system state configuration requires additional logic to be executed */
1708 if (asCubeADCS_SysStates[sSysStateCfg.eReqCfg].uAdditionalLogic)
1709 {
1710 CUBEADCS_TRACE_DBG("%s CUSTOM_STATE_ENTERED: %d\r", CUBEADCS_HEAD, sSysStateCfg.eReqCfg);
1711 /* Move the system state change operation to IN_PROGRESS */
1712 sSysStateCfg.eReqCfgStatus = eSYSSTATECFGSTATUS_IN_PROGRESS;
1713 }
1714 else
1715 {
1716 /* Signal that the system state change operation has completed */
1717 sSysStateCfg.eReqCfgStatus = eSYSSTATECFGSTATUS_OK;
1718 }
1719 }
1720 else
1721 {
1722 /* Increase the attempts counter */
1723 sSysStateCfg.eReqCfgStatus++;
1724 }
1725
1726 /* Update the system state configuration */
1727 fs_cubeadcs_update_adcs_sys_state_cfg_status(sSysStateCfg.eReqCfgStatus);
1728
1729 break;
1730 }
1731 }
1732
1733 CUBEADCS_TRACE_DBG("%s SYS_STATE_CHANGE_EXIT\r\r", CUBEADCS_HEAD);
1734 break;
1735 }
1736
1737 case eCUBEADCS_STATE_PREP_DETUMB:
1738 {
1739 CUBEADCS_TRACE_DBG("%s PREP_DETUMBLING_RUNS\r", CUBEADCS_HEAD);
1740

2481 }
