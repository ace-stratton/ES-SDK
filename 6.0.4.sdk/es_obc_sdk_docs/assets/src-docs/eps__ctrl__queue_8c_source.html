124eps_ctrl_result_t eps_ctrl_queue_add(eps_ctrl_channels_t ch, bool state, uint32_t type, uint32_t * const p_req_flag)
125{
126 eps_ctrl_result_t res = EPS_CTRL_DONE;
127 uint32_t tmp_flag;
128
129 QUEUE_CRITICAL_START
130 // Verify that the queue is not full
131 if ((true == is_full()) || (false == eps_ctrl_cfg_get_wait_flag(queue_ctx.tail_idx, &tmp_flag)))
132 {
133 EXEH_vException(eEXEHSeverity_Error, eEXEHModuleID_EPS_CTRL, eEXEH_EPS_CTRL_QUEUE_FULL, __LINE__);
134 res = EPS_CTRL_FAIL;
135 }
136
137 if (EPS_CTRL_DONE == res){
138 // Add the request to the queue
139 queue_buffer[queue_ctx.tail_idx].wait_flag = tmp_flag;
140 queue_buffer[queue_ctx.tail_idx].channel = ch;
141 queue_buffer[queue_ctx.tail_idx].type = type;
142 queue_buffer[queue_ctx.tail_idx].state = state;
143 queue_buffer[queue_ctx.tail_idx].wait_cancelled = false;
144
145 *p_req_flag = queue_buffer[queue_ctx.tail_idx].wait_flag;
146
147 // Increase the tail
148 queue_ctx.tail_idx = next_idx(queue_ctx.tail_idx);
149 }
150 QUEUE_CRITICAL_END
151
152 return res;
153}

155void eps_ctrl_queue_pop(void)
156{
157 QUEUE_CRITICAL_START
158 if(false == is_empty())
159 {
160 // Reset the values of the request
161 queue_buffer[queue_ctx.head_idx].channel = EPS_CTRL_CH_COUNT;
162 queue_buffer[queue_ctx.head_idx].type = EPS_CTRL_REQ_INVALID;
163
164 queue_ctx.head_idx = next_idx(queue_ctx.head_idx);
165 }
166 QUEUE_CRITICAL_END

169eps_ctrl_request_t* eps_ctrl_queue_find(const eps_ctrl_channels_t ch, const uint32_t flag)
170{
171 eps_ctrl_request_t* res = NULL;
172 uint32_t idx = queue_ctx.head_idx;
173 uint32_t end_idx = queue_ctx.tail_idx;
174 const uint32_t upper = CDEF_ELEMENT_COUNT(queue_buffer) - 1;
175 bool rollover = false;
176
177 QUEUE_CRITICAL_START
178
179 while ((idx != end_idx) && (NULL == res))
180 {
181 if (idx > upper)
182 {
183 // Index is out of the queue buffer's upper bound. Perform a roll-over in
184 // hope that the tail index of the queue is somewhere between 0 and the head one
185 if (false == rollover)
186 {
187 // This is the first time the index rolls over the buffer size.
188 // Continue searching for the tail index
189 rollover = true;
190 idx = 0;
191 }
192 else
193 {
194 // This is the second time the index rolls over the buffer size.
195 // Tail index is probably corrupted and the search shall be aborted
196 break;
197 }
198 }
199
200 if ((queue_buffer[idx].channel == ch) && (queue_buffer[idx].wait_flag == flag))
201 {
202 res = &queue_buffer[idx];
203 }
204
205 idx++;
206 }
207 QUEUE_CRITICAL_END
208
209 return res;
210}
