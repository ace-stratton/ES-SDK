55void sys_conf_init(void) {
56
57 // Calculate the CRC of the configuration data
58 uint32_t calculated_crc32 = MACCRC32_Calc32((unsigned char*)&p_system_configuration->data,
59 (uint32_t)p_system_configuration->size,
60 CRC32_SEED);
61 // Exit if the CRC doesn't match
62 if (calculated_crc32 != p_system_configuration->crc)
63 {
64 set_error_flag();
65 return;
66 }
67
68 // Verify the signature byte by byte
69 for(uint32_t i = 0; i < CDEF_ELEMENT_COUNT(CONFIG_SIGNATURE); i++)
70 {
71 if (p_system_configuration->sig[i] != CONFIG_SIGNATURE[i])
72 {
73 set_error_flag();
74 return;
75 }
76 }
77
78 // Possible version checks
79 const uint16_t minimum_version = 0x0002;
80 if (p_system_configuration->version < minimum_version)
81 {
82 set_error_flag();
83 return;
84 }
85
86 // Perform the module compatibility checks
87 if(!module_compatibility_check())
88 {
89 set_error_flag();
90 }
91
92 return;
93}

116// There should be a function for every kind of requirement or the if/else nesting will get too complex to follow
117static bool module_compatibility_check(void) {
118 // IMPORTANT: All checks should be done with IS_TRUE so in case of a single event upset (SEU)
119 // we don't assume a module is connected.
120 // i.e. if a value is set to safe true (0xFF), if a SEU changed bit, would return false
121 // on a IS_TRUE check. The safe assumption is to fall to false in case of doubt
122 bool b_compatible = true;
123
124 // Verify that the configuration is compatible
125 const safe_bool_t* p_modules = p_system_configuration->data.mod_activation_state;
126
127 //If the ES_ADCS is active, the EPS_I must be connected too.
128 if( b_compatible
129 && IS_TRUE(p_system_configuration->data.mod_activation_state[SYS_MODULE_ID_ES_ADCS])
130 && (!IS_TRUE(p_system_configuration->data.mod_activation_state[SYS_MODULE_ID_EPS_I])))
131 {
132 b_compatible = false;
133 }
134 // Verify the EPS modules compatibility
135 if(b_compatible)
136 {
137 b_compatible = eps_compatibility_check(p_modules);
138 }
139
140 // Verify the ADCS modules compatibility
141 if(b_compatible){
142 b_compatible = adcs_compatibility_check(p_modules);
143 }
144
145 return b_compatible;
146}

148// Example of a compatibility check: have one, but only one, type of EPS
149static bool eps_compatibility_check(const safe_bool_t* p_modules){
150 // IMPORTANT: All checks should be done with IS_TRUE so in case of a single event upset (SEU)
151 // we don't assume a module is connected.
152 // i.e. if a value is set to safe true (0xFF), if a SEU changed bit, would return false
153 // on a IS_TRUE check. The safe assumption is to fall to false in case of doubt
154
155 bool b_compatible = true;
156
157 if (IS_TRUE(p_modules[SYS_MODULE_ID_EPS_I]))
158 {
159 if ( IS_TRUE(p_modules[SYS_MODULE_ID_EPS_II]) || IS_TRUE(p_modules[SYS_MODULE_ID_EPS_M]) )
160 {
161 b_compatible = false; // We already have EPS I active
162 }
163 }
164 else // EPS I not active
165 {
166 if (IS_TRUE(p_modules[SYS_MODULE_ID_EPS_II]))
167 {
168 if (IS_TRUE(p_modules[SYS_MODULE_ID_EPS_M]))
169 {
170 b_compatible = false; // EPS II is active already
171 }
172 }
173 else // None EPS I and EPS II are active
174 {
175 if (IS_TRUE(p_modules[SYS_MODULE_ID_EPS_M]))
176 {
177 b_compatible = true; // Only third party EPS is active
178 }
179 else
180 {
181 b_compatible = false; // No EPS module active
182 }
183 }
184 }
185
186 return b_compatible;
187}

189static bool adcs_compatibility_check(const safe_bool_t * p_modules){
190 // IMPORTANT: All checks should be done with IS_TRUE so in case of a single event upset (SEU)
191 // we don't assume a module is connected.
192 // i.e. if a value is set to safe true (0xFF), if a SEU changed bit, would return false
193 // on a IS_TRUE check. The safe assumption is to fall to false in case of doubt
194
195 bool b_compatible = true;
196
197 if (IS_TRUE(p_modules[SYS_MODULE_ID_ES_ADCS]))
198 {
199 if (IS_TRUE(p_modules[SYS_MODULE_ID_CUBE_ADCS]) || IS_TRUE(p_modules[SYS_MODULE_ID_CUBE_ADCS_GEN2]))
200 {
201 b_compatible = false; // ES ADCS is already active
202 }
203 } else if (IS_TRUE(p_modules[SYS_MODULE_ID_CUBE_ADCS]) && IS_TRUE(p_modules[SYS_MODULE_ID_CUBE_ADCS_GEN2]))
204 {
205 b_compatible = false; // We cannot have both Gen1 and Gen2 CubeADCS modules active
206 }
207 else
208 {
209 // ADCS configuration is consistent
210 }
211
212 return b_compatible;
213}
