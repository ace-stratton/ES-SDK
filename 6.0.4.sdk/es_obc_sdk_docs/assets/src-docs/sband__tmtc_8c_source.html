41void sband_tmtc_init(void) {

56bool sband_tmtc_is_active(void) {
57 bool res = false;
58
59 CRIT_ASSERT(fs_lock != NULL);
60
61 if (NULL != fs_lock) {
62 if (osOK == osMutexAcquire(fs_lock, FS_MUTEX_TIMEOUT)) {
63 res = fs_is_active;
64 osMutexRelease(fs_lock);
65 }
66 }
67
68 return res;
69}

85static void fs_thread_routine(void *args) {
86 (void) args;
87
88 ReqContext_t ctx;
89 ctx.nInterfaceNumber = ESSASNI_SYS_PRI;
90 ctx.nAddr = sband_tmtc_cfg_get_mac_address();
91 ctx.netType = ESSASNETT_INTERNAL;
92 ctx.seqId = 0;
93
94 TaskMonitor_TaskInitialized(TASK_SBAND_TMTC);
95
96 while (1) {
97 osDelay(2000);
98
99
100 TaskMonitor_IamAlive(TASK_SBAND_TMTC);
101
102 (void) SBAND_TMTC_ReadUpTimeReq(&ctx);
103
104 bool received = false;
105
106 if (FS_RECEIVED_FLAG
107 == osThreadFlagsWait(FS_RECEIVED_FLAG, osFlagsWaitAny, 1000)) {
108 received = true;
109 }
110
111 if (osOK == osMutexAcquire(fs_lock, FS_MUTEX_TIMEOUT)) {
112 fs_is_active = received;
113 osMutexRelease(fs_lock);
114 }
115
116 vApplicationLowStackCheck(TASK_SBAND_TMTC);
117 }
