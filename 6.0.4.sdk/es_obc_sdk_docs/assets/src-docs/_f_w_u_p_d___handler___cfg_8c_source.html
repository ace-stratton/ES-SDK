61static uint32_t process_file_footer(const char * const pFileName, sTransferInfo_t * const pTransferInfo)
62{
63 CRIT_ASSERT(NULL != pTransferInfo);
64
65 FRESULT fres = FR_INT_ERR;
66 FIL fHnd;
67 uint32_t footer_size = 0U;
68
69 fres = SdMngr_f_open(&fHnd, pFileName, FA_READ | FA_OPEN_EXISTING);
70
71 if (FR_OK == fres)
72 {
73 if (SdMngr_f_size(&fHnd) >= sizeof(pTransferInfo->footer_ex))
74 {
75 fres = SdMngr_f_lseek(&fHnd, SdMngr_f_size(&fHnd) - sizeof(pTransferInfo->footer_ex));
76
77 if (FR_OK == fres)
78 {
79 UINT bytes_read = 0U;
80
81 fres = SdMngr_f_read(&fHnd, &pTransferInfo->footer_ex, sizeof(pTransferInfo->footer_ex), &bytes_read);
82
83 if ((FR_OK == fres) && (bytes_read == sizeof(pTransferInfo->footer_ex)))
84 {
85 bool ext_footer_sig_valid = (0 == memcmp(&pTransferInfo->footer_ex.ext_bun_sig[0], EXT_BUN_FOOTER_SIG, sizeof(pTransferInfo->footer_ex.ext_bun_sig)));
86
87 pTransferInfo->has_valid_footer = false;
88
89 if (ext_footer_sig_valid)
90 {
91 uint16_t ext_hdr_crc = crc16_ccitt_table((const uint8_t *) &pTransferInfo->footer_ex, sizeof(pTransferInfo->footer_ex) - sizeof(pTransferInfo->footer_ex.ext_bun_crc));
92
93 if (ext_hdr_crc == pTransferInfo->footer_ex.ext_bun_crc)
94 {
95 pTransferInfo->has_valid_footer = ('\0' != pTransferInfo->footer_ex.file_name[0]);
96 footer_size = sizeof(pTransferInfo->footer_ex);
97 }
98 }
99 }
100 }
101 }
102 }
103
104 (void) SdMngr_f_close(&fHnd);
105
106 return footer_size;
107}

134static void handle_file_ext(const char * const p_file_name)
135{
136 bool ext_handler_found = false;
137
138 if (NULL != p_file_name)
139 {
140 const char *p_dot_pos = strchr(p_file_name, '.');
141 const char *p_str_end = strchr(p_file_name, '\0');
142
143 if ((NULL != p_dot_pos) && (NULL != p_str_end) && (p_str_end > p_dot_pos) &&
144 (((uint32_t) p_str_end - (uint32_t) p_dot_pos - 1U) <= EXT_MAX_SIZE))
145 {
146 for (uint8_t file_ext_idx = 0; file_ext_idx < CDEF_ELEMENT_COUNT(file_ext_map); file_ext_idx++)
147 {
148 if ((0 == strcmp(p_dot_pos + 1, file_ext_map[file_ext_idx].uppercase_ext)) ||
149 (0 == strcmp(p_dot_pos + 1, file_ext_map[file_ext_idx].lowercase_ext)))
150 {
151 p_file_handler_t fhnd = file_ext_map[file_ext_idx].p_file_ext_handler;
152
153 if (NULL != fhnd)
154 {
155 (*fhnd)(p_file_name);
156 }
157
158 ext_handler_found = true;
159
160 break;
161 }
162 }
163 }
164
165 if (!ext_handler_found)
166 {
167 default_file_handler(p_file_name);
168 }
169 }
170}

229bool FWUPD_Handler_Cfg_HandleTransferComplete(sTransferInfo_t * const pTransferInfo,
230 const uint64_t u64ActiveTransferId,
231 const char * const pFileName)
232{
233 bool bTransferHandled = false;
234 eFwUpdPersistorResult finalizeTransferRes;
235
236 (void) u64ActiveTransferId;
237
238 if ((pTransferInfo != NULL) && (pFileName != NULL))
239 {
240 const SESSoftwareUPD_Bundle * const psHeader = &pTransferInfo->header;
241 uint32_t num_bytes_to_strip = 4U; // uint32_t FWUPD bundle CRC size at the end of the bundle file
242
243 // process and skip file footer if it exists
244 num_bytes_to_strip += process_file_footer(pFileName, pTransferInfo);
245
246 // remove any extra bytes added by GS in the update bundle
247 finalizeTransferRes = FWUPD_Persistor_FinalizeTransfer(num_bytes_to_strip);
248
249 if (finalizeTransferRes == EFWUPDPERSISTORRESULT_OK)
250 {
251 if ((psHeader->nModuleType == FW_OBC_MODULE_TYPE1) ||
252 (psHeader->nModuleType == FW_OBC_MODULE_TYPE2))
253 {
254 switch (psHeader->nSubModule)
255 {
256 // regular OBC update bundle received
257 case FW_OBC_SUBMODULE_SELF:
258 {
259 if (FWUPD_Persistor_PrepareImage(BOOTLDR_AUTOFLASH_FILE_PATH) == EFWUPDPERSISTORRESULT_OK)
260 {
261 // Switch to Bootloader mode to perform the actual flashing...
262 WriteHandler_ResetInBootOrAppMode(APPMODE_AUTO_FW_UPDATE);
263
264 bTransferHandled = true;
265 }
266 else
267 {
268 EXEH_HANDLE(eEXEHSeverity_Warning, EXC_FWUPD_COULD_NOT_PREPARE_IMAGE);
269 }
270
271 break;
272 }
273
274 // Specific files received
275 case FW_OBC_SUBMODULE_FILE:
276 {
277 char fileName[FWUPD_MAX_FILE_NAME_SIZE] = { '\0' };
278
279 if (true == pTransferInfo->has_valid_footer)
280 {
281 uint16_t hdr_ex_fname_sz = (uint16_t) strlen((const char *) pTransferInfo->footer_ex.file_name);
282
283 if (hdr_ex_fname_sz <= sizeof(fileName))
284 {
285 (void) memcpy(fileName, pTransferInfo->footer_ex.file_name, hdr_ex_fname_sz);
286 fileName[hdr_ex_fname_sz] = '\0';
287 }
288 else
289 {
290 (void) memcpy(fileName, pTransferInfo->footer_ex.file_name, sizeof(fileName));
291 fileName[sizeof(fileName) - 1] = '\0';
292 }
293 }
294
295 if (SESUL_FL_ARCHIVED == (psHeader->nFlags & SESUL_FL_ARCHIVED))
296 {
297 if (false == pTransferInfo->has_valid_footer)
298 {
299 snprintf(fileName, sizeof(fileName), "%sup_%05d.lz7", FWUPD_Persistor_GetStorageDir(), psHeader->nFWVerMaj);
300 }
301
302 LZ77_Init();
303 LZ77_Decompress((char *) aTempImageFileName, fileName, 60000);
304
305 // the LZ77 decompression will create the actual fileName, the following call performs only
306 // cleanup of temporary files when argument is set to NULL
307 (void) FWUPD_Persistor_PrepareImage(NULL);
308 } else if (SESUL_FL_FILE_TLE == (psHeader->nFlags & SESUL_FL_FILE_TLE))
309 {
310 if (false == pTransferInfo->has_valid_footer)
311 {
312 snprintf(fileName, sizeof(fileName), "%ssat.tle", FWUPD_Persistor_GetStorageDir());
313 }
314
315 (void) FWUPD_Persistor_PrepareImage(fileName);
316 TleUpdateFromFile(fileName);
317 } else if (SESUL_FL_PAYLOAD_SCHEDULE == (psHeader->nFlags & SESUL_FL_PAYLOAD_SCHEDULE))
318 {
319 if (false == pTransferInfo->has_valid_footer)
320 {
321 snprintf(fileName, sizeof(fileName), "%s%s", FWUPD_Persistor_GetStorageDir(), "paysched.bin");
322 }
323
324 (void) FWUPD_Persistor_PrepareImage(fileName);
325 PayloadScheduler_Reload(fileName);
326 } else if (SESUL_FL_FILE_ZIP == (psHeader->nFlags & SESUL_FL_FILE_ZIP))
327 {
328 if (false == pTransferInfo->has_valid_footer)
329 {
330 snprintf(fileName, sizeof(fileName), "%sup_%05d.zip", FWUPD_Persistor_GetStorageDir(), psHeader->nFWVerMaj);
331 }
332
333 (void) FWUPD_Persistor_PrepareImage(fileName);
334 } else if (SESUL_FL_FILE_CMD == (psHeader->nFlags & SESUL_FL_FILE_CMD))
335 {
336 if (false == pTransferInfo->has_valid_footer)
337 {
338 snprintf(fileName, sizeof(fileName), "%sup_%05d.cmd", FWUPD_Persistor_GetStorageDir(), psHeader->nFWVerMaj);
339 }
340
341 (void) FWUPD_Persistor_PrepareImage(fileName);
342 } else
343 {
344 if (false == pTransferInfo->has_valid_footer)
345 {
346 snprintf(fileName, sizeof(fileName), "%sup_%05d.upl", FWUPD_Persistor_GetStorageDir(), psHeader->nFWVerMaj);
347 }
348
349 (void) FWUPD_Persistor_PrepareImage(fileName);
350
351 handle_file_ext(fileName);
352 }
353
354 bTransferHandled = true;
355 break;
356 }
357
358 default:
359 EXEH_HANDLE(eEXEHSeverity_Warning, EXC_FWUPD_UNSUPPORTED_IMAGE_TYPE);
360 break;
361 }
362 }
363 }
364 else
365 {
366 EXEH_HANDLE(eEXEHSeverity_Warning, EXC_FWUPD_COULD_NOT_FINALIZE_IMAGE);
367 }
368 }
369 else
370 {
371 EXEH_HANDLE(eEXEHSeverity_Warning, EXC_FWUPD_INVALID_ARGS);
372 }
373
374 return bTransferHandled;
375}
