61static bool file_read(void) // we are counting on single-thread access to the routines in this module
62{
63 FIL file;
64 FRESULT res;
65 bool ret = false;
66
67 if ('\0' != sched_file_name[0])
68 {
69 (void) memset(&file, 0, sizeof(file));
70 res = SdMngr_f_open(&file, sched_file_name, FA_READ | FA_OPEN_EXISTING);
71 if (res == FR_OK)
72 {
73 UINT bytes_read;
74 SdMngr_f_lseek(&file, 0);
75 res = SdMngr_f_read(&file, &schedule_data, sizeof(schedule_data), &bytes_read);
76 SdMngr_f_close(&file);
77 if (bytes_read > 0)
78 {
79 schedule_dirty = false;
80 ret = true;
81 }
82 }
83 }
84
85 return ret;
86}

88static bool file_write(void) // we are counting on single-thread access to the routines in this module

94 memset(&file, 0, sizeof(file));
95 res = SdMngr_f_open(&file, sched_file_name, FA_WRITE | FA_CREATE_ALWAYS);

112static void records_clear(void)
113{
114 for (uint16_t i = 0; i < MAX_SCHEDULED_RECORDS; i++)
115 {
116 record_t *const pRecord = &(schedule_data.records[i]);
117 pRecord->payload_id = PAYLOAD_EMPTY_RECORD;
118 pRecord->time = 0;
119 pRecord->duration = 0;

128// order of checking "evaluate_XXX" functions is important - optimization issue
129static inline bool evaluate_expired(const record_time_t now, record_t *const pRecord)
130{
131 bool ret = false;
132 if (now > (pRecord->time + pRecord->duration))
133 {
134 // the record is expired - no sense to waste time on it!
135 ret = true;
136 }
137
138 return ret;
139}
140
141// order of checking "evaluate_XXX" functions is important - optimization issue
142static bool evaluate_time_action(const record_time_t now, record_t *const pRecord)
143{
144 if (now < (pRecord->time - pRecord->payload_boot_time - pRecord->adcs_args.adcs_prep_time))
145 {
146 return false;
147 }
148
149 return true;
150}

152bool PayloadScheduler_addRecord(const uint8_t payload_id, const record_time_t time, const uint16_t duration, record_adcs_t *const adcs_data, const uint8_t flags, const uint8_t *const pay_args)
153{
154 for (uint16_t i = 0; i < MAX_SCHEDULED_RECORDS; i++)
155 {
156 record_t *const pRecord = &(schedule_data.records[i]);
157
158 // if the record is currently not in use
159 if (pRecord->payload_id >= PAYLOAD_COUNT)
160 {
161 pRecord->duration = duration;
162 if (pRecord->duration < MINIMAL_POSSIBLE_DURATION)
163 {
164 pRecord->duration = MINIMAL_POSSIBLE_DURATION;
165 }
166
167 pRecord->time = time;
168 memcpy(&pRecord->adcs_args, adcs_data, sizeof(record_adcs_t));
169
170 pRecord->flags = flags;
171 memcpy(&pRecord->payload_args[0], pay_args, sizeof(uint8_t) * 16);
172
173 if (payload_id < PAYLOAD_COUNT)
174 {
175 pRecord->payload_id = payload_id;
176 schedule_reload = false;
177 return true;
178 }
179 else
180 {
181 // cannot map type! invalid input type!
182 return false;
183 }
184 }
185 }
186
187 return false;
188}

195static void psched_update_conops_flags(void)
196{
197 // we are counting on periodic calls of the following routine
198 PayloadScheduler_state_t conops_flag[PAYLOAD_COUNT];
199 uint16_t conops_flag_record_idx[PAYLOAD_COUNT];
200 memset(conops_flag, 0, sizeof(conops_flag));
201 memset(conops_flag_record_idx, 0, sizeof(conops_flag_record_idx));
202
203 const record_time_t now = (record_time_t) datetime_to_unix_now();
204
205 for (uint16_t i = 0; i < MAX_SCHEDULED_RECORDS; i++)
206 {
207 record_t *const pRecord = &(schedule_data.records[i]);
208 const uint8_t payload_id = pRecord->payload_id;
209
210 if (payload_id < PAYLOAD_COUNT)
211 {
212 // order of checking "evaluate_XXX" functions is important - optimization issue
213 if (!evaluate_expired(now, pRecord))
214 {
215 if (true == evaluate_time_action(now, pRecord))
216 {
217 conops_flag[payload_id] = SCHED_STATE_ACTIVE_PAYLOAD;
218 conops_flag_record_idx[payload_id] = i;
219 }
220 }
221 }
222 else
223 {
224 pRecord->payload_id = PAYLOAD_EMPTY_RECORD;
225 }
226 }
227
228 for (int i = 0; i < PAYLOAD_COUNT; i++)
229 {
230 if (conops_payload_flags[i].flag != conops_flag[i])
231 {
232 conops_payload_flags[i].flag = conops_flag[i];
233 memcpy(&(conops_payload_flags[i].sched_data), &(schedule_data.records[conops_flag_record_idx[i]]), sizeof(record_t));
234
235 /* Trigger an event to the ConOps SM */
236 conops_payload_trigger_event(i);
237 }
238 }
239
240 if (true == schedule_dirty)
241 {
242 file_write();
243 }
244}

263void PayloadScheduler_Init(void)
264{
265 (void) memset(sched_file_name, 0, sizeof(sched_file_name));
266
267 for (uint16_t i = 0; i < MAX_SCHEDULED_RECORDS; i++)
268 {
269 schedule_data.records[i].payload_id = 255; // Empty record

272 PayloadScheduler_TaskHandle = osThreadNew(psched_task, NULL, &PayloadScheduler_attributes);

276static void psched_task(void * argument)
277{
278 schedule_reload = true;
279
280 TaskMonitor_TaskInitialized(TASK_PAYLOAD_SCHEDULER);
281
282 for( ; ; )
283 {
284 if (true == schedule_reload)
285 {
286 if (true == file_read())
287 {
288 schedule_reload = false;
289 }
290 }
291 else
292 {
293 psched_update_conops_flags();
294 }
295
296 osDelay(1000);
297
298 TaskMonitor_IamAlive(TASK_PAYLOAD_SCHEDULER);
299 vApplicationLowStackCheck(TASK_PAYLOAD_SCHEDULER);
300 }
301}
