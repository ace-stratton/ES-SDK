109STATIC sds_op_status_t sensor_gyro_get_data_in_urad_per_s(sds_gyro_data_t * const p_data_dst)
110{
111 CRIT_ASSERT(p_data_dst != NULL);
112
113 AxesRaw_t raw_gyro_data;
114
115 sds_op_status_t ret_res = SDS_OP_STATUS_FAIL;
116
117 /* Get rate data from all gyroscopes */
118 if ( GYRO_SUCCESS == ADIS16265_GetAxesRate(PANLE_GYROS_AXIS_ALL, &raw_gyro_data) )
119 {
120 SDS_TRACE_DBG("%s GYRO RAW ANG RATE---X:%d, Y:%d, Z:%d\r", SDS_DBG_HEAD, raw_gyro_data.AXIS_X, raw_gyro_data.AXIS_Y, raw_gyro_data.AXIS_Z);
121
122 p_data_dst->gyro_1_urads = sensor_gyro_conv_gyro_raw_axis_data_to_urad_s(raw_gyro_data.AXIS_X);
123 p_data_dst->gyro_2_urads = sensor_gyro_conv_gyro_raw_axis_data_to_urad_s(raw_gyro_data.AXIS_Y);
124 p_data_dst->gyro_3_urads = sensor_gyro_conv_gyro_raw_axis_data_to_urad_s(raw_gyro_data.AXIS_Z);
125
126 SDS_TRACE_DBG("%s GYRO CONV ANG RATE---X:%d, Y:%d, Z:%d\r", SDS_DBG_HEAD, p_data_dst->gyro_1_urads, p_data_dst->gyro_2_urads, p_data_dst->gyro_3_urads);
127
128 gyro_sens_status = SDS_SENSOR_STATUS_INIT;
129 gyro_err_code = SDS_GYRO_ERR_CODE_GYRO_NO_ERR;
130 ret_res = SDS_OP_STATUS_OK;
131 }
132 else
133 {
134 gyro_sens_status = SDS_SENSOR_STATUS_ERR;
135 gyro_err_code = SDS_GYRO_ERR_CODE_GYRO_ERR;
136
137 SDS_TRACE_DBG("%s Gyroscopes read failed, please check last error\r", SDS_DBG_HEAD);
138 }
139
140 return ret_res;
141}

146STATIC sds_op_status_t sensor_gyro_get_temp_data_in_deg_c(sds_gyro_temp_data_t * const p_data_dst)
147{
148 CRIT_ASSERT(p_data_dst != NULL);
149
150 Temperature_t gyro_temp_data;
151
152 sds_op_status_t ret_res = SDS_OP_STATUS_FAIL;
153
154 if ( GYRO_SUCCESS == ADIS16265_GetTemperature(&gyro_temp_data) )
155 {
156 p_data_dst->gyro_1_deg_c = gyro_temp_data.Temp_X;
157 p_data_dst->gyro_2_deg_c = gyro_temp_data.Temp_Y;
158 p_data_dst->gyro_3_deg_c = gyro_temp_data.Temp_Z;
159
160 SDS_TRACE_DBG("%s GYRO TEMP DEG C --- 1:%d, 2:%d, 3:%d\r", SDS_DBG_HEAD, p_data_dst->gyro_1_deg_c, p_data_dst->gyro_2_deg_c, p_data_dst->gyro_3_deg_c);
161
162 gyro_sens_status = SDS_SENSOR_STATUS_INIT;
163 gyro_err_code = SDS_GYRO_ERR_CODE_GYRO_NO_ERR;
164 ret_res = SDS_OP_STATUS_OK;
165 }

168 gyro_sens_status = SDS_SENSOR_STATUS_ERR;
169 gyro_err_code = SDS_GYRO_ERR_CODE_GYRO_ERR;
170
171 SDS_TRACE_DBG("%s Gyroscopes read failed, please check last error\r", SDS_DBG_HEAD);
172 }
173
174 return ret_res;

220STATIC int32_t sensor_gyro_conv_gyro_raw_axis_data_to_urad_s(int16_t raw_axis_data)
221{
222 return (raw_axis_data <= ADIS16265_GYRO_OUT_POS_MAX_VAL) ?
223 (int32_t)((( raw_axis_data * + SDS_GYRO_ANG_RATE_LSB) * (M_PI / 180.0F)) * SDS_GYRO_URAD_MULT) :
224 (int32_t)((((SDS_GYRO_ZERO_VALUE - raw_axis_data) * - SDS_GYRO_ANG_RATE_LSB) * (M_PI / 180.0F)) * SDS_GYRO_URAD_MULT);
225}

233STATIC sds_op_status_t sensor_gyro_get_register(uint8_t sens_id, sds_axis_enum_t sens_axis, uint32_t reg_address, uint32_t const *reg_value)
234{
235 CRIT_ASSERT(sens_axis < SDS_SENS_AXIS_NUM);
236 CRIT_ASSERT(NULL != reg_value);
237
238 (void)sens_id;
239
240 sds_op_status_t ret_val = SDS_OP_STATUS_OK;
241
242 gyroscope_t *gyroscope_to_read_reg_from;
243
244 gyroscope_t gyro_x;
245 gyroscope_t gyro_y;
246 gyroscope_t gyro_z;
247
248 ADIS16265_get_full_gyro_info(&gyro_x, &gyro_y, &gyro_z);
249
250 if ( sens_axis == SDS_SENS_AXIS_X )
251 {
252 gyroscope_to_read_reg_from = &gyro_x;
253 }
254 else if ( sens_axis == SDS_SENS_AXIS_Y)
255 {
256 gyroscope_to_read_reg_from = &gyro_y;
257 }
258 else
259 {
260 gyroscope_to_read_reg_from = &gyro_z;
261 }
262
263 if ( GYRO_SUCCESS != ADIS16265_ReadReg16(gyroscope_to_read_reg_from, (uint8_t)reg_address, (uint16_t *)reg_value) )
264 {
265 ret_val = SDS_OP_STATUS_FAIL;
266 }
267
268 return ret_val;
269}
