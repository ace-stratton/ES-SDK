69void conops_payload_trigger_event(uint8_t u8PaySchedId)
70{
71 CRIT_ASSERT(u8PaySchedId < PAYLOAD_COUNT);
72
73 sConOps_pay_cfg_t sPayCfg;
74
75 sPayCfg.payload_id = u8PaySchedId;
76 memcpy(&sPayCfg.pointing_cfg, &conops_payload_flags[ u8PaySchedId ].sched_data.adcs_args, sizeof(record_adcs_t));
77 sPayCfg.payload_args.flags = (uint8_t) conops_payload_flags[ u8PaySchedId ].sched_data.flags;
78 memcpy(&sPayCfg.payload_args.pl_args, &conops_payload_flags[ u8PaySchedId ].sched_data.payload_args, sizeof(sPayCfg.payload_args.pl_args));
79
80 if ( SCHED_STATE_ACTIVE_PAYLOAD == conops_payload_flags[ u8PaySchedId ].flag )
81 conops_trigger_hsm(ev_conops_sm_payload_flag_set, &sPayCfg);
82 else
83 conops_trigger_hsm(ev_conops_sm_payload_flag_cleared, &sPayCfg);
84}

112STATIC void fs_conops_payload_op_callback(payload_ctrl_event_t *p_event)
113{
114 CRIT_ASSERT(NULL != p_event);
115
116 /* Kick the SM on payload error */
117 if ( PL_STATE_ERROR == p_event->op_state)
118 {
119 conops_set_logical_flag_value(eFLAG_PAYLOAD_ERR, true);
120
121 conops_trigger_hsm(ev_conops_sm_payload_flag_cleared, NULL);
122 }
123 else
124 {
125 /* Kick the SM on an ADCS error in the ADCS system state configuration request */
126 if ( PL_STATE_ADCS_TIMEOUT <= p_event->op_state )
127 {
128 CONOPS_TRACE_DBG("[AAA DBG]: ----- PL_STATE_ERROR: %d ---------------------------------- \r", p_event->op_state);
129 conops_set_logical_flag_value(eFLAG_ADCS_ERR, true);
130
131 conops_trigger_hsm(eHSM_StdEvent_Periodic, NULL);
132 }
133 }
134}
