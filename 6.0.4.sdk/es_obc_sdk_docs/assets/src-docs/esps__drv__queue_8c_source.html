33bool MAC_QUEUE_vTXQueueAdd(sMACInterfaceState_t * fp_psMACIntf, sMACTXPoolEntry_t * fp_psEntry)
34{
35 bool result = false;
36
37 sMACTXPoolEntry_t *psCurrent = NULL;
38
39 if((fp_psMACIntf == NULL) || (fp_psEntry == NULL))
40 {
41 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t) eMACException_InvalidParameter);
42 }
43 else
44 {
45 MAC_vEnterCriticalSection(fp_psMACIntf);
46
47 if((fp_psMACIntf->sTXCtx.psQueue == NULL) ||
48 (MAC_PRIORITY(fp_psMACIntf->sTXCtx.psQueue->sPrio) >= MAC_PRIORITY(fp_psEntry->sPrio)))
49 {
50 // Set the entry as the one to be processed next
51 fp_psEntry->psNext = fp_psMACIntf->sTXCtx.psQueue;
52 fp_psMACIntf->sTXCtx.psQueue = fp_psEntry;
53 }
54 else
55 {
56 // Place the entry depending on its priority
57 psCurrent = fp_psMACIntf->sTXCtx.psQueue;
58
59 while((psCurrent->psNext != NULL) && (MAC_PRIORITY(psCurrent->sPrio) < MAC_PRIORITY(fp_psEntry->sPrio)))
60 {
61 psCurrent = psCurrent->psNext;
62 }
63
64 fp_psEntry->psNext = psCurrent->psNext;
65 psCurrent->psNext = fp_psEntry;
66 }
67
68 result = true;
69
70 MAC_vLeaveCriticalSection(fp_psMACIntf);
71 }
72
73 if(result)
74 fp_psMACIntf->sTXCtx.u32QueueElementsCount++;
75
76 return result;
77}

80sMACTXPoolEntry_t *MAC_QUEUE_vTXQueuePop(sMACInterfaceState_t * fp_psMACIntf)
81{
82 MAC_vEnterCriticalSection(fp_psMACIntf);
83
84 sMACTXPoolEntry_t *psEntry = fp_psMACIntf->sTXCtx.psQueue;
85
86 if(fp_psMACIntf->sTXCtx.psQueue)
87 {
88 fp_psMACIntf->sTXCtx.psQueue = fp_psMACIntf->sTXCtx.psQueue->psNext;
89
90 if(fp_psMACIntf->sTXCtx.u32QueueElementsCount > 0U)
91 --fp_psMACIntf->sTXCtx.u32QueueElementsCount;
92 }
93
94 MAC_vLeaveCriticalSection(fp_psMACIntf);
95
96 return psEntry;
97}
