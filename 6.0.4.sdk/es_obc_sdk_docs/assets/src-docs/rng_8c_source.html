48bool rng_fill(uint8_t * const p_buf, const uint32_t buf_size)
49{
50 bool res = false;
51
52 CRIT_ASSERT(NULL != p_buf);
53 CRIT_ASSERT(buf_size > 0);
54
55 if ( (NULL == p_buf) || (0 == buf_size) )
56 {
57 ES_TRACE_ERROR("RNG wrong params error: %d, %d\n", p_buf, buf_size);
58 return false;
59 }
60
61 if (HAL_OK != rng_res)
62 {
63 ES_TRACE_ERROR("RNG not initialized: %d\n", rng_res);
64 return false;
65 }
66
67 uint32_t rand_word = 0U;
68 uint32_t rn_cnt = 0;
69 uint32_t *p_num = (uint32_t *) p_buf;
70 uint32_t max_cnt = buf_size / sizeof(rand_word);
71 bool err = false;
72
73 while (rn_cnt < max_cnt)
74 {
75 rng_res = HAL_RNG_GenerateRandomNumber(&rng_hnd, &rand_word);
76
77 if (HAL_OK == rng_res)
78 {
79 *(p_num++) = rand_word;
80 rn_cnt++;
81 }
82 else
83 {
84 err = true;
85 break;
86 }
87 }
88
89 if (false == err)
90 {
91 // Fill any remaining bytes
92 max_cnt = buf_size % sizeof(rand_word);
93
94 if (max_cnt > 0)
95 {
96 rng_res = HAL_RNG_GenerateRandomNumber(&rng_hnd, &rand_word);
97
98 if (HAL_OK == rng_res)
99 {
100 // Fill the remaining bytes from the generated 32-bit random number
101 (void) memcpy(p_num, &rand_word, max_cnt);
102 }
103 else
104 {
105 err = true;
106 }
107 }
108 }
109
110 res = !err;
111
112 if (false == res)
113 {
114 ES_TRACE_ERROR("RNG hardware generation error: %d\n", rng_res);
115 }
116
117 return res;
118}
