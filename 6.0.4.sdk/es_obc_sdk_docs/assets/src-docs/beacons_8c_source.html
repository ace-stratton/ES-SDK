80void beacons_init(void)
81{
82 EXEH_INIT_MODULE_FILENAME();
83
84 beacons_rt_context.os_task_hnd = NULL;
85 beacons_rt_context.beacon_frame_rolling_cntr = 0U;
86
87 timer_stop(&beacons_rt_context.send_timer);
88
89 if (NULL == beacons_lock_id)
90 {
91 beacons_lock_id = osMutexNew(&beacons_lock_attr);
92 }
93
94 beacons_sink.cmd(BEACONS_SINK_ADAPTER_CMD_INIT);
95
96 beacons_trigger_hsm(eHSM_StdEvent_Reset);
97 beacons_trigger_hsm(ev_beacons_sm_init);
98
99 // this condition ensures that at least one byte of beacon data will be
100 // transferable within the specified buffer size, otherwise, this
101 // implementation does not make much sense for such a configuration and
102 // the size of the active_beacon_frame buffer has to be increased
103 CRIT_ASSERT((sizeof(beacons_rt_context.active_beacon_frame) >
104 (sizeof(beacon_frame_hdr_t) +
105 sizeof(beacon_msg_hdr_t))));
106}

109void beacons_start(void)
110{
111 STATIC const osThreadAttr_t task_attr =
112 {
113 .attr_bits = osThreadDetached,
114 .name = "Beacons_Service_Task",
115 .priority = osPriorityNormal,
116 .cb_mem = NULL,
117 .cb_size = 0U,
118 .stack_mem = NULL,
119 .stack_size = 2500 // TODO: Measure worst-case stack usage!
120 };
121
122 // create thread only if not already existing...
123 if (NULL == beacons_rt_context.os_task_hnd)
124 {
125 beacons_rt_context.os_task_hnd = osThreadNew(beacons_task,
126 (void *) NULL,
127 &task_attr);
128
129 if (NULL == beacons_rt_context.os_task_hnd)
130 {
131 EXEH_HANDLE(eEXEHSeverity_Fatal,
132 (int32_t) EXEH_BCN_CANNOT_SPAWN_TASK);
133 }
134 else
135 {
136 TaskMonitor_TaskInitialized(TASK_BEACONS);
137
138 beacons_sink.cmd(BEACONS_SINK_ADAPTER_CMD_START);
139 }
140 }
141}

167STATIC void beacons_task(void * p_task_arg)
168{
169 static safe_bool_t last_beacon_en_status = SAFE_FALSE;
170 bool hsm_result = false;
171
172 // currently unused
173 (void) p_task_arg;
174
175 beacons_load_config();
176
177 for (;;)
178 {
179 // beacons_rt_context.gen_cfg will be refreshed as part of the HSM implementation
180 if (last_beacon_en_status != beacons_rt_context.gen_cfg.beacons_en)
181 {
182 switch (beacons_rt_context.gen_cfg.beacons_en)
183 {
184 case SAFE_TRUE:
185 {
186 hsm_result = beacons_trigger_hsm(ev_beacons_sm_start);
187
188 break;
189 }
190
191 case SAFE_FALSE:
192 {
193 hsm_result = beacons_trigger_hsm(ev_beacons_sm_stop);
194
195 break;
196 }
197
198 default:
199 {
200 CRIT_ASSERT(false);
201 break;
202 }
203 }
204
205 // make sure the HSM event will be fired until it gets processed by the HSM
206 if (hsm_result)
207 {
208 last_beacon_en_status = beacons_rt_context.gen_cfg.beacons_en;
209 }
210 }
211
212 // execute periodic SM activities...
213 (void) beacons_trigger_hsm(eHSM_StdEvent_Periodic);
214
215 osDelay(BEACON_TASK_PERIOD_MS);
216
217 TaskMonitor_IamAlive(TASK_BEACONS);
218 vApplicationLowStackCheck(TASK_BEACONS);
219 }
220}

222STATIC bool inline beacons_trigger_hsm(const uint32_t event_id)
223{
224 bool res = false;
225
226 osMutexAcquire(beacons_lock_id, osWaitForever);
227 res = HSM_TriggerEvent(beacons_sm_instance, event_id);
228 osMutexRelease(beacons_lock_id);
229
230 return res;
231}
