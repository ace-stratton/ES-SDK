64static inline void telemetry_file_sink_write_hdr(FIL * const f_handle, tlm_file_hdr_t * const p_hdr)
65{
66 CRIT_ASSERT(NULL != f_handle);
67 CRIT_ASSERT(NULL != p_hdr);
68 UINT bw = 0;
69
70 telemetry_file_sink_update_hdr_crc(p_hdr);
71
72 uint32_t prev_pos = SdMngr_f_tell(f_handle);
73 // set file pointer to beginning of file in order to update the header
74 FRESULT fr = SdMngr_f_lseek(f_handle, 0U);
75
76 if (FR_OK != fr)
77 {
78 ES_TRACE_ERROR("file seek error [%ld]", fr);
79 return;
80 }
81
82 fr = SdMngr_f_write(f_handle, (const void *) p_hdr, sizeof(tlm_file_hdr_t), &bw);
83
84 if (FR_OK != fr)
85 {
86 ES_TRACE_ERROR("tlm header write error [%ld]", fr);
87 return;
88 }
89
90 // restore file pointer where it was...
91 (void) SdMngr_f_lseek(f_handle, prev_pos);

175static void telemetry_file_sink_find_first_slot(void)
176{
177 FRESULT res;
178 tlm_file_hdr_t fhdr;
179 FIL fhnd;
180 uint16_t last_valid_file_index = tlm_filesink_rt_context.current_idx;
181 struct {
182 uint16_t last_invalid_file_index;
183 bool is_index_valid;
184 } last_invalid_file_info = { .last_invalid_file_index = 0, .is_index_valid = false };
185 UINT bytes_read = 0;
186
187 bool partial_match_found = false;
188
189 CRIT_ASSERT(tlm_filesink_rt_context.max_file_size_bytes >= TLM_MSG_SIZE_BYTES_LEAST);
190
191 tlm_filesink_rt_context.current_idx = 0U;
192
193 while (tlm_filesink_rt_context.current_idx < tlm_filesink_rt_context.cfg.max_log_files_cnt)
194 {
195 telemetry_file_sink_prepare_fname(tlm_filesink_rt_context.current_idx);
196
197 ES_TRACE_DEBUG("opening '%s'", tlm_filesink_rt_context.current_file_name);
198 res = SdMngr_f_open(&fhnd, tlm_filesink_rt_context.current_file_name, FA_OPEN_EXISTING | FA_READ);
199
200 if (FR_OK == res)
201 {
202 // Note: This initialization is important in case of a failing SdMngr_f_read() call
203 (void) memset(&fhdr, 0U, sizeof(fhdr));
204 res = SdMngr_f_read(&fhnd, (void *) &fhdr, sizeof(fhdr), &bytes_read);
205 (void) SdMngr_f_close(&fhnd);
206
207 if ((FR_OK == res) && (sizeof(fhdr) == bytes_read) && (telemetry_file_sink_is_fhdr_valid(&fhdr)))
208 {
209 // store last valid file index - it may be complete or partial...
210 last_valid_file_index = tlm_filesink_rt_context.current_idx;
211
212 // check if writing to this file can continue...
213 if (!telemetry_file_sink_is_file_complete(&fhdr, &fhnd))
214 {
215 ES_TRACE_DEBUG("[%u] partial match", tlm_filesink_rt_context.current_idx);
216 partial_match_found = true;
217 // we found the last partial file...
218 break;
219 }
220 else
221 {
222 ES_TRACE_DEBUG("[%u] file complete", tlm_filesink_rt_context.current_idx);
223 }
224
225 // if we reach a full or partial file, then we should not restore writing to the last invalid file, thus
226 // invalidate last invalid file index
227 last_invalid_file_info.is_index_valid = false;
228 last_invalid_file_info.last_invalid_file_index = tlm_filesink_rt_context.current_idx;
229 }
230 else
231 {
232 ES_TRACE_DEBUG("[%u] invalid file header", tlm_filesink_rt_context.current_idx);
233 // save last invalid file index to reuse if there are no any other partial slots after it in the sequence
234 last_invalid_file_info.last_invalid_file_index = tlm_filesink_rt_context.current_idx;
235 last_invalid_file_info.is_index_valid = true;
236 }
237 }
238 else if (FR_NO_FILE == res)
239 {
240 // we didn't find a partial file and there are still free slots and maybe an invalid file slot to reuse...
241 last_valid_file_index = (last_invalid_file_info.is_index_valid) ?
242 last_invalid_file_info.last_invalid_file_index :
243 tlm_filesink_rt_context.current_idx;
244 // in this case, we still don't have all slot files created on the SD card, so we must
245 // create a new slot file and continue from the current index instead of going back to 0 which
246 // will be done once we completely fill all slot files on the SD card
247 partial_match_found = true;
248
249 break;
250 }
251 else
252 {
253 partial_match_found = false;
254 }
255
256 // make sure that the telemetry index is always changed regardless of error to
257 // guarantee that the while loop is not infinite
258 tlm_filesink_rt_context.current_idx++;
259 }
260
261 // in case the file was not closed above...
262 (void) SdMngr_f_close(&fhnd);
263
264 // prepare last valid file for use... if the last valid file was partially filled, continue with it, otherwise starts from slot 0
265 tlm_filesink_rt_context.current_idx = partial_match_found ? last_valid_file_index : 0U;
266 telemetry_file_sink_prepare_fname(tlm_filesink_rt_context.current_idx);
267 ES_TRACE_DEBUG("preparing tlm file: '%s'", tlm_filesink_rt_context.current_file_name);
268}

314static void telemetry_file_sink_prepare_next_slot(void)
315{
316 switch (tlm_filesink_rt_context.cfg.cfg_type)
317 {
318 case TLM_FILE_CFG_TYPE_TIME_LIMIT:
319 {
320 break;
321 }
322
323 case TLM_FILE_CFG_TYPE_SIZE_LIMIT:
324 default:
325 {
326 tlm_filesink_rt_context.current_idx =
327 (tlm_filesink_rt_context.current_idx + 1U) % tlm_filesink_rt_context.cfg.max_log_files_cnt;
328
329 telemetry_file_sink_prepare_fname(tlm_filesink_rt_context.current_idx);
330
331 // reinitilize the new slot file header
332 FIL tlm_file_handle;
333 tlm_file_hdr_t fhdr;
334 UINT bytes_read = 0;
335 UINT bytes_written = 0;
336
337 FRESULT res = SdMngr_f_open(&tlm_file_handle,
338 tlm_filesink_rt_context.current_file_name,
339 FA_OPEN_ALWAYS | FA_READ | FA_WRITE);

353
354 if ((FR_OK == res) && (sizeof(fhdr) == bytes_written))
355 {
356 ES_TRACE_DEBUG("[%u] next slot header prepared for use", tlm_filesink_rt_context.current_idx);
357 }

362 }

590// GEN_OBJ_START::telemetry_file_sink_sm_action_store_telemetry
591void telemetry_file_sink_sm_action_store_telemetry(sState_t * const pSMRoot, const sHSM_EventType_t * const pEvent)
592{
593 // USER_CODE_START::action_store_telemetry
594 FRESULT res;
595 FIL tlm_file_handle;
596 tlm_file_hdr_t fhdr;
597 UINT bytes_read;
598
599 (void) pSMRoot;
600 (void) pEvent;
601
602 telemetry_queue_update_stats(TLM_STATS_ID_ACTIVE_SLOT, tlm_filesink_rt_context.current_idx);
603
604 (void) memset(&fhdr, 0U, sizeof(fhdr));
605 res = SdMngr_f_open(&tlm_file_handle,
606 tlm_filesink_rt_context.current_file_name,
607 FA_OPEN_ALWAYS | FA_READ | FA_WRITE);
608
609 if (FR_OK == res)
610 {
611 // verify file header...
612 res = SdMngr_f_read(&tlm_file_handle, &fhdr, sizeof(fhdr), &bytes_read);
613
614 if ((FR_OK == res) && (sizeof(fhdr) == bytes_read) && (telemetry_file_sink_is_fhdr_valid(&fhdr)))
615 {
616 fhdr.next_write_offset = (telemetry_file_sink_is_file_complete(&fhdr, &tlm_file_handle)) ?
617 (sizeof(tlm_file_hdr_t)) :
618 (fhdr.next_write_offset);
619
620 // restore file position pointer based on information from the header...
621 SdMngr_f_lseek(&tlm_file_handle,
622 fhdr.next_write_offset);
623 }
624 else
625 {
626 telemetry_file_sink_init_header(&fhdr);
627
628 telemetry_file_sink_write_hdr(&tlm_file_handle, &fhdr);
629
630 // skip header and set file pointer position for first message
631 (void) SdMngr_f_lseek(&tlm_file_handle, sizeof(fhdr));
632 }
633
634 uint16_t next_msg_size = telemetry_queue_peek();
635 uint32_t free_bytes = telemetry_file_sink_getfreebytes();
636 uint32_t batch_msg_count = 0U;
637
638 if (free_bytes > next_msg_size)
639 {
640 while (next_msg_size > 0U)
641 {
642 UINT bytes_written = 0U;
643
644 if (telemetry_file_sink_shall_write(fhdr.next_write_offset, next_msg_size))
645 {
646 uint16_t bytes_read_from_queue = telemetry_queue_pop(tlm_filesink_rt_context.tlm_msg_buffer,
647 sizeof(tlm_filesink_rt_context.tlm_msg_buffer));
648
649 CRIT_ASSERT(bytes_read_from_queue == next_msg_size);
650
651 res = SdMngr_f_write(&tlm_file_handle,
652 tlm_filesink_rt_context.tlm_msg_buffer,
653 bytes_read_from_queue,
654 &bytes_written);
655
656 fhdr.next_write_offset = SdMngr_f_tell(&tlm_file_handle);
657
658 if ((FR_OK != res) || (bytes_written != bytes_read_from_queue))
659 {
660 EXEH_HANDLE_EX(eEXEHSeverity_Error,
661 eEXEHModuleID_TELEMETRY_FILESINK,
662 (int32_t) EXEH_TLM_FS_FILE_ERROR);
663
664 ES_TRACE_DEBUG("[%u] tlm wrt error [%d]", tlm_filesink_rt_context.current_idx, res);
665
666 break;
667 }
668 else
669 {
670 fhdr.file_complete = false;
671
672 telemetry_file_sink_write_hdr(&tlm_file_handle, &fhdr);
673
674 next_msg_size = telemetry_queue_peek();
675
676 batch_msg_count++;
677
678 ES_TRACE_DEBUG("[%u] tlm wrt @ offs %lu", tlm_filesink_rt_context.current_idx, fhdr.next_write_offset);
679 }
680 }
681 else
682 {
683 ES_TRACE_DEBUG("no space in current file [%u]. switching to next slot...", tlm_filesink_rt_context.current_idx);
684
685 // TODO: timestamp is currently unused
686 fhdr.last_timestamp = 0U;
687 fhdr.file_complete = true;
688 // next writes in this file will start from
689 fhdr.next_write_offset = sizeof(tlm_file_hdr_t);
690
691 // Note: Order is important here in case of intermediate reset after this point! The next file slot must be
692 // prepared first before marking the current one as complete, otherwise we may end up with the current file complete and
693 // no partial slot to continue from next time after reset.
694 telemetry_file_sink_prepare_next_slot();
695
696 // call this after telemetry_file_sink_prepare_next_slot() to ensure correct recovery in case of CPU reset between these
697 // two calls
698 telemetry_file_sink_write_hdr(&tlm_file_handle, &fhdr);
699
700 break;
701 }
702 }
703
704 // report statistics on last run...
705 telemetry_queue_update_stats(TLM_STATS_ID_BATCH_COUNT, batch_msg_count);
706 }
707
708 (void) SdMngr_f_close(&tlm_file_handle);
709 }
710
711 // USER_CODE_END::action_store_telemetry
712}
