31#define Re ((sgp4_t)6378.137) // Earth Radius in [km]
32#define CK2 ((sgp4_t)5.413179e-4) // (1/2)*J2*aE^2 in Re^2 units (aE = 1) -> also known as k2

59sgp4_err_t sgp4_propagator(const sgp4_parsed_two_line_element_t* parsed_tle, const sgp4_t jd, sgp4_vec_3d_t* r, sgp4_vec_3d_t* v)
60{
61 if (parsed_tle == NULL) { return SGP4_ERR; }
62 if (r == NULL) { return SGP4_ERR; }
63 if (v == NULL) { return SGP4_ERR; }
64
65 const sgp4_err_t param_check_res = sgp4_check_orbit_parameters(parsed_tle);
66
67 if ( SGP4_OK != param_check_res )
68 {
69 return param_check_res;
70 }
71
72 const sgp4_t deltaT = (jd - parsed_tle->epoch + 1.0) * 1440.0;
73
74 // Preparation - convert angles to radians to be used in trig functions and assign
75 // some values to variables from TLE
76 sgp4_t INC = to_rad(parsed_tle->incl);
77 sgp4_t AOP = to_rad(parsed_tle->aop);
78 sgp4_t RAAN = to_rad(parsed_tle->raan);
79 sgp4_t MA0 = to_rad(parsed_tle->manomaly);
80 sgp4_t ECC = parsed_tle->ecc;
81 sgp4_t MM0 = parsed_tle->mmotion * TWO_PI / MpD; // Read in revolutions per day and renorm with n -> n*2*pi/MinDay (1440 sec assumed in a day)
82 sgp4_t Theta = cos(INC);
83
84 /************ WP1 - Recover original mean motion (n0) and semimajor axis (a0) from input elements through some helping constants */
85
86 sgp4_t SMA = pow(XKE / MM0, 2.0 / 3.0); // Original SMA from mean motion
87 sgp4_t TV1 = 1.5 * CK2 * (3.0 * pow2(Theta) - 1.0) / (pow2(SMA) * pow(1.0 - pow2(ECC), 1.5)); // d1) correction kept in TV1
88 SMA = (1.0 - TV1 / 3.0 - pow2(TV1) - 134.0 * pow3(TV1) / 81.0) * SMA; // a0 first correction from d1 kept in TV1
89 TV1 = 1.5 * CK2 * (3.0 * pow2(Theta) - 1.0) / (pow2(SMA) * pow(1.0 - pow2(ECC), 1.5)); // d0 correction kept in TV1
90 MM0 /= (1.0 + TV1); // Recovered original mean motion with d0 correction kept in TV1
91 SMA /= (1.0 - TV1); // Recovered original SMA with d0 correction kept in TV1
92
93 // Used to extract Keplerian elements at epoch after de-averaging operation - not required for the code !!!
94 //printf( "SMA as of epoch is %.10e \n" , SMA*Re );
95
96 /************ WP2 - work to be completed here for orbits going lower than 156 km altitude (final days) - not relevant for stable orbit */
97
98 // Implement checking of a0 to determine which s value should be used !!!
99 TV1 = ((1.0 - ECC) * SMA - 1.0) * Re; // Gives perigee height in [km] to be used for testing - kept in TV1
100
101 if ((TV1 <= 156.0) && (TV1 > 98.0))
102 {
103 PRINTF("Scenario 98 < hp < 156 km \n");
104 return SGP4_ERR;
105 }
106 else if (TV1 <= 98.0)
107 {
108 PRINTF("Scenario hp < 98 - your sat is basically down :/ \n");
109 return SGP4_ERR;
110 }
111 else
112 {
113 PRINTF("Sat is still high :P \n");
114 }
115 /************ WP3 - Calculate some constants to be used in propagation */
116
117 const sgp4_t s = 1.01222928;
118 const sgp4_t q0s = 1.88027916 * 1e-9; // default values of s and (q0-s)^4 in Earth radii and Earth Radii to power of 4
119
120 TV1 = 1.0 / (SMA - s); // Note that Xi is kept in TV1
121 sgp4_t TV2 = sqrt(1.0 - pow2(ECC)); // Note that B0 is kept in TV2
122 sgp4_t TV3 = SMA * ECC * TV1; // Note that Eta is kept in TV3
123
124 // TV4 holds the value of C4 constant
125 sgp4_t TV4 = q0s * pow4(TV1) * MM0 * pow(1.0 - pow2(TV3), -3.5) * (SMA * (1.0 + 1.5 * pow2(TV3) + 4.0 * ECC * TV3 + ECC * pow3(TV3)) + 1.5 * CK2 * TV1 * (-0.5 + 1.5 * pow2(Theta)) * (8.0 + 24.0 * pow2(TV3) + 3.0 * pow4(TV3)) / (1.0 - pow2(TV3)));
126 sgp4_t C1V = parsed_tle->bstar * TV4; // Note that TV4 holds the value of C2 for 2 rows (its only used do compute C1)
127 sgp4_t TV5a = q0s * pow5(TV1) * A30 * aE * MM0 * sin(INC) / (ECC * CK2); // Holds the value of C3 constant which is used only once further down
128
129 // TV4 is used to keep line reasonably long
130 TV4 = 3.0 * (1.0 - 3.0 * pow2(Theta)) * (1.0 + 1.5 * pow2(TV3) - 2.0 * ECC * TV3 - 0.5 * ECC * pow3(TV3)) + 0.75 * (1.0 - pow2(Theta)) * (2.0 * pow2(TV3) - ECC * TV3 - ECC * pow3(TV3)) * cos(2.0 * AOP);
131 TV4 = 2.0 * MM0 * q0s * pow4(TV1) * SMA * pow2(TV2) * pow(1.0 - pow2(TV3), -3.5) * ((2.0 * TV3 * (1.0 + ECC * TV3) + 0.5 * ECC + 0.5 * pow3(TV3)) - 2.0 * CK2 * TV1 * TV4 / (SMA * (1.0 - pow2(TV3))));
132
133 // TV4 is given the value of C4 to hold (one use further down)
134 sgp4_t TV5 = 2.0 * q0s * pow4(TV1) * SMA * pow2(TV2) * pow(1.0 - pow2(TV3), -3.5) * (1.0 + 11.0 * TV3 * (TV3 + ECC) / 4.0 + ECC * pow3(TV3)); // Holds value of C5 constant
135 sgp4_t D2V = 4.0 * SMA * TV1 * pow2(C1V); // D2 constant
136 sgp4_t D3V = 4.0 * SMA * pow2(TV1) * (17.00 * SMA + s) * pow3(C1V) / 3.0; // D3 constant
137 sgp4_t D4V = 2.0 * SMA * pow3(TV1) * (221.0 * SMA + 31.0 * s) * pow4(C1V) / 3.0; // D4 constant
138
139 /************ WP4 - Calculate secular effects of drag and gravity */
140
141 // Mean Anomaly at epoch first correction (M_DF)
142 sgp4_t MAn = MA0 + (1.0 + 3.0 * CK2 * (3.0 * pow2(Theta) - 1.0) / (2.0 * pow2(SMA) * pow3(TV2)) + 3.0 * pow2(CK2) * (13.0 - 78.0 * pow2(Theta) + 137.0 * pow4(Theta)) / (16.0 * pow4(SMA) * pow7(TV2))) * MM0 * deltaT;
143
144 // Argument of perigee intermediate value (w_DF)
145 AOP = AOP + (-3.0 * CK2 * (1.0 - 5.0 * pow2(Theta)) / (2.0 * pow2(SMA) * pow4(TV2)) + 3.0 * pow2(CK2) * (7.0 - 114.0 * pow2(Theta) + 395.0 * pow4(Theta)) / (16.0 * pow4(SMA) * pow8(TV2)) + 5.0 * CK4 * (3.0 - 36.00 * pow2(Theta) + 49.00 * pow4(Theta)) / (4.00 * pow4(SMA) * pow8(TV2))) * MM0 * deltaT;
146
147 // RAAN intermediate value (Om_DF) */
148 RAAN = RAAN + (-3.0 * CK2 * Theta / (pow2(SMA) * pow4(TV2)) + 3.0 * pow2(CK2) * (4.0 * Theta - 19.0 * pow3(Theta)) / (2.0 * pow4(SMA) * pow8(TV2)) + 5.0 * CK4 * Theta * (3.0 - 7.0 * pow2(Theta)) / (2.0 * pow4(SMA) * pow8(TV2))) * MM0 * deltaT;
149
150 sgp4_t TV6 = parsed_tle->bstar * TV5a * cos(to_rad(parsed_tle->aop)) * deltaT; // dw kept in TV6
151 sgp4_t TV7 = -2.0 * q0s * parsed_tle->bstar * pow4(TV1) * aE * (pow3(1.0 + TV3 * cos(MAn)) - pow3(1.0 + TV3 * cos(MA0))) / (3.0 * ECC * TV3); // dM kept in TV7
152 MAn = MAn + TV6 + TV7; // Mp = M_DF + dw + dM -> Mean Anomaly
153 AOP = AOP - TV6 - TV7; // w = w_DF - dw - dM -> Argument of Perigee
154 RAAN = RAAN - 10.5 * MM0 * CK2 * Theta * C1V * pow2(deltaT) / (pow2(SMA) * pow2(TV2)); // Final RAAN value -> Om
155 ECC = ECC - parsed_tle->bstar * (TV4 * deltaT + TV5 * (sin(MAn) - sin(MA0))); // Compute final eccentricity
156 SMA = SMA * pow2(1.0 - C1V * deltaT - D2V * pow2(deltaT) - D3V * pow3(deltaT) - D4V * pow4(deltaT)); // Compute final SMA
157
158 /************ WP5 - Long term periodic effects */
159
160 // Lb value is kept in TV1 variable
161 TV1 = MAn + AOP + RAAN + MM0 * (1.5 * C1V * pow2(deltaT) + (D2V + 2.0 * pow2(C1V)) * pow3(deltaT) + 0.25 * (3.0 * D3V + 12.0 * C1V * D2V + 10.0 * pow3(C1V)) * pow4(deltaT) + 0.20 * (3.0 * D4V + 12.0 * C1V * D3V + 6.00 * pow2(D2V) + 30.0 * pow2(C1V) * D2V + 15.0 * pow4(C1V)) * pow5(deltaT));
162
163 // Beta value is kept in TV2 variable
164 TV2 = sqrt(1.0 - pow2(ECC));
165
166 // Final Mean Motion (n)
167 MM0 = XKE / pow(SMA, 1.5); // multiply by 1440 (MinDay) and divide by 2*pi to get actual number of revolutions per day
168 TV3 = ECC * cos(AOP); // a_xN kept in TV3 from here
169 TV4 = A30 * sin(INC) * ECC * cos(AOP) * (3.0 + 5.0 * Theta) / (8.0 * CK2 * SMA * pow2(TV2) * (1.0 + Theta)); // Lb_L kept in TV4
170 TV5 = A30 * sin(INC) / (4.0 * CK2 * SMA * pow2(TV2)); // a_yNL kept in TV5
171 TV1 = TV4 + TV1; // Computing Lb_T = Lb + Lb_L which is now kept in TV1
172 TV5 = ECC * sin(AOP) + TV5; // Computing a_yN = a_yNL + e*sin( w )
173
174 // Now Solve Kepler's equation to obtain TA at required time
175 TV5a = TV1 - RAAN; // U is kept in TV5a
176 TV2 = TV5a; // TV2 will be used as variable in the loop -> seeded with Lb_T - Om value
177 TV4 = Ktol + 1.0; // TV4 will be used for TV4 variable which measures deviation from 2 consequent iterations
178
179 unsigned int i = 0; // Counter which stops the loop if it does not converge in Ntol times
180 while ((fabs(TV4) > Ktol) && (i < Ntol))
181 {
182 // delta(E+w) = ( U - ayN cos( E + w ) + axN sin( E + w ) - ( E + w ) )/( - ayN sin( E + w ) - axN cos( E + w ) + 1 )
183 // In these iterations TV1 takes values of E + w -> True Anomaly
184 TV4 = (TV5a - TV5 * cos(TV1) + TV3 * sin(TV1) - TV1) / (-TV5 * sin(TV1) - TV3 * cos(TV1) + 1.0);
185 TV1 += TV4;
186 i++;
187 }
188 if (i == Ntol)
189 {
190 PRINTF("The loop for Kepler's eq did not converge to required tolerance, mate! Check it perhaps!\n");
191 return SGP4_ERR;
192 }
193
194 // At this point TV1 holds the value of E + w and we can proceed to find final values
195
196 // Used to extract Keplerian elements at epoch after de-averaging operation - not required for the code !!!
197 //TV1 = SGP4_twoPiMod(TV1);
198 //printf("True anomaly at epoch desired is given by %.10e deg \n", TV1*180.0/PI);
199
200 /************ WP6 - Find final values of all orbital parameters at epoch */
201
202 TV2 = TV3 * cos(TV1) + TV5 * sin(TV1); // e*cos( E ) is held in TV2 variable
203 TV4 = TV3 * sin(TV1) - TV5 * cos(TV1); // e*sin( E ) is held in TV4 variable
204 TV5a = sqrt(pow2(TV3) + pow2(TV5)); // e_L is kept in TV5a
205 TV6 = SMA * (1.0 - pow2(TV5a)); // pL - semi-latus rectum is kept in TV6
206 sgp4_t RAD;
207 RAD = SMA * (1.0 - TV2); // Keeps preliminary radius value (r)
208 C1V = XKE * sqrt(SMA) * TV4 / RAD; // C1V keeps value of rdot -> radial component of velocity
209 D2V = XKE * sqrt(TV6) / RAD; // D2V keeps value of r*fdot -> angular component of velocity
210 D3V = (cos(TV1) - TV3 + TV5 * TV4 / (1.0 + sqrt(1.0 - pow2(TV5a)))) * SMA / RAD; // Cosine of eccentric anomaly cos( u )
211 D4V = (sin(TV1) - TV5 - TV3 * TV4 / (1.0 + sqrt(1.0 - pow2(TV5a)))) * SMA / RAD; // Sine of eccentric anomaly sin( u )
212 sgp4_t ECCA;
213 ECCA = atan(D4V / D3V); // u is inverted from tg(u) and kept for use further down in ECCA variable
214 TV1 = 2.0 * D3V * D4V; // sin (2u) is kept in variable TV1
215 TV2 = 2.0 * D3V * D3V - 1.0; // cos (2u) is kept in variable TV2
216 TV3 = CK2 * (1.0 - pow2(Theta)) * TV2 / (2.0 * TV6); // TV3 holds dr from this point
217 TV4 = -CK2 * (7.0 * pow2(Theta) - 1.0) * TV1 / (4.0 * pow2(TV6)); // TV4 holds du from this point
218 MAn = 1.5 * CK2 * Theta * TV1 / pow2(TV6); // Holds dOm value from this point
219 MA0 = 1.5 * CK2 * Theta * sin(INC) * TV2 / pow2(TV6); // Holds di value from this point
220 TV5 = -CK2 * MM0 * (1.0 - pow2(Theta)) * TV1 / TV6; // TV5 holds drdot from this point
221 TV7 = CK2 * MM0 * ((1.0 - pow2(Theta)) * TV2 - 1.5 * (1.0 - 3.0 * pow2(Theta))) / TV6; // TV7 holds drfdot from this point
222
223 // Final Values for parameters are
224 RAD = RAD * (1.0 - 1.5 * CK2 * sqrt(1.0 - pow2(TV5a)) * (3.0 * pow2(Theta) - 1.0) / pow2(TV6)) + TV3; // Radial distance ( r ) at desired time
225 ECCA = ECCA + TV4; // Final value for Elliptic Anomaly u at desired time
226 RAAN = RAAN + MAn; // Final value for RAAN at desired time
227 INC = INC + MA0; // Final value for inclination at desired time
228 C1V = C1V + TV5; // Final value for radial velocity at desired time
229 D2V = D2V + TV7; // Final value for angular component of velocity at desired time -> rfdot
230
231 // Reduce by 2*pi -> only useful if you need the actual angles and not necessary in general computations
232 //Om = SGP4_twoPiMod(Om);
233 //i0 = SGP4_twoPiMod(i0);
234 //w = SGP4_twoPiMod(w);
235
236 /************ WP7 - compute position and velocity at required Time and record them in output vectors */
237
238 TV5a = sin(INC); // TV5a keeps sine of inclination
239 Theta = cos(INC); // Theta once again keeps cosine of inclination
240 TV1 = cos(RAAN); // TV1 keeps cos( RAAN ) for final orbit orientation computations
241 TV2 = sin(RAAN); // TV2 keeps sin( RAAN ) for final orbit orientation computations
242 TV3 = Re * RAD / aE; // TV3 keeps radial norm of orbital position
243 TV4 = Re * MpD / StD; // TV4 keeps velocity norm -> up to this point it is in units of earth radii per minute -> correct to km/sec
244
245 // Used to extract Keplerian elements at epoch after de-averaging operation - not required for the code !!!
246 //printf("ECC, RAAN, INC and AOP are %.10e %.10e %.10e %.10e \n" , ECC , RAAN*180.0/PI , INC*180.0/PI , AOP*180.0/PI );
247
248 // Output vectors for the current moment */
249 // \vec{r} = r \vec{U} */
250 r->x = TV3 * (-TV2 * Theta * D4V + TV1 * D3V);
251 r->y = TV3 * ( TV1 * Theta * D4V + TV2 * D3V);
252 r->z = TV3 * TV5a * D4V;
253
254 //printf( "Final r has value %.10e \n" , r*Re );
255
256 // \dot{\vec{r}} = \dot{r}\vec{U} + r\dot{f}\vec{V}
257 v->x = (-(C1V * D4V + D2V * D3V) * TV2 * Theta + (C1V * D3V - D2V * D4V) * TV1) * TV4;
258 v->y = ( (C1V * D4V + D2V * D3V) * TV1 * Theta + (C1V * D3V - D2V * D4V) * TV2) * TV4;
259 v->z = ( (C1V * D4V + D2V * D3V) * TV5a) * TV4;
260
261 // In above notation U is unit radial vector and V is
262 // unit tangent vector to trajectory ( U x V would give 3d vector forming orbit local frame )
263
264 return SGP4_OK;
265}

267sgp4_err_t sgp4_check_orbit_parameters(const sgp4_parsed_two_line_element_t* const p_parsed_tle)
268{
269 if (NULL == p_parsed_tle) { return SGP4_ERR; }
270 if (p_parsed_tle->bstar > BSTAR_MAX_LIMIT) { return SGP4_BSTAR_ERR; }
271 if ((INC_MAX_LIMIT < p_parsed_tle->incl) || (INC_MIN_LIMIT > p_parsed_tle->incl)) { return SGP4_INC_ERR; }
272 if ((RAAN_MAX_LIMIT < p_parsed_tle->raan) || (RAAN_MIN_LIMIT > p_parsed_tle->raan)) { return SGP4_RAAN_ERR; }
273 if ((ECC_MAX_LIMIT < p_parsed_tle->ecc) || (ECC_MIN_LIMIT > p_parsed_tle->ecc)) { return SGP4_ECC_ERR; }
274 if ((AOP_MAX_LIMIT < p_parsed_tle->aop) || (AOP_MIN_LIMIT > p_parsed_tle->aop)) { return SGP4_AOP_ERR; }
275 if ((MA_MAX_LIMIT < p_parsed_tle->manomaly) || (MA_MIN_LIMIT > p_parsed_tle->manomaly)) { return SGP4_MA_ERR; }
276 if ((MM_MAX_LIMIT < p_parsed_tle->mmotion) || (MM_MIN_LIMIT > p_parsed_tle->mmotion)) { return SGP4_MM_ERR; }
277
278 return SGP4_OK;
279}
