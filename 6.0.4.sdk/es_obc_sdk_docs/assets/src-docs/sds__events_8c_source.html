96void sds_init_events_timer(const sds_sens_read_freq_t *sds_events_cfg)
97{
98 CRIT_ASSERT(NULL != sds_events_cfg);
99
100 RCC_ClkInitTypeDef clkconfig;
101
102 uint32_t uwTimclock;
103 uint32_t uwAPB1Prescaler;
104 uint32_t uwPrescalerValue;
105 uint32_t pFLatency;
106
107 /* Start by de-initilising the timer */
108 sds_deinit_events_timer();
109
110 for (uint8_t f_num = 0; f_num < SDS_SENSTYPE_MAX_CNT; f_num++)
111 {
112 sds_events_chan_freqs[f_num] = sds_events_cfg->sens_cfg[f_num].freq_cfg > 0 ? SDS_CONV_FREQ_TO_TICK_CNT(sds_events_cfg->sens_cfg[f_num].freq_cfg): 0;
113 sds_events_chan_events[f_num] = sds_events_cfg->sens_cfg[f_num].event_cfg;
114 }
115
116 /* Enable TIM3 and TIM4 clocks */
117 __HAL_RCC_TIM3_CLK_ENABLE();
118 __HAL_RCC_TIM4_CLK_ENABLE();
119
120 /* Get clock configuration */
121 HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
122
123 /* Get APB1 prescaler */
124 uwAPB1Prescaler = clkconfig.APB1CLKDivider;
125
126 /* Compute TIM3/4 clock */
127 if (uwAPB1Prescaler == RCC_HCLK_DIV1)
128 {
129 uwTimclock = HAL_RCC_GetPCLK1Freq();
130 }
131 else
132 {
133 uwTimclock = 2UL * HAL_RCC_GetPCLK1Freq();
134 }
135
136 /* Compute the prescaler value to have TIM3/4 counter clock equal to 10kHz */
137 uwPrescalerValue = ((uwTimclock / SDS_EVENTS_TIMER_TICK_FREQ) - 1U);
138
139 /* Initialise TIM3 */
140 htim3.Instance = TIM3;
141 htim3.Init.Period = SDS_EVENTS_TIMER_TICK_FREQ - 1U;
142 htim3.Init.Prescaler = uwPrescalerValue;
143 htim3.Init.ClockDivision = 0U;
144 htim3.Init.CounterMode = TIM_COUNTERMODE_UP;
145 htim3.Init.AutoReloadPreload = 0U;
146 htim3.Init.RepetitionCounter = 0U;
147
148 /* Initialise TIM4 */
149 htim4.Instance = TIM4;
150 htim4.Init.Period = SDS_EVENTS_TIMER_TICK_FREQ - 1U;
151 htim4.Init.Prescaler = uwPrescalerValue;
152 htim4.Init.ClockDivision = 0U;
153 htim4.Init.CounterMode = TIM_COUNTERMODE_UP;
154 htim4.Init.AutoReloadPreload = 0U;
155 htim4.Init.RepetitionCounter = 0U;
156
157 if ( (HAL_OK != HAL_TIM_Base_Init(&htim3)) || (HAL_OK != HAL_TIM_Base_Init(&htim4)) )
158 {
159 Error_Handler();
160 }
161
162 /* Set priority levels */
163 HAL_NVIC_SetPriority(TIM3_IRQn, 5, 0);
164 HAL_NVIC_EnableIRQ(TIM3_IRQn);
165
166 HAL_NVIC_SetPriority(TIM4_IRQn, 5, 0);
167 HAL_NVIC_EnableIRQ(TIM4_IRQn);
168
169 /* Initialise OC channels for TIM3 */
170 if ( (true != sds_events_init_oc_chan(&htim3, &htim3_oc_chan_1, TIM_CHANNEL_1, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_MTM])) ||
171 (true != sds_events_init_oc_chan(&htim3, &htim3_oc_chan_2, TIM_CHANNEL_2, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO])) ||
172 (true != sds_events_init_oc_chan(&htim3, &htim3_oc_chan_3, TIM_CHANNEL_3, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_CSS])) ||
173 (true != sds_events_init_oc_chan(&htim3, &htim3_oc_chan_4, TIM_CHANNEL_4, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_ACC])) )
174 {
175 Error_Handler();
176 }
177
178 /* Initialise OC channels for TIM4 */
179 if ( (true != sds_events_init_oc_chan(&htim4, &htim4_oc_chan_1, TIM_CHANNEL_1, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_PANEL_TEMP])) ||
180 (true != sds_events_init_oc_chan(&htim4, &htim4_oc_chan_2, TIM_CHANNEL_2, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO_TEMP])) ||
181 (true != sds_events_init_oc_chan(&htim4, &htim4_oc_chan_3, TIM_CHANNEL_3, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_0])) ||
182 (true != sds_events_init_oc_chan(&htim4, &htim4_oc_chan_4, TIM_CHANNEL_4, sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_1])) )
183 {
184 Error_Handler();
185 }
186
187 if ( (HAL_OK != HAL_TIM_OC_Init(&htim3)) || (HAL_OK != HAL_TIM_OC_Init(&htim4)) )
188 {
189 Error_Handler();
190 }
191
192 /* Register callback */
193 if ( (HAL_OK != HAL_TIM_RegisterCallback(&htim3, HAL_TIM_OC_DELAY_ELAPSED_CB_ID, sds_events_delay_elapsed_cb)) ||
194 (HAL_OK != HAL_TIM_RegisterCallback(&htim4, HAL_TIM_OC_DELAY_ELAPSED_CB_ID, sds_events_delay_elapsed_cb)) )
195 {
196 Error_Handler();
197 }
198
199 /* Configure TIM3 as master and TIM4 as slave */
200 htim3_master_config.MasterOutputTrigger = TIM_TRGO_ENABLE;
201 htim3_master_config.MasterSlaveMode = TIM_MASTERSLAVEMODE_ENABLE;
202
203 htim3_slave_config.InputTrigger = TIM_TS_TI1F_ED;
204 htim3_slave_config.SlaveMode = TIM_SLAVEMODE_TRIGGER;
205
206 htim4_slave_config.InputTrigger = TIM_TS_ITR2;
207 htim4_slave_config.SlaveMode = TIM_SLAVEMODE_TRIGGER;
208
209 if ( (HAL_OK != HAL_TIM_SlaveConfigSynchro(&htim3, &htim3_slave_config)) ||
210 (HAL_OK != HAL_TIMEx_MasterConfigSynchronization(&htim3, &htim3_master_config)) ||
211 (HAL_OK != HAL_TIM_SlaveConfigSynchro(&htim4, &htim4_slave_config)) )
212 {
213 Error_Handler();
214 }
215
216 /* We have a master-slave configuration with synchronisation and we want to start
217 * the master from its internal interrupt. However Base_Start does a slave configuration check and
218 * thus it will not be possible using the HAL library. Therefore we set the state of both timers to busy and
219 * go for the enable of the master. The configuration follows 39.3.19 Timer Synchronization - Figure 452 */
220 htim3.State = HAL_TIM_STATE_BUSY;
221 htim4.State = HAL_TIM_STATE_BUSY;
222 __HAL_TIM_ENABLE(&htim3);
223
224 /* Start in interrupt mode */
225 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_MTM], TIM_CHANNEL_1, &htim3);
226 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO], TIM_CHANNEL_2, &htim3);
227 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_CSS], TIM_CHANNEL_3, &htim3);
228 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_ACC], TIM_CHANNEL_4, &htim3);
229
230 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_PANEL_TEMP], TIM_CHANNEL_1, &htim4);
231 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO_TEMP], TIM_CHANNEL_2, &htim4);
232 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_0], TIM_CHANNEL_3, &htim4);
233 sds_events_start_oc_chan_it(sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_1], TIM_CHANNEL_4, &htim4);
234
235}

237void sds_deinit_events_timer(void)
238{
239 if ( ( HAL_OK != HAL_TIM_Base_Stop(&htim3) ) ||
240 ( HAL_OK != HAL_TIM_Base_Stop(&htim4) ) )
241 {
242 Error_Handler();
243 }
244
245 if ( (HAL_OK != HAL_TIM_OC_Stop_IT(&htim3, TIM_CHANNEL_1)) ||
246 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim3, TIM_CHANNEL_2)) ||
247 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim3, TIM_CHANNEL_3)) ||
248 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim3, TIM_CHANNEL_4)) ||
249 (HAL_OK != HAL_TIM_OC_DeInit(&htim3)) ||
250 (HAL_OK != HAL_TIM_Base_DeInit(&htim3)))
251 {
252 Error_Handler();
253 }
254
255 if ( (HAL_OK != HAL_TIM_OC_Stop_IT(&htim4, TIM_CHANNEL_1)) ||
256 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim4, TIM_CHANNEL_2)) ||
257 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim4, TIM_CHANNEL_3)) ||
258 (HAL_OK != HAL_TIM_OC_Stop_IT(&htim4, TIM_CHANNEL_4)) ||
259 (HAL_OK != HAL_TIM_OC_DeInit(&htim4)) ||
260 (HAL_OK != HAL_TIM_Base_DeInit(&htim4)))
261 {
262 Error_Handler();
263 }
264}

285STATIC void sds_events_delay_elapsed_cb(TIM_HandleTypeDef* htim)
286{
287 uint32_t cmp_val;
288
289 uint16_t freq_1;
290 uint16_t freq_2;
291 uint16_t freq_3;
292 uint16_t freq_4;
293
294 uint32_t flags_1;
295 uint32_t flags_2;
296 uint32_t flags_3;
297 uint32_t flags_4;
298
299 freq_1 = (&htim3 == htim) ? (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_MTM]) : (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_PANEL_TEMP]);
300 freq_2 = (&htim3 == htim) ? (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO]) : (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_GYRO_TEMP]);
301 freq_3 = (&htim3 == htim) ? (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_CSS]) : (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_0]);
302 freq_4 = (&htim3 == htim) ? (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_ACC]) : (sds_events_chan_freqs[SDS_SENSTYPECFGNUM_SENS_RES_1]);
303
304 flags_1 = (&htim3 == htim) ? (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_MTM]) : (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_PANEL_TEMP]);
305 flags_2 = (&htim3 == htim) ? (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_GYRO]) : (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_GYRO_TEMP]);
306 flags_3 = (&htim3 == htim) ? (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_CSS]) : (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_RES_0]);
307 flags_4 = (&htim3 == htim) ? (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_ACC]) : (sds_events_chan_events[SDS_SENSTYPECFGNUM_SENS_RES_1]);
308
309 if ( htim->Channel == HAL_TIM_ACTIVE_CHANNEL_1 )
310 {
311 cmp_val = (freq_1 + __HAL_TIM_GET_COUNTER(htim)) % SDS_EVENTS_TIMER_TICK_FREQ;
312
313 __HAL_TIM_SET_COMPARE(htim, TIM_CHANNEL_1, cmp_val);
314
315 (void) osThreadFlagsSet (sds_get_thread_id(), flags_1);
316 }
317
318 if ( htim->Channel == HAL_TIM_ACTIVE_CHANNEL_2 )
319 {
320 cmp_val = (freq_2 + __HAL_TIM_GET_COUNTER(htim)) % SDS_EVENTS_TIMER_TICK_FREQ;
321
322 __HAL_TIM_SET_COMPARE(htim, TIM_CHANNEL_2, cmp_val);
323
324 (void) osThreadFlagsSet (sds_get_thread_id(), flags_2);
325 }
326
327 if ( htim->Channel == HAL_TIM_ACTIVE_CHANNEL_3 )
328 {
329 cmp_val = (freq_3 + __HAL_TIM_GET_COUNTER(htim)) % SDS_EVENTS_TIMER_TICK_FREQ;
330
331 __HAL_TIM_SET_COMPARE(htim, TIM_CHANNEL_3, cmp_val);
332
333 (void) osThreadFlagsSet (sds_get_thread_id(), flags_3);
334 }
335
336 if ( htim->Channel == HAL_TIM_ACTIVE_CHANNEL_4 )
337 {
338 cmp_val = (freq_4 + __HAL_TIM_GET_COUNTER(htim)) % SDS_EVENTS_TIMER_TICK_FREQ;
339
340 __HAL_TIM_SET_COMPARE(htim, TIM_CHANNEL_4, cmp_val);
341
342 (void) osThreadFlagsSet (sds_get_thread_id(), flags_4);
343 }
344}

356STATIC bool sds_events_init_oc_chan(TIM_HandleTypeDef *htim, TIM_OC_InitTypeDef *htim_oc_chan_cfg, uint32_t channel, uint16_t freq)
357{
358 CRIT_ASSERT(NULL != htim);
359 CRIT_ASSERT(NULL != htim_oc_chan_cfg);
360
361 bool res = true;
362
363 htim_oc_chan_cfg->OCMode = TIM_OCMODE_TIMING;
364 htim_oc_chan_cfg->Pulse = freq;
365
366 if ( freq > 0 )
367 {
368 if ( HAL_OK != HAL_TIM_OC_ConfigChannel(htim, htim_oc_chan_cfg, channel) )
369 {
370 res = false;
371 }
372 }
373
374 return res;
375}
