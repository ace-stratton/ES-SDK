59void module_id_get_info(MODID_VersionInfo_t * const p_version_info)
60{
61 memset(p_version_info, 0, sizeof(MODID_VersionInfo_t));
62
63 if (p_version_info != NULL) {
64 Nvm_GetBlockById(NVM_BLOCK_MODULEID, &mod_id_settings);
65
66 p_version_info->u16ModuleType = mod_id_settings.nModuleType;
67 p_version_info->eActiveImageType = MODULEID_CFG_ACTIVE_IMAGE_TYPE;
68 p_version_info->u8ActiveImageId = module_id_active_img_id;
69 p_version_info->sFwVersion.u16Major = module_id_major_version;
70 p_version_info->sFwVersion.u16Minor = module_id_minor_version;
71 p_version_info->sFwVersion.u16Patch = module_id_patch_version;
72 p_version_info->sHwVersion.u16Major = (uint8_t) mod_id_settings.nModuleHWConfig >> 8U;
73 p_version_info->sHwVersion.u16Minor = (uint8_t) mod_id_settings.nModuleHWConfig;
74 p_version_info->sHwVersion.u16Patch = (uint8_t) 0U;
75 (void)memcpy(p_version_info->au8HwSerialNumber, mod_id_settings.szModuleSerial, sizeof(p_version_info->au8HwSerialNumber));
76 (void)memcpy(p_version_info->au8BuildTime, __TIME__, sizeof(__TIME__));
77 (void)memcpy(p_version_info->au8BuildDate, __DATE__, sizeof(__DATE__));
78 (void)memset(p_version_info->au8BuildDescription, 0x00, sizeof(module_id_label_version));
79 (void)memcpy(p_version_info->au8BuildDescription, module_id_label_version, sizeof(p_version_info->au8BuildDescription));
80 }
81}
