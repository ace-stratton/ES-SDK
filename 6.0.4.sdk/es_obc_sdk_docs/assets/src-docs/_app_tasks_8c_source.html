96static void app_task(void *argument)
97{
98 arm_fault_handler_env_t env_data = {
99 .rtc_time = 0U,
100 .uptime = 0U
101 };
102
103 static bool exceptions_persisted_to_file = false;
104
105 datetime_stamp_t timestamp_now;
106
107 (void)memset(&timestamp_now, 0U, sizeof(datetime_stamp_t));
108
109 TaskMonitor_TaskInitialized(TASK_MONITOR_APP_TASK); /* The task is initialized and is ready */
110
111 for (;;)
112 {
113 TaskMonitor_IamAlive(TASK_MONITOR_APP_TASK); /* Prevent from WatchDog reset */
114
115 if (false != sys_conf_is_valid())
116 {
117 /* blink the Green LED for 50ms to indicate the OBC is running */
118 drv_led_set(DRV_LED_ID_GREEN, DRV_LED_STATE_ON);
119 osDelay(APP_LED_ON_TIME);
120 drv_led_set(DRV_LED_ID_GREEN, DRV_LED_STATE_OFF);
121 }
122 else
123 {
124 /* blink the amber LED to indicate there is an error in the configuration validation */
125 drv_led_set(DRV_LED_ID_AMBER, DRV_LED_STATE_ON);
126 osDelay(APP_LED_ON_TIME);
127 drv_led_set(DRV_LED_ID_AMBER, DRV_LED_STATE_OFF);
128 }
129
130 // update general DataCache entries...
131 apptask_update_obc_data();
132
133 // update fault context
134 datetime_now(&timestamp_now);
135
136 env_data.uptime = osKernelGetTickCount();
137 env_data.rtc_time = (uint32_t)datetime_to_unix(&timestamp_now);
138 arm_fault_handler_set_env(&env_data);
139
140 if ((SdMngr_GetCardState() == SDMNGR_CARD_STATE_READY) && (!exceptions_persisted_to_file))
141 {
142 fault_persistor_dump_to_file("fault.dmp");
143 exceptions_persisted_to_file = true;
144 }
145
146 osDelay(APP_TASK_CALL_PERIOD); /* Give processing time for the other tasks */
147
148 update_uptime(); /* this is done after the delay to avoid incrementing the counter to 1 on first execution of the task */
149
150 vApplicationLowStackCheck(TASK_MONITOR_APP_TASK);
151 }
152}

162static void apptask_update_obc_data(void)
163{
164 DATA_CACHE_OBC_0_t obc_data;
165
166 conops_get_hsm_active_state(&obc_data.u8OpMode);
167 obc_data.u32UpTime = get_uptime_seconds();
168
169 // to avoid potential unaligned memory access, we use a stack variable to get the value and then assign it
170 // to obc_data entry
171 uint16_t u16ResetReasonBitField = 0U;
172 uint32_t total_reset_count = NvmResetCounters_GetTotalCount(&u16ResetReasonBitField);
173 obc_data.u16ResetReasonBitField = u16ResetReasonBitField;
174
175 obc_data.u16TotalResetCount = (total_reset_count <= UINT16_MAX) ? ((uint16_t)total_reset_count) : (UINT16_MAX);
176
177 obc_data.u16PayloadModesStatus = 0U;
178
179 // TODO: Data Transfer mode still in discussion and no status available at the moment
180
181 dc_set_obc_0_data(&obc_data);
182}
183
187static void update_uptime(void)
188{
189 static uint16_t delayPeriod = 0;
190
191 delayPeriod += SERVICE_TASK_PERIOD;
192
193 while (delayPeriod >= UPTIME_RESOLUTION_1S)
194 {
195 uptime_seconds++;
196
197 if (delayPeriod >= UPTIME_RESOLUTION_1S)
198 {
199 delayPeriod -= UPTIME_RESOLUTION_1S; /* - 1 second */
200 }
201 }
202}
