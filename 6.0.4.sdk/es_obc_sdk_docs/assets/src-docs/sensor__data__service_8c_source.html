133void sds_init(void)
134{
135 EXEH_INIT_MODULE_FILENAME();
136
137 sds_task_taskHandle = osThreadNew(fs_sds_task, NULL, &sds_task_attributes);
138
139 if ( NULL == sds_task_taskHandle )
140 {
141 EXEH_HANDLE(eEXEHSeverity_Fatal, (int32_t)SDS_Exception_OS_Task_Spawn_Error);
142 }
143 else
144 {
145 TaskMonitor_TaskInitialized(TASK_SDS);
146 }
147
148 fs_sds_spawn_locks();
149
150 for ( uint8_t sindx = 0; sindx < SENSOR_COUNT; ++sindx)
151 {
152 sensor_base_cfg[sindx].sens_if->init();
153 }
154
155 events_mask = (uint32_t)pow(2, SENSOR_COUNT) - 1U;
156
157 fs_sds_init_timer();
158
159 timer_start(&time_nvm_check);
160}

167bool sds_sens_lock(sensor_data_sensors_t s_type)
168{
169 bool res = true;
170
171 if ( osOK != osMutexAcquire(p_sens_lock[s_type], SDS_LOCK_TIMEOUT) )
172 {
173 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t)SDS_Exception_OS_Mutex_Lock_Error);
174
175 res = false;
176 }
177
178 return res;
179}

188
189 res = false;
190 }
191

312/**********************************************************************
313 *
314 * Static function definitions
315 *
316 **********************************************************************/
319STATIC void fs_sds_task(void *arg)
320{
321 uint32_t events;
322
323 uint32_t wait_time_ms = SDS_BASE_SAMPLING_PERIOD_MS;
324
325 static sds_all_sens_data_t sens_data;
326 static sds_all_sens_data_sat_step_t sens_data_sat;
327
328 const void *p_sens_data;
329 const void *p_sens_data_sat;
330
331 for(;;)
332 {
333 events = osThreadFlagsWait(events_mask, osFlagsWaitAny, wait_time_ms);
334
335 timer_start(&time_periodic_sample);
336
337 if ( (osFlagsErrorUnknown != events) &&
338 (osFlagsErrorResource != events) &&
339 (osFlagsErrorParameter != events) )
340 {
341 /* Default case, if no events have arrived during the SDS wait period, read all sensors */
342 if ( osFlagsErrorTimeout == events )
343 {
344 events = events_mask;
345 }
346
347 for ( uint8_t sindx = 0; sindx < SENSOR_COUNT; ++sindx )
348 {
349 if ( (false != (events & sds_chan_freq_cfgs.sens_cfg[sensor_base_cfg[sindx].freq_cfg_id].event_cfg)) &&
350 ( true == sds_sens_lock(sindx) ) )
351 {
352#if SDS_PIN_DBG
353 fs_dbg_out_pin_on(sds_chan_freq_cfgs.sens_cfg[sensor_base_cfg[sindx].freq_cfg_id].event_cfg, sindx);
354#endif /* SDS_PIN_DBG */
355 fs_sds_check_pwr_save_off(sindx);
356
357 if ( SDS_SENSOR_STATUS_INIT == sensor_base_cfg[sindx].sens_if->get_state() )
358 {
359 if ( SDS_OP_STATUS_OK == sds_get_sensor_data(sindx, &sens_data, &sens_data_sat, &p_sens_data, &p_sens_data_sat) )
360 {
361 dc_set_raw_data(DC_DATA_INPUT_INTERNAL,
362 sensor_base_cfg[sindx].dc_did_id_SI_sens_frame,
363 p_sens_data, sensor_base_cfg[sindx].dc_did_id_SI_sens_frame_size);
364
365 dc_set_raw_data(DC_DATA_INPUT_INTERNAL,
366 sensor_base_cfg[sindx].dc_did_id_SI_sat_frame,
367 p_sens_data_sat, sensor_base_cfg[sindx].dc_did_id_SI_sat_frame_size);
368 }
369 }
370
371 fs_sds_check_pwr_save_on(sindx);
372#if SDS_PIN_DBG
373 fs_dbg_out_pin_off(sds_chan_freq_cfgs.sens_cfg[sensor_base_cfg[sindx].freq_cfg_id].event_cfg, sindx);
374#endif /* SDS_PIN_DBG */
375 sds_sens_release(sindx);
376 }
377 }
378 }
379 if ( true == timer_is_elapsed(&time_nvm_check, SDS_NVM_CHECK_PERIOD_MS) )
380 {
381 /* Start next check */
382 timer_start(&time_nvm_check);
383
384 fs_sds_init_timer();
385 }
386
387 if ( SAFE_FALSE == sds_chan_freq_cfgs.enable)
388 {
389 uint32_t wait_time_ms_temp = SDS_BASE_SAMPLING_PERIOD_MS - timer_get_elapsed_time(&time_periodic_sample);
390 if(SDS_BASE_SAMPLING_PERIOD_MS < wait_time_ms_temp)
391 {
392 wait_time_ms = SDS_BASE_SAMPLING_PERIOD_MS;
393 }
394 else
395 {
396 wait_time_ms = wait_time_ms_temp;
397 }
398 }
399 else
400 {
401 wait_time_ms = SDS_BASE_SAMPLING_PERIOD_MS;
402 }
403
404 TaskMonitor_IamAlive(TASK_SDS);
405 vApplicationLowStackCheck(TASK_SDS);
406 }
407}

411STATIC void fs_sds_check_pwr_save_off(uint8_t sindx)
412{
413 if ( SDS_SENSOR_STATUS_INIT != sensor_base_cfg[sindx].sens_if->get_state() )
414 {
415 if ( !((sds_chan_freq_cfgs.enable == SAFE_TRUE) && (sds_chan_freq_cfgs.sens_cfg[sensor_base_cfg[sindx].freq_cfg_id].freq_cfg < 1)) )
416 {
417 sensor_base_cfg[sindx].sens_if->init();
418 }
419 }
420}
421
424STATIC void fs_sds_check_pwr_save_on(uint8_t sindx)
425{
426 if ( SDS_SENSOR_STATUS_INIT == sensor_base_cfg[sindx].sens_if->get_state() )
427 {
428 if ( ((sds_chan_freq_cfgs.enable == SAFE_TRUE) && (sds_chan_freq_cfgs.sens_cfg[sensor_base_cfg[sindx].freq_cfg_id].freq_cfg < 1)) ||
429 (sds_chan_freq_cfgs.enable == SAFE_FALSE) )
430 {
431 sensor_base_cfg[sindx].sens_if->deinit();
432 }
433 }
434}

452STATIC void fs_sds_spawn_locks(void)
453{
454 for ( uint8_t sindx = 0; sindx < SENSOR_COUNT; sindx++ )
455 {
456 char lock_name[10U];
457 snprintf(lock_name, sizeof(lock_name), "sens_%02d", sindx);
458 const osMutexAttr_t sens_lock_attr = { lock_name, osMutexRecursive | osMutexPrioInherit, NULL, 0U };
459 osMutexId_t lock_it = osMutexNew(&sens_lock_attr);
460
461 if ( NULL != lock_it)
462 {
463 p_sens_lock[sindx] = lock_it;
464 }
465 else
466 {
467 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t)SDS_Exception_OS_Mutex_Spawn_Error);
468 }
469 }
470}
