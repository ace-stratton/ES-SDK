98bool gnss_uart_init(const uint32_t baudrate)
99{
100 if (NULL == fs_rx_q)
101 {
102 fs_rx_q = osMessageQueueNew(FS_RX_Q_SZ, sizeof(gnss_uart_resp_t), NULL);
103
104 if (NULL == fs_rx_q)
105 {
106 return false;
107 }
108 }
109
110 fs_huart.Init.BaudRate = baudrate;
111
112 if (HAL_OK != HAL_UART_Init(&fs_huart))
113 {
114 return false;
115 }
116
117 __HAL_RCC_DMA2_CLK_ENABLE();
118
119 if (HAL_OK != HAL_DMA_Init(&fs_hdma_uart_rx))
120 {
121 return false;
122 }
123
124 __HAL_LINKDMA(&fs_huart, hdmarx, fs_hdma_uart_rx);
125
126 __HAL_DMA_DISABLE_IT(&fs_hdma_uart_rx, DMA_IT_HT);
127
128 if ((HAL_OK != HAL_UART_RegisterRxEventCallback(&fs_huart, fs_rx_event_callback)) ||
129 (HAL_OK != HAL_UART_AbortReceive(&fs_huart)) ||
130 (HAL_OK != HAL_UARTEx_ReceiveToIdle_DMA(&fs_huart, (uint8_t*)fs_rx_msg.data, sizeof(fs_rx_msg.data))))
131 {
132 return false;
133 }
134
135 HAL_NVIC_SetPriority(DMA2_Stream1_IRQn, 5, 0);
136 HAL_NVIC_EnableIRQ(DMA2_Stream1_IRQn);
137
138 HAL_NVIC_SetPriority(UART5_IRQn, 5, 0);
139 HAL_NVIC_EnableIRQ(UART5_IRQn);
140
141 return true;
142}

210static void fs_msp_init_callback(UART_HandleTypeDef* huart)
211{

221 gpio_init_struct.Speed = GPIO_SPEED_FREQ_LOW;

250static void fs_rx_event_callback(UART_HandleTypeDef* huart, uint16_t size)
251{
252 if ((UART5 == huart->Instance) && (huart == &fs_huart))

254 fs_rx_msg.data_sz = size;
255
256 if ((NULL != fs_rx_q) &&

260
261 fs_rx_user_callback();

263
264 (void)HAL_UARTEx_ReceiveToIdle_DMA(&fs_huart, (uint8_t*)fs_rx_msg.data, sizeof(fs_rx_msg.data));
