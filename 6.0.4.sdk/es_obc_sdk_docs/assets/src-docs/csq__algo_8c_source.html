48bool csq_algo_push(csq_algo_ctx_t* const p_ctx, const void* const p_data)
49{
50 bool res = false;
51
52 if (NULL != p_ctx)
53 {
54 if ((csq_algo_get_free(p_ctx) > 0) || (CSQ_ALGO_POLICY_OVERWRITE == p_ctx->user_params.algo_policy))
55 {
56 if (NULL != p_ctx->user_params.p_user_set_slot_data)
57 {
58 (*p_ctx->user_params.p_user_set_slot_data)(p_ctx->tail, p_data);
59 p_ctx->tail = cyc_get_next_idx(p_ctx->tail, p_ctx->user_params.queue_size);
60
61 if (p_ctx->slots_reserved < p_ctx->user_params.queue_size)
62 {
63 p_ctx->slots_reserved++;
64 }
65
66 res = true;
67 }
68 else
69 {
70 CRIT_ASSERT(false);
71 }
72 }
73 }
74
75 return res;
76}

78bool csq_algo_pop(csq_algo_ctx_t* const p_ctx, void * const p_popped_data)
79{
80 bool res = false;
81
82 if ((NULL != p_ctx) && (NULL != p_popped_data))
83 {
84 if ((csq_algo_get_free(p_ctx) != p_ctx->user_params.queue_size) && (p_ctx->slots_reserved > 0))
85 {
86 if (NULL != p_ctx->user_params.p_user_get_slot_data)
87 {
88 (*p_ctx->user_params.p_user_get_slot_data)(p_ctx->head, p_popped_data);
89 p_ctx->head = cyc_get_next_idx(p_ctx->head, p_ctx->user_params.queue_size);
90
91 p_ctx->slots_reserved--;
92
93 res = true;
94 }
95 }
96 }
97
98 return res;
99}

101bool csq_algo_reserve(csq_algo_ctx_t* const p_ctx, uint32_t * const p_reserve_idx)
102{
103 bool res = false;
104
105 if ((NULL != p_ctx) && (NULL != p_reserve_idx))
106 {
107 if ((csq_algo_get_free(p_ctx) > 0) || (CSQ_ALGO_POLICY_OVERWRITE == p_ctx->user_params.algo_policy))
108 {
109 *p_reserve_idx = p_ctx->tail;
110
111 p_ctx->tail = cyc_get_next_idx(p_ctx->tail, p_ctx->user_params.queue_size);
112
113 if (p_ctx->slots_reserved < p_ctx->user_params.queue_size)
114 {
115 p_ctx->slots_reserved++;
116 }
117
118 res = true;
119 }
120 }
121
122 return res;

140uint32_t csq_algo_get_free(const csq_algo_ctx_t* const p_ctx)
141{
142 uint32_t free_slots = 0;
