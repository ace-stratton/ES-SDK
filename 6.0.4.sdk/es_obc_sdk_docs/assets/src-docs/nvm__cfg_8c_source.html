43eCriticalSectionStatus_t
44NVM_Cfg_EnterCriticalSection(void)
45{
46 eCriticalSectionStatus_t res = ECRITICALSECTIONSTATUS_ERROR;
47 osStatus_t mutexAcqStatus;
48
49 if (osKernelGetState() == osKernelRunning)
50 {
51 // lazy init on first call if not already done
52 if (csMutexHandle == NULL)
53 csMutexHandle = osMutexNew((const osMutexAttr_t *)NULL);
54
55 mutexAcqStatus = osMutexAcquire(csMutexHandle, CS_TIMEOUT_MS);
56
57 switch (mutexAcqStatus)
58 {
59 case osOK:
60 res = ECRITICALSECTIONSTATUS_OK;
61 break;
62
63 case osErrorTimeout:
64 res = ECRITICALSECTIONSTATUS_TIMEOUT;
65 break;
66
67 default:
68 res = ECRITICALSECTIONSTATUS_ERROR;
69 break;
70 }
71 }
72 else
73 {
74 __disable_irq();
75 res = ECRITICALSECTIONSTATUS_OK;
76 }
77
78 return res;
79}
80
81eCriticalSectionStatus_t
82NVM_Cfg_LeaveCriticalSection(void)
83{
84 eCriticalSectionStatus_t res = ECRITICALSECTIONSTATUS_ERROR;
85 osStatus_t mutexRelStatus;
86
87 if (osKernelGetState() == osKernelRunning)
88 {
89 if (csMutexHandle != NULL)
90 {
91 mutexRelStatus = osMutexRelease(csMutexHandle);
92
93 switch (mutexRelStatus)
94 {
95 case osOK:
96 res = ECRITICALSECTIONSTATUS_OK;
97 break;
98
99 default:
100 res = ECRITICALSECTIONSTATUS_ERROR;
101 break;
102 }
103 }
104 }
105 else
106 {
107 __enable_irq();
108 res = ECRITICALSECTIONSTATUS_OK;
109 }
110
111 return res;
112}
