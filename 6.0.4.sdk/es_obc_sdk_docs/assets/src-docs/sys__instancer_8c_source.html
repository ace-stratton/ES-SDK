25bool sys_inst_init(sys_init_level_id_t level) {
26 mod_op_status_id_t ret = STATUS_OK;
27
28 // Iterate the g_p_if_sys_mod_list
29 for (sys_module_id_t mod=0; mod < SYS_MODULE_ID_MAX; mod++)
30 {
31 if (sys_conf_is_module_active(mod))
32 {
33 // Make sure the instancer has a valid instance defined for the module
34 if (g_p_modules_interface_list[mod] == NULL)
35 {
36 ret = STATUS_FAILED;
37 break;
38 }
39 // Call the init method if it is not null
40 if (g_p_modules_interface_list[mod]->init)
41 {
42 ret = g_p_modules_interface_list[mod]->init(level);
43 if (ret == STATUS_FAILED)
44 {
45 // Stop initialisation sequence
46 break;
47 }
48 }
49 }
50 }
51
52 // Return true only if all the operations succeeded
53 // NOTE: It is ok if a module doesn't support a level of initialisation
54 return (ret != STATUS_FAILED);
55}

69 // Call the deinit method
70 if (ret != STATUS_FAILED)
71 {
72 if (g_p_modules_interface_list[mod]->deinit)
73 {
74 // Keep the STATUS_FAILED if any module failed
75 ret = g_p_modules_interface_list[mod]->deinit(level);
76 }
77 }
78 }
79
80 // Return true only if all the operations succeeded
81 // NOTE: It is ok if a module doesn't support a level of deinitialisation
82 return (ret != STATUS_FAILED);
83}
