108static void fs_adcs_tlm_task(void *arg)
109{
110 for(;;)
111 {
112 const uint8_t data_buff_size = sys_conf_is_module_active(SYS_MODULE_ID_CUBE_ADCS) ? 100 : 50;
113 uint8_t u8Data[data_buff_size];
114 uint8_t *pu8Data;
115
116 uint16_t u16CmdRspLen;
117
118 CubeADCS_CmdCfg_t sCmdStruct;
119
120 if ( (eCUBEADCS_STATE_IDLE == CubeADCS_GetState() ) &&
121 (eCUBEADCS_FTRANSF_STATE_IDLE == CubeADCS_FTransfer_GetTaskState() ) )
122 {
123 uint8_t u8TLMIndx;
124
125 for (uint8_t u8Indx= 0; u8Indx < ( (uint8_t)(sizeof(au8CubeADCS_TLM)/sizeof(au8CubeADCS_TLM[0])) ); u8Indx++)
126 {
127 pu8Data = &u8Data[0];
128 for (u8TLMIndx = 0; u8TLMIndx < au8CubeADCS_TLM[u8Indx].u8CmdListSize; u8TLMIndx++)
129 {
130 CubeADCS_PREPCmd(
131 &sCmdStruct,
132 au8CubeADCS_TLM[u8Indx].pu8CmdList[u8TLMIndx].u8ID,
133 NULL,
134 0,
135 (void *)pu8Data,
136 au8CubeADCS_TLM[u8Indx].pu8CmdList[u8TLMIndx].u8Size,
137 &u16CmdRspLen);
138
139 if ( eCUBEADCS_COMM_OK != CubeADCS_SendCmd(&sCmdStruct, eCUBEADCS_COMMREQTYPE_FETCH) )
140 {
141 break;
142 }
143 else
144 {
145 /* Each telemetry is received prefixed with the telemetry ID from the ADCS.
146 * It has to be removed as it is not part of the data cache relevant data */
147 for ( uint8_t u8BIndx = 0; u8BIndx < au8CubeADCS_TLM[u8Indx].pu8CmdList[u8TLMIndx].u8Size - 1; u8BIndx++ )
148 {
149 *pu8Data = *(pu8Data + 1);
150 pu8Data++;
151 }
152 }
153 }
154
155 if ( u8TLMIndx == au8CubeADCS_TLM[u8Indx].u8CmdListSize )
156 {
157
158 ADCS_TLM_CFG_Update_DCache(au8CubeADCS_TLM[u8Indx].eDCacheType, &u8Data[0]);
159 }
160 }
161 }
162
163 osDelay(ADCS_TLM_TASK_PERIOD_MS);
164
165 TaskMonitor_IamAlive(TASK_ADCS_TLM);
166 vApplicationLowStackCheck(TASK_ADCS_TLM);
167 }
168}
