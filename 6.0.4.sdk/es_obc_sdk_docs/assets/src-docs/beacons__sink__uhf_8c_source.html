41bool
42beacons_uhf_send(uint8_t *const data, const uint8_t size, uint8_t *const p_id)
43{
44 CRIT_ASSERT(NULL != data);
45 CRIT_ASSERT(size > 0U);
46
47 if (NULL != p_id)
48 *p_id = beacon_id++;
49
50 ReqContext_t esps_req_ctx = { .nAddr = BEACONS_UHF_SINK_ESPS_ADDRESS,
51 .nInterfaceNumber = ESSASNI_SYS_PRI,
52 .netType = ESSASNETT_INTERNAL };
53
54 UHF_sExBeacon_t bcn_data = { .u8Size =
55 MIN(size, sizeof(bcn_data.au8Data)) };
56
57 (void)memcpy(bcn_data.au8Data, data, MIN(size, sizeof(bcn_data.au8Data)));
58
59 // fill-up remaining bytes in beacon frame with zeroes
60 if (size <= sizeof(bcn_data.au8Data))
61 (void)memset(&bcn_data.au8Data[size], INVALID_BEACON_SLOT_ASSIGNMENT,
62 sizeof(bcn_data.au8Data) - size);
63
64 // NOTE: internal seqId from ESPS FP layer cannot be obtained with the
65 // current version of the
66 // generators, therefore we use our own beacon_id internally
67 ESSATMAC_ErrCodes res = UHF_ExBeaconSetSendReq(&esps_req_ctx, &bcn_data);
68
69 // TODO: replace by false when tests with real UHF are performed
70 return (ESSATMAC_EC_OK == res);
71}
