83static void on_send_result_cbk(eObcPortResult_t result, const uint32_t duration_in_ms,
84 const uint8_t script_id)
85{
86 sScriptData_t script_data;
87
88 eObcPortResult_t res = upy_get_script_status(script_id, &script_data);
89 // update the status if the element is still in the status array
90 if (UPY_OK == res)
91 {
92 script_data.eScriptStatus = result;
93 script_data.u32DurationInMs = duration_in_ms;
94
95 res = upy_set_script_status(script_id, &script_data);
96 }
97
98 if (UPY_OK != res)
99 {
100 if (UPY_SCRIPT_NOT_FOUND == res)
101 {
102 // status is ignored as the element in the script array is already
103 // overwritten
104 EXEH_HANDLE_EX(eEXEHSeverity_Warning, eEXEHModuleID_UPYTHON,
105 EXEH_UPYTHON_CANT_FIND_SCRIPT_CONTEXT)
106
107 ES_TRACE_DEBUG("Script not found ScriptID = %d", script_id);
108 }
109 else
110 {
111 EXEH_HANDLE_EX(eEXEHSeverity_Error, eEXEHModuleID_UPYTHON,
112 EXEH_UPYTHON_ERROR)
113 ES_TRACE_DEBUG(
114 "Error Getting/Setting the status of a script. ScriptID = %d, "
115 "ErrorID "
116 "= %d",
117 script_id, res);
118 }
119 }
120}

172static void obc_micropython_GetScriptExecStatusRequestHandlerImpl(ReqContext_t *pReqCtx,
173 const obc_micropythonGetScriptExecStatusRequestData_t* pRequestData)
174{
175 ESSATMAC_ErrCodes respResult;
176 RespContext_t respCtx;
177 char strScript_full_name[47];
178 uint8_t u8Script_full_nameSize;
179 OBC_MICROPYTHON_Result_t eOp_result;
180 OBC_MICROPYTHON_stime_t sRequested_time;
181 OBC_MICROPYTHON_stime_t sTime_started;
182 uint32_t u32Duration_ms;
183
184 // @USER_VAR_SECTION_START@obc_micropython::GetScriptExecStatus@
185 // Put your local variables in this section to preserve during merge!
186 // @USER_VAR_SECTION_END@obc_micropython::GetScriptExecStatus@
187
188 if ((pReqCtx != NULL) && (pRequestData != NULL))
189 {
190 respCtx.nInterfaceNumber = pReqCtx->nInterfaceNumber;
191 respCtx.netType = pReqCtx->netType;
192 respCtx.nAddr = pReqCtx->nAddr;
193 respCtx.seqId = pReqCtx->seqId;
194
195 // @USER_CODE_SECTION_START@obc_micropython::GetScriptExecStatus@
196
197 sScriptData_t script_data;
198
199 eObcPortResult_t res = upy_get_script_status(pRequestData->u8Script_id, &script_data);
200
201 // check if the fidl command size is correct(equal to FATFS configuration for the system)
202 static_assert(FF_MAX_LFN == sizeof(strScript_full_name));
203
204 if (UPY_OK == res)
205 {
206 eOp_result = aResultMap[script_data.eScriptStatus];
207
208 uint32_t name_length = strnlen(script_data.pName, FF_MAX_LFN);
209
210 CRIT_ASSERT(name_length < UINT8_MAX);
211 u8Script_full_nameSize = (uint8_t)name_length;
212
213 memset(strScript_full_name, '\0', sizeof(strScript_full_name));
214 strncpy(strScript_full_name, script_data.pName, FF_MAX_LFN); // NOSONAR src will never be longer than dest
215
216 static_assert(sizeof(sys_time_t) == sizeof(OBC_MICROPYTHON_stime_t));
217
218 memcpy(&sRequested_time, &script_data.sTimeRequested, sizeof(OBC_MICROPYTHON_stime_t));
219 memcpy(&sTime_started, &script_data.sTimeStarted, sizeof(OBC_MICROPYTHON_stime_t));
220 u32Duration_ms = script_data.u32DurationInMs;
221 }
222 else
223 {
224 eOp_result = OBC_MICROPYTHON_RESULT_NOT_FOUND_IN_QUEUE;
225 strScript_full_name[0] = '\0';
226 u8Script_full_nameSize = 1;
227 memset(&sRequested_time, 0, sizeof(OBC_MICROPYTHON_stime_t));
228 memset(&sTime_started, 0, sizeof(OBC_MICROPYTHON_stime_t));
229 u32Duration_ms = 0;
230 }
231
232 // @USER_CODE_SECTION_END@obc_micropython::GetScriptExecStatus@
233
234 respResult = obc_micropython_GetScriptExecStatusResp(
235 &respCtx,
236 strScript_full_name,
237 u8Script_full_nameSize,
238 eOp_result,
239 &sRequested_time,
240 &sTime_started,
241 u32Duration_ms
242 );
243
244 if (respResult != ESSATMAC_EC_OK)
245 TRACE_ERROR(ES_SAT_FUNC_PROTOCOL_ID_OBC_MICROPYTHON, OBC_MICROPYTHON_GETSCRIPTEXECSTATUS_FUNCRESP_ID, respResult);
246 }
247}
