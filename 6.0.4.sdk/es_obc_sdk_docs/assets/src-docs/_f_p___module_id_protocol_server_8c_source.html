78{
79 ESSA_Stack_FP_MsgHdr_t *pHdr = NULL;
80 bool bIsFuncSupported = false;
81 uint8_t i;
82
83 // Basic header validity check
84 if ((fp_DataInfo == NULL) ||
85 (fp_DataInfo->u16DataSize < sizeof(ESSA_Stack_FP_MsgHdr_t)) ||
86 (fp_DataInfo->pu8Data == NULL))
87 {
88 return false;
89 }
90
91 pHdr = (ESSA_Stack_FP_MsgHdr_t *) fp_DataInfo->pu8Data;
92
93 if (IS_REQUEST(*pHdr))
94 {
95 for (i = 0; i < COUNT_OF(fs_aFuncArray); i++)
96 {
97 if (fs_aFuncArray[i].funcId == pHdr->funcId)
98 {
99 if (fs_aFuncArray[i].pfFunc != NULL)
100 {
101 bIsFuncSupported = true;
102 fs_aFuncArray[i].pfFunc(fp_DataInfo);
103 }
104
105 break;
106 }
107 }
108
109 if (!bIsFuncSupported)
110 (void) ProtocolSendErrorResp(fp_DataInfo, (uint8_t) ESSA_FP_ERRCODE_FUNC_NOT_SUPPORTED);
111 }
112
113 return bIsFuncSupported;
114}
