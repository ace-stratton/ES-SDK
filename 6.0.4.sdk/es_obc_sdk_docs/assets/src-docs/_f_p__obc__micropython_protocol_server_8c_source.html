90static bool fs_HandleData(ESSA_Stack_DataDispatchInfo_t *fp_DataInfo)
91{
92 ESSA_Stack_FP_MsgHdr_t *pHdr = NULL;
93 bool bIsFuncSupported = false;
94 uint8_t i;
95
96 // Basic header validity check
97 if ((fp_DataInfo == NULL) ||
98 (fp_DataInfo->u16DataSize < sizeof(ESSA_Stack_FP_MsgHdr_t)) ||
99 (fp_DataInfo->pu8Data == NULL))
100 {
101 return false;
102 }
103
104 pHdr = (ESSA_Stack_FP_MsgHdr_t *) fp_DataInfo->pu8Data;
105
106 if (IS_REQUEST(*pHdr))
107 {
108 for (i = 0; i < COUNT_OF(fs_aFuncArray); i++)
109 {
110 if (fs_aFuncArray[i].funcId == pHdr->funcId)
111 {
112 if (fs_aFuncArray[i].pfFunc != NULL)
113 {
114 bIsFuncSupported = true;
115 fs_aFuncArray[i].pfFunc(fp_DataInfo);
116 }
117
118 break;
119 }
120 }
121
122 if (!bIsFuncSupported)
123 (void) ProtocolSendErrorResp(fp_DataInfo, (uint8_t) ESSA_FP_ERRCODE_FUNC_NOT_SUPPORTED);
124 }
125
126 return bIsFuncSupported;
127}
