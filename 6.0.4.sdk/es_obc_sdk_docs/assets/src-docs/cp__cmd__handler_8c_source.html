128// this function will be called by the ESSA Stack when the FP layer replies back to the received CP request
129ESSATMAC_ErrCodes cp_cmd_handler_mac_fp_gw_enq_message(ESSATMAC_Context *p_mac_ctx, uint8_t *p_msg_data, uint8_t msg_data_sz)
130{
131 ESSATMAC_ErrCodes result = ESSATMAC_EC_ERROR;
132
133 if ((p_mac_ctx != NULL) && (p_msg_data != NULL) && (msg_data_sz >= sizeof(mac_fp_gw_hdr_t)))
134 {
135 if ((p_mac_ctx->netType == ESSASNETT_GS_MACFP_GW) &&
136 (p_mac_ctx->nInterfaceNumber == MACFPGW_IF_ID) &&
137 (p_mac_ctx->nPayloadProtocol == ES_SAT_MAC_PROTOCOL_ID_FP_LAYER) &&
138 (user_resp_ctx.pu8DataBuf != NULL) &&
139 (user_resp_ctx.u32MaxDataSize >= msg_data_sz))
140 {
141 // transfer message data back to CP buffer
142 (void)memcpy(user_resp_ctx.pu8DataBuf, p_msg_data, msg_data_sz);
143
144 // store actual data size for CP command reply dispatch in cp_cmd_handler_mac_fp_gw_send_cmd(...)
145 user_resp_ctx.u32ResponseSize = msg_data_sz;
146
147 ESSA_Stack_TP_Layer_ReleaseThreadWait();
148
149 result = ESSATMAC_EC_OK;
150 }
151 else
152 {
153 result = ESSATMAC_EC_INVALID_PARAMS;
154 }
155 }
156
157 return result;
158}

160eCPDispatchResult_t cp_cmd_handler_mac_fp_gw_send_cmd(sCPDispatchContext_t *const p_disp_ctx)
161{
162 if (!CPCmdHandler_IsDispContextValid(p_disp_ctx, CP_CMD_MACFP_GWSEND))
163 {
164 return ECPDISPATCHRESULT_ERROR;
165 }
166
167 ESSATMAC_Context mac_ctx;
168 const mac_fp_gw_hdr_t *p_fp_gw_hdr = (const mac_fp_gw_hdr_t *)p_disp_ctx->pu8CmdData;
169
170 if (p_disp_ctx->u32CmdDataSize >= p_fp_gw_hdr->payload_size)
171 {
172 (void)memset((void *)&mac_ctx, (uint8_t)0, sizeof(ESSATMAC_Context));
173
174 // setup response buffer for ESSA Stack
175 user_resp_ctx.pu8DataBuf = p_disp_ctx->pu8ResponseBuf;
176 user_resp_ctx.u32MaxDataSize = p_disp_ctx->u32ResponseBufSize;
177 user_resp_ctx.u32ResponseSize = 0U;
178 user_resp_ctx.onCmdResponseReady = NULL;
179 user_resp_ctx.pUserCtx = NULL;
180
181 mac_ctx.nInterfaceNumber = MACFPGW_IF_ID;
182 mac_ctx.nPayloadProtocol = ES_SAT_MAC_PROTOCOL_ID_FP_LAYER;
183 mac_ctx.nSourceAddr = 0x00U;
184 mac_ctx.nTargetAddr = 0x00U;
185 mac_ctx.netType = ESSASNETT_GS_MACFP_GW;
186
187 ESSA_Stack_TP_Layer_PrepareThreadWait();
188
189 // inject message to ESSA Stack for FP dispatching...
190 ESSA_Stack_DataDispatch(&mac_ctx,
191 (uint8_t *)&p_disp_ctx->pu8CmdData[sizeof(mac_fp_gw_hdr_t)],
192 p_fp_gw_hdr->payload_size);
193
194 // wait for cp_cmd_handler_mac_fp_gw_enq_message() to be called or timeout...
195 eThreadBlockEvt_t waitResult = ESSA_Stack_TP_Layer_ThreadWaitForEvent(MACFPGW_RESPONSE_TIMEOUT_MS);
196
197 if (waitResult == THREADBlOCKEVT_WAITFINISHED)
198 {
199 if (p_disp_ctx->u32ResponseBufSize >= sizeof(ESSA_Stack_FP_MsgHdr_t))
200 {
201 // user-data is already written in the response buffer
202 p_disp_ctx->u32ResponseActualSize = user_resp_ctx.u32ResponseSize;
203 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
204 }
205 else
206 {
207 p_disp_ctx->u32ResponseActualSize = 0U;
208 p_disp_ctx->u8TLError = EESTL_SFERR_OUT_OF_MEM;
209 }
210 }
211 else if (waitResult == THREADBLOCKEVT_WAITTIMEOUT)
212 {
213 p_disp_ctx->u32ResponseActualSize = 0U;
214 p_disp_ctx->u8TLError = EESTL_SFERR_RECEIVE_TIMEOUT;
215 }
216 else
217 {
218 (void)memcpy(p_disp_ctx->pu8ResponseBuf, &waitResult, sizeof(waitResult));
219 p_disp_ctx->u32ResponseActualSize = sizeof(waitResult);
220 p_disp_ctx->u8TLError = EESTL_SFERR_UNKNOWN;
221 }
222 }
223 else
224 {
225 p_disp_ctx->u32ResponseActualSize = 0U;
226 p_disp_ctx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
227 }
228
229 return ECPDISPATCHRESULT_RESULT_OK;
230}

232eCPDispatchResult_t cp_cmd_handler_cubeadcs_gw_send_cmd(sCPDispatchContext_t *const p_disp_ctx)
233{
234 if (!CPCmdHandler_IsDispContextValid(p_disp_ctx, CP_CMD_CUBEADCS_GWSEND))
235 {
236 return ECPDISPATCHRESULT_ERROR;
237 }
238
239 p_disp_ctx->u32ResponseActualSize = 0U;
240
241 if (sys_conf_is_module_active(SYS_MODULE_ID_CUBE_ADCS))
242 {
243 bool bSendCmdStatus = true;
244
245 CubeADCS_CmdCfg_t sCmd;
246
247 sCmd.u8CmdID = p_disp_ctx->pu8CmdData[0];
248
249 sCmd.u16CmdDataSize = (uint16_t)(p_disp_ctx->u32CmdDataSize - sizeof(sCmd.u8CmdID));
250
251 if (p_disp_ctx->u32CmdDataSize == sizeof(sCmd.u8CmdID))
252 {
253 sCmd.pu8CmdData = NULL;
254 }
255 else
256 {
257 sCmd.pu8CmdData = (uint8_t *)&p_disp_ctx->pu8CmdData[sizeof(sCmd.u8CmdID)];
258 }
259
260 if (p_disp_ctx->u32CmdDataSize >= sCmd.u16CmdDataSize + sizeof(sCmd.u8CmdID))
261 {
262 user_resp_ctx.pu8DataBuf = p_disp_ctx->pu8ResponseBuf;
263 user_resp_ctx.u32MaxDataSize = p_disp_ctx->u32ResponseBufSize;
264 user_resp_ctx.u32ResponseSize = 0U;
265 user_resp_ctx.onCmdResponseReady = on_cmd_response_ready_cbk;
266 user_resp_ctx.pUserCtx = (void *)&user_resp_ctx;
267
268 sCmd.pvCmdResDst = (void *)user_resp_ctx.pu8DataBuf;
269 sCmd.u16CmdResDstSize = (uint16_t)user_resp_ctx.u32MaxDataSize;
270 sCmd.u16CmdResSize = (uint16_t *)&p_disp_ctx->u32ResponseActualSize;
271
272 ESSA_Stack_TP_Layer_PrepareThreadWait();
273
274 bSendCmdStatus = CubeADCS_UsrCmd_CmdHandler(eCUBEADCS_USR_CMDS_GENERIC, &sCmd, &user_resp_ctx);
275
276 if (!bSendCmdStatus)
277 {
278 // block here to wait answer from CUBE_ADCS...
279 eThreadBlockEvt_t waitResult = ESSA_Stack_TP_Layer_ThreadWaitForEvent(CUBEADCS_RESPONSE_TIMEOUT_MS);
280
281 if (waitResult == THREADBlOCKEVT_WAITFINISHED)
282 {
283 if (!bSendCmdStatus)
284 {
285 if (user_resp_ctx.u32ResponseSize <= p_disp_ctx->u32ResponseBufSize)
286 {
287 // user-data is already written in the response buffer which was passed before the call to
288 // CubeADCS_UsrCmd_CmdHandler(...)
289 p_disp_ctx->u32ResponseActualSize = user_resp_ctx.u32ResponseSize;
290 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
291 }
292 else
293 {
294 p_disp_ctx->u8TLError = EESTL_SFERR_OUT_OF_MEM;
295 }
296 }
297 }
298 else if (waitResult == THREADBLOCKEVT_WAITTIMEOUT)
299 {
300 p_disp_ctx->u8TLError = EESTL_SFERR_RECEIVE_TIMEOUT;
301 }
302 else
303 {
304 (void)memcpy(p_disp_ctx->pu8ResponseBuf, &waitResult, sizeof(waitResult));
305 p_disp_ctx->u32ResponseActualSize = sizeof(waitResult);
306 p_disp_ctx->u8TLError = EESTL_SFERR_UNKNOWN;
307 }
308 }
309 else
310 {
311 p_disp_ctx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
312 }
313 }
314 else
315 {
316 p_disp_ctx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
317 }
318 }
319 else // not SYS_MODULE_ID_CUBE_ADCS
320 {
321 p_disp_ctx->u8TLError = EESTL_SFERR_CMD_NOT_SUPP;
322 }
323
324 return ECPDISPATCHRESULT_RESULT_OK;
325}

331eCPDispatchResult_t cp_cmd_handler_download_file(sCPDispatchContext_t *const p_disp_ctx)
332{
333 eCPDispatchResult_t res = ECPDISPATCHRESULT_ERROR;
334 uint32_t file_size = 0U;
335
336 if (CPCmdHandler_IsDispContextValid(p_disp_ctx, CP_CMD_DBG_DOWNLOADFILE))
337 {
338 // 4 bytes for file_offset parameter and a valid file name(1 symbol and Null terminator)
339 if (p_disp_ctx->u32CmdDataSize >=
340 sizeof(dw_file_by_chunks_in_args.file_offset) + SHORTEST_POSSIBLE_FNAME_LEN)
341 {
342 dw_file_by_chunks_in_args = *((dw_file_by_chunks_in_args_t *)p_disp_ctx->pu8CmdData);
343
344 // clear any pending errors
345 dw_file_error_result.error = FR_OK;
346 dw_file_error_result.src_line_no = 0U;
347
348 eFileManagerOpResult_t fres = fileManager_FGetSize(dw_file_by_chunks_in_args.file_path,
349 &file_size);
350 if (FR_OK == fres)
351 {
352 if (file_size > dw_file_by_chunks_in_args.file_offset)
353 {
354 file_size -= dw_file_by_chunks_in_args.file_offset;
355 // does the file fit in a single response buffer?
356 if (file_size <= p_disp_ctx->u32ResponseBufSize)
357 {
358 // file fits, just read it in the current buffer and respond back
359 // NOTE:
360 // Ensure that dw_file_by_chunks_in_args already contains the file path information
361 // because the following function doesn't accept a context parameter and shall somehow
362 // know which file to attempt to read
363 uint8_t cbk_res = dw_file_by_chunks_cbk(0U,
364 CP_CMD_DBG_DOWNLOADFILE,
365 file_size,
366 dw_file_by_chunks_in_args.file_offset,
367 p_disp_ctx->pu8ResponseBuf);
368
369 if (cbk_res == EESTL_SFERR_SUCCESS)
370 {
371 p_disp_ctx->u32ResponseActualSize = file_size;
372 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
373 res = ECPDISPATCHRESULT_RESULT_OK;
374 }
375 }
376 else
377 { // file doesn't fit, therefore request chunk-based transfer to GS
378 // prepare file CRC before triggering chunk-based transfer
379 uint8_t crc_tmp_buff[FILEDOWNLOAD_CRCREAD_BUF_SIZE];
380 if (b_calc_file_crc(dw_file_by_chunks_in_args.file_offset, crc_tmp_buff, sizeof(crc_tmp_buff)))
381 {
382 p_disp_ctx->pChunkRequestCbk = dw_file_by_chunks_cbk;
383 p_disp_ctx->u32ResponseActualSize = file_size;
384 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
385 res = ECPDISPATCHRESULT_RESULT_BY_CHUNKS;
386 }
387 }
388 }
389 else
390 {
391 p_disp_ctx->u32ResponseActualSize = 0;
392 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
393 res = ECPDISPATCHRESULT_RESULT_OK;
394 }
395 }
396 else
397 {
398 dw_store_error_res(fres, __LINE__);
399 }
400 }
401 else
402 {
403 p_disp_ctx->u32ResponseActualSize = 0U;
404 p_disp_ctx->u8TLError = EESTL_SFERR_BAD_CMD_PARAMS;
405 res = ECPDISPATCHRESULT_SIMPLERESULT_OK;
406 }
407 }
408
409 if (ECPDISPATCHRESULT_ERROR == res)
410 {
411 // attempt to make a human-readable error message
412 int32_t ch_count = (int32_t)snprintf((char *volatile)p_disp_ctx->pu8ResponseBuf,
413 (size_t)(p_disp_ctx->u32ResponseBufSize),
414 "%s: %d @ line: %lu\n",
415 FILE_DW_ERROR_MSG,
416 dw_file_error_result.error,
417 dw_file_error_result.src_line_no);
418
419 if (ch_count > 0)
420 {
421 p_disp_ctx->u32ResponseActualSize = (uint32_t)ch_count;
422 }
423 else
424 {
425 (void)memcpy(p_disp_ctx->pu8ResponseBuf, &dw_file_error_result, sizeof(dw_file_error_result));
426 p_disp_ctx->u32ResponseActualSize = sizeof(dw_file_error_result);
427 }
428
429 p_disp_ctx->u8TLError = EESTL_SFERR_SUCCESS;
430 res = ECPDISPATCHRESULT_RESULT_OK;
431 }
432
433 return res;
434}

455static uint8_t dw_file_by_chunks_cbk(const uint64_t cmd_id,
456 const uint32_t cmd_type,
457 const uint32_t required_chunk_size,
458 const uint32_t offset,
459 uint8_t *p_chunk)
460{
461 uint8_t res = EESTL_SFERR_UNKNOWN;
462 FileManagerFHnd h_file;
463
464 (void)cmd_id;
465 (void)cmd_type;
466
467 eFileManagerOpResult_t fres = fileManager_FOpen(dw_file_by_chunks_in_args.file_path,
468 eFileManagerFileModeId_OPEN_EXISTING | eFileManagerFileModeId_READ,
469 &h_file);
470
471 if (fres == FR_OK)
472 {
473 uint32_t bytes_read = 0U;
474
475 fres = fileManager_FRead(h_file, dw_file_by_chunks_in_args.file_offset + offset, p_chunk, required_chunk_size, &bytes_read);
476
477 if ((fres == FR_OK) && (bytes_read == required_chunk_size))
478 {
479 res = EESTL_SFERR_SUCCESS;
480 }
481 else
482 {
483 dw_store_error_res(fres, __LINE__);
484 }
485
486 (void)fileManager_FClose(h_file);
487 }
488 else
489 {
490 dw_store_error_res(fres, __LINE__);
491 }
492
493 return res;
494}

496static bool b_calc_file_crc(const uint32_t file_offset, uint8_t *const p_buff, const uint32_t buff_size)
497{
498 uint32_t offset = file_offset;
499 uint32_t bytes_read = 0U;
500 bool b_first_block = true;
501 bool b_success = true;
502 FileManagerFHnd h_file;
503
504 eFileManagerOpResult_t fres = fileManager_FOpen(dw_file_by_chunks_in_args.file_path,
505 eFileManagerFileModeId_OPEN_EXISTING | eFileManagerFileModeId_READ,
506 &h_file);
507
508 if (fres == FR_OK)
509 {
510 do
511 {
512 fres = fileManager_FRead(h_file, offset, p_buff, buff_size, &bytes_read);
513
514 if (fres == FR_OK)
515 {
516 ESCP_ModDrv_CalcBlockCRC(p_buff, bytes_read, b_first_block);
517
518 b_first_block = false;
519
520 offset += bytes_read;
521 }
522 else
523 {
524 dw_store_error_res(fres, __LINE__);
525 b_success = false;
526 break;
527 }
528 } while (bytes_read > 0U);
529
530 (void)fileManager_FClose(h_file);
531 }
532 else
533 {
534 dw_store_error_res(fres, __LINE__);
535 }
536
537 return b_success;
538}
