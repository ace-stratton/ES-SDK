145void conops_trigger_hsm(const uint32_t event_id, const sConOps_pay_cfg_t *const pay_cfg)
146{
147 sConOps_event_t trigger_event;
148
149 osMutexAcquire(conops_ev_mutex, osWaitForever);
150
151 trigger_event.event_id.id = event_id;
152
153 if ( NULL != pay_cfg )
154 {
155 memcpy(&trigger_event.event_data, pay_cfg, sizeof(sConOps_pay_cfg_t));
156 }
157 else
158 {
159 memset(&trigger_event.event_data, 0, sizeof(sConOps_pay_cfg_t));
160 }
161
162 (void)CONOPS_TRIGGER_HSM(&trigger_event.event_id);
163
164 osMutexRelease(conops_ev_mutex);
165}

174bool conops_set_thresh_val(int32_t const * const p_thres_val, uint8_t index,
175 CONOPS_ThresholdValueTypes_t thres_type)
176{
177 CRIT_ASSERT(NULL != p_thres_val);
178
179 // Return true unless an error is detected
180 bool bRes = true;
181
182 switch (thres_type)
183 {
184 case CONOPS_THRESHOLDVALUETYPES_CBATT_SAFE:
185 sConOps_thresh_cfg.cfg.i32CBattSafe = *p_thres_val;
186 break;
187 case CONOPS_THRESHOLDVALUETYPES_S_SAFE:
188 sConOps_thresh_cfg.cfg.i32SSafe = *p_thres_val;
189 break;
190 case CONOPS_THRESHOLDVALUETYPES_S_DET:
191 sConOps_thresh_cfg.cfg.i32SDet = *p_thres_val;
192 break;
193 case CONOPS_THRESHOLDVALUETYPES_S_PAY:
194 if (index >= CDEF_ELEMENT_COUNT(sConOps_thresh_cfg.cfg.ai32SPay))
195 {
196 bRes = false;
197 }
198 else
199 {
200 sConOps_thresh_cfg.cfg.ai32SPay[index] = *p_thres_val;
201 }
202 break;
203 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_MAX:
204 sConOps_thresh_cfg.cfg.i32AngVelMax = *p_thres_val;
205 break;
206 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_MIN:
207 sConOps_thresh_cfg.cfg.i32AngVelLow = *p_thres_val;
208 break;
209 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_RANGE:
210 if (index >= CDEF_ELEMENT_COUNT(sConOps_thresh_cfg.cfg.ai32AngVelRange))
211 {
212 bRes = false;
213 }
214 else
215 {
216 sConOps_thresh_cfg.cfg.ai32AngVelRange[index] = *p_thres_val;
217 }
218 break;
219 default:
220 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t)eConOpsException_Runtime_Error);
221
222 bRes = false;
223 break;
224 }
225
226 if (true == bRes)
227 {
228 Nvm_SetBlockById(NVM_APP_BLOCK_CONOPS_CONFIG, &sConOps_thresh_cfg, NVMSETBLOCK_STORE_IMMEDIATELY);
229 Nvm_GetBlockById(NVM_APP_BLOCK_CONOPS_CONFIG, &sConOps_thresh_cfg);
230 }
231
232 return bRes;
233}

240 // Return true unless an error is detected

249 *p_thres_val = sConOps_thresh_cfg.cfg.i32SSafe;
250 break;
251 case CONOPS_THRESHOLDVALUETYPES_S_DET:
252 *p_thres_val = sConOps_thresh_cfg.cfg.i32SDet;
253 break;
254 case CONOPS_THRESHOLDVALUETYPES_S_PAY:
255 if (index >= CDEF_ELEMENT_COUNT(sConOps_thresh_cfg.cfg.ai32SPay))
256 {
257 bRes = false;
258 }
259 else
260 {
261 *p_thres_val = sConOps_thresh_cfg.cfg.ai32SPay[index];
262 }
263 break;
264 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_MAX:
265 *p_thres_val = sConOps_thresh_cfg.cfg.i32AngVelMax;
266 break;
267 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_MIN:
268 *p_thres_val = sConOps_thresh_cfg.cfg.i32AngVelLow;
269 break;
270 case CONOPS_THRESHOLDVALUETYPES_ANGVEL_RANGE:

284 EXEH_HANDLE(eEXEHSeverity_Error, (int32_t)eConOpsException_Runtime_Error);

389bool conops_is_omega_above_thresh(const aocs_cntrl_sys_state_types_t aocs_state)
390{
391 bool bRes;
392
393 int32_t i32OmegaThresh;
394 DATA_CACHE_AOCS_CNTRL_TLM_t sADCSData;
395
396 bRes = false;
397
398 if ( aocs_state > AOCS_CNTRL_SYS_STATE_VERY_FAST_DETUMBLING )
399 {
400 return false;
401 }
402
403 if ( true != conops_get_thresh_val(&i32OmegaThresh, 0, CONOPS_THRESHOLDVALUETYPES_ANGVEL_MAX ) )
404 {
405 return false;
406 }
407
408 if ( DC_DATA_STATUS_OK == dc_get_aocs_cntrl_tlm_data(&sADCSData) )
409 {
410 if ( (AOCS_CNTRL_SYS_STATE_Y_THOMSON == aocs_state) || (AOCS_CNTRL_SYS_STATE_Y_THOMSON_MEMS_RATE == aocs_state) )
411 {
412 sConOps_3axis_vec_t sAngRateVecTmp;
413
414 sAngRateVecTmp.X = sADCSData.ai32EstAngRateVec[0];
415 sAngRateVecTmp.Y = sADCSData.ai32EstAngRateVec[1];
416 sAngRateVecTmp.Z = sADCSData.ai32EstAngRateVec[2];
417
418 if ( false == fs_conops_chk_stable_y_thomson(&sAngRateVecTmp) )
419 {
420 bRes = true;
421 }
422 }
423 else if ( sADCSData.i32EstAngRateNorm > i32OmegaThresh )
424 {
425 bRes = true;
426 }
427 else
428 {
429 bRes = false;
430 }
431 }
432
433 CONOPS_TRACE_DBG("%s: conops_is_omega_above_thresh: %d\r", CONOPS_SM_TEST_COND, bRes);
434
435 return bRes;
436}

484bool conops_is_detumb_completed(void)
485{
486 bool bRes = false;
487
488 DATA_CACHE_AOCS_CNTRL_TLM_t sADCSData;
489
490 if ( DC_DATA_STATUS_OK != dc_get_aocs_cntrl_tlm_data(&sADCSData) )
491 {
492 return bRes;
493 }
494
495 aocs_cntrl_sys_state_types_t aocs_state;
496 aocs_cntrl_sys_state_req_state_t aocs_state_req_state;
497
498 if ( conops_get_aocs_sys_state(&aocs_state, &aocs_state_req_state) &&
499 (AOCS_CNTRL_SYS_STATE_STATE_REQ_COMPLETED == aocs_state_req_state) )
500 {
501 if ( AOCS_CNTRL_SYS_STATE_Y_THOMSON == aocs_state )
502 {
503 sConOps_3axis_vec_t sAngRateVecTmp;
504
505 sAngRateVecTmp.X = sADCSData.ai32EstAngRateVec[0];
506 sAngRateVecTmp.Y = sADCSData.ai32EstAngRateVec[1];
507 sAngRateVecTmp.Z = sADCSData.ai32EstAngRateVec[2];
508
509 bRes = fs_conops_chk_stable_y_thomson(&sAngRateVecTmp);
510 }
511 else
512 {
513 int32_t i32OmegaThresh;
514
515 if ( true != conops_get_thresh_val(&i32OmegaThresh, 0, CONOPS_THRESHOLDVALUETYPES_ANGVEL_MIN ) )
516 {
517 return bRes;
518 }
519
520 if ( i32OmegaThresh > sADCSData.i32EstAngRateNorm )
521 {
522 bRes = true;
523 }
524 }
525
526 if ( true == bRes)
527 {
528 conops_set_logical_flag_value(eFLAG_DETUMB_COMPLETED, true);
529 }
530 }
531
532 CONOPS_TRACE_DBG("%s: conops_is_detumb_completed: %d\r", CONOPS_SM_TEST_COND, bRes);
533
534 return bRes;
535}

576bool conops_is_3axis_allowed(void)
577{
578 bool bRes;
579
580 DATA_CACHE_AOCS_CNTRL_TLM_t sAdcsData;
581 uint8_t u8Chk = 0;
582
583 bRes = false;
584
585 if ( DC_DATA_STATUS_OK == dc_get_aocs_cntrl_tlm_data(&sAdcsData) )
586 {
587 if ( (CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH >= sAdcsData.ai32EstAngRateVec[X]) && (-CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH <= sAdcsData.ai32EstAngRateVec[X]) )
588 {
589 ++u8Chk;
590 }
591
592 if ( (CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH >= sAdcsData.ai32EstAngRateVec[Y]) && (-CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH <= sAdcsData.ai32EstAngRateVec[Y]) )
593 {
594 ++u8Chk;
595 }
596
597 if ( (CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH >= sAdcsData.ai32EstAngRateVec[Z]) && (-CONOPS_3AXIS_CONTROL_ANG_VEL_SAFETY_THRESH <= sAdcsData.ai32EstAngRateVec[Z]) )
598 {
599 ++u8Chk;
600 }
601
602 if ( CONOPS_3AXIS_CHECK_OK == u8Chk)
603 {
604 bRes = true;
605 }
606 }
607
608 return bRes;
609}

653STATIC bool fs_conops_cmp_cbatt(CONOPS_ThresholdValueTypes_t eThreshValType, uint8_t u8PayNum, eConOps_cmp_type_t eCmpType)
654{
655 bool bRes;
656
657 int32_t i32CSafe;
658 int32_t i32Mult;
659
660 DATA_CACHE_EpsOverallBatteryInfoType_t sEPS_Data;
661
662 bRes = false;
663
664 if ( (true != conops_get_thresh_val(&i32CSafe, 0, CONOPS_THRESHOLDVALUETYPES_CBATT_SAFE )) ||
665 (true != conops_get_thresh_val(&i32Mult, u8PayNum, eThreshValType)) )
666 {
667 return bRes;
668 }
669
670 if ( DC_DATA_STATUS_OK == dc_get_epsoverallbatteryinfo_data(&sEPS_Data) ) {
671 switch (eCmpType) {
672 case eCMPTYPE_ABOVE:
673 if ( sEPS_Data.u32Capacity > (((uint32_t)i32CSafe * i32Mult) / 1000) )
674 {
675 bRes = true;
676 }
677 break;
678 case eCMPTPYE_BELOW:
679 if ( sEPS_Data.u32Capacity < (((uint32_t)i32CSafe * i32Mult) / 1000) )
680 {
681 bRes = true;
682 }
683 break;
684 default:
685 EXEH_HANDLE(eEXEHSeverity_Error,
686 (int32_t) eConOpsException_Runtime_Error);
687 break;
688 }
689
690 }
691
692 CONOPS_TRACE_DBG("%s: fs_conops_cmp_cbatt: %d\r", CONOPS_SM_TEST_COND, bRes);
693
694 return bRes;
695}

705STATIC bool fs_conops_chk_raised_adcs_err_flags(uint32_t u32ErrFlags)
