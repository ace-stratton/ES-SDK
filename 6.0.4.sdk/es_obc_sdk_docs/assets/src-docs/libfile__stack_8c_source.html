26file_stack_res_t file_stack_open(file_stack_ctx_t * const p_ctx, const TCHAR * const p_stack_fname, const uint32_t stack_elem_size)
27{
28 file_stack_res_t res = FR_INVALID_PARAMETER;
29
30 if (NULL != p_ctx)
31 {
32 p_ctx->stack_elem_size = stack_elem_size;
33 (void) strncpy(p_ctx->stack_file_name, p_stack_fname, FF_LFN_BUF);
34 p_ctx->stack_file_name[FF_LFN_BUF - 1] = '\0';
35 p_ctx->stack_elem_cnt = 0U;
36
37 res = SdMngr_f_open(&p_ctx->stack_file_hnd, p_ctx->stack_file_name, FA_CREATE_ALWAYS | FA_READ | FA_WRITE);
38 }
39
40 return res;
41}

43file_stack_res_t file_stack_push(file_stack_ctx_t * const p_ctx, const void * const p_elem)
44{
45 file_stack_res_t res = FR_INVALID_PARAMETER;
46
47 if ((NULL == p_ctx) || (NULL == p_elem))
48 {
49 return res;
50 }
51
52 uint32_t push_pos = p_ctx->stack_elem_size * p_ctx->stack_elem_cnt;
53 res = SdMngr_f_lseek(&p_ctx->stack_file_hnd, push_pos);
54
55 if (FR_OK == res)
56 {
57 UINT bytes_written = 0U;
58 res = SdMngr_f_write(&p_ctx->stack_file_hnd, p_elem, p_ctx->stack_elem_size, &bytes_written);
59
60 if ((FR_OK == res) && (bytes_written == p_ctx->stack_elem_size))
61 {
62 ++(p_ctx->stack_elem_cnt);
63 }
64 }
65
66 return res;
67}

69file_stack_res_t file_stack_pop(file_stack_ctx_t * const p_ctx, void * const p_elem)
70{
71 file_stack_res_t res = FR_INVALID_PARAMETER;
72
73 if ((NULL == p_ctx) || (NULL == p_elem) || (0 == p_ctx->stack_elem_cnt))
74 {
75 return res;
76 }
77
78 uint32_t pop_pos = (p_ctx->stack_elem_cnt - 1) * p_ctx->stack_elem_size;
79
80 res = SdMngr_f_lseek(&p_ctx->stack_file_hnd, pop_pos);
81
82 if (FR_OK == res)
83 {
84 UINT bytes_read = 0U;
85 res = SdMngr_f_read(&p_ctx->stack_file_hnd, p_elem, p_ctx->stack_elem_size, &bytes_read);
86
87 if ((FR_OK == res) && (bytes_read == p_ctx->stack_elem_size))
88 {
89 --(p_ctx->stack_elem_cnt);
90 }
91 }
92
93 return res;
94}

103

106 res = SdMngr_f_unlink(p_ctx->stack_file_name);

108 }
109
110 return res;
