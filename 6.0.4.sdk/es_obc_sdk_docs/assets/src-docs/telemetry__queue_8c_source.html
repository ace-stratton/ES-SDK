81bool telemetry_queue_push(uint8_t * const p_data, const uint16_t data_size)
82{
83 bool success = false;
84
85 CRIT_ASSERT(data_size <= sizeof(queue_rt.qbuf));
86
87 if (data_size < queue_rt.free_bytes)
88 {
89 TLM_QUEUE_LOCK();
90
91 uint8_t *p_queue_pos = &queue_rt.qbuf[queue_rt.tail_idx];
92 volatile uint16_t queue_idx = queue_rt.tail_idx;
93 uint8_t *p_data_pos = p_data;
94 uint16_t bytes_written = 0U;
95
96 while (bytes_written < data_size)
97 {
98 *p_queue_pos = *p_data_pos++;
99
100 queue_idx = (queue_idx + 1) % TLM_QUEUE_SIZE;
101 p_queue_pos = &queue_rt.qbuf[queue_idx];
102
103 bytes_written++;
104
105 // did we end up at the first message to be read?
106 if (((queue_idx + 1U) % TLM_QUEUE_SIZE) == queue_rt.head_idx)
107 {
108 break;
109 }
110 }
111
112 queue_rt.free_bytes -= bytes_written;
113
114 if (queue_rt.stats.free_bytes_min > queue_rt.free_bytes)
115 {
116 queue_rt.stats.free_bytes_min = queue_rt.free_bytes;
117 }
118
119 queue_rt.tail_idx = (queue_rt.tail_idx + bytes_written) % TLM_QUEUE_SIZE;
120 success = true;
121
122 if (queue_rt.stats.msg_pushed_cnt < ULONG_MAX)
123 {
124 queue_rt.stats.msg_pushed_cnt++;
125 }
126
127 TLM_QUEUE_UNLOCK();
128 }
129
130 if ((!success) && (queue_rt.stats.discarded_messages_cnt < ULONG_MAX))
131 {
132 queue_rt.stats.discarded_messages_cnt++;
133 }
134
135 return success;
136}

138uint16_t telemetry_queue_pop(uint8_t * const p_data_buf, const uint16_t buf_size)
139{
140 uint16_t bytes_written = 0U;
141
142 // are there any messages in the queue?
143 if (queue_rt.free_bytes < TLM_QUEUE_SIZE)
144 {
145 TLM_QUEUE_LOCK();
146
147 uint8_t *p_queue_pos = &queue_rt.qbuf[queue_rt.head_idx];
148 uint16_t queue_idx = queue_rt.head_idx;
149 uint8_t *p_data_pos = p_data_buf;
150
151 while ((bytes_written < buf_size) &&
152 (TLM_FRM_DELIMITER != *p_queue_pos))
153 {
154 *p_data_pos++ = *p_queue_pos;
155 *p_queue_pos = TLM_FRM_DELIMITER;
156 queue_idx = (queue_idx + 1) % TLM_QUEUE_SIZE;
157 p_queue_pos = &queue_rt.qbuf[queue_idx];
158
159 bytes_written++;
160 }
161
162 // did we actually reach the end of a frame?
163 if (TLM_FRM_DELIMITER != *p_queue_pos)
164 {
165 // no, probably some invalid data was placed in the queue
166 bytes_written = 0U;
167 }
168 else
169 {
170 *p_data_pos++ = *p_queue_pos;
171 bytes_written++;
172
173 queue_rt.free_bytes += bytes_written;
174 queue_rt.head_idx = (queue_rt.head_idx + bytes_written) % TLM_QUEUE_SIZE;
175
176 if (queue_rt.stats.msg_popped_cnt < ULONG_MAX)
177 {
178 queue_rt.stats.msg_popped_cnt++;
179 }
180 }
181
182 TLM_QUEUE_UNLOCK();
183 }
184
185 return bytes_written;
186}

188uint16_t telemetry_queue_peek(void)
189{
190 uint16_t bytes_read = 0U;
191
192 if (queue_rt.free_bytes < TLM_QUEUE_SIZE)
193 {
194 TLM_QUEUE_LOCK();
195
196 uint8_t *p_queue_pos = &queue_rt.qbuf[queue_rt.head_idx];
197 uint16_t queue_idx = queue_rt.head_idx;
198
199 while ((queue_idx != queue_rt.tail_idx) &&
200 (TLM_FRM_DELIMITER != *p_queue_pos))
201 {
202 queue_idx = (queue_idx + 1) % TLM_QUEUE_SIZE;
203 p_queue_pos = &queue_rt.qbuf[queue_idx];
204
205 bytes_read++;
206 }
207
208 // did we actually reach the end of a frame?
209 if (TLM_FRM_DELIMITER != *p_queue_pos)
210 {
211 // no, probably some invalid data was placed in the queue
212 bytes_read = 0U;
213 }
214 else
215 {
216 bytes_read++;
217 }
218
219 TLM_QUEUE_UNLOCK();
220 }
221
222 return bytes_read;
223}

237void telemetry_queue_update_stats(const tlm_queue_stats_id_t stat_id, const uint32_t value)
238{
239 TLM_QUEUE_LOCK();
240
241 switch (stat_id)
242 {
243 case TLM_STATS_ID_BATCH_COUNT:
244 {
245 if (queue_rt.stats.max_number_of_batch_msg < value)
246 {
247 queue_rt.stats.max_number_of_batch_msg = value;
248 }
249
250 break;
251 }
252
253 case TLM_STATS_ID_ACTIVE_SLOT:
254 {
255 queue_rt.stats.active_slot_id = value;
256 break;
257 }
258
259 default:
260 {
261 CRIT_ASSERT(false);
262 break;
263 }
264 }
265
266 TLM_QUEUE_UNLOCK();
267}
