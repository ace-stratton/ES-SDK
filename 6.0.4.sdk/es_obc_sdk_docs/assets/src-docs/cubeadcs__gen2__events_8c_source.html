66bool cubeadcs_gen2_events_get_marker(void)
67{
68 bool op_res = false;
69
70 static cubeadcs_gen2_event_dwn_progress_t evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_IDLE;
71
72 if ( true == fs_cubeadcs_gen2_events_lock_ev() )
73 {
74 switch (evt_dwn_progress)
75 {
76 case CUBEADCS_GEN2_EVENT_DWN_IDLE:
77 {
78 if (true == fs_cubeadcs_gen2_events_get_marker_start())
79 {
80 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_STARTED;
81 }
82 }
83 break;
84
85 case CUBEADCS_GEN2_EVENT_DWN_STARTED:
86 {
87 cubeadcs_gen2_transfer_evt_status_t get_marker_status;
88 const ErrorCode err = cubeadcs_gen2_transfer_evt_status(&get_marker_status);
89
90 if (CUBEOBC_ERROR_OK == err)
91 {
92 if (CUBEOBC_ERROR_OK == get_marker_status.err_code)
93 {
94 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_POLL;
95 }
96 else
97 {
98 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_IDLE;
99 }
100 }
101 }
102 break;
103
104 case CUBEADCS_GEN2_EVENT_DWN_POLL:
105 {
106 if (true == fs_cubeadcs_gen2_events_get_marker_poll())
107 {
108 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_FINISHED;
109 }
110 else
111 {
112 // Retry the event log download sequence
113 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_IDLE;
114 }
115 }
116 break;
117
118 case CUBEADCS_GEN2_EVENT_DWN_FINISHED:
119 {
120 op_res = true;
121
122 // Back to vanilla state
123 evt_dwn_progress = CUBEADCS_GEN2_EVENT_DWN_IDLE;
124 }
125 break;
126
127 default:
128 break;
129 }
130
131 (void)fs_cubeadcs_gen2_events_unlock_ev();
132 }
133
134 return op_res;
135}

137bool cubeadcs_gen2_events_check_for_new_crit_events(bool * const is_new_crit_events)
138{
139 CRIT_ASSERT(NULL != is_new_crit_events);
140
141 bool op_res = false;
142
143 TctlmCubeComputerCommon3_EventLogStatus event_log_status;
144
145 if ( (true == fs_cubeadcs_gen2_events_lock_ev()) &&
146 (CUBEOBC_ERROR_OK == CubeADCS_Gen2_CubeComputerCommon3_getEventLogStatusRequestGw(TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &event_log_status, CUBEADCS_GEN2_DEFAULT_CRIT_SECTION_TIMEOUT)))
147 {
148 // It is possible that the events log gets deleted by the user ...
149 if ( (events_ctx.last_major_events_num != event_log_status.numberOfMajorWarningEvents) ||
150 (events_ctx.last_crit_events_num != event_log_status.numberOfCriticalEvents) )
151 {
152 events_ctx.events_to_download = abs((event_log_status.numberOfMajorWarningEvents - events_ctx.last_major_events_num) +
153 (event_log_status.numberOfCriticalEvents - events_ctx.last_crit_events_num));
154
155 TODO("TEST how much time it takes to download events and modify the limit accordingly")
156 if ( events_ctx.events_to_download > CUBEADCS_GEN2_EVENTS_MAX_EVENTS_TO_DOWNLOAD )
157 {
158 events_ctx.events_to_download = CUBEADCS_GEN2_EVENTS_MAX_EVENTS_TO_DOWNLOAD;
159 }
160
161 events_ctx.last_major_events_num = event_log_status.numberOfMajorWarningEvents;
162 events_ctx.last_crit_events_num = event_log_status.numberOfCriticalEvents;
163
164 *is_new_crit_events = true;
165 }
166 else
167 {
168 *is_new_crit_events = false;
169 }
170
171 (void)fs_cubeadcs_gen2_events_unlock_ev();
172
173 op_res = true;
174 }
175
176 return op_res;

179bool cubeadcs_gen2_events_dwn_new_crit_events(bool * const p_is_crit_evt)
180{
181 CRIT_ASSERT(NULL != p_is_crit_evt);
182
183 bool op_res = false;
184
185 static bool is_new_crit_event = false;
186
187 static cubeadcs_gen2_event_crit_dwn_states_t dwn_state = CUBEADCS_GEN2_EVENT_CRIT_IDLE;
188
189 if ( true == fs_cubeadcs_gen2_events_lock_ev() )
190 {
191 switch (dwn_state)
192 {
193 case CUBEADCS_GEN2_EVENT_CRIT_IDLE:
194 if ( true == cubeadcs_gen2_events_check_for_new_crit_events(&is_new_crit_event))
195 {
196 dwn_state = CUBEADCS_GEN2_EVENT_CRIT_CHECK;
197 }
198 break;
199 case CUBEADCS_GEN2_EVENT_CRIT_CHECK:
200 if ( true == is_new_crit_event )
201 {
202 if ( true == fs_cubeadcs_gen2_events_dwn_new_crit_events_start() )
203 {
204 dwn_state = CUBEADCS_GEN2_EVENT_CRIT_START_DWN;
205 }
206 }
207 else
208 {
209 dwn_state = CUBEADCS_GEN2_EVENT_CRIT_FINISHED;
210 }
211 break;
212 case CUBEADCS_GEN2_EVENT_CRIT_START_DWN:
213 if ( true == fs_cubeadcs_gen2_events_dwn_new_crit_events_poll() )
214 {
215 dwn_state = CUBEADCS_GEN2_EVENT_CRIT_FINISHED;
216 }
217 break;
218 case CUBEADCS_GEN2_EVENT_CRIT_FINISHED:
219
220 *p_is_crit_evt = is_new_crit_event;
221
222 op_res = true;
223
224 // Return to vanilla state
225 dwn_state = CUBEADCS_GEN2_EVENT_CRIT_IDLE;
226 is_new_crit_event = false;
227
228 break;
229
230 default:
231 break;
232 }
233
234 (void)fs_cubeadcs_gen2_events_unlock_ev();
235 }
236
237 return op_res;

289STATIC bool fs_cubeadcs_gen2_events_parse_events_log(const char * const p_f_name, const uint32_t ev_cnt)
290{
291 CRIT_ASSERT(NULL != p_f_name);
292
293 FIL f_handle;
294
295 bool op_res = false;
296
297 cubeadcs_gen2_event_t event;
298 UINT read_bytes;
299
300 if ( (FR_OK == SdMngr_f_open(&f_handle, p_f_name, FA_READ)) &&
301 (FR_OK == SdMngr_f_lseek(&f_handle, 0)) &&
302 (true == fs_cubeadcs_gen2_events_lock_ev()))
303 {
304 for (uint32_t ev = 0; ev < ev_cnt; )
305 {
306 if ( (FR_OK == SdMngr_f_read(&f_handle, (void *)&event, sizeof(cubeadcs_gen2_event_t), &read_bytes)) &&
307 (read_bytes == sizeof(cubeadcs_gen2_event_t)))
308 {
309 fs_cubeadcs_gen2_events_update_events_cache(&event);
310 }
311 else
312 {
313 // We want to get all events and FDIR on them, otherwise just break and try again
314 break;
315 }
316
317 if ( ++ev == ev_cnt )
318 {
319 events_ctx.last_read_counter = event.counter;
320
321 for (uint8_t ev_indx = 0; ev_indx < CUBEADCS_GEN2_EVENTS_MAX; ev_indx++)
322 {
323 events_ctx.read_events.all_read_events[ev_indx] += events_ctx.read_events.last_read_events[ev_indx];
324 }
325
326 op_res = true;
327 }
328 }
329
330 (void)SdMngr_f_close(&f_handle);
331 (void)fs_cubeadcs_gen2_events_unlock_ev();
332 }
333
334 return op_res;
335}

343STATIC void fs_cubeadcs_gen2_events_update_events_cache(const cubeadcs_gen2_event_t * const p_ev)
344{
345 CRIT_ASSERT(NULL != p_ev);
346
347 if ( (CUBEADCS_GEN2_EVENT_CLASS_CRITICAL == p_ev->identifier.event_class) ||
348 (CUBEADCS_GEN2_EVENT_CLASS_MAJOR == p_ev->identifier.event_class))
349 {
350 switch (p_ev->identifier.event_type)
351 {
352 case CUBEADCS_GEN2_EVENTS_ADCS_CONFIG_VALIDATION_ERROR_ID:
353 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_ADCS_CONFIG_VALIDATION_ERROR] += 1;
354 break;
355 case CUBEADCS_GEN2_EVENTS_NODE_SETUP_ERROR_ID:
356 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_NODE_SETUP_ERROR] += 1;
357 break;
358 case CUBEADCS_GEN2_EVENTS_NODE_JUMP_ERROR_ID:
359 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_NODE_JUMP_ERROR] += 1;
360 break;
361 case CUBEADCS_GEN2_EVENTS_MAG_DEPLOY_ERROR_ID:
362 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_MAG_DEPLOY_ERROR] += 1;
363 break;
364 case CUBEADCS_GEN2_EVENTS_CONTROL_LOOP_FAILURE_ID:
365 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_CONTROL_LOOP_FAILURE] += 1;
366 break;
367 case CUBEADCS_GEN2_EVENTS_NODE_FAILURE_ID:
368 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_NODE_FAILURE] += 1;
369 break;
370 case CUBEADCS_GEN2_EVENTS_CONFIG_PERSIST_ERROR_ID:
371 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_CONFIG_PERSIST_ERROR] += 1;
372 break;
373 case CUBEADCS_GEN2_EVENTS_VIRTUAL_WATCHDOG_WARNING_ID:
374 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_VIRTUAL_WATCHDOG_WARNING] += 1;
375 break;
376 case CUBEADCS_GEN2_EVENTS_PORT_VALIDATION_ERROR_ID:
377 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_PORT_VALIDATION_ERROR] += 1;
378 break;
379 case CUBEADCS_GEN2_EVENTS_AUTO_DISCOVERY_ERROR_ID:
380 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_AUTO_DISCOVERY_ERROR] += 1;
381 break;
382 case CUBEADCS_GEN2_EVENTS_CONTROL_LOOP_CRITICAL_ID:
383 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_CONTROL_LOOP_CRITICAL] += 1;
384 break;
385 case CUBEADCS_GEN2_EVENTS_GYRO_VOLTAGE_ERROR_ID:
386 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_GYRO_VOLTAGE_ERROR] += 1;
387 break;
388 case CUBEADCS_GEN2_EVENTS_GYRO_CRITICAL_ID:
389 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_GYRO_CRITICAL] += 1;
390 break;
391 case CUBEADCS_GEN2_EVENTS_VIRTUAL_WATCHDOG_ERROR_ID:
392 events_ctx.read_events.last_read_events[CUBEADCS_GEN2_EVENTS_VIRTUAL_WATCHDOG_ERROR] += 1;
393 break;
394 default:
395 break;
396 }
397 }
