54ErrorCode cubeadcs_gen2_transfer_init(void)
55{
56 ErrorCode ret = CUBEOBC_ERROR_OK;
57
58 if (NULL == fs_lock)
59 {
60 fs_evt.err_code = CUBEOBC_ERROR_USAGE;
61 fs_tlm.err_code = CUBEOBC_ERROR_USAGE;
62 fs_img_stored.err_code = CUBEOBC_ERROR_USAGE;
63 fs_csf_bl.err_code = CUBEOBC_ERROR_USAGE;
64 fs_csf_ap.err_code = CUBEOBC_ERROR_USAGE;
65
66 osMutexAttr_t attr =
67 {
68 .name = "cube_transfer",
69 .attr_bits = osMutexRecursive | osMutexPrioInherit,
70 .cb_mem = NULL,
71 .cb_size = 0U
72 };
73
74 fs_lock = osMutexNew(&attr);
75
76 if (NULL == fs_lock)
77 {
78 ret = CUBEOBC_ERROR_USAGE;
79 }
80 }
81
82 if ((CUBEOBC_ERROR_OK == ret) && (NULL == fs_task))
83 {
84 osThreadAttr_t attr =
85 {
86 .attr_bits = osThreadDetached,
87 .name = "cube_transfer",
88 .priority = osPriorityNormal,
89 .cb_mem = NULL,
90 .cb_size = 0U,
91 .stack_mem = NULL,
92 .stack_size = 2048,
93 .tz_module = 0U,
94 .reserved = 0U,
95 };
96
97 fs_task = osThreadNew(fs_transfer_task, NULL, &attr);
98
99 if (NULL == fs_task)
100 {
101 ret = CUBEOBC_ERROR_USAGE;
102 }
103 }
104
105 return ret;
106}

109ErrorCode cubeadcs_gen2_transfer_evt(const char* file_name, const TctlmCubeComputerCommon3_EventLogFilterTransferSetup* setup)
110{
111 ErrorCode ret = CUBEOBC_ERROR_USAGE;
112 uint32_t len = strnlen(file_name, CUBEADCS_GEN2_BDT_FILE_NAME_SZ);
113
114 if ((NULL != file_name) &&
115 (0 < len) &&
116 (NULL != setup))
117 {
118 if (osOK == osMutexAcquire(fs_lock, FS_LOCK_TIMEOUT))
119 {
120 memset(&fs_evt, 0, sizeof(fs_evt));
121 memset(&fs_file_ctx, 0, sizeof(fs_file_ctx));
122 memcpy(fs_evt.file_name, file_name, len);
123 memcpy(&fs_evt.setup, setup, sizeof(fs_evt.setup));
124
125 // null termination already added by memset()
126 memcpy(fs_file_ctx.file_name, file_name, len);
127
128 fs_file_ctx.context = CUBEADCS_GEN2_BDT_CTX_EVT_CC;
129
130 // create the file
131 if (FR_OK == SdMngr_f_open(&fs_file_ctx.file_handle, fs_file_ctx.file_name, FA_CREATE_ALWAYS | FA_WRITE))
132 {
133 ret = CUBEOBC_ERROR_OK;
134
135 (void)SdMngr_f_close(&fs_file_ctx.file_handle);
136 (void)osThreadFlagsSet(fs_task, FS_START_TRANSFER_FLAG);
137 }
138 else
139 {
140 ret = CUBEOBC_ERROR_UNKNOWN;
141 }
142
143 (void)osMutexRelease(fs_lock);
144 }
145 else
146 {
147 ret = CUBEOBC_ERROR_BUSY;
148 }
149 }
150
151 return ret;
152}

453static ErrorCode fs_start_blocking_upload_to_boot(TypeDef_TctlmEndpoint* endpoint)
454{
455 CRIT_ASSERT(NULL != endpoint);
456
457 ErrorCode status;
458 FILINFO fno_cp;
459 FILINFO fno_cfg;
460 TctlmCommonFramework1_Identification identity;
461
462 // Check if the file exists in OBC.
463 // Since we are uploading, it must exist in our file system.
464 if (FR_OK != SdMngr_f_stat(fs_csf_bl.cp_file_name, &fno_cp))
465 {
466 // The file doesn't exist...
467 status = CUBEOBC_ERROR_EXIST;
468 }
469 else
470 {
471 // The file exists, we can continue with next steps.
472 status = CUBEOBC_ERROR_OK;
473 }
474
475 // Check if the file exists in OBC.
476 // Since we are uploading, it must exist in our file system.
477 if (CUBEOBC_ERROR_OK == status)
478 {
479 if (FR_OK != SdMngr_f_stat(fs_csf_bl.cfg_file_name, &fno_cfg))
480 {
481 // The file doesn't exist...
482 status = CUBEOBC_ERROR_EXIST;
483 }
484 else
485 {
486 // The file exists, we can continue with next steps.
487 status = CUBEOBC_ERROR_OK;
488 }
489 }
490
491 if (CUBEOBC_ERROR_OK == status)
492 {
493 status = CubeADCS_Gen2_CommonFramework1_getIdentificationRequestGw(
494 TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &identity, CUBEOBC_CRITICAL_WAIT_FOREVER);
495 }
496
497 if ((CUBEOBC_ERROR_OK == status) &&
498 (TCTLM_COMMON_FRAMEWORK_ENUMS__PROGRAM_TYPE_BOOTLOADER == identity.programType))
499 {
500 // we are already in bootloader. go directly to upload.
501 }
502 else if ((CUBEOBC_ERROR_OK == status) &&
503 (true == fs_csf_bl.force) &&
504 (TCTLM_COMMON_FRAMEWORK_ENUMS__PROGRAM_TYPE_CONTROL == identity.programType))
505 {
506 // reset to bootloader
507 (void)CubeADCS_Gen2_CommonFramework1_setResetRequestGw(
508 TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, TCTLM_COMMON_FRAMEWORK_1__SOFT, CUBEOBC_CRITICAL_WAIT_FOREVER);
509
510 // wait for boot
511 cubeObc_time_delay(1000u);
512
513 status = CubeADCS_Gen2_CommonFramework1_getIdentificationRequestGw(
514 TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, &identity, CUBEOBC_CRITICAL_WAIT_FOREVER);
515
516 if ((CUBEOBC_ERROR_OK == status) &&
517 ((TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_TYPE_CUBE_COMPUTER != identity.nodeType) ||
518 (TCTLM_COMMON_FRAMEWORK_ENUMS__PROGRAM_TYPE_BOOTLOADER != identity.programType)))
519 {
520 // failed to enter bootloader
521 status = CUBEOBC_ERROR_UNKNOWN;
522 }
523
524 if (CUBEOBC_ERROR_OK == status)
525 {
526 // stay in bootloader. do not jump to application
527 status = CubeADCS_Gen2_CubeCommonBaseBootloader5_setHaltRequestGw(
528 TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, CUBEOBC_CRITICAL_WAIT_FOREVER);
529 }
530 }
531 else if (CUBEOBC_ERROR_OK == status)
532 {
533 // we do not support uploads in any other program type
534 status = CUBEOBC_ERROR_UNKNOWN;
535 }
536 else
537 {
538 // nothing to do
539 }
540
541 // save context type
542 U32 ctx = fs_file_ctx.context;
543
544 if (CUBEOBC_ERROR_OK == status)
545 {
546 // reset whole file context
547 memset(&fs_file_ctx, 0, sizeof(fs_file_ctx));
548
549 (void)memcpy(
550 fs_file_ctx.file_name,
551 fs_csf_bl.cp_file_name,
552 strnlen(fs_csf_bl.cp_file_name, CUBEADCS_GEN2_BDT_FILE_NAME_SZ));
553
554 // restore context type
555 fs_file_ctx.context = ctx;
556
557 status = cubeObc_bootloader_uploadCubeSpaceFile(
558 endpoint, fno_cp.fsize, &fs_file_ctx, &fs_csf_bl.status);
559 }
560
561 if (CUBEOBC_ERROR_OK == status)
562 {
563 // reset whole file context
564 memset(&fs_file_ctx, 0, sizeof(fs_file_ctx));
565
566 (void)memcpy(
567 fs_file_ctx.file_name,
568 fs_csf_bl.cfg_file_name,
569 strnlen(fs_csf_bl.cfg_file_name, CUBEADCS_GEN2_BDT_FILE_NAME_SZ));
570
571 // restore context type
572 fs_file_ctx.context = ctx;
573
574 status = cubeObc_bootloader_uploadCubeSpaceFile(
575 endpoint, fno_cfg.fsize, &fs_file_ctx, &fs_csf_bl.status);
576 }
577
578 if ((CUBEOBC_ERROR_OK == status) && (CUBEOBC_ERROR_OK == fs_csf_bl.status.result))
579 {
580 status = CubeADCS_Gen2_CubeCommonBaseBootloader5_setJumpToDefaultAppRequestGw(
581 TCTLM_COMMON_FRAMEWORK_ENUMS__NODE_COMPUTER, CUBEOBC_CRITICAL_WAIT_FOREVER);
582 }
583
584 return status;
585}

693
694 (void)memcpy(
695 fs_file_ctx.file_name,
696 fs_csf_ap.cfg_file_name,
697 strnlen(fs_csf_ap.cfg_file_name, CUBEADCS_GEN2_BDT_FILE_NAME_SZ));
698
699 // restore context type
700 fs_file_ctx.context = ctx;
701
702 status = cubeObc_cubeComputer_uploadCubeSpaceFile(
703 endpoint, fno_cfg.fsize, &fs_file_ctx, &fs_csf_ap.status);
704 }
705
706 if ((CUBEOBC_ERROR_OK == status) && (CUBEOBC_ERROR_OK == fs_csf_ap.status.errorCode))
707 {
708 status = cubeObc_cubeComputer_upgrade(
709 endpoint,
710 fs_csf_ap.node_type,
711 fs_csf_ap.serial_int,
712 TCTLM_COMMON_FRAMEWORK_ENUMS__PROGRAM_TYPE_CONTROL,
713 TCTLM_CUBE_COMPUTER_CONTROL_PROGRAM_8__PORT_NONE,
714 &fs_csf_ap.status);
715 }
716
717 return status;
718}

736

741 }

754 cubeObc_cubeComputer_tlmDownload(

756 } break;

760 cubeObc_cubeComputer_imageDownloadStored(

762 } break;
763 case CUBEADCS_GEN2_BDT_CTX_FTP_BL:
764 {
765 fs_csf_bl.err_code = (CUBEOBC_ERROR_OK != status) ? status :
766 fs_start_blocking_upload_to_boot(&endpoint);
767 } break;
768 case CUBEADCS_GEN2_BDT_CTX_FTP_AP:
769 {
770 fs_csf_ap.err_code = (CUBEOBC_ERROR_OK != status) ? status :
771 fs_start_blocking_upload_to_appl(&endpoint);
772 } break;
773 default:
774 {
775 // no more contexts
776 CRIT_ASSERT(false);
777 } break;
778 }
