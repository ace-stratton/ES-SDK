148void SystemInit (void)
149{
150#if defined (DATA_IN_D2_SRAM)
151 __IO uint32_t tmpreg;
152#endif /* DATA_IN_D2_SRAM */
153
154 /* FPU settings ------------------------------------------------------------*/
155 #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
156 SCB->CPACR |= ((3UL << (10*2))|(3UL << (11*2))); /* set CP10 and CP11 Full Access */
157 #endif
158 /* Reset the RCC clock configuration to the default reset state ------------*/
159
160 /* Increasing the CPU frequency */
161 if(FLASH_LATENCY_DEFAULT > (READ_BIT((FLASH->ACR), FLASH_ACR_LATENCY)))
162 {
163 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
164 MODIFY_REG(FLASH->ACR, FLASH_ACR_LATENCY, (uint32_t)(FLASH_LATENCY_DEFAULT));
165 }
166
167 /* Set HSION bit */
168 RCC->CR |= RCC_CR_HSION;
169
170 /* Reset CFGR register */
171 RCC->CFGR = 0x00000000;
172
173 /* Reset HSEON, HSECSSON, CSION, HSI48ON, CSIKERON, PLL1ON, PLL2ON and PLL3ON bits */
174 RCC->CR &= 0xEAF6ED7FU;
175
176 /* Decreasing the number of wait states because of lower CPU frequency */
177 if(FLASH_LATENCY_DEFAULT < (READ_BIT((FLASH->ACR), FLASH_ACR_LATENCY)))
178 {
179 /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
180 MODIFY_REG(FLASH->ACR, FLASH_ACR_LATENCY, (uint32_t)(FLASH_LATENCY_DEFAULT));
181 }
182
183#if defined(D3_SRAM_BASE)
184 /* Reset D1CFGR register */
185 RCC->D1CFGR = 0x00000000;
186
187 /* Reset D2CFGR register */
188 RCC->D2CFGR = 0x00000000;
189
190 /* Reset D3CFGR register */
191 RCC->D3CFGR = 0x00000000;
192#else
193 /* Reset CDCFGR1 register */
194 RCC->CDCFGR1 = 0x00000000;
195
196 /* Reset CDCFGR2 register */
197 RCC->CDCFGR2 = 0x00000000;
198
199 /* Reset SRDCFGR register */
200 RCC->SRDCFGR = 0x00000000;
201#endif
202 /* Reset PLLCKSELR register */
203 RCC->PLLCKSELR = 0x02020200;
204
205 /* Reset PLLCFGR register */
206 RCC->PLLCFGR = 0x01FF0000;
207 /* Reset PLL1DIVR register */
208 RCC->PLL1DIVR = 0x01010280;
209 /* Reset PLL1FRACR register */
210 RCC->PLL1FRACR = 0x00000000;
211
212 /* Reset PLL2DIVR register */
213 RCC->PLL2DIVR = 0x01010280;
214
215 /* Reset PLL2FRACR register */
216
217 RCC->PLL2FRACR = 0x00000000;
218 /* Reset PLL3DIVR register */
219 RCC->PLL3DIVR = 0x01010280;
220
221 /* Reset PLL3FRACR register */
222 RCC->PLL3FRACR = 0x00000000;
223
224 /* Reset HSEBYP bit */
225 RCC->CR &= 0xFFFBFFFFU;
226
227 /* Disable all interrupts */
228 RCC->CIER = 0x00000000;
229
230#if (STM32H7_DEV_ID == 0x450UL)
231 /* dual core CM7 or single core line */
232 if((DBGMCU->IDCODE & 0xFFFF0000U) < 0x20000000U)
233 {
234 /* if stm32h7 revY*/
235 /* Change the switch matrix read issuing capability to 1 for the AXI SRAM target (Target 7) */
236 *((__IO uint32_t*)0x51008108) = 0x000000001U;
237 }
238#endif
239
240#if defined (DATA_IN_D2_SRAM)
241 /* in case of initialized data in D2 SRAM (AHB SRAM) , enable the D2 SRAM clock (AHB SRAM clock) */
242#if defined(RCC_AHB2ENR_D2SRAM3EN)
243 RCC->AHB2ENR |= (RCC_AHB2ENR_D2SRAM1EN | RCC_AHB2ENR_D2SRAM2EN | RCC_AHB2ENR_D2SRAM3EN);
244#elif defined(RCC_AHB2ENR_D2SRAM2EN)
245 RCC->AHB2ENR |= (RCC_AHB2ENR_D2SRAM1EN | RCC_AHB2ENR_D2SRAM2EN);
246#else
247 RCC->AHB2ENR |= (RCC_AHB2ENR_AHBSRAM1EN | RCC_AHB2ENR_AHBSRAM2EN);
248#endif /* RCC_AHB2ENR_D2SRAM3EN */
249
250 tmpreg = RCC->AHB2ENR;
251 (void) tmpreg;
252#endif /* DATA_IN_D2_SRAM */
253
254#if defined(DUAL_CORE) && defined(CORE_CM4)
255 /* Configure the Vector Table location add offset address for cortex-M4 ------------------*/
256#ifdef VECT_TAB_SRAM
257 SCB->VTOR = D2_AXISRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal SRAM */
258#else
259 SCB->VTOR = FLASH_BANK2_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
260#endif /* VECT_TAB_SRAM */
261
262#else
263
264 /*
265 * Disable the FMC bank1 (enabled after reset).
266 * This, prevents CPU speculation access on this bank which blocks the use of FMC during
267 * 24us. During this time the others FMC master (such as LTDC) cannot use it!
268 */
269 FMC_Bank1_R->BTCR[0] = 0x000030D2;
270
271 /* Configure the Vector Table location add offset address for cortex-M7 ------------------*/
272#ifdef VECT_TAB_SRAM
273 SCB->VTOR = D1_AXISRAM_BASE | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal AXI-RAM */
274#else
275 // if this image doesn't support a bootloader, VTOR register must be set by the application,
276 // otherwise the bootloader will take care to set it when it performs the application jump at boot up
277 #if defined(BOOTLOADER_DISABLED)
278 SCB->VTOR = ((uint32_t) &_Application_Start_Addr) | VECT_TAB_OFFSET; /* Vector Table Relocation in Internal FLASH */
279 #endif // #if defined(BOOTLOADER_DISABLED)
280#endif
281
282#endif /*DUAL_CORE && CORE_CM4*/
283
284}

323void SystemCoreClockUpdate (void)
324{
325 uint32_t pllp, pllsource, pllm, pllfracen, hsivalue, tmp;
326 uint32_t common_system_clock;
327 float_t fracn1, pllvco;
328
329
330 /* Get SYSCLK source -------------------------------------------------------*/
331
332 switch (RCC->CFGR & RCC_CFGR_SWS)
333 {
334 case RCC_CFGR_SWS_HSI: /* HSI used as system clock source */
335 common_system_clock = (uint32_t) (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3));
336 break;
337
338 case RCC_CFGR_SWS_CSI: /* CSI used as system clock source */
339 common_system_clock = CSI_VALUE;
340 break;
341
342 case RCC_CFGR_SWS_HSE: /* HSE used as system clock source */
343 common_system_clock = HSE_VALUE;
344 break;
345
346 case RCC_CFGR_SWS_PLL1: /* PLL1 used as system clock source */
347
348 /* PLL_VCO = (HSE_VALUE or HSI_VALUE or CSI_VALUE/ PLLM) * PLLN
349 SYSCLK = PLL_VCO / PLLR
350 */
351 pllsource = (RCC->PLLCKSELR & RCC_PLLCKSELR_PLLSRC);
352 pllm = ((RCC->PLLCKSELR & RCC_PLLCKSELR_DIVM1)>> 4) ;
353 pllfracen = ((RCC->PLLCFGR & RCC_PLLCFGR_PLL1FRACEN)>>RCC_PLLCFGR_PLL1FRACEN_Pos);
354 fracn1 = (float_t)(uint32_t)(pllfracen* ((RCC->PLL1FRACR & RCC_PLL1FRACR_FRACN1)>> 3));
355
356 if (pllm != 0U)
357 {
358 switch (pllsource)
359 {
360 case RCC_PLLCKSELR_PLLSRC_HSI: /* HSI used as PLL clock source */
361
362 hsivalue = (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3)) ;
363 pllvco = ( (float_t)hsivalue / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
364
365 break;
366
367 case RCC_PLLCKSELR_PLLSRC_CSI: /* CSI used as PLL clock source */
368 pllvco = ((float_t)CSI_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
369 break;
370
371 case RCC_PLLCKSELR_PLLSRC_HSE: /* HSE used as PLL clock source */
372 pllvco = ((float_t)HSE_VALUE / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
373 break;
374
375 default:
376 hsivalue = (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3)) ;
377 pllvco = ((float_t)hsivalue / (float_t)pllm) * ((float_t)(uint32_t)(RCC->PLL1DIVR & RCC_PLL1DIVR_N1) + (fracn1/(float_t)0x2000) +(float_t)1 );
378 break;
379 }
380 pllp = (((RCC->PLL1DIVR & RCC_PLL1DIVR_P1) >>9) + 1U ) ;
381 common_system_clock = (uint32_t)(float_t)(pllvco/(float_t)pllp);
382 }
383 else
384 {
385 common_system_clock = 0U;
386 }
387 break;
388
389 default:
390 common_system_clock = (uint32_t) (HSI_VALUE >> ((RCC->CR & RCC_CR_HSIDIV)>> 3));
391 break;
392 }
393
394 /* Compute SystemClock frequency --------------------------------------------------*/
395#if defined (RCC_D1CFGR_D1CPRE)
396 tmp = D1CorePrescTable[(RCC->D1CFGR & RCC_D1CFGR_D1CPRE)>> RCC_D1CFGR_D1CPRE_Pos];
397
398 /* common_system_clock frequency : CM7 CPU frequency */
399 common_system_clock >>= tmp;
400
401 /* SystemD2Clock frequency : CM4 CPU, AXI and AHBs Clock frequency */
402 SystemD2Clock = (common_system_clock >> ((D1CorePrescTable[(RCC->D1CFGR & RCC_D1CFGR_HPRE)>> RCC_D1CFGR_HPRE_Pos]) & 0x1FU));
403
404#else
405 tmp = D1CorePrescTable[(RCC->CDCFGR1 & RCC_CDCFGR1_CDCPRE)>> RCC_CDCFGR1_CDCPRE_Pos];
406
407 /* common_system_clock frequency : CM7 CPU frequency */
408 common_system_clock >>= tmp;
409
410 /* SystemD2Clock frequency : AXI and AHBs Clock frequency */
411 SystemD2Clock = (common_system_clock >> ((D1CorePrescTable[(RCC->CDCFGR1 & RCC_CDCFGR1_HPRE)>> RCC_CDCFGR1_HPRE_Pos]) & 0x1FU));
412
413#endif
414
415#if defined(DUAL_CORE) && defined(CORE_CM4)
416 SystemCoreClock = SystemD2Clock;
417#else
418 SystemCoreClock = common_system_clock;
419#endif /* DUAL_CORE && CORE_CM4 */
420}
