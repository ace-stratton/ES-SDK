59uint8_t beacons_cfg_get_active_preset(void)
60{
61 uint8_t conops_preset_id = 0U;
62
63 if (IS_TRUE(sim_mode_ctx.sim_mode_active))
64 {
65 if (sim_mode_ctx.req_preset_id >= BEACON_PRESETS_COUNT)
66 {
67 sim_mode_ctx.req_preset_id = 0U;
68 }
69
70 conops_preset_id = sim_mode_ctx.req_preset_id;
71 }
72 else
73 {
74 uint8_t state_id = 0U;
75
76 conops_get_hsm_active_state(&state_id);
77
78 switch (state_id)
79 {
80 case CONOPS_OPMODES_STATE_SAFE:
81 {
82 conops_preset_id = 0U;
83
84 break;
85 }
86
87 case CONOPS_OPMODES_STATE_IDLE:
88 {
89 conops_preset_id = 1U;
90
91 break;
92 }
93
94 case CONOPS_OPMODES_STATE_DETUMBLING:
95 {
96 conops_preset_id = 2U;
97
98 break;
99 }
100
101 case CONOPS_OPMODES_STATE_PAYLOAD:
102 {
103 conops_preset_id = 3U;
104
105 break;
106 }
107
108 default:
109 {
110 break;
111 }
112 }
113
114 }
115
116 return conops_preset_id;
117}

125bool beacons_cfg_set_active_preset(uint8_t preset_id)
126{
127 bool valid_cfg = false;
128
129 if (preset_id < BEACON_PRESETS_COUNT)
130 {
131 sim_mode_ctx.req_preset_id = preset_id;
132 SET_TRUE(sim_mode_ctx.sim_mode_active);
133
134 valid_cfg = true;
135 }
136
137 return valid_cfg;
138}
