75void CommandsApi_GPIO_Init(void)
76{
77 GPIO_InitTypeDef GPIO_InitStruct = { 0 };
78
79 enable_gpio_port_clocks();
80
81 // prepare common configuration for all ports
82 GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
83 GPIO_InitStruct.Pull = GPIO_NOPULL;
84 GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
85
86 for (uint8_t idx = 0; idx < CDEF_ELEMENT_COUNT(pinCfg); idx++)
87 {
88 GPIO_InitStruct.Pin = pinCfg[idx].pinMask;
89 HAL_GPIO_Init(pinCfg[idx].outPortAddr, &GPIO_InitStruct);
90 }
91}

93GpioStat_t ReadHandler_GpOutputState(void)
94{
95 GpioStat_t retValue = { .gpioStatusBitField = 0x00 };
96
97 retValue.gpioStatusBitField = 0U;
98
99 for (uint8_t idx = 0; idx < CDEF_ELEMENT_COUNT(pinCfg); idx++)
100 {
101 retValue.gpioStatusBitField |=
102 HAL_GPIO_ReadPin(pinCfg[idx].outPortAddr, pinCfg[idx].pinMask) << pinCfg[idx].bitPos;
103 }
104
105 return retValue;
106}

108HwRes_t WriteHandler_GpOutputState(uint8_t pin, uint8_t value)
109{
110 HwRes_t status = HWRES_ERROR;
111
112 if (pin < CDEF_ELEMENT_COUNT(pinCfg))
113 {
114 if ((GPIO_TypeDef *) NULL != pinCfg[pin].outPortAddr)
115 {
116 HAL_GPIO_WritePin(pinCfg[pin].outPortAddr, pinCfg[pin].pinMask, value);
117 status = HWRES_SUCCESS;
118 }
119 }
120
121 return status;
122}

138StdResult_t WriteHandler_ResetInBootOrAppMode(const AppMode_t appMode)
139{
140 StdResult_t res = STDRESULT_SUCCESS;
141 boot_struct BootData;
142
143 Nvm_GetBlockById(NVM_BLOCK_BOOTDATA, &BootData);
144 switch (appMode)
145 {
146 case APPMODE_APPLICATION:
147 BootData.Mailbox = MAILBOX_VAL_APPL;
148 break;
149
150 case APPMODE_BOOTLOADER:
151 BootData.Mailbox = MAILBOX_VAL_BOOT;
152 break;
153
154 case APPMODE_AUTO_FW_UPDATE:
155 BootData.Mailbox = MAILBOX_VAL_AUTO_FLASH;
156 break;
157 default:
158 res = STDRESULT_INVALID_ARGS;
159 break;
160 }
161
162 if(res == STDRESULT_SUCCESS)
163 {
164 NvmResetCounters_ClearCounter(RSTCOUNTERID_ALL);
165 BootData.RebootRequest = pdTRUE;
166 // set valid checksum for the RTC backup registers after changing the mailbox
167 Nvm_SetBlockById(NVM_BLOCK_BOOTDATA, &BootData, NVMSETBLOCK_STORE_IMMEDIATELY);
168
169 // Reset the OBC
170 TaskMonitor_TriggerDelayedReset(1000U);
171 }
172
173 return res;
174}
