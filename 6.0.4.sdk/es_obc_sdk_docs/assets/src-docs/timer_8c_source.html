34static inline uint32_t calc_delta(timer_handle_t * const p_handle)
35{
36 CRIT_ASSERT(IS_TRUE(is_initialized));
37 CRIT_ASSERT((timer_handle_t * const) NULL != p_handle);
38
39 uint32_t delta = 0U;
40
41 if (IS_TRUE(p_handle->is_running))
42 {
43 uint32_t curr_tick_count = timer_cfg_sys_get_tick_count();
44 delta = (curr_tick_count >= p_handle->timestamp) ?
45 (curr_tick_count - p_handle->timestamp) :
46 (curr_tick_count + ULONG_MAX - p_handle->timestamp);
47 }
48
49 return delta;
50}
