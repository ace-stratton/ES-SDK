62upy_scr_mgr_return_t upy_scr_mgr_add(const char *const p_name, uint8_t len, uint32_t time_out,
63 pfMicropythonExecScript_Cbk_t p_on_send_result_cbk,
64 uint8_t *const p_script_id)
65{
66 upy_scr_mgr_return_t result = UPY_SCR_MGR_ERROR;
67 sys_time_t time_of_request;
68
69 if ((NULL != p_name) && (NULL != p_script_id) &&
70 (NULL != p_on_send_result_cbk))
71 {
72 g_current_script_id += 1;
73
74 // make sure invalid script id is not used.
75 if (INVALID_SCRIPT_ID == g_current_script_id)
76 {
77 g_current_script_id = 0;
78 }
79
80 (void)sys_time_get(&time_of_request);
81
82 uint8_t script_idx = g_current_script_id % UPY_SCRIPT_QUEUE_LEN;
83
84 osStatus_t eMutexRes = osMutexAcquire(upy_scr_lock_id, osWaitForever);
85
86 if (osOK == eMutexRes)
87 {
88 sScriptData_t *const psScriptData =
89 &ScriptHndls[script_idx].sScriptData;
90 if (0 < len)
91 {
92 memcpy(psScriptData->pName, p_name, len);
93 psScriptData->pName[len] = '\0';
94 }
95
96 ScriptHndls[script_idx].u8ScriptId = g_current_script_id;
97 psScriptData->u32TimeOut = time_out;
98 psScriptData->pfOnSendResultCbk = p_on_send_result_cbk;
99 psScriptData->eScriptStatus = UPY_PENDING;
100 psScriptData->u32DurationInMs = 0;
101
102 memcpy(&psScriptData->sTimeRequested, &time_of_request,
103 sizeof(sys_time_t));
104 memset(&psScriptData->sTimeStarted, 0, sizeof(sys_time_t));
105
106 osMutexRelease(upy_scr_lock_id);
107 *p_script_id = g_current_script_id;
108 result = UPY_SCR_MGR_OK;
109 }
110 else
111 {
112 g_current_script_id = INVALID_SCRIPT_ID;
113 result = UPY_SCR_MGR_MUTEX_ERROR;
114 }
115 }
116 else
117 {
118 result = UPY_SCR_MGR_WRONG_PARAMS;
119 }
120
121 return result;
122}

124upy_scr_mgr_return_t upy_scr_mgr_get_status(uint8_t script_id, sScriptData_t *const p_script_data)
125{
126 upy_scr_mgr_return_t res = UPY_SCR_MGR_ERROR;
127 uint8_t u8ScriptIdx = script_id % UPY_SCRIPT_QUEUE_LEN;
128
129 if (NULL != p_script_data)
130 {
131 osStatus_t eMutexRes = osMutexAcquire(upy_scr_lock_id, osWaitForever);
132
133 if (osOK == eMutexRes)
134 {
135 // make sure that the element in the array is still not overwritten
136 // by new element
137 if (ScriptHndls[u8ScriptIdx].u8ScriptId == script_id)
138 {
139 memcpy(p_script_data, &ScriptHndls[u8ScriptIdx].sScriptData,
140 sizeof(sScriptData_t));
141 res = UPY_SCR_MGR_OK;
142 }
143 else
144 {
145 res = UPY_SCR_MGR_SCRIPT_NOT_FOUND;
146 }
147
148 osMutexRelease(upy_scr_lock_id);
149 }
150 else
151 {
152 res = UPY_SCR_MGR_MUTEX_ERROR;
153 }
154 }
155 else
156 {
157 res = UPY_SCR_MGR_WRONG_PARAMS;
158 }
159
160 return res;
161}
