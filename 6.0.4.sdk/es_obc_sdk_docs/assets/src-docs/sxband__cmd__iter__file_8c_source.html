55static bool read_data_and_verify(sxband_cmd_file_iter_ctx_t * const p_file_ctx, void * const p_data, const uint32_t exp_size)
56{
57 uint32_t bytes_read = 0U;
58
59 p_file_ctx->fs_last_res = SdMngr_f_read(&p_file_ctx->f_hnd, p_data, exp_size, (UINT *) &bytes_read);
60
61 return (FR_OK == p_file_ctx->fs_last_res) && (bytes_read == exp_size);
62}

64sxband_cmd_iter_init_status_t sxband_cmd_iter_file_init_ctx(const char * const cmd_file_name, const uint8_t cmd_file_name_size, sxband_cmd_file_iter_ctx_t * const p_file_ctx)
65{
66 sxband_cmd_iter_init_status_t res = SXB_INIT_STATUS_MAX;
67
68 CRIT_ASSERT(NULL != p_file_ctx);
69
70 if (NULL != p_file_ctx)
71 {
72 // if the function is called twice, the file handle shall be closed before using it again
73 (void) SdMngr_f_close(&p_file_ctx->f_hnd);
74
75 (void) memset(p_file_ctx->cmd_file_name, 0, sizeof(p_file_ctx->cmd_file_name));
76 (void) memcpy(p_file_ctx->cmd_file_name, cmd_file_name, min(cmd_file_name_size, sizeof(p_file_ctx->cmd_file_name)));
77 p_file_ctx->cmd_file_name[sizeof(p_file_ctx->cmd_file_name) - 1] = '\0';
78
79 p_file_ctx->fs_last_res = SdMngr_f_open(&p_file_ctx->f_hnd, p_file_ctx->cmd_file_name, FA_READ | FA_OPEN_EXISTING);
80
81 if (FR_OK == p_file_ctx->fs_last_res)
82 {
83 if (read_data_and_verify(p_file_ctx, (void *) &p_file_ctx->cmd_list_hdr, sizeof(cmd_list_hdr_t)))
84 {
85 if (sxband_cmd_iter_cmn_verify_hdr(&p_file_ctx->cmd_list_hdr))
86 {
87 res = SXB_INIT_STATUS_OK;
88 }
89 else
90 {
91 res = SXB_INIT_STATUS_INVALID_FILE_FORMAT;
92 }
93 }
94 else
95 {
96 res = SXB_INIT_STATUS_INVALID_FILE_FORMAT;
97 }
98 }
99 else
100 {
101 res = SXB_INIT_STATUS_FS_ERROR;
102 }
103 }
104
105 return res;
106}

108pl_ctrl_iter_status_t file_iter_reset(void * const p_ctx)
109{
110 pl_ctrl_iter_status_t res = PL_CTRL_ITER_STATUS_NOK;
111
112 CRIT_ASSERT(NULL != p_ctx);
113
114 if (NULL != p_ctx)
115 {
116 sxband_cmd_file_iter_ctx_t *p_file_ctx = (sxband_cmd_file_iter_ctx_t *) p_ctx;
117
118 // skip file header and position on the first command entry
119 p_file_ctx->fs_last_res = SdMngr_f_lseek(&p_file_ctx->f_hnd, sizeof(cmd_list_hdr_t));
120
121 p_file_ctx->cmd_idx = 0U;
122
123 res = PL_CTRL_ITER_STATUS_OK;
124 }
125
126 return res;
127}

129bool file_iter_has_next(void * const p_ctx)
130{
131 bool has_next = false;
132
133 CRIT_ASSERT(NULL != p_ctx);
134
135 if (NULL != p_ctx)
136 {
137 const sxband_cmd_file_iter_ctx_t *p_file_ctx = (sxband_cmd_file_iter_ctx_t *) p_ctx;
138
139 has_next = (FR_OK == p_file_ctx->fs_last_res) ?
140 (SdMngr_f_tell(&p_file_ctx->f_hnd) < SdMngr_f_size(&p_file_ctx->f_hnd)) : false;
141 }
142
143 return has_next;
144}

146pl_ctrl_iter_status_t file_iter_next(void * const p_ctx, sxband_sched_cmd_t * const p_cmd)
147{
148 pl_ctrl_iter_status_t res = PL_CTRL_ITER_STATUS_NOK;
149
150 CRIT_ASSERT(NULL != p_ctx);
151 CRIT_ASSERT(NULL != p_cmd);
152
153 p_cmd->p_cmd_hdr = NULL;
154 p_cmd->p_data = NULL;
155
156 if ((NULL != p_ctx) && (NULL != p_cmd))
157 {
158 sxband_cmd_file_iter_ctx_t *p_file_ctx = (sxband_cmd_file_iter_ctx_t *) p_ctx;
159
160 // perform all reads regardless of status to ensure that the file position offset will be updated and iteration can
161 // continue to the end of the file even if there are invalid commands inside
162 bool read_ok = read_data_and_verify(p_file_ctx, (void *) &p_file_ctx->iter_cmd_entry.hdr, sizeof(cmd_list_command_hdr_t));
163 read_ok = read_ok && read_data_and_verify(p_file_ctx, (void *) &p_file_ctx->iter_cmd_entry.data, p_file_ctx->iter_cmd_entry.hdr.size);
164 read_ok = read_ok && read_data_and_verify(p_file_ctx, (void *) &p_file_ctx->iter_cmd_entry.footer, sizeof(cmd_list_command_footer_t));
165
166 if (read_ok && (sxband_cmd_iter_cmn_verify_cmd(&p_file_ctx->iter_cmd_entry.hdr, p_file_ctx->iter_cmd_entry.data, p_file_ctx->iter_cmd_entry.footer.cmd_crc)))
167 {
168 p_cmd->cmd_idx = p_file_ctx->cmd_idx++;
169 p_cmd->p_cmd_hdr = &p_file_ctx->iter_cmd_entry.hdr;
170 p_cmd->p_data = p_file_ctx->iter_cmd_entry.data;
171
172 res = PL_CTRL_ITER_STATUS_OK;
173 }
174 else
175 {
176 if (FR_OK == p_file_ctx->fs_last_res)
177 {
178 res = PL_CTRL_ITER_REACHED_END;
179 }
180 }
181 }
182
183 return res;
184}
