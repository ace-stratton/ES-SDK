30__weak uint8_t BSP_SD_Init(void)
31{
32 uint8_t sd_state = MSD_OK;
33 /* Check if the SD card is plugged in the slot */
34 if (BSP_SD_IsDetected() != SD_PRESENT)
35 {
36 return MSD_ERROR_SD_NOT_PRESENT;
37 }
38 /* HAL SD initialization */
39 sd_state = HAL_SD_Init(&hsd1);
40 /* Configure SD Bus width (4 bits mode selected) */
41 if (sd_state == MSD_OK)
42 {
43 /* Enable wide operation */
44 if (HAL_SD_ConfigWideBusOperation(&hsd1, SDMMC_BUS_WIDE_4B) != HAL_OK)
45 {
46 sd_state = MSD_ERROR;
47 }
48 }
49
50 return sd_state;
51}
