222{
223 ESSA_Stack_FP_MsgHdr_t *pHdr = NULL;

225 uint8_t i;
226
227 // Basic header validity check
228 if ((fp_DataInfo == NULL) ||
229 (fp_DataInfo->u16DataSize < sizeof(ESSA_Stack_FP_MsgHdr_t)) ||
230 (fp_DataInfo->pu8Data == NULL))
231 {
232 return false;
233 }
234
235 pHdr = (ESSA_Stack_FP_MsgHdr_t *) fp_DataInfo->pu8Data;
236
237 if (IS_RESPONSE(*pHdr))
238 {
239 pHdr->funcId = GET_FUNC_ID(*pHdr);
240
241 for (i = 0; i < COUNT_OF(fs_aFuncArray); i++)
242 {
243 if (fs_aFuncArray[i].funcId == pHdr->funcId)
244 {
245 if (fs_aFuncArray[i].pfFunc != NULL)
246 {
247 fs_aFuncArray[i].pfFunc(fp_DataInfo);
248
249 bIsFuncSupported = true;
250 }
251
252 break;
253 }
254 }
255 }
256
257 return bIsFuncSupported;
