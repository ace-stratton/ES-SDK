47HAL_StatusTypeDef HAL_InitTick(uint32_t TickPriority)
48{
49 RCC_ClkInitTypeDef clkconfig;
50 uint32_t uwTimclock, uwAPB1Prescaler;
51
52 uint32_t uwPrescalerValue;
53 uint32_t pFLatency;
54/*Configure the TIM6 IRQ priority */
55 if (TickPriority < (1UL << __NVIC_PRIO_BITS))
56 {
57 HAL_NVIC_SetPriority(TIM6_DAC_IRQn, TickPriority ,0U);
58
59 /* Enable the TIM6 global Interrupt */
60 HAL_NVIC_EnableIRQ(TIM6_DAC_IRQn);
61 uwTickPrio = TickPriority;
62 }
63 else
64 {
65 return HAL_ERROR;
66 }
67
68 /* Enable TIM6 clock */
69 __HAL_RCC_TIM6_CLK_ENABLE();
70
71 /* Get clock configuration */
72 HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
73
74 /* Get APB1 prescaler */
75 uwAPB1Prescaler = clkconfig.APB1CLKDivider;
76 /* Compute TIM6 clock */
77 if (uwAPB1Prescaler == RCC_HCLK_DIV1)
78 {
79 uwTimclock = HAL_RCC_GetPCLK1Freq();
80 }
81 else
82 {
83 uwTimclock = 2UL * HAL_RCC_GetPCLK1Freq();
84 }
85
86 /* Compute the prescaler value to have TIM6 counter clock equal to 1MHz */
87 uwPrescalerValue = (uint32_t) ((uwTimclock / 1000000U) - 1U);
88
89 /* Initialize TIM6 */
90 htim6.Instance = TIM6;
91
92 /* Initialize TIMx peripheral as follow:
93 + Period = [(TIM6CLK/1000) - 1]. to have a (1/1000) s time base.
94 + Prescaler = (uwTimclock/1000000 - 1) to have a 1MHz counter clock.
95 + ClockDivision = 0
96 + Counter direction = Up
97 */
98 htim6.Init.Period = (1000000U / 1000U) - 1U;
99 htim6.Init.Prescaler = uwPrescalerValue;
100 htim6.Init.ClockDivision = 0;
101 htim6.Init.CounterMode = TIM_COUNTERMODE_UP;
102
103 if(HAL_TIM_Base_Init(&htim6) == HAL_OK)
104 {
105 /* Start the TIM time Base generation in interrupt mode */
106 return HAL_TIM_Base_Start_IT(&htim6);
107 }
108
109 /* Return function status */
110 return HAL_ERROR;
111}
