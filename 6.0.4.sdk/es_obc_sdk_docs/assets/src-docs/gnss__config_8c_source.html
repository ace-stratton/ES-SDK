262static void send_cmd_to_gen2_cubeadcs(const gnss_time_t * const p_time, const gnss_xyz_t* const p_xyz)
263{
264 if ((NULL != p_time) &&
265 (NULL != p_xyz) &&
266 (GNSS_XYZ_SOL_STAT_COMPUTED == p_xyz->p_sol_status) && // Is the BESTXYZ position vector data reliable?
267 (GNSS_XYZ_SOL_STAT_COMPUTED == p_xyz->v_sol_status) && // Is the BESTXYZ velocity vector data reliable?
268 (GNSS_TIM_STAT_FINE <= p_xyz->header.time_status) && // Is the quality of GPS time measurements sufficient?
269 (GNSS_UTC_STATUS_VALID == p_time->utc_status)) // Can we trust the UTC time data?
270 {
271
272 // Accumulate Unix seconds by beginning with the whole weeks since the GPS epoch.
273 // We use 64-bit integer to avoid multiple overflow checks
274 uint64_t unix_sec_long = ((uint64_t)p_time->header.week * NUM_SEC_IN_WEEK) + GPS_TO_UNIX_OFFSET;
275
276 // Add the seconds in current week.
277 // Compensate deviation (dOffset) of GPS receiver clock from GPS system time to increase accuracy.
278 // Also adjust with UTC offset to get the UTC time encoded in the Unix timestamp.
279 // Double type is used to avoid overflow checks.
280 const double unix_sec_dbl = (double)unix_sec_long + p_time->header.seconds + p_time->utc_offset - p_time->offset;
281
282 if (unix_sec_dbl < ((double)UINT_MAX + 1)) // Because check for equality is a bit tricky with doubles
283 {
284 // Calculate the milliseconds fractional part.
285 const uint32_t msec = (uint32_t)((unix_sec_dbl - (uint32_t)unix_sec_dbl) * GPS_SEC_ACCURACY_MULT);
286
287 // Prepare command to ADCS
288 TctlmCubeComputerControlProgram8_GnssSensorCmd adcs_cmd;
289 // No overflow expected for the position coordinates, because they are normally between -7000000m and 7000000m
290 adcs_cmd.gnssSatPosX = (S32)(p_xyz->px * NUM_CM_IN_M);
291 adcs_cmd.gnssSatPosY = (S32)(p_xyz->py * NUM_CM_IN_M);
292 adcs_cmd.gnssSatPosZ = (S32)(p_xyz->pz * NUM_CM_IN_M);
293 // No overflow for the velocity vectors, because they cannot exceed 8000m/s
294 adcs_cmd.gnssSatVelX = (S32)(p_xyz->vx * NUM_CM_IN_M);
295 adcs_cmd.gnssSatVelY = (S32)(p_xyz->vy * NUM_CM_IN_M);
296 adcs_cmd.gnssSatVelZ = (S32)(p_xyz->vz * NUM_CM_IN_M);
297 adcs_cmd.gnssTimeSeconds = (U32) unix_sec_dbl;
298 adcs_cmd.gnssTimeNs = msec * NUM_NS_IN_MS;
299
300 adcs_cmd.syncTime = ADCS_RTC_SYNCH;
301
302 // Send command to ADCS
303 if (true == cubeObc_critical_enter(ADCS_CRITICAL_SECTION_TIMEOUT))
304 {
305 TypeDef_TctlmEndpoint endpoint;
306 cubeadcs_gen2_setup_endpoint(&endpoint, TYPEDEF__COMMS_ENDPOINT_UART, ADCS_COMM_TIMEOUT);
307 const ErrorCode err = tctlmCubeComputerControlProgram8_setGnssSensorCmd(&endpoint, &adcs_cmd);
308 cubeObc_critical_exit();
309 if (err != CUBEOBC_ERROR_OK)
310 {
311 ES_TRACE_WARN("Could not send GNSS data to Gen2 CubeADCS!");
312 }
313 }
314
315 }
316 else
317 {
318 ES_TRACE_ERROR("Overflow error!");
319 }
320 }
321}

331static void gnss_config_adcs_sync_callback(void)
332{
333 if (false != sys_conf_is_module_active(SYS_MODULE_ID_CUBE_ADCS_GEN2))
334 {
335 DATA_CACHE_GNSS_TIME_t time_data;
336 (void)dc_get_gnss_time_data(&time_data);
337 static_assert(sizeof(DATA_CACHE_GNSS_TIME_t) == sizeof(gnss_time_t));
338 const gnss_time_t * const p_time = (const gnss_time_t* const)&time_data;
339 if (true == gnss_is_tm_valid(p_time))
340 {
341 DATA_CACHE_GNSS_BESTXYZ_t pos_data;
342 (void)dc_get_gnss_bestxyz_data(&pos_data);
343 static_assert(sizeof(DATA_CACHE_GNSS_BESTXYZ_t) == sizeof(gnss_xyz_t));
344 const gnss_xyz_t * const p_xyz = (const gnss_xyz_t * const)&pos_data;
345
346 if (fabs(p_xyz->header.seconds - p_time->header.seconds) < REF_TIME_EPSILON ) // Are TIME and BESTXYZ logs for the same second?)
347 {
348 send_cmd_to_gen2_cubeadcs(p_time, p_xyz);
349 }
350 }
351 }
352}
