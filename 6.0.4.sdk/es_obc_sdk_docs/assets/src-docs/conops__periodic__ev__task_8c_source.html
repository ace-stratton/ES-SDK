165STATIC void fs_conops_p_ev_task(void *arg)
166{
167 for(;;)
168 {
169 conops_refresh_runtime_nvm_copies();
170
171 conops_trigger_hsm(eHSM_StdEvent_Periodic, NULL);
172
173 fs_conops_p_ev_track_emergency_shutdown();
174
175 fs_conops_p_ev_sun_track_sequence();
176
177 TaskMonitor_IamAlive(TASK_CONOPS_PERIODIC_EV);
178 vApplicationLowStackCheck(TASK_CONOPS_PERIODIC_EV);
179
180 osDelay(CONOPS_PER_EV_TASK_PERIOD);
181 }
182}

191STATIC void fs_conops_p_ev_sun_track_sequence(void)
192{
193 static eSTT_states_t eTState = STT_None;
194
195 uint8_t eOpModeNum;
196 aocs_cntrl_sys_state_types_t aocs_state;
197 aocs_cntrl_sys_state_req_state_t aocs_state_req_state;
198
199 conops_get_hsm_active_state(&eOpModeNum);
200
201 if ( (false != conops_get_aocs_sys_state(&aocs_state, &aocs_state_req_state)) &&
202 (AOCS_CNTRL_SYS_STATE_UNDEFINED != aocs_state) &&
203 (AOCS_CNTRL_SYS_STATE_STATE_REQ_COMPLETED == aocs_state_req_state) )
204 {
205 if ( STATE_SAFE == eOpModeNum )
206 {
207 /* This check assumes that we have completed detumbling in the context of Detumbling Mode */
208 if ( AOCS_CNTRL_SYS_STATE_Y_THOMSON == aocs_state ) {
209 eTState = STT_Y_Thomson;
210
211 CONOPS_TRACE_INFO("%s: STT Y-Thomson Started: Last STT: %s\r", CONOPS_SM_STT_DBG, stt_state_strings[eTState]);
212 }
213
214 if ( AOCS_CNTRL_SYS_STATE_NO_CONTROL == aocs_state ) {
215 eTState = STT_None;
216
217 CONOPS_TRACE_INFO("%s: NEW STT Started: Last STT: %s\r", CONOPS_SM_STT_DBG, stt_state_strings[eTState]);
218 }
219 }
220
221 CONOPS_TRACE_INFO("%s: OpMode: %d ---- STT_State: %s\r", CONOPS_SM_STT_DBG, eOpModeNum, stt_state_strings[eTState]);
222
223 aocs_cms_status_t eCmsStatus;
224
225 if ( (STATE_SAFE == eOpModeNum) && aocs_cntrl_get_cms_status(&eCmsStatus) )
226 {
227 switch ( eTState )
228 {
229 case STT_None:
230 {
231 if ( CMS_COMPLETE == eCmsStatus ) {
232 conops_set_aocs_sys_state(AOCS_CNTRL_SYS_STATE_NORMAL_DETUMBLING, NULL);
233
234 eTState = STT_Y_Thomson;
235 }
236
237 break;
238 }
239
240 case STT_Y_Thomson:
241 {
242 if ( CMS_COMPLETE == eCmsStatus ) {
243 conops_set_aocs_sys_state(AOCS_CNTRL_SYS_STATE_Y_MOMENTUM, NULL);
244
245 eTState = STT_Y_Momentum_Initial;
246 }
247
248 break;
249 }
250
251 case STT_Y_Momentum_Initial:
252 {
253 if ( true == conops_get_enable_sun_tracking_flag() ) {
254 if ( true == fs_conops_p_ev_is_y_momentum_rdy() ) {
255 if ( CMS_COMPLETE == eCmsStatus ) {
256 const aocs_cntrl_reference_params_t rpy_params = {0.0f, 0.0f, 0.0f};
257 conops_set_aocs_sys_state(AOCS_CNTRL_SYS_STATE_3AXIS, &rpy_params);
258
259 eTState = STT_3Axis;
260 }
261 }
262 }
263
264 break;
265 }
266
267 case STT_3Axis:
268 {
269 if ( true == conops_get_enable_sun_tracking_flag() ) {
270 if ( true == fs_conops_p_ev_is_3axis_cntrl_rdy() ) {
271 if ( CMS_COMPLETE == eCmsStatus ) {
272 conops_set_aocs_sys_state(AOCS_CNTRL_SYS_STATE_SUN_TRACKING, NULL);
273
274 eTState = STT_Sun_Tracking;
275 }
276 }
277 }
278
279 break;
280 }
281
282 case STT_Sun_Tracking:
283 {
284 eTState = STT_Completed;
285
286 break;
287 }
288
289 default:
290 break;
291 }
292
293 CONOPS_TRACE_INFO("%s: OpMode: %d ---- NEW STT_State: %s\r", CONOPS_SM_STT_DBG, eOpModeNum, stt_state_strings[eTState]);
294 }
295 }
296}

315STATIC bool fs_conops_p_ev_is_y_momentum_rdy(void)
316{
317 bool bRes;
318
319 DATA_CACHE_AOCS_CNTRL_TLM_t sAdcsData;
320 uint8_t u8Chk = 0;
321
322 bRes = false;
323
324 if ( DC_DATA_STATUS_OK == dc_get_aocs_cntrl_tlm_data(&sAdcsData) )
325 {
326 if ( (CONOPS_RPY_MDEG_THRESH > sAdcsData.ai32EstAttAngles[Roll]) && (-CONOPS_RPY_MDEG_THRESH < sAdcsData.ai32EstAttAngles[Roll]) )
327 {
328 ++u8Chk;
329 }
330
331 if ( (CONOPS_RPY_MDEG_THRESH > sAdcsData.ai32EstAttAngles[Pitch]) && (-CONOPS_RPY_MDEG_THRESH < sAdcsData.ai32EstAttAngles[Pitch]) )
332 {
333 ++u8Chk;
334 }
335
336 if ( (CONOPS_RPY_MDEG_THRESH > sAdcsData.ai32EstAttAngles[Yaw]) && (-CONOPS_RPY_MDEG_THRESH < sAdcsData.ai32EstAttAngles[Yaw]) )
337 {
338 ++u8Chk;
339 }
340
341 if ( CONOPS_RPY_CHECK_OK == u8Chk)
342 {
343 bRes = true;
344 }
345 }
346
347 return bRes;
348}

357STATIC bool fs_conops_p_ev_is_3axis_cntrl_rdy(void)
358{
359 bool bRes;
360
361 bRes = true;
362
363 DATA_CACHE_AOCS_CNTRL_TLM_t sAdcsData;
364
365 if ( DC_DATA_STATUS_OK == dc_get_aocs_cntrl_tlm_data(&sAdcsData) )
366 {
367 for (uint32_t idx = 0; idx < CDEF_ELEMENT_COUNT(sAdcsData.ai16MeasWheelSpeed); idx++)
368 {
369 if ( (CONOPS_Y_WHEEL_RPM_THRESH <= sAdcsData.ai16MeasWheelSpeed[idx]) &&
370 (-CONOPS_Y_WHEEL_RPM_THRESH >= sAdcsData.ai16MeasWheelSpeed[idx]) )
371 {
372 bRes = false;
373 break;
374 }
375 }
376 }
377
378 return bRes;
379}
