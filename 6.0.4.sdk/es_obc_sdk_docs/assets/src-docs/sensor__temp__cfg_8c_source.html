85STATIC sds_op_status_t sensor_panel_temp_get_data_in_deg_c(sds_panel_temp_data_t * const p_data_dst)
86{
87 CRIT_ASSERT(NULL != p_data_dst);
88
89 int16_t panel_temp_data_raw[MAX_PAN] = {0, 0, 0, 0, 0, 0};
90
91 for ( uint8_t i = 0; i < MAX_PAN; i++ )
92 {
93 if ( SEN_SUCCESS != TMP122_GetTemperatureFromPanel(i, &panel_temp_data_raw[i]) )
94 {
95 panel_temp_err_code = SDS_PANEL_TEMP_ERR_CODE_ERR;
96 }
97 else
98 {
99 panel_temp_data_raw[i] = TMP122_getTemperatureInDegC(panel_temp_data_raw[i]);
100 }
101 }
102
103 p_data_dst->pan_temp_1_deg_c = panel_temp_data_raw[0];
104 p_data_dst->pan_temp_2_deg_c = panel_temp_data_raw[1];
105 p_data_dst->pan_temp_3_deg_c = panel_temp_data_raw[2];
106 p_data_dst->pan_temp_4_deg_c = panel_temp_data_raw[3];
107 p_data_dst->pan_temp_5_deg_c = panel_temp_data_raw[4];
108 p_data_dst->pan_temp_6_deg_c = panel_temp_data_raw[5];
109
110 SDS_TRACE_DBG("%s PANEL TEMP DATA IN DEG C---1:%d, 2:%d, 3:%d, 4:%d, 5:%d, 6:%d\r", SDS_DBG_HEAD,
111 p_data_dst->pan_temp_1_deg_c, p_data_dst->pan_temp_2_deg_c, p_data_dst->pan_temp_3_deg_c,
112 p_data_dst->pan_temp_4_deg_c, p_data_dst->pan_temp_5_deg_c, p_data_dst->pan_temp_6_deg_c);
113
114 return SDS_OP_STATUS_OK;
115}
