161void cubeadcs_gen2_fdir_init_os(void)
162{
163 cubeadcs_gen2_fdir_err_lock = osMutexNew(&cubeadcs_gen2_fdir_err_lock_attr);
164
165 if ( NULL == cubeadcs_gen2_fdir_err_lock )
166 {
167 EXEH_HANDLE_EX(eEXEHSeverity_Error, eEXEHModuleID_CUBEADCS_GEN2,
168 EXEH_CUBEADCS_GEN2_OS_OBJ_ERR);
169 }

172bool cubeadcs_gen2_fdir_exec(void)
173{
174 bool op_res = false;
175
176 cubeadcs_gen2_events_cnts_t events;
177
178 if ( (true == cubeadcs_gen2_events_get_events_stats(&events)) &&
179 (true == fs_cubeadcs_gen2_fdir_lock_err()))
180 {
181 // Clear the last events and only then work on the local copy
182 if ( true == cubeadcs_gen2_events_clear_event_stats(CUBEADCS_GEN2_EVENTS_CLEAR_LAST_READ_EVENTS) )
183 {
184 for (uint8_t ev = 0; ev < CUBEADCS_GEN2_EVENTS_MAX; ev++)
185 {
186 if ( (0 != events.last_read_events[ev]) &&
187 (fdir_ctx.fdir_handlers[ev].failure_event_handler_exec < fdir_ctx.fdir_handlers[ev].failure_event_handler_max_exec) )
188 {
189 if (true == fdir_ctx.fdir_handlers[ev].failure_event_handler())
190 {
191 fdir_ctx.fdir_handlers[ev].failure_event_handler_exec_op_res[fdir_ctx.fdir_handlers[ev].failure_event_handler_exec] = true;
192 }
193 else
194 {
195 // If error was not handled or we failed to handle it, update the unrecoverable errors
196 fdir_ctx.unrecoverable_errors |= fdir_ctx.fdir_handlers[ev].failure_event_fdir_error;
197 }
198
199 fdir_ctx.fdir_handlers[ev].failure_event_handler_exec++;
200 }
201
202 }
203
204 op_res = true;
205 }
206
207 (void)fs_cubeadcs_gen2_fdir_unlock_err();
208 }
209
210 return op_res;
211}

265bool cubeadcs_gen2_fdir_clear_fdir_ctx_handlers(void)
266{
267 bool op_res = false;
268
269 if ( true == fs_cubeadcs_gen2_fdir_lock_err() )
270 {
271 for (uint8_t ev = 0; ev < CUBEADCS_GEN2_EVENTS_MAX; ev++)
272 {
273 fdir_ctx.fdir_handlers[ev].failure_event_handler_exec = 0;
274
275 for (uint8_t ev_h = 0; ev < fdir_ctx.fdir_handlers[ev].failure_event_handler_max_exec; ev_h++)
276 {
277 fdir_ctx.fdir_handlers[ev].failure_event_handler_exec_op_res[ev_h] = false;
278 }
279 }
280
281 (void)fs_cubeadcs_gen2_fdir_unlock_err();
282
283 op_res = true;
284 }
285
286 return op_res;

320STATIC bool fs_cubeadcs_gen2_fdir_lock_err(void)
321{
322 bool op_res = true;
323
324 if ( osOK != osMutexAcquire(cubeadcs_gen2_fdir_err_lock, osWaitForever) )
325 {
326 op_res = false;
327
328 EXEH_HANDLE_EX(eEXEHSeverity_Warning, eEXEHModuleID_CUBEADCS_GEN2,
329 EXEH_CUBEADCS_GEN2_OS_OBJ_ERR)
330 }
331
332 return op_res;
333}
