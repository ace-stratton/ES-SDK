150static bool SD_IsReadyWithTimeout(uint32_t timeout)
151{
152 uint32_t timer;
153 bool bRes = false;
154 /* block until SDIO peripheral is ready again or a timeout occur */
155#if (osCMSIS <= 0x20000U)
156 timer = osKernelSysTick();
157 while( osKernelSysTick() - timer < timeout)
158#else
159 timer = osKernelGetTickCount();
160 while( osKernelGetTickCount() - timer < timeout)
161#endif
162 {
163 if (BSP_SD_GetCardState() == SD_TRANSFER_OK)
164 {
165 bRes = true;
166 break;
167 }
168 }
169
170 return bRes;
171}

183bool SD_initialize(void)
184{
185 bool bIsSdReady = false;
186
187 bIsSdInitialized = false;
188
189 /*
190 * check that the kernel has been started before continuing
191 * as the osMessage API will fail otherwise
192 */
193#if (osCMSIS <= 0x20000U)
194 if(osKernelRunning())
195#else
196 if(osKernelGetState() == osKernelRunning)
197#endif
198 {
199#if !defined(DISABLE_SD_INIT)
200
201 if(BSP_SD_Init() == MSD_OK)
202 {
203 bIsSdReady = SD_IsReady();
204 }
205
206#else
207 bIsSdReady = SD_IsReady();
208#endif
209
210 /*
211 * if the SD is correctly initialized, create the operation queue
212 * if not already created
213 */
214
215 if (bIsSdReady)
216 {
217 if (SDQueueID == NULL)
218 {
219 #if (osCMSIS <= 0x20000U)
220 osMessageQDef(SD_Queue, QUEUE_SIZE, uint16_t);
221 SDQueueID = osMessageCreate (osMessageQ(SD_Queue), NULL);
222#else
223 SDQueueID = osMessageQueueNew(QUEUE_SIZE, 2, NULL);
224#endif
225 }
226
227 if (SDQueueID != NULL)
228 {
229 bIsSdInitialized = true;
230 }
231 }
232 }
233
234 return bIsSdInitialized;
235}

240static void SD_reinit(void)
241{
242 HAL_GPIO_WritePin(SD_EN_GPIO_Port, SD_EN_Pin, GPIO_PIN_SET);
243
244 __HAL_RCC_SDMMC1_CLK_DISABLE();
245
246 //RCC SDIO1 RESET
247 __HAL_RCC_SDMMC1_FORCE_RESET();
248 while( ( RCC->AHB3RSTR & RCC_AHB3RSTR_SDMMC1RST ) != RCC_AHB3RSTR_SDMMC1RST ){
249 //wait for reset to happen
250 }
251 osDelay(1);
252 __HAL_RCC_SDMMC1_RELEASE_RESET();
253 while( ( RCC->AHB3RSTR & RCC_AHB3RSTR_SDMMC1RST ) == RCC_AHB3RSTR_SDMMC1RST ){
254 //wait for reset to happen
255 }
256
257 osDelay(150);
258
259 HAL_GPIO_WritePin(SD_EN_GPIO_Port, SD_EN_Pin, GPIO_PIN_RESET);
260
261 __HAL_RCC_SDMMC1_CLK_ENABLE();
262 osDelay(1);
263
264 (void)HAL_SD_Init(&hsd1);
265
266 osDelay(10);
267}

270SDRESULT SD_read_drv(uint8_t *buff, uint32_t sector, uint32_t count)
271{
272 uint8_t ret = MSD_ERROR;
273 SDRESULT res = eRES_ERROR;
274 uint32_t timer;
275#if (osCMSIS < 0x20000U)
276 osEvent event;
277#else
278 uint16_t event;
279 osStatus_t status;
280#endif
281#if (ENABLE_SD_DMA_CACHE_MAINTENANCE == 1)
282 uint32_t alignedAddr;
283#endif
284 /*
285 * ensure the SDCard is ready for a new operation
286 */
287
288 if (!SD_IsReadyWithTimeout(SD_TIMEOUT))
289 {
290 return eRES_NOT_READY;
291 }
292
293#if defined(ENABLE_SCRATCH_BUFFER)
294 // for some reason, reading directly to memory in the D2 RAM region fails with SD DMA,
295 // hence all reads to such regions are routed through the scratch buffer for now
296 if ((uint32_t)0 == ((uint32_t)buff & (uint32_t)0x3)
297 && ((uint32_t) buff < (uint32_t) &RAM_D2_START_ADDRESS))
298 {
299#endif
300 /* Fast path cause destination buffer is correctly aligned */
301 ret = BSP_SD_ReadBlocks_DMA((uint32_t*)buff, sector, count);
302
303 if (ret == MSD_OK) {
304#if (osCMSIS < 0x20000U)
305 /* wait for a message from the queue or a timeout */
306 event = osMessageGet(SDQueueID, SD_TIMEOUT);
307
308 if (event.status == osEventMessage)
309 {
310 if (event.value.v == READ_CPLT_MSG)
311 {
312 timer = osKernelSysTick();
313 /* block until SDIO IP is ready or a timeout occur */
314 while(osKernelSysTick() - timer <SD_TIMEOUT)
315#else
316 status = osMessageQueueGet(SDQueueID, (void *)&event, NULL, SD_TIMEOUT);
317 if ((status == osOK) && (event == READ_CPLT_MSG))
318 {
319 timer = osKernelGetTickCount();
320 /* block until SDIO IP is ready or a timeout occur */
321 while(osKernelGetTickCount() - timer <SD_TIMEOUT)
322#endif
323 {
324 if (BSP_SD_GetCardState() == SD_TRANSFER_OK)
325 {
326 res = eRES_OK;
327#if (ENABLE_SD_DMA_CACHE_MAINTENANCE == 1)
328 /*
329 the SCB_InvalidateDCache_by_Addr() requires a 32-uint8_t aligned address,
330 adjust the address and the D-Cache size to invalidate accordingly.
331 */
332 alignedAddr = (uint32_t)buff & ~0x1F;
333 SCB_InvalidateDCache_by_Addr((uint32_t*)alignedAddr, count*BLOCKSIZE + ((uint32_t)buff - alignedAddr));
334#endif
335 break;
336 }
337 }
338#if (osCMSIS < 0x20000U)
339 }
340 }
341#else
342 }
343#endif
344 }
345
346#if defined(ENABLE_SCRATCH_BUFFER)
347 }
348 else
349 {
350 /* Slow path, fetch each sector a part and memcpy to destination buffer */
351 uint32_t i;
352
353 for (i = 0; i < count; i++)
354 {
355 ret = BSP_SD_ReadBlocks_DMA((uint32_t*)scratch, sector, 1);
356 sector++;
357 if (ret == MSD_OK )
358 {
359 /* wait until the read is successful or a timeout occurs */
360#if (osCMSIS < 0x20000U)
361 /* wait for a message from the queue or a timeout */
362 event = osMessageGet(SDQueueID, SD_TIMEOUT);
363
364 if (event.status == osEventMessage)
365 {
366 if (event.value.v == READ_CPLT_MSG)
367 {
368 timer = osKernelSysTick();
369 /* block until SDIO IP is ready or a timeout occur */
370 while(osKernelSysTick() - timer <SD_TIMEOUT)
371#else
372 status = osMessageQueueGet(SDQueueID, (void *)&event, NULL, SD_TIMEOUT);
373 if ((status == osOK) && (event == READ_CPLT_MSG))
374 {
375 timer = osKernelGetTickCount();
376 /* block until SDIO IP is ready or a timeout occur */
377 ret = MSD_ERROR;
378 while(osKernelGetTickCount() - timer < SD_TIMEOUT)
379#endif
380 {
381 ret = BSP_SD_GetCardState();
382
383 if (ret == MSD_OK)
384 {
385 break;
386 }
387 }
388
389 if (ret != MSD_OK)
390 {
391 break;
392 }
393#if (osCMSIS < 0x20000U)
394 }
395 }
396#else
397 }
398#endif
399#if (ENABLE_SD_DMA_CACHE_MAINTENANCE == 1)
400 /*
401 *
402 * invalidate the scratch buffer before the next read to get the actual data instead of the cached one
403 */
404 SCB_InvalidateDCache_by_Addr((uint32_t*)scratch, BLOCKSIZE);
405#endif
406 memcpy(buff, scratch, BLOCKSIZE);
407 buff += BLOCKSIZE;
408 }
409 else
410 {
411 break;
412 }
413 }
414
415 if ((i == count) && (ret == MSD_OK ))
416 {
417 res = eRES_OK;
418 }
419 }
420#endif
421 return res;
422}

428SDRESULT SD_read(uint8_t* const buff, uint32_t sector, uint32_t count)
429{
430 SDRESULT res = eRES_ERROR;
431
432 for (uint32_t block = 0; block < count; block++)
433 {
434 for (uint8_t retry = 0; retry < SD_REINIT_MAX_RETRIES; retry++)
435 {
436 res = SD_read_drv(&buff[block*hsd1.SdCard.BlockSize], sector+block, 1);
437 if (res != eRES_OK)
438 {
439 SD_reinit();
440 } else
441 {
442 break;
443 }
444 }
445 if (eRES_OK != res)
446 {
447 break;
448 }
449 }
450
451 return res;
452}

455static SDRESULT SD_write_drv(const uint8_t *buff, uint32_t sector, uint32_t count)
456{
457 SDRESULT res = eRES_ERROR;
458 uint32_t timer;
459
460#if (osCMSIS < 0x20000U)
461 osEvent event;
462#else
463 uint16_t event;
464 osStatus_t status;
465#endif
466
467#if defined(ENABLE_SCRATCH_BUFFER)
468 uint8_t ret = MSD_ERROR;
469#endif
470
471 /*
472 * ensure the SDCard is ready for a new operation
473 */
474
475 if (!SD_IsReadyWithTimeout(SD_TIMEOUT))
476 {
477 return eRES_NOT_READY;
478 }
479
480#if defined(ENABLE_SCRATCH_BUFFER)
481 if ((uint32_t)0 == ((uint32_t)buff & (uint32_t)0x3))
482 {
483#endif
484#if (ENABLE_SD_DMA_CACHE_MAINTENANCE == 1)
485 uint32_t alignedAddr;
486 /*
487 the SCB_CleanDCache_by_Addr() requires a 32-uint8_t aligned address
488 adjust the address and the D-Cache size to clean accordingly.
489 */
490 alignedAddr = (uint32_t)buff & ~0x1F;
491 SCB_CleanDCache_by_Addr((uint32_t*)alignedAddr, count*BLOCKSIZE + ((uint32_t)buff - alignedAddr));
492#endif
493
494 if(BSP_SD_WriteBlocks_DMA((uint32_t*)buff,
495 sector,
496 count) == MSD_OK)
497 {
498#if (osCMSIS < 0x20000U)
499 /* Get the message from the queue */
500 event = osMessageGet(SDQueueID, SD_TIMEOUT);
501
502 if (event.status == osEventMessage)
503 {
504 if (event.value.v == WRITE_CPLT_MSG)
505 {
506#else
507 status = osMessageQueueGet(SDQueueID, (void *)&event, NULL, SD_TIMEOUT);
508 if ((status == osOK) && (event == WRITE_CPLT_MSG))
509 {
510#endif
511 #if (osCMSIS < 0x20000U)
512 timer = osKernelSysTick();
513 /* block until SDIO IP is ready or a timeout occur */
514 while(osKernelSysTick() - timer < SD_TIMEOUT)
515#else
516 timer = osKernelGetTickCount();
517 /* block until SDIO IP is ready or a timeout occur */
518 while(osKernelGetTickCount() - timer < SD_TIMEOUT)
519#endif
520 {
521 if (BSP_SD_GetCardState() == SD_TRANSFER_OK)
522 {
523 res = eRES_OK;
524 break;
525 }
526 }
527#if (osCMSIS < 0x20000U)
528 }
529 }
530#else
531 }
532#endif
533 }
534#if defined(ENABLE_SCRATCH_BUFFER)
535 }
536 else
537 {
538 /* Slow path, fetch each sector a part and memcpy to destination buffer */
539 uint32_t i;
540
541#if (ENABLE_SD_DMA_CACHE_MAINTENANCE == 1)
542 /*
543 * invalidate the scratch buffer before the next write to get the actual data instead of the cached one
544 */
545 SCB_InvalidateDCache_by_Addr((uint32_t*)scratch, BLOCKSIZE);
546#endif
547 for (i = 0; i < count; i++)
548 {
549 memcpy((void *)scratch, buff, BLOCKSIZE);
550 buff += BLOCKSIZE;
551
552 ret = BSP_SD_WriteBlocks_DMA((uint32_t*)scratch, sector, 1);
553 sector++;
554 if (ret == MSD_OK )
555 {
556 /* wait until the read is successful or a timeout occurs */
557#if (osCMSIS < 0x20000U)
558 /* wait for a message from the queue or a timeout */
559 event = osMessageGet(SDQueueID, SD_TIMEOUT);
560
561 if (event.status == osEventMessage)
562 {
563 if (event.value.v == READ_CPLT_MSG)
564 {
565 timer = osKernelSysTick();
566 /* block until SDIO IP is ready or a timeout occur */
567 while(osKernelSysTick() - timer <SD_TIMEOUT)
568#else
569 status = osMessageQueueGet(SDQueueID, (void *)&event, NULL, SD_TIMEOUT);
570 if ((status == osOK) && (event == READ_CPLT_MSG))
571 {
572 timer = osKernelGetTickCount();
573 /* block until SDIO IP is ready or a timeout occur */
574 ret = MSD_ERROR;
575 while(osKernelGetTickCount() - timer < SD_TIMEOUT)
576#endif
577 {
578 ret = BSP_SD_GetCardState();
579
580 if (ret == MSD_OK)
581 {
582 break;
583 }
584 }
585
586 if (ret != MSD_OK)
587 {
588 break;
589 }
590#if (osCMSIS < 0x20000U)
591 }
592 }
593#else
594 }
595#endif
596 }
597 else
598 {
599 break;
600 }
601 }
602
603 if ((i == count) && (ret == MSD_OK ))
604 {
605 res = eRES_OK;
606 }
607 }
608
609#endif
610
611 return res;
612}
