112uint8_t CubeADCS_CRC_8(uint8_t *buf, uint16_t size)
113{
114 if ( size == 0 )
115 return 0xFF;
116
117 uint16_t i;
118 uint8_t crc = 0;
119
120 for ( i = 0; i < size; i++ ) {
121 crc = CubeADCS_CRC_8_Tab[ crc ^ buf[i] ];
122 }
123
124 return crc;
125
126}

128uint16_t CubeADCS_CRC_16(uint8_t *u8Start, uint32_t u8Len)
129{
130 uint8_t *u8Data;
131 uint8_t *u8End = u8Start + u8Len;
132
133 uint16_t u16Crc = 0;
134
135 for (u8Data = u8Start; u8Data < u8End; u8Data++)
136 CubeADCS_Calculate_Single_CRC_16(u16Crc, *u8Data);
137
138 return u16Crc;
139}
140
141uint16_t CubeADCS_Calculate_Single_CRC_16(uint16_t u16Crc, uint8_t u8Data)
142{
143 u16Crc = (u16Crc >> 8) | (u16Crc << 8);
144 u16Crc ^= u8Data;
145 u16Crc ^= (u16Crc & 0xff) >> 4;
146 u16Crc ^= u16Crc << 12;
147 u16Crc ^= (u16Crc & 0xff) << 5;
148
149 return u16Crc;
150}
