123// Request EPS I module to switch on or off a channel
124bool eps_i_set_channel_output(EPS1_QueryControlsIndex_t channel, EPS1_QueryState_t state){
125
126 // Channel mask should be a valid value from the enum type
127 if(EPS1_QUERYCONTROLSINDEX_MAX_CNT <= channel)
128 {
129 CRIT_ASSERT(false); // Catch this error while in debug
130 return false;
131 }
132
133 // Initialize the input to 0
134 EPS1_QueryState_t req_data [EPS1_QUERYCONTROLSINDEX_MAX_CNT] = {0};
135
136 // Populate the request information
137 req_data[channel] = state;
138
139 // This is the esps request
140 return (ESSATMAC_EC_OK == EPS1_queryControlsReq(&fp_context, req_data));
141
142}

192// Process response GetDistributionInfo
193void eps_i_info_to_datacache(const EPS1getBatteryInfoResponseData_t* battery_info){
194 static DATA_CACHE_EPS_0_t eps_0_data = {0};
195
196 // Convert the raw value from the EPS 1 into human readable values (mv, ma)
197 int32_t batt_voltage = conv_batt_volt_mv * (int32_t) battery_info->sBattery.sBatt.u16Volt;
198 int32_t batt_current = conv_batt_curr_ma * (int32_t) battery_info->sBattery.sBatt.u16Curr;
199
200 // We cannot obtain all the values from the EPS I, some of them must be estimated somewhere else
201 // Populate the DATA_CACHE_EPS_0_t struct
202 eps_0_data.i64BattEnergy = (int64_t) 0x0;
203 eps_0_data.i64BattCharge = (int64_t) 0x0;
204 eps_0_data.i64BattChargeCapacity = (int64_t) 0x0;
205 eps_0_data.i64BattPercent = (int64_t) 0x0;
206 eps_0_data.i32BattVoltage = batt_voltage;
207 eps_0_data.i32BattCurrent = batt_current;
208 eps_0_data.i32BattTemperature = (int32_t) 0x0;
209
210 // Update the datacache
211 dc_set_eps_0_data(&eps_0_data);
212}

258static void eps_i_task(void * args){
259 (void)args;
260
261 for(;;) {
262 if (SAFE_TRUE == ESSA_Stack_Facade_Is_Init())
263 {
264 // Send request to the EPS I
