87bool eps_ctrl_init(void)
88{
89 bool res = true;
90
91 // Initialize the requests queue
92 if (EPS_CTRL_DONE != eps_ctrl_queue_init())
93 {
94 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_EPS_CTRL_OS_ERROR);
95 res = false;
96 }
97
98 // Initialize the eps_ctrl_config
99 if (EPS_CTRL_DONE != eps_ctrl_cfg_init())
100 {
101 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_EPS_CTRL_OS_ERROR);
102 res = false;
103 }
104
105 // Initialize the event flag
106 eps_ctrl_request_evt = osEventFlagsNew(NULL);
107 if(NULL == eps_ctrl_request_evt)
108 {
109 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_EPS_CTRL_OS_ERROR);
110 res = false;
111 }
112
113 // Initialize the task
114 h_eps_ctrl_task = osThreadNew(&eps_ctrl_task, NULL, &eps_ctrl_task_attributes);
115 if(NULL == h_eps_ctrl_task)
116 {
117 EXEH_HANDLE(eEXEHSeverity_Error, eEXEH_EPS_CTRL_OS_ERROR);
118 res = false;
119 }
120
121 task_mon_set_task_policy(TASK_EPS_CTRL, TASK_MON_POLICY_MONITOR);
122
123 // Initialize the state machine
124 trigger_hsm_event(eHSM_StdEvent_Reset);
125
126 return res;
127}

142eps_ctrl_result_t eps_ctrl_set_channel_output(eps_ctrl_channels_t channel, bool state)
143{
144 eps_ctrl_result_t res = EPS_CTRL_FAIL;
145 uint32_t wait_flag = 0;
146
147 // Protect the queue data structures
148 res = eps_ctrl_queue_add(channel, state, EPS_CTRL_REQ_SET_CH, &wait_flag);
149
150 if (EPS_CTRL_DONE == res)
151 {
152 // Send state machine event
153 trigger_hsm_event(ev_eps_ctrl_sm_new_request);
154
155 // Wait for the channel flag to be set from the ESPS response callback
156 uint32_t wait_res = osEventFlagsWait(eps_ctrl_request_evt, wait_flag,
157 osFlagsWaitAny, EPS_CTRL_REQ_TIMEOUT);
158
159 // Check for os Errors
160 if (osFlagsErrorTimeout == wait_res)
161 {
162 eps_ctrl_cancel_req_wait(channel, wait_flag);
163 res = EPS_CTRL_TIMEOUT;
164 }
165 else if (osFlagsError == (osFlagsError & wait_res))
166 {
167 eps_ctrl_cancel_req_wait(channel, wait_flag);
168 res = EPS_CTRL_FAIL;
169 }
170 else
171 {
172 // Check if the last operation failed
173 bool dummy_state = false; // The SET commands don't return the current state
174 bool b_success = eps_ctrl_cfg_get_last_req(channel, &dummy_state);
175 if(false == b_success)
176 {
177 res = EPS_CTRL_FAIL;
178 }
179 }
180 }
181 return res;
182}

184eps_ctrl_result_t eps_ctrl_get_channel_output(eps_ctrl_channels_t channel, bool* p_state)
185{
186 eps_ctrl_result_t res = EPS_CTRL_FAIL;
187 uint32_t wait_flag = 0;
188
189 // Protect the queue data structures
190 res = eps_ctrl_queue_add(channel, true, EPS_CTRL_REQ_GET_CH, &wait_flag);
191
192 if (EPS_CTRL_DONE == res)
193 {
194 // Send state machine event
195 trigger_hsm_event(ev_eps_ctrl_sm_new_request);
196
197 // Wait for the channel flag to be set from the ESPS response callback
198 uint32_t wait_res = osEventFlagsWait(eps_ctrl_request_evt, wait_flag,
199 osFlagsWaitAll, EPS_CTRL_REQ_TIMEOUT);
200
201 // Check for os Errors
202 if (osFlagsErrorTimeout == wait_res)
203 {
204 eps_ctrl_cancel_req_wait(channel, wait_flag);
205 res = EPS_CTRL_TIMEOUT;
206 }
207 else if (osFlagsError == (osFlagsError & wait_res))
208 {
209 eps_ctrl_cancel_req_wait(channel, wait_flag);
210 res = EPS_CTRL_FAIL;
211 }
212 else
213 {
214 // Check if the last operation failed and update the p_state
215 bool b_success = eps_ctrl_cfg_get_last_req(channel, p_state);
216 if(false == b_success)
217 {
218 res = EPS_CTRL_FAIL;
219 }
220 }
221 }
222 return res;

230void eps_ctrl_unblock_user_thread(void)
231{
232 if (false != eps_ctrl_cfg_service_lock_get())
233 {
234 eps_ctrl_channels_t active_request_channel;
235 uint32_t active_wait_flag;
236
237 if (false != eps_ctrl_cfg_get_channel_and_flag(&active_request_channel, &active_wait_flag))
238 {
239 const eps_ctrl_request_t* p_active_request = eps_ctrl_queue_find(active_request_channel, active_wait_flag);
240 if ((NULL != p_active_request) && (false == p_active_request->wait_cancelled))
241 {
242 // Trigger the active request flag
243 osEventFlagsSet(eps_ctrl_request_evt, active_wait_flag);
244 }
245 else
246 {
247 // There is no thread waiting on the currently processed flag with a request for the currently processed channel
248 }
249 }
250
251 eps_ctrl_cfg_service_lock_free();
252 }

277static void eps_ctrl_task(void * args)
278{
279 (void) args;
280 // Loop triggering the periodic events in the state machine
281
282 // Initialize the State Machine once the OS Kernel is initialized
283 trigger_hsm_event(ev_eps_ctrl_sm_initialized);
284
285 // Trigger new_request event if there are untreated requests in the queue
286 if(false == eps_ctrl_queue_is_empty())
287 {
288 trigger_hsm_event(ev_eps_ctrl_sm_new_request);
289 }
290
291 for (;;)
292 {
293 // execute periodic SM activities...
294 (void) trigger_hsm_event(eHSM_StdEvent_Periodic);
295
296 osDelay(EPS_CTRL_TASK_PERIOD_MS);
297
298 TaskMonitor_IamAlive(TASK_EPS_CTRL);
299 vApplicationLowStackCheck(TASK_EPS_CTRL);
300 }
301}

313static bool eps_ctrl_cancel_req_wait(const eps_ctrl_channels_t channel, const uint32_t wait_flag)
314{
315 bool res = false;
316 if (false != eps_ctrl_cfg_service_lock_get())
317 {
318 eps_ctrl_request_t* p_req = eps_ctrl_queue_find(channel, wait_flag);
319 if (NULL != p_req)
320 {
321 p_req->wait_cancelled = true;
322 // This is a robustness measure for the case when a notification is sent before the "cancelled"
323 // flag for the request has been updated. In this case the requesting thread is no longer waiting
324 // on the event which is already set
325 (void)osEventFlagsClear(eps_ctrl_request_evt, wait_flag);
326
327 res = true;
328 }
329
330 eps_ctrl_cfg_service_lock_free();
331 }
332
333 return res;
334}
